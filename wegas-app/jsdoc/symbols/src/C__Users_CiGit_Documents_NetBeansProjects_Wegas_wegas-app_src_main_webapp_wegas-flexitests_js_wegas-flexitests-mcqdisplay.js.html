 13 YUI.add("wegas-flexitests-mcqdisplay", function(Y) {
 14     "use strict";
 15     var INITIALVALUE = "flexi_initial_value";
 16 
 17     Y.namespace('Wegas').FlexitestsMCQ = Y.Base.create("wegas-flexitests-mcqdisplay", Y.Widget,
 18             [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 19 
 20         /**
 21          * Lifecycle method
 22          * @function
 23          * @private
 24          * @returns {undefined}
 25          */
 26         initializer: function() {
 27             this.events = [];
 28         },
 29         /**
 30          * Lifecycle method
 31          * @function
 32          * @private
 33          * @returns {undefined}
 34          */
 35         renderUI: function() {
 36             var cb = this.get("contentBox");
 37             cb.append("<div class='" + this.getClassName("feedback") + "'></div>");
 38             cb.append("<div class='" + this.getClassName("input") + "'></div>");
 39         },
 40         /**
 41          * Lifecycle method
 42          * @function
 43          * @private
 44          * @returns {undefined}
 45          */
 46         syncUI: function() {
 47             if (this.get("variable.evaluated")) {
 48                 this.generators[this.get("responseType")].apply(this);
 49             } else {
 50                 // Display warning
 51             }
 52         },
 53         /**
 54          * Lifecycle method
 55          * @function
 56          * @private
 57          * @returns {undefined}
 58          */
 59         bindUI: function() {
 60             this.events.push(this.get("contentBox").delegate("click", function(e) {
 61                 e.halt(true);
 62                 this.fire("clientResponse", {value: e.currentTarget.getData("reference")});
 63             }, "." + this.getClassName("input") + " > span", this));
 64             this.events.push(this.get("contentBox").one("." + this.getClassName("input")).delegate("change", function(e) {
 65                 e.halt(true);
 66                 this.fire("clientResponse", {value: e.target.getDOMNode().value});
 67                 e.target.getDOMNode().value = INITIALVALUE;
 68             }, "select", this));
 69             this.events.push(this.get("contentBox").one("." + this.getClassName("input")).delegate("change", function(e) {
 70                 e.halt(true);
 71                 if (e.target.getDOMNode().value) {
 72                     this.fire("clientResponse", {value: e.target.getDOMNode().value});
 73                     e.target.getDOMNode().checked = false;
 74                 }
 75             }, "form", this));
 76             this.events.push(this.get("root").get("boundingBox").after("keypress", function(e) {
 77                 this.keyPressed(String.fromCharCode(e.keyCode));
 78             }, this));
 79         },
 80         keyPressed: function(charCode) {
 81             var choice, counter = 0, i;
 82             if (this.get("responseType") !== "key") {
 83                 return;
 84             }
 85             switch (charCode) {
 86                 case "f":
 87                     choice = 0;
 88                     break;
 89                 case "h":
 90                     choice = 1;
 91                     break;
 92                 default:
 93                     return;
 94             }
 95             for (i in this.get("variable.evaluated").get("properties")) {
 96                 if (counter === choice) {
 97                     this.fire("clientResponse", {value: i});
 98                     break;
 99                 }
100                 counter+=1;
101             }
102 
103         },
104         success: function() {
105             if (+this.get("feedback") > 0) {
106                 this.get("contentBox").one("." + this.getClassName("feedback")).setContent("<span class='flexitests-mcq-good'></span>");
107                 this.get("contentBox").one("." + this.getClassName("input")).hide();
108                 Y.later(+this.get("feedback"), this, this.clearFeedBack);
109             }
110         },
111         error: function() {
112             if (+this.get("feedback") > 0) {
113                 this.get("contentBox").one("." + this.getClassName("feedback")).setContent("<span class='flexitests-mcq-wrong'></span>");
114                 this.get("contentBox").one("." + this.getClassName("input")).hide();
115                 Y.later(+this.get("feedback"), this, this.clearFeedBack);
116             }
117         },
118         clearFeedBack: function() {
119             try {
120                 this.get("contentBox").one("." + this.getClassName("feedback")).empty();
121                 this.get("contentBox").one("." + this.getClassName("input")).show();
122             } catch (e) {
123                 //page changed?
124             }
125         },
126         save: function(el) {
127             var id = el.index;
128             delete el.index;
129             Y.Wegas.Facade.VariableDescriptor.sendRequest({
130                 request: "/Script/Run/" + Y.Wegas.Facade.Game.get('currentPlayerId'),
131                 cfg: {
132                     method: "POST",
133                     data: Y.JSON.stringify({
134                         "@class": "Script",
135                         language: "JavaScript",
136                         content: "store(" + this.get("variable.evaluated").get("name") + ",'" + id + "','" + Y.JSON.stringify(el) + "');"
137                     })
138                 },
139                 on: {
140                     failure: Y.bind(function(e) {
141                         Y.log("error", "Failed to store data", "Y.Wegas.FlexitestsMCQ");
142                     }, this)
143                 }
144             });
145         },
146         generators: {
147             link: function() {
148                 var cb = this.get("contentBox"),
149                         inputDiv = cb.one("." + this.getClassName("input")),
150                         question = this.get("variable.evaluated");
151                 inputDiv.empty();
152                 for (var i in question.get("properties")) {
153                     inputDiv.append("<span data-reference='" + i + "'>" + question.get("properties")[i] + "</span><br/>");
154                 }
155             },
156             selector: function() {
157                 var cb = this.get("contentBox"),
158                         inputDiv = cb.one("." + this.getClassName("input")),
159                         question = this.get("variable.evaluated"),
160                         engine = new Y.Template(Y.Template.Micro),
161                         render = engine.compile("<select><option value='" + INITIALVALUE + "'>Choose:</option>" +
162                         "<% for(var i in this.get('properties')){ %>" +
163                         "<option value='<%= i%>'><%= this.get('properties')[i] %></option>" +
164                         "<% } %></select>");
165                 inputDiv.empty();
166                 inputDiv.append(render(question));
167             },
168             key: function() {
169                 this.get("contentBox").one("." + this.getClassName("input")).empty();
170             },
171             radio: function() {
172                 var cb = this.get("contentBox"),
173                         inputDiv = cb.one("." + this.getClassName("input")),
174                         question = this.get("variable.evaluated"),
175                         engine = new Y.Template(Y.Template.Micro),
176                         render = engine.compile("<form>" +
177                         "<% for(var i in this.get('properties')){ %>" +
178                         "<label><input type='radio' name='mcq-flexi-radio' value='<%= i %>'><span><%= this.get('properties')[i] %></span></label>" +
179                         "<% } %>" +
180                         "</form>");
181                 inputDiv.empty();
182                 inputDiv.append(render(question));
183             }
184         },
185         /**
186          * Lifecycle method
187          * @function
188          * @private
189          * @returns {undefined}
190          */
191         destructor: function() {
192             for (var i in this.events) {
193                 this.events[i].detach();
194             }
195         }
196     }, {
197         EDITORNAME: "Flexitests MCQ Display",
198         ATTRS: {
199             feedback: {
200                 value: 1000,
201                 type: "number",
202                 _inputex: {
203                     label: "Feedback duration (ms)",
204                     description: "0 or less to disable."
205                 }
206             },
207             responseType: {
208                 value: "link",
209                 type: "string",
210                 _inputex: {
211                     label: "Response layout"
212                 },
213                 choices: [
214                     {
215                         value: "link",
216                         label: "Link"
217                     },
218                     {
219                         value: "selector",
220                         label: "Selector"
221                     },
222                     {
223                         value: "key",
224                         label: "Keyboard F/H"
225                     },
226                     {
227                         value: "radio",
228                         label: "Horizontal scale"
229                     }
230                 ]
231             },
232             variable: {
233                 /**
234                  * The target variable, returned either based on the name attribute,
235                  * and if absent by evaluating the expr attribute.
236                  */
237                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
238                 _inputex: {
239                     _type: "variableselect",
240                     legend: "Store results",
241                     classFilter: ["ObjectDescriptor"]
242                 }
243             }
244         }
245     });
246 });
