 14 YUI.add('wegas-app', function(Y) {
 15     "use strict";
 16 
 17     var Wegas = Y.namespace('Wegas');
 18     /**
 19      * Create a new wegas-app
 20      *
 21      * <p><strong>Attributes</strong></p>
 22      * <ul>
 23      *    <li>base {String} base url for app</li>
 24      *    <li>dataSources {Object[]} the list of datasource to be loaded on startup</li>
 25      * </ul>
 26      *
 27      * @name Y.Wegas.App
 28      * @class Base class for wegas, handle initialisation of datasources and rendering
 29      * @extends Y.Base
 30      * @constructor
 31      * @param Object Will be used to fill attributes field
 32      */
 33     Wegas.App = Y.Base.create("wegas-app", Y.Base, [], {
 34         /** @lends Y.Wegas.App# */
 35         // ** Public methods ** //
 36         /**
 37          * Lifecycle methods
 38          * @function
 39          * @private
 40          */
 41         initializer: function() {
 42             /**
 43              * @name render
 44              * @event
 45              */
 46             this.publish("render");
 47             /**
 48              * @name render
 49              * private
 50              */
 51             this.dataSources = {};
 52 
 53             Wegas.app = this;                                                   // Setup global references to the app
 54             Wegas.Facade = this.dataSources;                                    // and the data sources
 55         },
 56         /**
 57          * Render function
 58          * @function
 59          * @public
 60          */
 61         render: function() {
 62             var ds, dsClass, widgetCfg,
 63                     dataSources = this.get('dataSources'),                      // Data sources cfg objects
 64                     requestCounter = 0,                                         // Request counter 
 65                     onRequest = function() {                                    // When a response to initial requests is received
 66                         requestCounter -= 1;
 67                         if (requestCounter === 0) {                             // If all initial request arrived,
 68                             var widget = Wegas.Widget.create(widgetCfg);        // instantiate the root widget
 69                             widget.render();                                    // render it
 70                             this.widget = widget;                               // push a reference
 71                             this.fire("render");                                // fire a render event for some eventual post processing
 72                         }
 73                     };
 74 
 75             Y.io.header("Accept-Language", Y.config.lang);                      // Set up the language for all requests
 76             Y.on("io:failure", this.globalFailureHandler, this);                // Set up a default failure handler
 77 
 78             // Send data sources initial requests
 79             Wegas.use(Y.Object.values(dataSources), Y.bind(function(Y) {        // Retrieve data sources dependencies (e.g. Pusher)
 80                 Y.Object.each(dataSources, function(cfg, name) {                // For each data source,       
 81                     cfg.source = this.get("base") + (cfg.source || "");         // Set up datasource path
 82                     dsClass = Wegas[cfg.type] || Wegas.DataSource;              // Determine which class to use (default is Y.Wegas.DataSource)
 83                     ds = new dsClass(cfg);                                      // Instantiate the datasource
 84                     if (ds.hasInitialRequest()) {                               // If the data source has an initial request,
 85                         requestCounter += 1;                                    // increment request counter
 86                         ds.sendInitialRequest();                                // send it
 87                         ds.once("response", onRequest, this);                   // and increment the request counter
 88                     }
 89                     this.dataSources[name] = ds;                                // Push to data source list
 90                 }, this);
 91 
 92                 requestCounter += 1;
 93                 this.dataSources.Page.once("response", function(e) {            // When page data source response arrives,
 94                     widgetCfg = e.response.results;                             // store the result for later use
 95                     Wegas.use(widgetCfg, Y.bind(onRequest, this));              // Optim: Load pages dependencies as soon as the data is received
 96                 }, this);
 97             }, this));
 98             
 99             // Post render events
100             this.on("render", function() {                                      // When the first page is rendered,
101                 var body = Y.one("body");
102                 body.removeClass("wegas-loading-overlay");                      // Remove loading overlay on render
103                 body.on("key", function() {                                     // Add shortcut to activate developper mode
104                     body.toggleClass("wegas-stdmode");                          // Toggle stdmode class on body (hides any wegas-advancedfeature)
105                     Y.config.win.Y = Y;                                         // Allow access to Y instance
106                 }, "167", this);                                                // on key '�' pressed
107             }, this);
108         },
109         /**
110          * Destructor methods.
111          * @function
112          * @public
113          */
114         destructor: function() {
115             Y.Object.each(this.dataSources, function(i) {
116                 i.destroy();
117             });
118         },
119         // ** Private methods ** //
120         /**
121          * 
122          * @param {type} tId
123          * @param {type} req
124          * @param {type} e
125          * @returns {undefined}
126          */
127         globalFailureHandler: function(tId, req, e) {                           // Add a global io failure listener
128             var response, msg;
129             try {
130                 msg = "Error sending " + e.cfg.method + " request : " + e.target.get("source") + e.request
131                         + ", " + e.cfg.data + ": ";
132             } catch (e) {
133                 msg = "Error sending request: ";
134             }
135             try {
136                 response = Y.JSON.parse(req.responseText);
137                 msg += "\n Server reply " + Y.JSON.stringify(response, null, "\t");
138 
139                 if (response.exception === "org.apache.shiro.authz.UnauthenticatedException") {// If the user session has timed out,
140                     new Wegas.Panel({                                           // show a message that invites to reconnect
141                         content: "<div class='icon icon-info'>You have been logged out.</div>",
142                         modal: true,
143                         centered: true,
144                         buttons: {
145                             footer: [{
146                                     label: 'Click here to reconnect',
147                                     action: Y.config.win.location.reload
148                                 }]
149                         }
150                     }).render();
151                     e.preventDefault();
152                 } //else if (r.exception === "org.apache.shiro.authz.UnauthorizedException") {
153                 // @todo Do something?
154                 //}
155             } catch (e) {                                                       // Error while parsing json
156                 msg += "\n Server reply " + (req && req.responseText);
157             }
158             Y.log(msg, "error", "Y.Wegas.App");                                 // Log the error
159         }
160     }, {
161         /** @lends Y.Wegas.App */
162         /**
163          * @field
164          * @static
165          */
166         ATTRS: {
167             dataSources: {
168                 value: {}
169             },
170             /**
171              * Base url for app
172              */
173             base: {
174                 getter: function() {
175                     return Y.config.groups.wegas.base.replace("wegas-app/", "");
176                 }
177             }
178         }
179     });
180 });
