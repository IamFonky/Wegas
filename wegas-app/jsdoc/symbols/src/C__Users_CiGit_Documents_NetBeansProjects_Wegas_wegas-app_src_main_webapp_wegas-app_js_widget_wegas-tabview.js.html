177     /**
178      * Extension enabling a Tab to be a parent of another Widget.
179      *
180      * @modified from original WidgetParent module
181      *
182      * @module widget-parent
183      * @param {Object} config

185     function Parent(config) {
186         //Y.WidgetParent.call(this, config);
187         this.publish("addChild", {
188             defaultTargetOnly: true,
189             defaultFn: this._defAddChildFn
190         });
191         this.publish("removeChild", {
192             defaultTargetOnly: true,
193             defaultFn: this._defRemoveChildFn
194         });
195 
196         this._items = [];
197 
198         var children, handle;
199 
200         if (config && config.children) {
201 
202             children = config.children;
203 
204             handle = this.after("initializedChange", function(e) {
205                 this._add(children);
206                 handle.detach();
207             });
208 
209         }
210 
211         //  Widget method overlap
212         Y.after(this._renderChildren, this, "renderUI");
213         Y.after(this._bindUIParent, this, "bindUI");
214 
215         //this.after("selectionChange", this._afterSelectionChange);
216         //this.after("selectedChange", this._afterParentSelectedChange);
217         //this.after("activeDescendantChange", this._afterActiveDescendantChange);
218 
219         this._hDestroyChild = this.after("*:destroy", this._afterDestroyChild);
220         this.after("*:focusedChange", this._updateActiveDescendant);
221     }

230     /**
231      * @name Y.Wegas.Tab
232      * @extends Y.Tab

238     Tab = Y.Base.create("tab", Y.Tab, [Parent, Y.WidgetChild, Wegas.Editable, Wegas.Parent], {
239         /** @lends Y.Wegas.Tab# */
240         PANEL_TEMPLATE: '<div><div class=\"panel-inner\"></div></div>',
241         // *** Private Fields *** //
242         // *** Lifecycle Methods *** //
243         /**
244          * @function
245          * @private
246          * @param cfg
247          * @description Set variables with initials values.
248          * call explicitly initializer method of widget parent.
249          * assign this tab in owner TabView.
250          */
251         initializer: function(cfg) {
252             Tab.superclass.initializer.apply(this, arguments);
253             TabView.tabs[cfg.id || cfg.label] = this;
254             this._witems = [];
255             this.on("addChild", function(e) {
256                 this._witems.push(e.child);
257             });
258 
259             //this.plug(Closable);
260             this.plug(Plugin.PopupListener, {
261                 targetAttr: "panelNode",
262                 alignAttr: "panelNode"
263             });
264         },
265         /**
266          * @function
267          * @private
268          * @description call render method of widget parent.
269          */
270         renderUI: function() {
271             Tab.superclass.renderUI.apply(this, arguments);
272         },
273         /**
274          * @function
275          * @private
276          * @description call sync method of widget parent.
277          */
278         syncUI: function() {
279             Tab.superclass.syncUI.apply(this, arguments);
280         },
281         /**
282          * @function
283          * @private
284          * @description delete Tab in owner Tabview.
285          *  call explicitly destructor method of widget parent.
286          */
287         destructor: function() {
288             var i;
289             for (i = 0; i < this._witems.length; i += 1) {
290                 this._witems[i].destroy();
291             }
292 
293             delete TabView.tabs[this.get("id")];
294             Tab.superclass.destructor.apply(this, arguments);
295         },
296         _onActivate: function(e) {
297             if (e.target.get("disabled")) {
298                 return;
299             }
300             Tab.superclass._onActivate.apply(this, arguments);
301         },
302         // *** Private Methods *** //
303         /**
304          * @function
305          * @private
306          * @param cfg
307          * @param callback
308          * @description Retrieves the given widget configuration and add it
309          * to the tab
310          */
311         load: function(cfg, callback) {
312             this.showOverlay();
313             Wegas.Widget.use(cfg, Y.bind(function(cfg, callback) {            // Load the subpage dependencies
314                 this.hideOverlay();
315                 var widgets = this.add(cfg);                                    // Render the subpage
316                 if (callback) {
317                     callback(widgets.item(0), this);                            // Trigger the callback
318                 }
319             }, this, cfg, callback));
320         },
321         /**
322          * @function
323          * @private
324          * @description Override Y.WidgetParent to render children in the panel node;
325          * to the tab
326          */
327         _renderChildren: function() {
328             var renderTo = this._childrenContainer || this.get("panelNode").one("div");    // @modified
329 
330             this._childrenContainer = renderTo;
331 
332             this.each(function(child) {
333                 child.render(renderTo);
334             }, this);
335         },
336         /**
337          * @function
338          * @private
339          * @param index
340          * @description return the widget item at given index
341          */
342         witem: function(index) {
343             return this._witems[index];
344         }
345     }, {
346         CSS_PREFIX: "yui3-tab",
347         /** @lends Y.Wegas.Tab */
348 
349         /**
350          * @field
351          * @static

362             label: {

365             content: {
366                 "transient": true,
367                 setter: function() {
368                 }
369             },
370             panelNode: {
371                 "transient": true
372             }
373         }

442     /**
443      * Plugin to toggle visibility of a tab
444      *
445      * @name Y.Plugin.LayoutToggleTab
446      * @extends Y.Plugin.Base
447      * @class plugin to toggle visibility of a tab
448      * @constructor
449      */
450     var LayoutToggleTab = function() {
451         LayoutToggleTab.superclass.constructor.apply(this, arguments);
452     };
453     Y.extend(LayoutToggleTab, Plugin.Base, {
454         /** @lends Y.Wegas.LayoutToggleTab# */
455 
456         // *** Private fields *** //
457         /**
458          * @function
459          * @private
460          * @description if "removeChild" is fired by host, hide tab.
461          *  If "addChild" is fired by host, show tab.
462          */
463         initializer: function() {
464             this.afterHostEvent("removeChild", function() {
465                 Y.later(100, this, function() {
466                     if (this.get("host").isEmpty()) {
467                         Wegas.app.widget.hidePosition("right");
468                     }
469                 });
470             });
471             this.onHostEvent("addChild", function() {
472                 if (this.get("host").isEmpty()) {
473                     Wegas.app.widget.showPosition("right");
474                 }
475             });
476         }
477     }, {
478         NS: "LayoutToggleTab",
479         NAME: "LayoutToggleTab"
480     });
481     Y.namespace("Plugin").LayoutToggleTab = LayoutToggleTab;

483     /**
484      * Override to add support for disabled tabs
485      * @param {type} e
486      */
487     Y.Tab.prototype._onActivate = function(e) {
488         if (e.target === this && !this.get("disabled")) {   /* modified */
489             //  Prevent the browser from navigating to the URL specified by the
490             //  anchor's href attribute.
491             e.domEvent.preventDefault();
492             e.target.set('selected', 1);
493         }
