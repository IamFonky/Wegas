307         /**
308          * @function
309          * @private
310          * @param email
311          * @param password
312          * @param remember
313          * @description Send REST request to login a user with form's values
314          *  informations.
315          */
316         login: function(email, password, remember) {
317             this.loginRequest("/Authenticate/?email=" + email + "&password=" + password + "&remember=" + remember);
318         },
319         loginRequest: function(url) {
320             Wegas.Facade.User.sendRequest({
321                 request: url,
322                 cfg: {
323                     method: "POST"
324                 },
325                 on: {
326                     success: Y.bind(function(e) {
327                         this.showMessage("success", "Login successful");
328                         window.location = Wegas.Helper.getURLParameter("redirect") || Wegas.app.get("base");
329                     }, this),
330                     failure: Y.bind(function(e) {
331                         this.showMessage("error", e.response.results.message || "Email/password combination not found", 6000);
332                         //this.showMessage("error", e.response.results.message || "Guest login failed", 6000);
333                     }, this)
334                 }
335             });
336         },
337         /**
338          * @function
339          * @private
340          * @param data
341          * @description Send REST request to create a new account based on the
342          *  form's values
343          */
344         createAccount: function(data) {
345             Wegas.Facade.User.sendRequest({
346                 request: "/Signup/",
347                 cfg: {
348                     method: "POST",
349                     data: data
350                 },
351                 on: {
352                     success: Y.bind(function(data, e) {
353                         //this.showMessage("success", "User created", 6000);
354                         //this.set("mode", "login");
355                         this.login(data.email, data.password, false);
356                     }, this, data),
357                     failure: Y.bind(function(e) {
358                         this.showMessage("error", e.response.results.message || "Error creating user");
359                     }, this)
360                 }
361             });
362         },
363         /**
364          * @function
365          * @private
366          * @param email
367          * @description Send REST request to set and set a new password for
368          *  the user.
369          */
370         sendNewPassword: function(email) {
371             Wegas.Facade.User.sendRequest({
372                 request: "/SendNewPassword/?email=" + email,
373                 cfg: {
374                     method: "POST"
375                 },
376                 on: {
377                     success: Y.bind(function(e) {
378                         this.hideOverlay();
379                         this.showMessage("success", "A new password has been sent", 4000);
380                         this.changeRightForms(false);
381                     }, this),
382                     failure: Y.bind(function(e) {
383                         this.hideOverlay();
384                         this.showMessage("error", e.response.results.message || "Error sending new password", 4000);
385                     }, this)
386                 }
387             });
388         }

392     /**
393      * @Hack because "typeInvite" and password don't work (typeInvite is hidden)
394      * Change color property (in grey) when input is not focused (black else)
395      */
396     Y.inputEx.PasswordField.prototype.updateTypeInvite = function() {
397         var divEl = Y.one(this.divEl);
398         if (!this._fakePass && this.options.typeInvite) {
399             this._fakePass = Y.Node.create("<input type='text'>");
400             this._fakePass.set("value", this.options.typeInvite);
401             this._fakePass.setAttribute("class", this.el.getAttribute("class"));
402             this._fakePass.hide();
403             Y.one(this.el).ancestor().prepend(this._fakePass);
404             this._fakePass.on("focus", function() {
405                 Y.one(this.el).show();
406                 this._fakePass.hide();
407                 Y.one(this.el).focus();
408             }, this);
409         }
410         // field not focused
411         if (!divEl.hasClass("inputEx-focused")) {
412             // show type invite if field is empty
413             if (this.isEmpty()) {
414                 divEl.addClass("inputEx-typeInvite");
415                 Y.one(this.el).hide();
416                 this._fakePass.show();
417                 this.el.value = this.options.typeInvite;
418                 // important for setValue to work with typeInvite
419             } else {
420                 Y.one(this.el).show();
421                 this._fakePass.hide();
422                 divEl.removeClass("inputEx-typeInvite");
423             }
424             // field focused : remove type invite
425         } else {
426             if (divEl.hasClass("inputEx-typeInvite")) {
427                 // remove text
428                 this.el.value = "";
429                 // remove the "empty" state and class
430                 this.previousState = null;
431                 Y.one(this.el).show();
432                 this._fakePass.hide();
433                 divEl.removeClass("inputEx-typeInvite");
434             }
435         }
