 11 YUI.add('wegas-widgetmenu', function(Y) {
 12     "use strict";
 13 
 14     var HOST = "host", BOUNDINGBOX = "boundingBox",
 15             WidgetMenu, Menu, MenuBar;
 16 
 17     /**
 18      * @name Y.Plugin.WidgetMenu
 19      * @extends Y.Plugin.Base
 20      * @class Plugin to be attached to a Y.Widget, adds a menu
 21      *  that shows up when the target event is triggered (by default click). Uses
 22      *  a Y.Wegas.Menu to display the menu.
 23      * @constructor
 24      */
 25     WidgetMenu = function() {
 26         WidgetMenu.superclass.constructor.apply(this, arguments);
 27     };
 28 
 29     Y.extend(WidgetMenu, Y.Plugin.Base, {
 30         /** @lends Y.Plugin.WidgetMenu# */
 31 
 32         // *** Lifecycle methods *** //
 33         /**
 34          * @function
 35          * @private
 36          */
 37         initializer: function() {
 38             var host = this.get(HOST);
 39 
 40             if (host instanceof Y.Node) {                                       // 1st case: host is a Y.Node
 41                 this.bind();
 42             } else if (host instanceof Y.Widget) {                              // 2nd case: host is a Y.Widiget
 43                 this.afterHostEvent("render", this.bind);
 44             }
 45 
 46             this.after("childrenChange", function() {                           // On children attribute update
 47                 if (this.menu) {
 48                     this.menu.destroyAll();
 49                     this.menu.add(this.get("children"));                        // update menu items
 50                 }
 51             });
 52         },
 53         destructor: function() {
 54             if (this.menu) {
 55                 this.menu.destroy();
 56             }
 57         },
 58         bind: function() {
 59             var node = this.get("targetNode");
 60             node.delegate(this.get("event"), function(e) {                      // Target event listener
 61                 this.show(e.target);
 62                 e.halt(true);                                                   // Prevent event from bubbling
 63                 this.fire("menuOpen", {domEvent: e});                           // Notify the parent the menu has been opened
 64             }, this.get("selector"), this);
 65 
 66             node.addClass("wegas-widgetmenu-hassubmenu");                       // Add submenu class
 67             if (this.get("menuCfg.points") && this.get("menuCfg.points")[0].indexOf("b") < 0) {
 68                 node.addClass("wegas-widgetmenu-hassubmenuright");
 69             }
 70         },
 71         add: function(widget, index) {
 72             var children = this.get("children"),
 73                     i = (typeof index === 'number') ? index : children.length,
 74                     w = Y.Lang.isArray(widget) ? widget : [widget];
 75             Array.prototype.splice.apply(children, [i, 0].concat(w));
 76             this.set("children", children);
 77         },
 78         size: function() {
 79             return this.get("children").length;
 80         },
 81         /**
 82          * @function
 83          * @private
 84          */
 85         show: function(node) {
 86             var menu = this.getMenu();
 87             node = node || this.get("targetNode");
 88             menu.attachTo(node);                                                // /Get a menu instance and attach it to the target node
 89             menu.focus();
 90         },
 91         // *** Private methods *** //
 92 
 93         /**
 94          * @function
 95          * @private
 96          */
 97         getMenu: function() {
 98             if (!this.menu) {
 99                 var cfg = this.get("menuCfg"),
100                         host = this.get(HOST),
101                         parent = host.get("parent"),
102                         menu;
103                 cfg.zIndex = WidgetMenu.menuIndex;                              // Better exemple for this at http://yuilibrary.com/yui/docs/overlay/overlay-stack.html
104                 WidgetMenu.menuIndex += 1;
105                 cfg.children = this.get("children");
106 
107                 menu = new Menu(cfg);
108                 menu.addTarget(this);                                           // Catch any event generated by the menu
109                 menu._menuPlugin = this;
110                 this.menu = menu;                                               // Set up reference
111 
112                 if (this.get("isSubmenu")) {                                    // Handle nested menus, events are forwarded to the parent widget
113                     if (parent._menuPlugin) {
114                         menu.addTarget(parent._menuPlugin);                     // Forward events to parent plugin
115                     }
116                     menu.on("timerCanceled", parent.cancelMenuTimer, parent);
117                     menu.on("timerStarted", parent.startMenuHideTimer, parent);
118                 }
119 
120                 (host.get("contentBox") || host).delegate("mouseleave", function() {
121                     this.menu.startMenuHideTimer(false);
122                 }, this.get("selector"), this);
123             }
124             return this.menu;
125         }
126     }, {
127         menuIndex: 50,
128         /** @lends Y.Plugin.WidgetMenu */
129 
130         NS: "menu",
131         NAME: "widgetmenu",
132         ATTRS: {
133             isSubmenu: {
134                 getter: function() {
135                     var parent = this.get(HOST).get("parent");
136                     return (parent && parent instanceof Menu);
137                 }
138             },
139             targetNode: {
140                 getter: function() {
141                     var host = this.get(HOST),
142                             node = (host instanceof Y.Widget) ?
143                             host.get(BOUNDINGBOX) : host;
144                     return node;
145                 }
146             },
147             children: {
148                 value: []
149             },
150             selector: {
151                 value: "*"
152             },
153             menuCfg: {
154                 value: {}
155             },
156             event: {
157                 value: "click"
158             }
159         }
160     });
161     Y.namespace('Plugin').WidgetMenu = WidgetMenu;
162 
163     /**
164      *
165      */
166     MenuBar = Y.Base.create("menubar", Y.Widget, [Y.WidgetParent], {
167         CONTENT_TEMPLATE: null
168     }, {
169         ATTRS: {
170             defaultChildType: {
171                 value: Y.Wegas.Button
172             }
173         }
174     });
175     Y.namespace('Wegas').MenuBar = MenuBar;
176     /**
177      * @name Y.Wegas.Menu
178      * @class Menu Widget, an positionnalbe overlay, intendend to be used by the menu plugin.
179      * @contstructor
180      * @extends Y.Widget
181      * @augments Y.WidgetPosition
182      * @augments Y.WidgetPositionAlign
183      * @augments Y.WidgetStack
184      * @augments Y.WidgetParent
185      * @augments Y.WidgetPositionConstrain
186      */
187     Menu = Y.Base.create("menu", MenuBar, [Y.WidgetPosition, Y.WidgetPositionAlign, Y.WidgetPositionConstrain, Y.WidgetStack], {
188         /** @lends Y.Wegas.Menu# */
189         // *** Lifecycle methods *** //
190         initializer: function() {
191             this.publish("cancelMenuTimer", {
192                 emitFacade: true
193             });
194         },
195         renderUI: function() {
196             var bb = this.get(BOUNDINGBOX);
197 
198             bb.on("clickoutside", this.clickOutside, this);
199             bb.on("click", this.hide, this);
200             bb.on("mouseenter", this.cancelMenuTimer, this);
201             bb.on("mouseleave", this.startMenuHideTimer, this);
202         },
203         hide: function() {
204             this.get("boundingBox").transition({
205                 duration: 0.1,
206                 easing: 'ease-out',
207                 opacity: 0
208             }, Y.bind(Menu.superclass.hide, this));
209         },
210         show: function() {
211             Menu.superclass.show.call(this);
212             this.get("boundingBox").transition({
213                 duration: 0.1,
214                 easing: 'ease-out',
215                 opacity: 1
216             });
217         },
218         // *** Public methods *** //
219         /**
220          *
221          *  Displays the menu next to the provided node and add mouseenter and
222          *  mouseleave callbacks to the node
223          *
224          * @function
225          */
226         attachTo: function(node) {
227             this.currentNode = node;
228 
229             this.set("align", {
230                 node: node,
231                 points: this.get("points")
232             });
233             this.cancelMenuTimer();
234             this.show();
235 
236             //node.on("mouseenter", this.show, this);
237             //node.on("mouseleave", this.hide, this);
238         },
239         // *** Private methods *** //
240         clickOutside: function(e) {
241             if (this.currentNode !== e.target) {
242                 this.hide();
243             }
244         },
245         startMenuHideTimer: function(fireEvent) {
246             //console.log("startMenuHideTimer",this.get("contentBox").one("button").getHTML());
247             this.cancelMenuTimer();
248             this.timer = Y.later(200, this, this.hide);
249 
250             if (!!fireEvent) {
251                 this.fire("timerStarted");
252             }
253         },
254         cancelMenuTimer: function() {
255             //console.log("cancelMenuTimer", this.get("contentBox").one("button").getHTML());
256             if (this.timer) {
257                 this.timer.cancel();
258             }
259             this.fire("timerCanceled");
260         }
261 
262     }, {
263         /** @lends Y.Wegas.Menu */
264         ATTRS: {
265             points: {
266                 value: ["tl", "bl"]
267             },
268             constrain: {
269                 value: true
270             },
271             preventOverlap: {
272                 value: true
273             },
274             zIndex: {
275                 value: 50
276             },
277             render: {
278                 value: true
279             },
280             visible: {
281                 value: false
282             }
283         }
284     });
285     Y.namespace('Wegas').Menu = Menu;
286 
287 });
