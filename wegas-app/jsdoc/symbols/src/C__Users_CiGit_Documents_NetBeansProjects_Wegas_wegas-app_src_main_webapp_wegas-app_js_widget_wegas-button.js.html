 15     "use strict";

 17     var CONTENTBOX = 'contentBox',
 18             BOUNDINGBOX = 'boundingBox',
 19             Wegas = Y.Wegas, Button;

 21     /**
 22      * @name Y.Wegas.Button
 23      * @extends Y.Button
 24      * @borrows Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable
 25      * @class Custom Button implementation.
 26      * @constructor
 27      * @description Custom Button implementation. Adds Y.WidgetChild and
 28      * Y.Wegas.Widget extensions to the original Y.Button
 29      */
 30     Button = Y.Base.create("button", Y.Button, [Y.WidgetChild, Wegas.Widget, Wegas.Editable], {
 31         /** @lends Y.Wegas.Button# */
 32         // *** Private fields *** //
 33 
 34         // *** Lifecycle Methods *** //
 35         /**
 36          * @function
 37          * @private
 38          * @description Set variable with initials values.
 39          * Plug tooltip add given (by ATTRS) css class to contentbox
 40          */
 41         initializer: function() {
 42             Button.superclass.initializer.apply(this, arguments);
 43 
 44             this.publish("click", {
 45                 emitFacade: true,
 46                 bubbles: true,
 47                 defaultFn: function() {                                         // Force event activation by default
 48                 }
 49             });
 50             //this.constructor.CSS_PREFIX = "yui3-button";                      // Revert changes done by Y.Wegas.Widget so styling will work
 51             this._cssPrefix = "yui3-button";
 52 
 53             if (this.get("cssClass")) {
 54                 this.get(CONTENTBOX).addClass(this.get("cssClass"));
 55             }
 56 
 57             if (this.get("tooltip")) {
 58                 this.plug(Y.Plugin.Tooltip, {
 59                     content: this.get("tooltip")
 60                 });
 61             }
 62         },
 63         getEditorLabel: function() {
 64             return Wegas.Helper.stripHtml(this.get("label"));
 65         },
 66         /**
 67          * @function
 68          * @private
 69          * @description Call widget parent to execute its proper render function.
 70          * add "wegas-button" class to bounding box.
 71          */
 72         renderUI: function() {
 73             Button.superclass.renderUI.apply(this, arguments);
 74             this.get(BOUNDINGBOX).addClass("wegas-button");
 75         },
 76         _getLabel: function(value) {
 77             return value;
 78         }
 79     }, {
 80         /**
 81          * @lends Y.Wegas.Button
 82          */
 83         EDITORNAME: "Button",
 84         /**
 85          * @field
 86          * @static
 87          * @description
 88          * <p><strong>Attributes</strong></p>
 89          * <ul>
 90          *    <li>label: the label of the button</li>
 91          *    <li>data: the data used by the button</li>
 92          *    <li>tooltip: the tooltip of the button</li>
 93          *    <li>disabled: boolean to choose state of the button</li>
 94          *    <li>cssClass: cssClass of the button</li>
 95          *    <li>plugins: impact to bind at the buttons</li>
 96          * </ul>
 97          */
 98         ATTRS: {
 99             label: {
100                 type: "string",
101                 optional: true
102             },
103             labelHTML: {
104                 "transient": true
105             },
106             data: {},
107             tooltip: {
108                 type: "string",
109                 optional: true,
110                 "transient": true
111             },
112             disabled: {
113                 type: "boolean",
114                 _inputex: {
115                     wrapperClassName: 'inputEx-fieldWrapper wegas-advanced-feature'
116                 }
117             },
118             cssClass: {
119                 value: null
120             }
121         }
122     });
123     Wegas.Button = Button;

133     /**
134      * Plugin which adds an unread message counter to a widget.
135      *
136      * @class Y.Wegas.UnreadCount
137      * @extends Y.Plugin.Base
138      * @borrows Y.Wegas.Plugin, Y.Wegas.Editable
139      */
140     var UnreadCount = Y.Base.create("wegas-unreadCount", Y.Plugin.Base, [Wegas.Plugin, Wegas.Editable], {
141         /** @lends Y.Plugin.UnreadCount# */
142 
143         // *** Private fields *** //
144         // *** Lifecycle methods *** //
145         /**
146          * @function
147          * @private
148          * @description Set variable with initials values.
149          */
150         initializer: function() {
151             this.handlers = {};
152             this.bindUI();
153         },
154         /**
155          * @function
156          * @private
157          * @description bind function to events.
158          * When VariableDescriptorFacade is updated, do sync.
159          * When plugin's host is render, do sync.
160          */
161         bindUI: function() {
162             this.handlers.update = Wegas.Facade.VariableDescriptor.after("update", this.syncUI, this);
163             this.afterHostEvent("render", this.syncUI, this);
164         },
165         /**
166          * @function
167          * @private
168          * @description call function 'getUnreadCount' to set the number of
169          * unread on the host.
170          */
171         syncUI: function() {
172             var cb = this.get('host').get(CONTENTBOX),
173                     target = cb.one(".wegas-unreadcount"),
174                     unreadCount = this.getUnreadCount();
175 
176             if (!target) {                                                      // If the counter span has not been rendered, do it
177                 cb.append('<span class="wegas-unreadcount"></span>');
178                 target = cb.one(".wegas-unreadcount");
179             }
180 
181             if (unreadCount > 0) {                                              // Update the content
182                 target.setContent("<span class='value'>" + unreadCount + "</span>");
183             } else {
184                 target.setContent("");
185             }
186         },
187         /**
188          * @function
189          * @private
190          * @description Detach all functions created by this widget
191          */
192         destructor: function() {
193             for (var k in this.handlers) {
194                 this.handlers[k].detach();
195             }
196         },
197         // *** Private methods *** //
198         /**
199          * @function
200          * @private
201          * @return Number of unread.
202          * @description Count the number of unread reply in given variable.
203          */
204         getUnreadCount: function() {
205             var i, instance, messages, items, count = 0,
206                     descriptor = this.get('variable.evaluated');
207 
208             if (!descriptor) {
209                 return 0;
210             }
211 
212             if (descriptor instanceof Wegas.persistence.ListDescriptor) {     // For ListDescriptors, we count the children instance's
213                 items = descriptor.flatten();
214                 for (i = 0; i < items.length; i = i + 1) {
215                     if (Wegas.persistence.QuestionDescriptor
216                             && items[i] instanceof Wegas.persistence.QuestionDescriptor) {
217                         instance = items[i].getInstance();
218                         //count += instance.get("unread") ? 1 : 0;
219                         if (instance.get("replies")) {
220                             count += instance.get("replies").length === 0 && instance.get("active") ? 1 : 0; // only count if it is active
221                         }
222                     }
223                 }
224             }
225 
226             if (descriptor instanceof Wegas.persistence.InboxDescriptor) {
227                 messages = descriptor.getInstance().get("messages");            // For InboxVariableDescriptors, we count the replies
228                 count = descriptor.getInstance().get("unreadCount");
229             }
230 
231             return count;
232         }
233     }, {
234         NS: "UnreadCount",
235         NAME: "UnreadCount",
236         /**
237          * @lends Y.Plugin.UnreadCount
238          */
239         /**
240          * @field
241          * @static
242          * @description
243          * <p><strong>Attributes</strong></p>
244          * <ul>
245          *    <li>variable: The target variable, returned either based on the name
246          *     attribute, and if absent by evaluating the expr attribute.</li>
247          * </ul>
248          */
249         ATTRS: {
250             /**
251              * The target variable, returned either based on the variableName attribute,
252              * and if absent by evaluating the expr attribute.
253              */
254             variable: {
255                 getter: Wegas.Widget.VARIABLEDESCRIPTORGETTER,
256                 _inputex: {
257                     _type: "variableselect",
258                     label: "Unread count",
259                     classFilter: ["ListDescriptor", "InboxDescriptor"]
260                 }
261             }
262         }
263     });
264     Y.Plugin.UnreadCount = UnreadCount;
