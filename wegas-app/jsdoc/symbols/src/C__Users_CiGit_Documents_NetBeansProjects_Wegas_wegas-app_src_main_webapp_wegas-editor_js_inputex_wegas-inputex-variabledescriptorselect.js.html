 21     /**
 22      * @name Y.inputEx.Wegas.VariableDescriptorSelect
 23      * @class
 24      * @constructor
 25      * @extends Y.inpuEx.Group
 26      * @param {Object} options InputEx definition object
 27      */
 28     VariableDescriptorSelect = function(options) {
 29         VariableDescriptorSelect.superclass.constructor.call(this, options);
 30     };
 31 
 32     Y.extend(VariableDescriptorSelect, inputEx.Group, {
 33         currentEntityField: null,
 34         /** @lends Y.Wegas.VariableDescriptorSelect# */
 35         /**
 36          * Setup the options.fields from the availableFields option
 37          * @function
 38          */
 39         setOptions: function(options) {
 40             options.fields = options.fields || [];
 41             VariableDescriptorSelect.superclass.setOptions.call(this, options);
 42             this.options.className = options.className || 'wegas-inputex-variabledescriptorselect-group inputEx-Group';
 43             this.options.label = options.label;
 44             if (options.classFilter) {
 45                 this.options.classFilter = options.classFilter;
 46                 this.options.classFilter.push("ListDescriptor");
 47             }
 48         },
 49         /**
 50          * @function
 51          */
 52         render: function() {
 53             VariableDescriptorSelect.superclass.render.call(this);
 54             this.fieldset.classList.add("wegas-inputex-variabledescriptorselect");
 55             this.syncUI();
 56             if (this.options.label) {
 57                 var node = new Y.Node(this.fieldset);
 58                 node.get("parentNode").prepend("<label>" + this.options.label + "</label>");
 59             }
 60         },
 61         /**
 62          * @function
 63          */
 64         setValue: function(val) {
 65             // Set value should not ba called directly
 66             //Y.log("VariableDescriptorSelect.setValue", val);
 67             VariableDescriptorSelect.superclass.setValue.apply(this, arguments);
 68             this.options.value = val;
 69             this.syncUI();
 70         },
 71         /**
 72          * @function
 73          */
 74         getValue: function() {
 75             return "Variable.find(gameModel, \"" + this.currentEntityField.getValue() + "\")";
 76         },
 77         /**
 78          * @function
 79          */
 80         syncUI: function() {
 81             this.empty();
 82 
 83             var ret = [], rootEntities = Y.Wegas.Facade.VariableDescriptor.cache.findAll(),
 84                     currentEntity = Y.Wegas.Facade.VariableDescriptor.cache.find('name', this.options.value) || rootEntities[0];
 85 
 86             if (currentEntity) {
 87                 this.currentEntity = currentEntity;                                 // Keeps a reference to the current entity
 88 
 89                 while (currentEntity.parentDescriptor) {                            // Add the current entity hierarchy
 90                     ret.push(this.generateSelectConfig(currentEntity.parentDescriptor,
 91                             currentEntity, currentEntity.parentDescriptor.get("items")));
 92                     currentEntity = currentEntity.parentDescriptor;
 93                 }
 94                 ret.push(this.generateSelectConfig(null, currentEntity, rootEntities));// Add the root context (entities that are at the root of the gameModel
 95 
 96                 Y.Array.each(ret.reverse(), this.addField, this);
 97                 this.currentEntityField = this.inputs[this.inputs.length - 1];
 98             } else {
 99                 // @todo Show error message
100             }
101         },
102         /**
103          *
104          * @overrride Y.inputEx.Group.onChange()
105          */
106         onChange: function(fieldValue, fieldInstance) {
107             if (fieldValue === "----------") {
108                 return; /* @fixme */
109             }
110             var entity = Y.Wegas.Facade.VariableDescriptor.cache.find('name', fieldValue);
111             if (entity) {                                                       // An entity was found, it is the new current entity
112                 // if (Y.Lang.isNumber(fieldValue)) {
113                 this.options.value = fieldValue;
114                 this.options.method = null;
115                 this.options.methodCfg = null;
116                 this.options.arguments = null;
117                 this.syncUI();
118             } else if (Y.Lang.isString(fieldValue)) {                           // The id is a method, it's the new current mehtod
119                 this.options.value = fieldInstance.options.parentEntity.get("name");
120                 this.options.method = fieldValue;
121                 this.options.arguments = null;
122                 this.syncUI();
123             } else {                                                            // Otherwise current args value
124                 this.options.arguments = this.inputs[this.inputs.length - 1].getValue();
125             }
126             this.fireUpdatedEvt();
127         },
128         /**
129          * @function
130          */
131         empty: function() {
132             while (this.inputs.length > 0) {
133                 this.inputs.pop().destroy();
134             }
135             (new Y.Node(this.fieldset)).empty();
136         },
137         /**
138          * Overriden to add reference to parententity
139          * @function
140          */
141         addField: function(fieldOptions) {
142             VariableDescriptorSelect.superclass.addField.call(this, fieldOptions);
143             this.inputs[this.inputs.length - 1].options.parentEntity = fieldOptions.parentEntity;
144         },
145         /**
146          * Generate
147          * @function
148          */
149         generateSelectConfig: function(entity, selectedEntity, items) {
150             var value;
151             if (entity === this.currentEntity) {
152                 value = this.options.method;
153             }
154             if (selectedEntity) {
155                 value = selectedEntity.get("name");
156             }
157             return {
158                 type: 'select',
159                 choices: this.genChoices(entity, items),
160                 value: value,
161                 parentEntity: entity
162             };
163         },
164         /**
165          * @function
166          */
167         genChoices: function(entity, items) {
168             if (items) {
169                 return Y.Array.map(Y.Array.filter(items, function(i) {        // Apply class filter
170                     return !this.options.classFilter || Y.Array.indexOf(this.options.classFilter, i.get("@class")) > -1;
171                 }, this), function(i) {
172                     return{
173                         value: i.get("name"),
174                         label: i.getEditorLabel()
175                     };
176                 }, this);
177             }
178             return [];
179         }
180     });

195     Y.extend(VariableDescriptorMethod, VariableDescriptorSelect, {
196         /** @lends Y.inputEx.Wegas.VariableDescriptorMethod# */
197         GLOBALMETHODS: {},
198         generateSelectConfig: function(entity, selectedEntity, items) {
199             if (Y.Lang.isString(this.options.value)
200                     && Y.Object.hasKey(this.GLOBALMETHODS, this.options.value.replace("GLOBAL", ""))) {
201                 return {
202                     type: 'select',
203                     choices: this.genChoices(entity, items),
204                     value: this.options.value,
205                     parentEntity: entity
206                 };
207             } else {
208                 return VariableDescriptorMethod.superclass.generateSelectConfig.apply(this, arguments);
209             }
210         },
211         syncUI: function() {
212             var args, methods, cMethod,
213                     rootEntities = Y.Wegas.Facade.VariableDescriptor.cache.findAll(),
214                     currentEntity = Y.Wegas.Facade.VariableDescriptor.cache.find('name', this.options.value) || rootEntities[0];
215 
216             if (Y.Lang.isString(this.options.value)
217                     && Y.Object.hasKey(this.GLOBALMETHODS, this.options.value.replace("GLOBAL", ""))) {
218                 VariableDescriptorSelect.prototype.syncUI.apply(this, arguments);
219 
220                 var cMethod = this.GLOBALMETHODS[this.options.value.replace("GLOBAL", "")];
221                 this.addField(Y.mix({
222                     type: "combine",
223                     fields: cMethod.arguments,
224                     value: this.options.arguments,
225                     label: null
226                 }, cMethod));
227 
228             } else {
229 
230                 while (this.getMethods(currentEntity).length === 0              // If the current entity has no methods,
231                         && currentEntity.get("items") && currentEntity.get("items").length > 0) { // but it has a child
232                     currentEntity = currentEntity.get("items")[0];              // select its first child
233                     this.options.value = currentEntity.get("name");
234                 }
235 
236                 cMethod = this.options.methodCfg;                               //assign cMethod after set this.options.methodCfg by this.getMethods()
237 
238                 VariableDescriptorMethod.superclass.syncUI.call(this);
239 
240                 this.addField(this.generateSelectConfig(this.currentEntity,
241                         null, this.currentEntity.get("items")));                // Pushes the current entity methods and children to the stack
242 
243                 methods = this.getMethods(this.currentEntity);
244 
245                 if (!cMethod && Y.Object.values(methods).length > 0) {
246                     cMethod = this.options.methodCfg = Y.Object.values(methods)[0];// By default select the first method available
247                 }
248 
249                 args = (cMethod && cMethod.arguments) ? cMethod.arguments : [];
250 
251                 Y.Array.each(args, function(a) {
252                     a.entity = this.currentEntity;
253                 }, this);                                                       // Adds a reference to the target entity to the argument Fields;
254 
255                 this.currentMethod = cMethod;
256 
257                 this.addField(Y.mix({
258                     type: "combine",
259                     fields: args,
260                     value: this.options.arguments,
261                     label: null
262                 }, cMethod));
263 
264                 // Same as above, but using json object format for method definitions
265                 //var schemaMap = {
266                 //    Entity: {
267                 //        type: "array",
268                 //        items: cMethod.arguments
269                 //    }
270                 //}, builder = new Y.inputEx.JsonSchema.Builder({
271                 //    'schemaIdentifierMap': schemaMap,
272                 //    'defaultOptions':{
273                 //        'showMsg':true
274                 //    }
275                 //}), field  = builder.schemaToInputEx(schemaMap.Entity);
276             }
277         },
278         setOptions: function(options) {
279             VariableDescriptorMethod.superclass.setOptions.call(this, options);
280             this.options.method = options.method;
281             this.options.arguments = options.arguments;
282             this.argsOffset = 1;
283         },
284         getValue: function() {
285             var l = this.inputs.length,
286                     args = this.inputs[l - this.argsOffset].getValue(),
287                     method = this.inputs[l - this.argsOffset - 1].getValue();
288 
289             if (Y.Lang.isString(this.options.value)
290                     && Y.Object.hasKey(this.GLOBALMETHODS, this.options.value.replace("GLOBAL", ""))) {
291                 var k = this.options.value.replace("GLOBAL", ""),
292                         cMethod = this.GLOBALMETHODS[this.options.value.replace("GLOBAL", "")];
293                 return k + "(" + this.encodeArgs(this.inputs[1].getValue(), cMethod.arguments) + ")";
294             } else {
295                 if (!method) {
296                     return "true";
297                 }
298                 return "Variable.find(gameModel, \"" + this.inputs[l - this.argsOffset - 2].getValue() + "\")"
299                         + "." + method + "(" + this.encodeArgs(args, this.currentMethod.arguments) + ")";
300             }
301         },
302         encodeArgs: function(args, argsCfg) {
303             var i, j;
304             for (i = 0; i < args.length; i = i + 1) {
305                 if (argsCfg[i].scriptType === "string") {
306                     if (Y.Lang.isArray(args[i])) {
307                         for (j = 0; j < args[i].length; j++) {
308                             args[i][j] = Y.Wegas.Helper.escapeJSString(args[i][j]);
309                         }
310                         args[i] = Y.JSON.stringify(args[i]);
311                     } else if (Y.Lang.isObject(args[i])) {
312                         for (j in args[i]) {
313                             args[i][j] = Y.Wegas.Helper.escapeJSString(args[i][j]);
314                         }
315                         args[i] = Y.JSON.stringify(args[i]);
316                     } else {
317                         args[i] = '"' + Y.Wegas.Helper.escapeJSString(args[i]) + '"';
318                     }
319                 }
320             }
321             return args.join(", ");
322         },
323         onChange: function(fieldValue) {
324             if (Y.Lang.isString(fieldValue)
325                     && Y.Object.hasKey(this.GLOBALMETHODS, fieldValue.replace("GLOBAL", ""))) {
326                 this.options.value = fieldValue;
327                 this.options.arguments = null;
328                 this.options.methodCfg = null;
329                 this.syncUI();
330                 return;
331             }
332             VariableDescriptorMethod.superclass.onChange.apply(this, arguments);
333         },
334         /**
335          * Generate choices for a given entity: add it's methods and then pass up to parent class
336          * to add children variable descriptors.
337          */
338         genChoices: function(entity, items) {
339             var i, choices = [];
340             if (entity) {
341                 choices = choices.concat(this.getMethods(entity));              // Push the methods to the select choices
342             }
343 
344             if (items && choices.length > 0) {                                  // If required, push separator
345                 choices.push({
346                     value: "----------"
347                 });
348             }
349             choices = choices.concat(VariableDescriptorMethod.superclass.genChoices.apply(this, arguments));
350 
351             if (!entity) {                                                      // If the entity is at root level of its hierarchy (game model level)
352                 for (i in this.GLOBALMETHODS) {                                 // Adds all global methods
353                     choices.push({
354                         label: this.GLOBALMETHODS[i].label,
355                         value: "GLOBAL" + i
356                     });
357                 }
358             }
359             return choices;
360         },
361         getMethods: function(entity) {
362             var i, methods = entity.getMethodCfgs(), ret = [];
363             for (i in methods) {
364                 if (!this.options.returnsFilter                                 // Apply filter on the method return type
365                         || Y.Array.indexOf(this.options.returnsFilter, methods[i].returns || "void") >= 0) {
366 
367                     methods[i].value = i;
368                     methods[i].label = methods[i].label || i;
369                     ret.push(methods[i]);
370                     if (i === this.options.method) {
371                         this.options.methodCfg = methods[i];
372                     }
373                 }
374             }
375             return ret;
376         }
377     });

558     /**
559      * @name Y.inputEx.Wegas.EntityArrayFieldSelect

566         EntityArrayFieldSelect.superclass.constructor.call(this, options);
567     };
568     Y.extend(EntityArrayFieldSelect, inputEx.SelectField, {
569         /** @lends Y.Wegas.EntityArrayFieldSelect# */
570         /**
571          * Set the ListField classname
572          * @param {Object} options Options object as passed to the constructor
573          */
574         setOptions: function(options) {
575             var results;
576             if (options.scope !== "instance") {
577                 results = options.entity ? options.entity.get(options.field) :
578                         Y.Plugin.EditEntityAction.currentEntity.get(options.field);
579             } else {
580                 results = options.entity ? options.entity.getInstance().get(options.field) :
581                         Y.Plugin.EditEntityAction.currentEntity.getInstance().get(options.field);
582             }
583             options.choices = Y.Array.map(results, function(r) {
584                 return {
585                     value: r.get(options.returnAttr || "name"),
586                     label: r.getEditorLabel() || this.optionNameToString(r, options)
587                 };
588             }, this);
589 
590             EntityArrayFieldSelect.superclass.setOptions.call(this, options);
591             this.options.entity = options.entity;
592         },
593         optionNameToString: function(result, options) {
594             var string = [],
595                     separator = (options.name) ? options.name.separator || "," : ",";
596             if (!options.name || !options.name.values || options.name.values.length <= 0) {
597                 string.push("undefined");
598             } else {
599                 string = Y.Array.map(options.name.values, function(v) {
600                     return result.get(v);
601                 });
602             }
603             return string.join(separator);
604         }
605     });
