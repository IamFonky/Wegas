 12 YUI.add('wegas-join-token', function(Y) {
 13     "use strict";
 14 
 15     var CONTENTBOX = 'contentBox', Wegas = Y.Wegas, TokenJoin;
 16 
 17     /**
 18      * @name Y.Wegas.TokenJoin
 19      * @extends Y.Widget
 20      * @class  class for join a game and a team
 21      * @borrows Y.WidgetChild, Y.Wegas.Widget
 22      * @constructor
 23      * @description Allows to join a game by token or a public game. Then you can
 24      * join or create a new team
 25      */
 26     TokenJoin = Y.Base.create("wegas-join-token", Y.Widget, [Y.WidgetChild, Wegas.Widget], {
 27         /**
 28          * @lends Y.Wegas.TokenJoin#
 29          */
 30         // *** Private fields *** //
 31 
 32         /**
 33          * @function
 34          * @private
 35          * @description All button and fields are created.
 36          * For creating the field inputEx libary is used
 37          */
 38         renderUI: function() {
 39             var cb = this.get(CONTENTBOX),
 40                     tokenParameter = Wegas.Helper.getURLParameter("token");
 41 
 42             this.tokenField = new Y.inputEx.StringField({//                     // Render
 43                 required: true,
 44                 parentEl: cb,
 45                 label: "Enrolment key",
 46                 typeInvite: "type here"
 47             });
 48 
 49             this.button = new Y.Button({
 50                 label: "Submit",
 51                 render: cb
 52             });
 53 
 54             if (tokenParameter) {
 55                 this.isInitialTokenTry = true;
 56                 this.sendTokenJoin(tokenParameter);
 57             }
 58         },
 59         /**
 60          * @function
 61          * @private
 62          * @description All events are added to the buttons
 63          * Create team button call rest url : rest/GameModel/{gameModelID}/Game/{gameID}/CreateTeam/{teamName}
 64          */
 65         bindUI: function() {
 66             this.button.on("click", function() {                                // join a game based on a token
 67                 if (this.tokenField.validate()) {
 68                     this.sendTokenJoin(this.tokenField.getValue());
 69                     //} else if (this.selectPublicGame.getValue() !== "") {
 70                     // @Todo
 71                     // this.joinTeam = new Wegas.SelectTeam({
 72                     //    entity: this.selectPublicGame.getValue()
 73                     //}
 74                 } else {
 75                     this.showMessage("error", "Enter a valid token");
 76                     return;
 77                 }
 78             }, this);
 79         },
 80         destructor: function() {
 81             if (this.teamWidget) {
 82                 this.teamWidget.destroy();
 83             }
 84             this.button.destroy();
 85             this.tokenField.destroy();
 86         },
 87         /**
 88          * @function
 89          * @private
 90          * @description Method for join a game by token
 91          * Call rest request for join the game : rest/GameModel/1/Game/{gameModelID}/JoinGame/{token}
 92          */
 93         sendTokenJoin: function(token) {
 94             this.showOverlay();
 95             Y.log("sendTokenJoin()", "info", "Wegas.TokenJoin");
 96 
 97             Wegas.Facade.Game.sendRequest({
 98                 request: "/JoinGame/" + token + "?view=Extended",
 99                 cfg: {
100                     updateCache: false
101                 },
102                 on: {
103                     success: Y.bind(function(e) {
104                         var cb = this.get(CONTENTBOX),
105                                 entity = e.response.entity;
106 
107                         this.hideOverlay();
108 
109                         if (entity === "Team token required") {                 // Team token is required, game token was provided
110                             // @fixme should show a message when this happens from the lobby
111 
112                             //} else if (entity instanceof Wegas.persistence.Game) {
113                             //    gm = entity.get("gameModel");
114                             //            &&
115                             //        !(gm.get("properties.allowCreateTeam") || gm.get("properties.allowJoinTeam"))) {
116                             if (!this.isInitialTokenTry) {
117                                 this.showMessageBis("error", "No game found for this key");
118                             }
119                         } else if (e.response.entities[0] instanceof Wegas.persistence.Team
120                                 && !(e.response.entities[1].get("gameModel").get("properties.freeForAll")
121                                 || e.response.entities[0].get("players").length === 0)) {// If the token is already in use
122 
123                             this.showMessageBis("error",
124                                     "This team has already been created. You can contact it's members so they can join you in.");
125                         } else {
126                             Y.log("sendTokenJoin(): Rendering team widget", "info", "Wegas.TokenJoin");
127                             this.destructor();
128                             this.teamWidget = new Wegas.JoinTeam({//            // Player can choose or create its team
129                                 entity: e.response.entities,
130                                 render: cb
131                             });
132                             this.teamWidget.addTarget(this);                    // So overlay and message events will be forwarded
133                         }
134                     }, this),
135                     failure: Y.bind(this.defaultFailureHandler, this)
136                 }
137             });
138         }
139     });
140     Y.namespace('Wegas').TokenJoin = TokenJoin;
141 });
