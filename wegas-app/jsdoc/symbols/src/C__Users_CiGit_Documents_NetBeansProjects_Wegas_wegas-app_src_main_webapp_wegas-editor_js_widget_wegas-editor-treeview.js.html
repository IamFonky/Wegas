 12 YUI.add('wegas-editor-treeview', function(Y) {
 13     "use strict";
 14 
 15     var CONTENTBOX = 'contentBox', DATASOURCE = "dataSource", ID = "id",
 16             CLASS = "@class", NAME = "name", HOST = "host",
 17             Plugin = Y.Plugin, Wegas = Y.Wegas, EditorTreeView;
 18 
 19     /**
 20      * @name Y.Wegas.EditorTreeView
 21      * @extends Y.Widget
 22      * @augments Y.WidgetChild
 23      * @augments Y.Wegas.Widget
 24      * @constructor
 25      * @class Widget that renders the content of a Y.Wegas.Datasource in a Y.Treeview.Widget,
 26      * generating requires nodes.
 27      */
 28     EditorTreeView = Y.Base.create("wegas-editor-treeview", Y.Widget, [Y.WidgetChild, Wegas.Widget], {
 29         /** @lends Y.Wegas.EditorTreeView# */
 30         // *** Private fields ** //
 31         // ** Lifecycle methods ** //
 32         /**
 33          *
 34          */
 35         renderUI: function() {
 36             this.treeView = new Y.TreeView({
 37                 emptyMsg: this.get("emptyMessage")
 38             });                                                                 // Instantiate treeview
 39             this.treeView.addTarget(this);                                      // Listen to treeview's events
 40             this.treeView.render(this.get(CONTENTBOX));                         // Render treeview                   
 41 
 42             this.plug(Plugin.EditorTVContextMenu);                              // Open context menu on right click
 43             this.plug(Plugin.RememberExpandedTreeView);                         // Selected node is preserved across requests
 44         },
 45         /**
 46          * @function
 47          * @private
 48          */
 49         bindUI: function() {
 50             var ds = this.get(DATASOURCE),
 51                     request = this.get("request");
 52             if (ds) {
 53                 this.updateHandler = ds.after("update", this.syncUI, this);     // Listen updates on the target datasource
 54                 this.failureHandler = ds.after("failure", this.defaultFailureHandler, this);// GLOBAL error message
 55 
 56                 this.addedHandler = ds.after("added", function(e) {
 57                     this.currentSelection = e.entity.get("id");
 58                 }, this);
 59 
 60                 if (request) {
 61                     ds.sendRequest(request);
 62                 }
 63             }
 64         },
 65         /**
 66          * @function
 67          * @private
 68          */
 69         syncUI: function() {
 70             Y.log("sync()", "info", "Wegas.EditorTreeView");
 71 
 72             if (!this.get(DATASOURCE)) {
 73                 this.get(CONTENTBOX).append("Unable to find datasource");
 74                 return;
 75             }
 76 
 77             this.treeView.destroyAll();
 78             this.treeView.add(this.getNodes());
 79             this.treeView.syncUI();
 80 
 81             this.hideOverlay();
 82 
 83         },
 84         destructor: function() {
 85             this.treeView.destroy();
 86             this.updateHandler.detach();
 87             this.failureHandler.detach();
 88         },
 89         // *** Private Methods *** //
 90         /**
 91          * @function
 92          * @private
 93          */
 94         getNodes: function() {
 95             var ds = this.get(DATASOURCE),
 96                     selector = this.get("dataSelector"),
 97                     entities = (selector) ? ds.cache.find(selector.key, selector.val) : ds.cache.findAll();
 98 
 99             return this.genTreeViewElements(entities);
100         },
101         genTreeViewElements: function(elements) {
102             return Y.Array.filter(Y.Array.map(elements, this.genTreeViewElement, this), Y.Lang.isObject);
103         },
104         genTreeViewElement: function(entity) {
105             var children = entity.get("players");                               // @hack so it works for team
106 
107             if (entity instanceof Wegas.persistence.User) {                     // @hack
108                 entity = entity.getMainAccount();
109             }
110 
111             return {
112                 type: (children) ? "TreeNode" : "TreeLeaf",
113                 label: entity.get(NAME),
114                 //label: el.get(CLASS) + ': ' + el.get(NAME),
115                 selected: (this.currentSelection === entity.get(ID)) ? 2 : 0,
116                 collapsed: !this.isNodeExpanded(entity),
117                 children: (children) ? this.genTreeViewElements(children) : [],
118                 data: {
119                     entity: entity
120                 },
121                 iconCSS: 'wegas-icon-' + entity.get(CLASS).toLowerCase()
122             };
123         },
124         /**
125          * @function
126          * @private
127          */
128         isNodeExpanded: function(entity) {
129             return this.RememberExpandedTreeView.expandedIds[entity.get(ID)] || false;
130         }
131     }, {
132         /** @lends Y.Wegas.EditorTreeView */
133 
134         /**
135          * <p><strong>Attributes</strong></p>
136          * <ul>
137          *    <li>includeClasses a list of entity classes names that will be included</li>
138          *    <li>excludeClasses a list of entity classes that will be excluded</li>
139          *    <li>emptyMessage string message to display if there are no entity
140          *    to display <i>default: "No data to display"</i></li>
141          *    <li>dataSelector</li>
142          *    <li>dataSource</li>
143          *    <li>request</li>
144          * </ul>
145          *
146          * @field
147          * @static
148          */
149         ATTRS: {
150             emptyMessage: {
151                 value: "No data to display"
152             },
153             dataSelector: {},
154             dataSource: {
155                 getter: function(val) {
156                     if (Y.Lang.isString(val)) {
157                         return Wegas.Facade[val];
158                     }
159                     return val;
160                 }
161             },
162             request: {}
163         }
164     });
165     Y.namespace('Wegas').EditorTreeView = EditorTreeView;

170     var TeamTreeView = Y.Base.create("wegas-editor-treeview", EditorTreeView, [], {
171         CONTENT_TEMPLATE: '<div class="wegas-editor-treeview-team">'
172                 + '<div class="yui3-g wegas-editor-treeview-table wegas-editor-treeview-tablehd" style="padding-right: 255px">'
173                 + '<div class="yui3-u yui3-u-col1">Name</div>'
174                 //+ '<div class="yui3-u yui3-u-col2 yui3-g" style="margin-right: -250px;width:250px">'
175                 //+ '<div class="yui3-u">Players</div></div>'
176                 + '</div>'
177                 + '<div class="treeview"></div>'
178                 + '</div>',
179         renderUI: function() {
180             this.treeView = new Y.TreeView({
181                 emptyMsg: this.get("emptyMessage")
182             });                                                                 // Render the treeview
183             this.treeView.addTarget(this);
184             this.treeView.render(this.get(CONTENTBOX).one(".treeview"));
185 
186             this.plug(Plugin.RememberExpandedTreeView);
187             this.plug(Plugin.WidgetToolbar);
188 
189             //this.plug(Plugin.EditorTVToggleClick);
190             //if (this.isFreeForAll()) {                                        // @hack Change the display if the gamemodel is freeforall
191             //    this.get("parent").set("label", "Players");
192             //}
193         },
194         getNodes: function() {
195             var entity = this.get("entity"), acc = [],
196                     nodes = this.genTreeViewElements(entity.get("teams"));
197 
198             if (entity.get("properties.freeForAll")) {                          // Do not display teams in free for all game
199                 Y.Array.each(nodes, function(i) {
200                     acc = acc.concat(i.children);
201                 });
202                 return acc;
203             } else
204                 return nodes;
205         },
206         genTreeViewElement: function(entity) {
207             var elClass = entity.get(CLASS),
208                     collapsed = !this.isNodeExpanded(entity);
209             //selected = (this.currentSelection === entity.get(ID)) ? 2 : 0;
210 
211             switch (elClass) {
212                 case 'Team':
213                     var children = this.genTreeViewElements(entity.get("players")),
214                             expanded = Y.Array.find(children, function(p) {
215                         return p.selected;
216                     }) || !collapsed;
217 
218                     return {
219                         type: 'TreeNode',
220                         collapsed: !expanded,
221                         selected: entity.get("id") === Wegas.Facade.Game.get('currentTeamId') ? 2 : 0,
222                         //selected: selected,
223                         label: entity.get(NAME),
224                         children: children,
225                         data: {
226                             entity: entity
227                         },
228                         iconCSS: 'wegas-icon-team'
229                     };
230 
231                 case 'Player':
232                     return {
233                         label: entity.get(NAME),
234                         selected: entity.get("id") === Wegas.Facade.Game.get('currentPlayerId') ? 2 : 0,
235                         // selected: selected,
236                         data: {
237                             entity: entity
238                         },
239                         iconCSS: 'wegas-icon-player'
240                     };
241             }
242         }
243     }, {
244         ATTRS: {
245             dataSource: {
246                 value: "Game"
247             },
248             entity: {
249                 getter: function(val) {
250                     if (!val)
251                         return Wegas.Facade.Game.cache.getCurrentGame();
252                     else
253                         return val;
254                 }
255             },
256             emptyMessage: {
257                 value: "No player has joined yet"
258             }
259         }
260     });

265      * collapsed nodes.

268     Plugin.RememberExpandedTreeView = Y.Base.create("wegas-rememberexpandedtreeview", Plugin.Base, [], {
269         expandedIds: null,
270         initializer: function() {
271             this.expandedIds = {};
272             this.onHostEvent("*:nodeExpanded", function(e) {
273                 this.expandedIds[e.node.get("data").entity.get(ID)] = true;
274             });
275             this.onHostEvent("*:nodeCollapsed", function(e) {
276                 delete this.expandedIds[e.node.get("data").entity.get(ID)];
277             });
278         }
279     }, {
280         NS: "RememberExpandedTreeView"
281     });

283     /**
284      * @class Open a menu on click, containing the admin edition field
285      * @constructor
286      */
287     Plugin.EditorTVToolbarMenu = Y.Base.create("admin-menu", Plugin.Base, [], {
288         initializer: function() {
289             this.afterHostEvent("*:click", this.onTreeViewSelection);
290         },
291         onTreeViewSelection: function(e) {
292             var menuItems = this.getMenuItems(e.target.get("data")),
293                     host = this.get(HOST);
294 
295             if (menuItems) {
296                 host.toolbar.destroyAll();
297                 host.toolbar.add(menuItems);                                    // Populate the menu with the elements associated to the
298             } else {
299                 Y.log("Menu item has no target entity", "info", "Y.Plugin.EditorTVToolbarMenu");
300                 host.currentSelection = null;
301             }
302         },
303         getMenuItems: function(data) {
304             var menuItems = this.get("children"), entity,
305                     host = this.get(HOST);
306 
307             if (data) {
308                 entity = data.entity || data.widget;
309                 if (entity) {
310                     host.currentSelection = entity.get(ID);
311                 }
312                 data.dataSource = host.get(DATASOURCE);
313 
314                 if (menuItems) {
315                     Wegas.Editable.mixMenuCfg(menuItems, data);
316                 } else {
317                     menuItems = entity.getMenuCfg(data).slice(0);               // If no menu is provided, use a clone of the entity default value
318                 }
319 
320                 Y.Array.each(menuItems, function(i) {                           // @hack add icons to some buttons
321                     switch (i.label) {
322                         case "Delete":
323                         case "New":
324                         case "Add":
325                         case "Copy":
326                         case "View":
327                         case "Open in editor":
328                         case "Open":
329                         case "Edit":
330                             i.label = '<span class="wegas-icon wegas-icon-' + i.label.replace(/ /g, "-").toLowerCase() + '"></span>' + i.label;
331                     }
332                 });
333             }
334             return menuItems;
335         }
336     }, {
337         NS: "menu",
338         ATTRS: {
339             children: {}
340         }
341     });

362     /**
363      * @class Open a menu on right click, containing the admin edition field
364      * @constructor
365      */
366     Plugin.EditorTVContextMenu = Y.Base.create("admin-menu", Plugin.Base, [], {
367         initializer: function() {
368             this.onHostEvent("contextmenu", this.onTreeViewClick, this);
369 
370             this.menu = new Wegas.Menu();
371             this.menu.addTarget(this.get(HOST));
372             this.menu.render();
373         },
374         onTreeViewClick: function(e) {
375             var targetWidget = Y.Widget.getByNode(e.domEvent.target),
376                     menuItems = this.getMenuItems(targetWidget.get("data"), targetWidget);    // Fetch menu items
377 
378             //menuItems.splice(0, 1);                                           // Remove "Edit" button
379 
380             Y.Array.each(menuItems, function(i, itemIndex) {                    // @HACK Fix the submenu positioning
381                 Y.Array.each(i.plugins, function(p, index) {
382                     if (p.fn === "WidgetMenu") {
383                         menuItems[itemIndex] = Y.mix({}, menuItems[itemIndex]);
384                         menuItems[itemIndex].plugins = menuItems[itemIndex].plugins.slice(0);
385                         menuItems[itemIndex].plugins[index] = {
386                             fn: "WidgetMenu",
387                             cfg: Y.mix({
388                                 menuCfg: {
389                                     points: ["tl", "tr"]
390                                 },
391                                 event: "mouseenter"
392                             }, p.cfg)
393                         };
394                     }
395                 });
396             });
397 
398             if (menuItems) {
399                 e.domEvent.preventDefault();
400                 this.menu.destroyAll();
401                 this.menu.add(menuItems);                                       // Populate the menu with the elements associated to the
402                 this.menu.show();                                               // Move the right click position
403                 this.menu.set("xy", [e.domEvent.pageX, e.domEvent.pageY]);
404             } else {
405                 Y.log("Menu item has no target entity", "info", "Y.Plugin.EditorTVToolbarMenu");
406             }
407         },
408         getMenuItems: function(data) {
409             return Plugin.EditorTVToolbarMenu.prototype.getMenuItems.call(this, data);
410         }
411     }, {
412         NS: "contextmenu",
413         ATTRS: {
414             children: {}
415         }
416     });
