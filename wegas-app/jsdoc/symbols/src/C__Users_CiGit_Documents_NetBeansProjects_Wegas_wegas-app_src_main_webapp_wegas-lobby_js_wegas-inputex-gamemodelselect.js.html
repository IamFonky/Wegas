 12 YUI.add("wegas-inputex-gamemodelselect", function(Y) {
 13     "use strict";

 15     /**
 16      * @class Y.inputEx.Wegas.GameModelSelect
 17      * @extends Y.inputEx.SelectField
 18      * @constructor
 19      * @param {Object} options  options object
 20      * <ul>
 21      *   <li></li>
 22      * </ul>
 23      */
 24     var GameModelSelect = function(options) {
 25         GameModelSelect.superclass.constructor.call(this, options);
 26     };
 27 
 28     Y.extend(GameModelSelect, Y.inputEx.SelectField, {
 29         setOptions: function(options) {
 30             var gameModels = Y.Wegas.Facade.GameModel.cache.findAll();
 31 
 32             options.choices = [];
 33             options.filters = options.filters || {};
 34 
 35             Y.Array.each(gameModels, function(gm) {
 36                 if (Y.Object.some(options.filters, function(value, key) {
 37                     return gm.get(key) === value;
 38                 })) {                                                           // If the game model does not match any filter
 39                     options.choices.push({// add it to available games
 40                         label: gm.get("name"),
 41                         value: gm.get("id")
 42                     });
 43                 }
 44             });
 45             GameModelSelect.superclass.setOptions.call(this, options);
 46         },
 47         setValue: function(value, fireUpdateEvent) {
 48             if ((!value || value === "")
 49                     && !!Y.Plugin.EditorDTMenu.currentGameModel) {
 50                 value = Y.Plugin.EditorDTMenu.currentGameModel.get("id");
 51             }
 52             GameModelSelect.superclass.setValue.call(this, value, fireUpdateEvent);
 53         }
 54     });
 55     Y.inputEx.registerType("gamemodelselect", GameModelSelect);                 // Register this class
 56     Y.namespace("inputEx.Wegas").GameModelSelect = GameModelSelect;
 57 
 58     /**
 59      * @class Y.inputEx.Wegas.EnrolmentKeyList
 60      * @extends Y.inputEx.ListField
 61      * @constructor
 62      * @param {Object} options  options object
 63      * <ul>
 64      *   <li></li>
 65      * </ul>
 66      */
 67     var EnrolmentKeyList = function(options) {
 68         EnrolmentKeyList.superclass.constructor.call(this, options);
 69     };
 70     Y.extend(EnrolmentKeyList, Y.inputEx.ListField, {
 71         setOptions: function(options) {
 72             options.elementType = options.elementType || {
 73                 type: "group",
 74                 label: "",
 75                 required: true,
 76                 fields: [{
 77                         name: "@class",
 78                         type: "hidden",
 79                         value: "GameEnrolmentKey"
 80                     }, {
 81                         name: "key",
 82                         type: "uneditable"
 83                                 //type: "string"
 84                     }, {
 85                         name: "used",
 86                         value: false,
 87                         type: "hidden"
 88                     }]
 89             };
 90 
 91             EnrolmentKeyList.superclass.setOptions.call(this, options);
 92         },
 93         addElement: function(value) {
 94             var subfield = EnrolmentKeyList.superclass.addElement.call(this, value),
 95                     node = new Y.Node(subfield.divEl);
 96 
 97             node.all(".inputEx-Field").each(function(n) {
 98                 n.setContent("<span class='wegas-enrolmentkey'>" + n.getContent() + "</span>, ");
 99             });
100             (new Y.Node(this.divEl)).all(".inputEx-ListField-delButton").remove(true); // Remove delete button
101             (new Y.Node(this.divEl)).all(".inputEx-ListField-childContainer > div").setStyle("float", "left");
102 
103             if (value.used) {
104                 node.all(".inputEx-Field span").setStyle("textDecoration", "line-through");// strike through used tokens
105                 //f.disable();
106                 //(new Y.Node(this.subFields[i].divEl)).all("input").setStyle("textDecoration", "line-through");
107             }
108             return subfield;
109         },
110         /**
111          * Add a new element to the list and fire updated event
112          * @method onAddButton
113          * @param {Event} e The original click event
114          */
115         onAddButton: function(e) {
116             e.halt();
117             var i, total = this.promptMessage(),
118                     game = this.parentField.parentWidget.get("entity"),
119                     //teamCount = game.get("teams").length+1,
120                     teamCount = this.subFields.length + 1,
121                     prefix = game.get("token");
122 
123             for (i = 0; i < total; i += 1) {                                    // Add key fields
124                 this.addElement({
125                     key: prefix + "-" + teamCount
126                 });
127                 teamCount += 1;
128             }
129             this.fireUpdatedEvt();                                              // Fire updated !
130         },
131         promptMessage: function() {
132             return prompt("How many keys do you want to generate?");
133         }
134     });
135     Y.inputEx.registerType("enrolmentkeylist", EnrolmentKeyList);               // Register this class
