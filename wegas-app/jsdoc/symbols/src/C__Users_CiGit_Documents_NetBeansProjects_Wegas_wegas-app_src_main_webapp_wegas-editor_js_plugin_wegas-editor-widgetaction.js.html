 16             WidgetAction;

 18     /**
 19      * @class
 20      * @name Y.Plugin.WidgetAction
 21      * @extends Y.Plugin.Action
 22      * @constructor
 23      */
 24     WidgetAction = function() {
 25         WidgetAction.superclass.constructor.apply(this, arguments);
 26     };
 27     Y.extend(WidgetAction, Action, {}, {
 28         /** @lends Y.Wegas.EntityAction */
 29         NS: "WidgetAction",
 30         NAME: "WidgetAction",
 31         ATTRS: {
 32             widget: {},
 33             dataSource: {
 34                 getter: function(val) {
 35                     if (!val) {
 36                         return Wegas.Facade.Page;
 37                     }
 38                     return val;
 39                 }
 40             }
 41         }
 42     });

 45     /**
 46      * @name Y.Plugin.EditWidgetAction
 47      * @extends Y.Plugin.WidgetAction
 48      * @constructor
 49      */
 50     var EditWidgetAction = function() {
 51         EditWidgetAction.superclass.constructor.apply(this, arguments);
 52     };
 53     Y.extend(EditWidgetAction, WidgetAction, {
 54         /**
 55          * @function
 56          * @private
 57          */
 58         execute: function() {
 59             Plugin.EditEntityAction.hideRightTabs();
 60             var widget = this.get("widget"),
 61                     form = Plugin.EditEntityAction.showEditForm(widget, Y.bind(function(val, entity) {
 62                         Plugin.EditEntityAction.showEditFormOverlay();
 63                         var i, plugins = {}, plugin, cfg, oldCfg = entity.get("root").toObject();
 64                         entity.setAttrs(val);
 65                         for (i = 0; i < val.plugins.length; i += 1) {
 66                             plugin = Y.Plugin[Y.Wegas.Plugin.getPluginFromName(val.plugins[i].fn)];
 67                             if (!Y.Lang.isUndefined(entity._plugins[plugin.NS])) {      //that plugin exists on target
 68                                 entity[plugin.NS].setAttrs(val.plugins[i].cfg);
 69                                 plugins[plugin.NS] = true;                              //store namespace as treated
 70                             } else {
 71                                 entity.plug(plugin, val.plugins[i].cfg);
 72                                 plugins[plugin.NS] = true;                              //store namespace as treated
 73                             }
 74                         }
 75                         for (i in entity.get("plugins")) {                                    // remove
 76                             if (Y.Lang.isUndefined(plugins[entity.get("plugins")[i].fn])) {                       //An inexistant namespace
 77                                 entity.unplug(entity.get("plugins")[i].fn);
 78                             }
 79                         }
 80                         cfg = entity.get("root").toObject();
 81                         if (Y.JSON.stringify(cfg) !== Y.JSON.stringify(oldCfg)) {
 82                             this.get("dataSource").cache.patch(cfg, Y.bind(function() {
 83                                 entity.fire("AttributesChange", {attrs: val});
 84                                 Plugin.EditEntityAction.hideEditFormOverlay();
 85                                 Plugin.EditEntityAction.showFormMessage("success", "Item has been saved.");
 86                                 this.highlight(Plugin.EditEntityAction.currentEntity, true);
 87                             }, this));
 88                         } else {
 89                             Plugin.EditEntityAction.hideEditFormOverlay();
 90                         }
 91                     }, this), Y.bind(function(entity) {
 92                         if (entity) {
 93                             this.highlight(entity, false);
 94                         }
 95                     }, this)),
 96                     menuItems = Y.Array.filter(widget.getMenuCfg().slice(0), function(i) {
 97 
 98                         switch (i.label) {                                              // @hack add icons to some buttons
 99                             case "Delete":
100                             case "Edit":
101                                 i.label = '<span class="wegas-icon wegas-icon-' + i.label.replace(/ /g, "-").toLowerCase() + '"></span>' + i.label;
102                         }
103 
104                         // return (!i.label || (i.label.indexOf("New") < 0 && i.label.indexOf("Edit") < 0));
105                         return (!i.label || (i.label !== "New" && i.label.indexOf("Edit") < 0));
106                     });                                                                 // Retrieve menu and remove the first item
107 
108             this.highlight(widget, true);
109             form.toolbar.add(menuItems).item(0).get("contentBox").setStyle("marginLeft", "10px");
110         },
111         highlight: function(widget, val) {
112             if (!widget.get("destroyed")) {
113                 if (val || Y.Lang.isUndefined(val)) {
114                     widget.get("boundingBox").addClass("highlighted");
115                 }else{
116                     widget.get("boundingBox").removeClass("highlighted");
117                 }
118             }
119         }
120     }, {
121         NS: "EditWidgetAction",
122         NAME: "EditWidgetAction"
123     });
124     Plugin.EditWidgetAction = EditWidgetAction;
125 
126     /**
127      * @class
128      * @name Y.Plugin.AddChildWidgetAction
129      * @extends Y.Plugin.WidgetAction
130      * @constructor
131      */
132     var AddChildWidgetAction = function() {
133         AddChildWidgetAction.superclass.constructor.apply(this, arguments);
134     };
135     Y.extend(AddChildWidgetAction, WidgetAction, {
136         execute: function() {
137             Wegas.Editable.use(this.get("childCfg"), Y.bind(function() { // Load target widget dependencies
138                 var newWidget = Y.Wegas.Widget.create(this.get("childCfg"));
139 
140                 Plugin.EditEntityAction.showEditForm(newWidget, Y.bind(function(val) {
141                     Plugin.EditEntityAction.showEditFormOverlay();
142                     var targetWidget = this.get("widget"), widget = Y.Wegas.Widget.create(val);
143                     targetWidget.add(widget);
144 
145                     this.get("dataSource").cache.patch(targetWidget.get("root").toObject(), Y.bind(function() {
146                         var tw = new Y.Wegas.Text();
147                         Plugin.EditEntityAction.showFormMessage("success", "Element has been saved");
148                         Plugin.EditEntityAction.hideEditFormOverlay();
149                         tw.plug(Plugin.EditWidgetAction, {widget: this});
150                         tw.EditWidgetAction.execute();
151                     }, widget));
152                 }, this));
153             }, this));
154         }
155     }, {
156         NS: "AddChildWidgetAction",
157         NAME: "AddChildWidgetAction",
158         ATTRS: {
159             childType: {},
160             childCfg: {
161                 value: {},
162                 getter: function(v) {
163                     if (!v.type) {
164                         v.type = this.get("childType");
165                     }
166                     return v;
167                 }
168             }
169         }
170     });
171     Plugin.AddChildWidgetAction = AddChildWidgetAction;
172 
173     /**
174      * @class
175      * @name Y.Plugin.DeleteWidgetAction
176      * @extends Y.Plugin.WidgetAction
177      * @constructor
178      */
179     var DeleteWidgetAction = function() {
180         DeleteWidgetAction.superclass.constructor.apply(this, arguments);
181     };
182     Y.extend(DeleteWidgetAction, WidgetAction, {
183         execute: function() {
184             if (confirm("Are your sure you want to delete this element ?")) {
185                 var targetWidget = this.get("widget"),
186                         root = targetWidget.get("root");
187                 targetWidget.destroy();
188                 this.get("dataSource").cache.patch(root.toObject());
189             }
190         }
191     }, {
192         NS: "DeleteWidgetAction",
193         NAME: "DeleteWidgetAction"
194     });
195     Plugin.DeleteWidgetAction = DeleteWidgetAction;
196     /**
197      * @class
198      * @name Y.Plugin.DeleteLayoutWidgetAction
199      * @extends Y.Plugin.WidgetAction
200      * @constructor
201      */
202     Plugin.DeleteLayoutWidgetAction = function() {
203         Plugin.DeleteLayoutWidgetAction.superclass.constructor.apply(this, arguments);
204     };
205     Y.extend(Plugin.DeleteLayoutWidgetAction, WidgetAction, {
206         execute: function() {
207             var targetWidget = this.get("widget"),
208                     root = targetWidget.get("root");
209             if (targetWidget.size() > 0) {
210                 alert("Please delete content first");
211             } else if (confirm("Are your sure your want to delete this widget and all of its content ?")) {
212                 if (root !== targetWidget) {
213                     targetWidget.destroy();
214                 } else if (targetWidget.item && targetWidget.item(0)) { // @TODO: Panic mode, to change
215                     targetWidget.destroyAll();
216                 }
217 
218                 this.get("dataSource").cache.patch(root.toObject());
219             }
220         }
221     }, {
222         NS: "DeleteLayoutWidgetAction",
223         NAME: "DeleteLayoutWidgetAction"
224     });
