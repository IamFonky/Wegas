 14 YUI.add("wegas-template", function(Y) {
 15     "use strict";
 16 
 17     var Micro = Y.Template.Micro, Wegas = Y.Wegas, AbstractTemplate;
 18     /**
 19      * @name Y.Wegas.AbstractTemplate
 20      * @extends Y.Widget
 21      * @borrows Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable
 22      * @class
 23      * @constructor
 24      * @description  Display  Wegas variables instance (or/and descriptor) under
 25      * specifique templates : text, title, box, fraction and valuebox.
 26      * It is also possible to create a custom template.
 27      */
 28     AbstractTemplate = Y.Base.create("wegas-template", Y.Widget, [Y.WidgetChild, Wegas.Widget, Wegas.Editable], {
 29         /*@lends Y.Wegas.AbstractTemplate#*/
 30         syncUI: function() {
 31             Y.log("syncUI()", "log", "Wegas.AbstractTemplate");
 32             var template = this.getTemplate(),
 33                     data = this.computeData();
 34 
 35             try {
 36                 this.get("contentBox").setHTML(template(data));
 37             } catch (e) {
 38                 Y.log("Error rendering template: " + template, "error", "Wegas.AbstractTemplate");
 39             }
 40         },
 41         bindUI: function() {
 42             this.after(["dataChange", "variableChange"], this.syncUI);
 43             this.vdUpdateHandler = Wegas.Facade.VariableDescriptor.after("update", this.syncUI, this);
 44         },
 45         getTemplate: function() {
 46             return this.TEMPLATE;
 47         },
 48         computeData: function() {
 49             var data = {}, desc = this.get("variable.evaluated");
 50 
 51             if (desc) {
 52                 if (desc instanceof Wegas.persistence.ListDescriptor && desc.get("currentItem")) {       // If the widget is a list,
 53                     desc = desc.get("currentItem");                             // display it with the current list and the current element
 54                 }
 55 
 56                 data.label = this.undefinedToEmpty(desc.getLabel());
 57                 data.value = this.undefinedToEmpty(desc.getInstance().get("value"));
 58                 data.maxValue = this.undefinedToEmpty(desc.get("maxValue"));
 59                 data.minValue = this.undefinedToEmpty(desc.get("minValue"));
 60                 data.defaultValue = this.undefinedToEmpty(desc.get("defaultValue"));
 61                 data.variable = desc;
 62             }
 63             return Y.mix(Y.merge(this.get("data")), data, false, null, 0, true);
 64         },
 65         getEditorLabel: function() {
 66             var variable = this.get("variable.evaluated");
 67             if (variable) {
 68                 return variable.getEditorLabel();
 69             }
 70             return null;
 71         },
 72         destructor: function() {
 73             this.vdUpdateHandler.detach();
 74         },
 75         undefinedToEmpty: function(value) {
 76             return Y.Lang.isUndefined(value) ? "" : "" + value;
 77         }
 78     }, {
 79         /*@lends Y.Wegas.AbstractTemplate*/
 80         EDITORNAME: "Variable template",
 81         ATTRS: {
 82             /**
 83              * The target variable, returned either based on the variableName attribute,
 84              * and if absent by evaluating the expr attribute.
 85              */
 86             variable: {
 87                 getter: Wegas.Widget.VARIABLEDESCRIPTORGETTER,
 88                 _inputex: {
 89                     _type: "variableselect",
 90                     label: "variable"
 91                 }
 92             },
 93             data: {
 94                 value: {},
 95                 _inputex: {
 96                     wrapperClassName: 'inputEx-fieldWrapper wegas-advanced-feature',
 97                     _type: "object",
 98                     useButtons: true,
 99                     required: false
100                 }
101             }
102         }
103     });
104 
105     Wegas.Template = Y.Base.create("wegas-template", AbstractTemplate, [], {
106         /*@lends Y.Wegas.Template#*/
107         TEMPLATES: {},
108         getTemplate: function() {
109             var template = this.get("custom"),
110                     hashCode = "" + Wegas.Helper.hashCode(template);
111 
112             if (Y.Lang.isUndefined(this.TEMPLATES[hashCode])) {
113                 this.TEMPLATES[hashCode] = Micro.compile(template);
114             }
115             return this.TEMPLATES[hashCode];
116         }
117     }, {
118         /*@lends Y.Wegas.Template*/
119         EDITORNAME: "Custom template",
120         ATTRS: {
121             custom: {
122                 type: "text",
123                 value: "<%= this.variable.getValue() || 'Undefined' %>",
124                 _inputex: {
125                     label: "Template"
126                 }
127             }
128         }
129     });
130     Wegas.ValueboxTemplate = Y.Base.create("wegas-template", AbstractTemplate, [], {
131         TEMPLATE: Micro.compile("<div class='wegas-template-valuebox'><label><%= this.label || '{label}'%></label</div><br/></label><div class='wegas-template-valuebox-units'><% for(var i=+this.minValue; i < +this.maxValue + 1; i+=1){%>" +
132                 "<div class='wegas-template-valuebox-unit <%= +i < +this.value ? ' wegas-template-valuebox-previous' : '' %><%= +i === +this.value ? ' wegas-template-valuebox-selected' : '' %>'><%= ''+i %></div><% } %></span>" +
133                 "</div></div>")
134     });
135     Wegas.BoxTemplate = Y.Base.create("wegas-template", AbstractTemplate, [], {
136         TEMPLATE: Micro.compile("<div class='wegas-template-box'><label><%= this.label || '{label}'%></label><br/>"
137                 + "<div class='wegas-template-box-units'><% for(var i=0; i < this.value; i+=1){%>" +
138                 "<div class='wegas-template-box-unit'></div><% } %></div>" +
139                 "<span class='wegas-template-box-value'>"
140                 + "(<%= this.value || '{value}' %>"
141                 //+ "<% if(this.defaultValue != ''){ %><%= '/' + (this.defaultValue || '{defaultValue}') %><% } %>"
142                 + ")</span></div>")
143     });
144     Wegas.NumberTemplate = Y.Base.create("wegas-template", AbstractTemplate, [], {
145         TEMPLATE: Micro.compile("<div class='wegas-template-text'><span><%= this.label || '{label}' %></  span><br/><span><%= this.value || '{value}' %></span></div>")
146     });
147     Wegas.TitleTemplate = Y.Base.create("wegas-template", AbstractTemplate, [], {
148         TEMPLATE: Micro.compile("<div class='wegas-template-title'><%= this.label || '{label}'%></div>")
149     });
150     Wegas.FractionTemplate = Y.Base.create("wegas-template", AbstractTemplate, [], {
151         TEMPLATE: Micro.compile("<div class='wegas-template-fraction'>"
152                 + "<label><%= this.label || '{label}'%></label>"
153                 //+ "<%= (this.minValue || '{minValue}')%> /"
154                 + "<%= (this.value || '{label}') + '/' + (this.maxValue || '{maxValue}') %></div>")
155     });
156     Wegas.TextTemplate = Y.Base.create("wegas-template", AbstractTemplate, [], {
157         TEMPLATE: Micro.compile("<div><%== this.value || 'Unable to find text' %></div>")
158     });
159 
160 });
