 19     /**
 20      * @name Y.Wegas.Widget
 21      * @class Extension common to all wegas widgets
 22      */
 23     function Widget() {
 24         this.after("render", function() {
 25             var bb = this.get(BOUNDING_BOX);
 26             bb.addClass("wegas-widget");
 27             if (this.get("cssClass")) {
 28                 bb.addClass(this.get("cssClass"));
 29             }
 30         });
 31         this.constructor.CSS_PREFIX = this.constructor.CSS_PREFIX               // If no prefix is set, use the name (without
 32                 || this.constructor.NAME.toLowerCase();                         // the usual "yui3-" prefix)
 33         this._cssPrefix = this.constructor.CSS_PREFIX;
 34 
 35         this.publish("showOverlay", {// Add custom event
 36             emitFacade: true
 37         });
 38         this.publish("hideOverlay", {
 39             emitFacade: true
 40         });
 41         this.publish("message", {
 42             emitFacade: true
 43         });
 44         this.publish("AttributesChange", {
 45             bubbles: false,
 46             defaultFn: function() {
 47                 var widget = this.rebuild();
 48                 if (Y.Plugin.EditEntityAction.currentEntity === this) {         // @FIXME @fx wtf?
 49                     Y.Plugin.EditEntityAction.currentEntity = widget;
 50                 }
 51             }
 52         });
 53     }

 58          * @function
 59          * @private
 60          */
 61         defaultFailureHandler: function(e) {
 62             this.hideOverlay();
 63             var error = e.response.message || e.response.results.message || "Error during request.",
 64                     test = error.match(/ConstraintViolationException: (.*) is out of bound/),
 65                     stringMessage = error.match(/Error: StringMessage: (.*)/);
 66             if (test) {
 67                 this.showMessageBis("error", "You don't have enough " + test[1] + ".");
 68             } else if (stringMessage) {
 69                 this.showMessageBis("error", stringMessage[1].split(" in <")[0]);
 70             } else if (error.match(/ERROR: duplicate key value violates unique constraint "unq_game_0"/)) {
 71                 this.showMessageBis("error", "This name is already in use");
 72             } else {
 73                 this.showMessageBis("error", error);
 74             }
 75             // e.halt(true);
 76         },
 77         /**
 78          * @function
 79          * @private
 80          * @description show an loading - overlay on all the screen.
 81          */
 82         showOverlay: function() {
 83             this.fire("wegas:showOverlay")
 84         },
 85         /**
 86          * @function
 87          * @private
 88          * @description hide overlay (see function showOverlay).
 89          */
 90         hideOverlay: function() {
 91             this.fire("wegas:hideOverlay");
 92         },
 93         /**
 94          * Display a closable message with a status-image.
 95          * Status-image of message depends of level parameters
 96          * Txt parameters is the displayed text.
 97          * Timeout is the displaying time of this message.
 98          * @function
 99          * @private
100          * @param level
101          * @param txt
102          * @param timeout
103          * @description
104          *
105          * @deprecated Will be replace by showMessageBis
106          */
107         showMessage: function(level, txt, timeout) {
108             this.emitDOMMessage(level, {content: txt, timeout: timeout});
109         },
110         emitDOMMessage: function() {
111             var bb = this.get("boundingBox");
112             bb.emitDOMMessage.apply(bb, arguments);
113         },
114         /**
115          * Will replace the original showMessage
116          */
117         showMessageBis: function(level, msg, timeout) {
118             this.fire("wegas:message", {
119                 level: level,
120                 content: msg,
121                 timeout: timeout
122             });
123         },
124         rebuild: function() {
125             var parent, index, cfg;
126             if (this.isRoot()) {
127                 parent = Y.Widget.getByNode(this.get(BOUNDING_BOX).get("parentNode"));
128                 parent.reload();
129                 return parent.get("widget"); // dependencies should (and must) be loaded by now that way we obtain the new widget
130             }
131             parent = this.get("parent");
132             index = parent.indexOf(this);
133             cfg = this.toObject();
134             this.destroy();
135             return parent.add(cfg, index).item(0);
136         }

519         /**
520          * @function

522          * @param config
523          * @return child
524          * @description function to create and return a widget with the given
525          *  configuration. Log an exception if creation isn't possible.
526          */
527         create: function(config) {
528             var child, Fn, type = config.childType || config.type;
529             if (type) {
530                 Fn = Lang.isString(type) ? Wegas[type] || Y[type] : type;
531             }
532 
533             if (Lang.isFunction(Fn)) {
534                 child = new Fn(config);
535             } else {
536                 Y.log("Could not create a child widget because its constructor is either undefined or invalid(" + type + ").", "error", "Wegas.Widget");
537             }
538 
539             return child;
540         },

552          *
553          *  This getter is to be used for any object attribute that references a VariableDescriptor and
554          *  has either an name, id or expr parameter.
555          *
556          */
557         /**
558          * @function
559          * @private
560          * @param cfg, cb
561          * @return value
562          * @description This getter is to be used for any object attribute
563          *  that references a VariableDescriptor and has either an name, id
564          *  or expr parameter.
565          */
566         VARIABLEDESCRIPTORGETTER: function(val, fullName) {
567             var ds = Wegas.Facade.VariableDescriptor;
568 
569 
570             if (val && fullName.split(".")[1] === "evaluated") {                // If evaluated value is required
571 
572                 if (val.content) {                                              // Eval based on the field (new pattern)
573                     try {
574                         val.evaluated = Wegas.Facade.VariableDescriptor.script.localEval(val.content);
575                     } catch (e) {
576                         Y.log("Unable to read expression: " + val.expr, "error", "Wegas.Widget");
577                         val.evaluated = null;
578                     }
579                 } else if (val.name) {                                          // @backwardcompatibility
580                     val.evaluated = ds.cache.find('name', val.name);
581 
582                 } else if (val.expr) {                                          // @backwardcompatibility if absent evaluate the expr field
583                     try {
584                         val.evaluated = ds.cache.findById(Wegas.Facade.VariableDescriptor.script.localEval(val.expr));
585                     } catch (e) {
586                         Y.log("Unable to read expression: " + val.expr, "error", "Wegas.Widget");
587                         val.evaluated = null;
588                     }
589                 } else if (val.id) {
590                     val.evaluated = ds.cache.findById(val.id);
591                 }
592             }
593 
594             if (val && fullName.indexOf(".") < 0) {                             // If the getter requires the full object (e.g. serialisation)
595                 delete val.evaluated;                                           // Remove the ref to the evaluated descriptor
596             }
597 
598             return val;
599         },

663     /**
664      * Simulate a DOM Event bubbling up to a listener and stops.
665      * @param {String} type
666      * @param {Object} data
667      */
668     Y.Node.prototype.emitDOMMessage = function(type, data) {
669         var ev = "dom-message:" + type;
670         data = data || {};
671         data.type = type;
672         try {
673             this.ancestor(function(node) {
674                 return node.getEvent(ev) ? true : false;
675             }, true).fire(ev, data);
676         } catch (e) {
677             //no ancestor found
678         }
