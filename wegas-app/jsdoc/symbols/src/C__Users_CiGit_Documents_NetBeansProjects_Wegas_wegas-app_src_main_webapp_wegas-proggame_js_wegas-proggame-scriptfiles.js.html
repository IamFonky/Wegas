 12 YUI.add('wegas-proggame-scriptfiles', function(Y) {
 13     "use strict";
 14     var CONTENTBOX = 'contentBox';
 15     /**
 16      *  @class Display the proggame files
 17      *  @name Y.Plugin.ScriptFiles
 18      *  @extends Y.Widget
 19      *  @constructor
 20      */
 21     var ScriptFiles = Y.Base.create("wegas-proggame-scriptfiles", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 22         handlers: null,
 23         initializer: function() {
 24             this.handlers = {};
 25 
 26             this.publish('openFile', {
 27                 broadcast: true,
 28                 emitFacade: true
 29             });
 30         },
 31         renderUI: function() {
 32             var cb = this.get(CONTENTBOX);
 33             this.addButton = new Y.Wegas.Button({
 34                 label: "Add file"
 35             });
 36             this.addButton.render(cb);
 37             cb.append("<div class='fileName'></div>");
 38 
 39             this.createPanel();
 40         },
 41         bindUI: function() {
 42             this.handlers.sync = Y.Wegas.Facade.VariableDescriptor.after("update", this.syncUI, this);
 43 
 44             this.addButton.on("click", function(e) {
 45                 //this.newFilePanel.show();
 46                 this.createPanel().show();
 47             }, this);
 48         },
 49         syncUI: function() {
 50 
 51             Y.Wegas.Facade.VariableDescriptor.cache.getWithView(this.get('variable.evaluated').getInstance(), "Extended", {
 52                 on: {
 53                     success: Y.bind(function(e) {
 54                         var node, i,
 55                                 messages = e.response.entity.get("messages"),
 56                                 fileNameDiv = this.get(CONTENTBOX).one(".fileName");
 57 
 58                         fileNameDiv.get('childNodes').remove();
 59 
 60                         if (messages.length === 0) {
 61                             fileNameDiv.append("<p>No files to display</p>");
 62                         } else {
 63                             for (i = 0; i < messages.length; i++) {
 64                                 node = Y.Node.create('<p>' + messages[i].get("subject") + '</p>');
 65                                 node.data = messages[i];
 66                                 fileNameDiv.append(node);
 67 
 68                                 node.on("click", function(e) {
 69                                     //console.log("openFile event");
 70                                     this.fire("openFile", {file: e.target.data});
 71                                 }, this);
 72                             }
 73                         }
 74                     }, this)
 75                 }
 76             });
 77         },
 78         createPanel: function() {
 79             var node = Y.Node.create('<p><input type="text" name="fileName" placeholder="File name"/></p>'), that = this;
 80             this.newFilePanel = new Y.Wegas.Panel({
 81                 srcNode: node,
 82                 headerContent: 'Add a new file',
 83                 width: 250,
 84                 modal: true,
 85                 centered: true,
 86                 visible: false,
 87                 render: true,
 88                 buttons: [{
 89                         value: 'Add',
 90                         section: Y.WidgetStdMod.FOOTER,
 91                         action: function(e) {
 92                             e.preventDefault();
 93                             this.exit();
 94                             that.addFile(this.get("srcNode").one("input").get("value"));
 95                             this.get("srcNode").one("input").set("value", "");
 96                         }
 97                     }, {
 98                         value: 'Cancel',
 99                         section: Y.WidgetStdMod.FOOTER,
100                         action: function(e) {
101                             e.preventDefault();
102                             this.exit();
103                         }
104                     }
105                 ]
106             });
107             return this.newFilePanel;
108         },
109         addFile: function(fileName) {
110             Y.Wegas.Facade.VariableDescriptor.sendRequest({
111                 request: "/Script/Run/" + Y.Wegas.Facade.Game.get('currentPlayerId'),
112                 //'Managed-Mode': 'false',
113                 cfg: {
114                     method: "POST",
115                     data: {
116                         "@class": "Script",
117                         language: "JavaScript",
118                         content: "Variable.find(gameModel, 'files').sendMessage(self, '', '" + fileName + ".js', '', []);"
119                     }
120                 },
121                 on: {
122                     success: Y.bind(function(e) {
123                         this.fire("openFile", {file: e.response.entity});
124                     }, this)
125                 }
126             });
127         },
128         destructor: function() {
129             var k;
130             for (k in this.handlers) {
131                 this.handlers[k].detach();
132             }
133             this.addButton.destroy();
134         }
135     }, {
136         ATTRS: {
137             variable: {
138                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
139                 _inputex: {
140                     _type: "variableselect",
141                     label: "variable",
142                     classFilter: ["InboxDescriptor"]
143                 }
144             }
145         }
146     });
147     Y.namespace('Wegas').ScriptFiles = ScriptFiles;
148 });
