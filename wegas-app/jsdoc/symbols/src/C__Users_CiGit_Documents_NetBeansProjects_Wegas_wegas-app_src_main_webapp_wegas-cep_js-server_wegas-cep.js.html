 13 function passPeriod() {
 14     var currentTime = phases.descriptor.items.get(currentPhase.value - 1),
 15             currentTimeInstance = currentTime.getInstance(self);
 16     if (currentTimeInstance.value == currentTime.maxValue) {
 17         phases.value += 1;
 18         currentPhase.value += 1;
 19         phases.descriptor.items.get(currentPhase.value - 1).getInstance(self).value += 1;
 20     } else {
 21         currentTimeInstance.value += 1;
 22     }
 23     humanResources.value = humanResources.descriptor.defaultInstance.value;
 24 }

 30 function setTeamMotivation() {
 31     var i, gm = self.getGameModel(),
 32             listEmployees = Variable.findByName(gm, 'employees'),
 33             employeeInstance,
 34             teamMotivation = Variable.findByName(gm, 'teamMotivation'),
 35             morals = [],
 36             mSum = 0,
 37             mAverage,
 38             mGap = [],
 39             tmpVal,
 40             SumOfSquareOfMGap = 0,
 41             standardDeviation,
 42             newTeamMotivation;
 43     if (!listEmployees || !teamMotivation) {
 44         return;
 45     }
 46 
 47     // calcul arithmetic average of morals (on actives employees only)
 48     for (i = 0; i < listEmployees.items.size(); i++) {
 49         employeeInstance = listEmployees.items.get(i).getInstance(self);
 50         if (employeeInstance.getActive() == true) {
 51             tmpVal = parseInt(employeeInstance.getMoral());
 52             //Bound moral between teamMotivation Min val and max val
 53             if (boundConstrain(tmpVal, teamMotivation.getMinValue(), teamMotivation.getMaxValue()) !== tmpVal) {
 54                 tmpVal = boundConstrain(tmpVal, teamMotivation.getMinValue(), teamMotivation.getMaxValue());
 55                 employeeInstance.setMoral(tmpVal);
 56             }
 57             morals.push(tmpVal);
 58             mSum += tmpVal;
 59         }
 60     }
 61     mAverage = mSum / morals.length;
 62 
 63     //For each moral calcul gap between moral and average (= moral - average);
 64     //take the sum of each square of gaps (= Sum(n_gaps * n_gaps)).
 65     for (i = 0; i < morals.length; i++) {
 66         mGap.push(morals[i] - mAverage);
 67         SumOfSquareOfMGap += Math.pow(mGap[i], 2);
 68     }
 69 
 70     // calcul the standard deviation
 71     standardDeviation = Math.sqrt(SumOfSquareOfMGap / morals.length);
 72 
 73     //calcul the new Team Motivation
 74     newTeamMotivation = (mAverage + (mAverage - standardDeviation)) / 2;
 75     if (newTeamMotivation < 0) { //in extrems cases
 76         newTeamMotivation = 0;
 77     }
 78 
 79     //set teamMotivation
 80     teamMotivation.getInstance(self).setValue(Math.round(newTeamMotivation));
 81 }
 82 
 83 function boundConstrain(val, lowerBound, upperBound) {
 84     return Math.max(lowerBound, Math.min(val, upperBound));
 85 }

 87 /**
 88  * Set picture depending of resource's current moral.
 89  */
 90 function changePicture() {
 91     var i, j, valueInst, valueDescr, gm = self.getGameModel(), oldImg, newImg, moral,
 92             listEmployees = Variable.findByName(gm, 'employees'),
 93             imgSuffixe = ['Triste', 'Neutre', 'Joie'];
 94     if (!listEmployees) {
 95         return;
 96     }
 97     for (i = 0; i < listEmployees.items.size(); i++) {
 98         valueDescr = listEmployees.items.get(i);
 99         valueInst = valueDescr.getInstance(self);
100         moral = parseInt(valueInst.getMoral());
101         oldImg = valueInst.getProperty('picture');
102         newImg = null;
103         switch (true) {
104             case moral < 40 :
105                 for (j = 0; j < imgSuffixe.length; j++) {
106                     if (oldImg.indexOf(imgSuffixe[j]) > -1) {
107                         newImg = oldImg.replace(imgSuffixe[j], imgSuffixe[0]);
108                         break;
109                     }
110                 }
111                 break;
112             case moral < 75 :
113                 for (j = 0; j < imgSuffixe.length; j++) {
114                     if (oldImg.indexOf(imgSuffixe[j]) > -1) {
115                         newImg = oldImg.replace(imgSuffixe[j], imgSuffixe[1]);
116                         break;
117                     }
118                 }
119                 break;
120             default :
121                 for (j = 0; j < imgSuffixe.length; j++) {
122                     if (oldImg.indexOf(imgSuffixe[j]) > -1) {
123                         newImg = oldImg.replace(imgSuffixe[j], imgSuffixe[2]);
124                         break;
125                     }
126                 }
127                 break;
128         }
129         if (newImg) {
130             valueInst.setProperty('picture', newImg);
131         }
132     }
133 }
