 27     Y.extend(ResetAction, Action, {
 28         /** @lends Y.Plugin.ResetAction# */
 29 
 30         /**
 31          * @function
 32          * @private
 33          */
 34         execute: function() {
 35             if (confirm("This will restart for every player. Are you sure?")) {
 36                 var host = this.get("host");
 37                 host.showOverlay();
 38                 Wegas.Facade.VariableDescriptor.sendRequest({
 39                     request: '/Reset/',
 40                     on: {
 41                         success: Y.bind(host.hideOverlay, host),
 42                         failure: Y.bind(host.defaultFailureHandler, host)
 43                     }
 44                 });
 45             }
 46         }
 47     }, {
 48         NS: "wegas",
 49         NAME: "ResetAction"
 50     });

 62     Y.extend(OpenTabAction, Action, {
 63         /** @lends Y.Plugin.OpenTabAction# */
 64 
 65         /**
 66          * @function
 67          * @private
 68          */
 69         execute: function() {
 70             var label = this.get("label") || this.get("host").get("label");
 71             if (this.get("emptyTab")) {
 72                 //if (this.get("tabSelector") === "#rightTabView") {
 73                 //        && !Wegas.TabView.findTab(label)) {
 74                 Y.Widget.getByNode("#rightTabView").destroyAll();
 75             }
 76 
 77             var tab = Wegas.TabView.findTabAndLoadWidget(label, this.get("tabSelector"),
 78                     {}, this.get("wchildren"));                                 // Forward plugin data to the target widget
 79 
 80 
 81             tab.set("selected", this.get("selected"));
 82 
 83             if (this.get("emptyTab") || this.get("tabSelector") !== "#rightTabView") {// @hack
 84                 tab.set("selected", 2);
 85             }
 86 
 87             //if (this.get("tabSelector") !== "#rightTabView") {                          // @hack
 88             tab.plug(Y.Plugin.Removeable);
 89             //}
 90         }

 99          * <ul>
100          *    <li>tabSelector: the id of a Y.Wegas.TabView where a new tab will be created if none
101          *    was found <i>default: #centerTabView</i></li>
102          *    <li>wchildren: the element to render in the opened tab</li>
103          * </ul>
104          *
105          * @field
106          * @static
107          */

111                 value: false

114                 value: '#centerTabView'

250     /**
251      * Class for display the player link in menu's
252      *
253      * @name Y.Wegas.Linkwidget
254      * @extends Y.Widget
255      * @class  Allows to display the player link in a menu.
256      * the link is in a textField. For this field inputEx is used.
257      * @constructor
258      * @param Object Will be used to fill attributes field
259      */
260     Linkwidget = Y.Base.create("wegas-playerlink-buttons", Y.Widget, [Wegas.Widget, Wegas.Editable, Y.WidgetChild], {
261         /** @lends Y.Wegas.Linkwidget# */
262         CONTENT_TEMPLATE: '<div><div class="playerlink-label"><p>Link</p><div></div>',
263         /**
264          * 1) Add a <div class="playerlink-label"><p>Player link</p><div> node fordisplay a label in the menu
265          * 2) Add the inputeExStringField
266          * 3) Stop the click event on this contentbox
267          * @function
268          * @private
269          */
270         renderUI: function() {
271             var cb = this.get(CONTENTBOX);
272             this.textField = new Y.inputEx.StringField({
273                 parentEl: cb
274             });
275             cb.on("click", function(e) {
276                 e.halt(true);
277                 this.textField.el.select();
278             }, this);
279         },
280         /**
281          * Add the new url
282          * @function
283          * @private
284          */
285         syncUI: function() {
286             var gm = this.get("entity");
287             if (!gm) {
288                 gm = Y.Wegas.Facade.Game.cache.getCurrentGame();
289             }
290             var url = Wegas.app.get("base") + "game.html?token=" + gm.get("token");
291             this.textField.setValue(url);
292         }
293     }, {
294         /** @lends Y.Wegas.Linkwidget */
295 
296         /**
297          * <p><strong>Attributes</strong></p>
298          * <ul>
299          *    <li>entity: the gamemodel or team entity, which the link will point to.</li>
300          * </ul>
301          *
302          * @field
303          * @static
304          */
305         ATTRS: {
306             entity: {
307                 "transient": true
308             }
309         }
310     });
311     Y.namespace("Wegas").Linkwidget = Linkwidget;

314     /**
315      * Class for display the player link in menu's
316      *
317      * @name Y.Wegas.Linkwidget
318      * @extends Y.Widget
319      * @class  Allows to display the player link in a menu.
320      * the link is in a textField. For this field inputEx is used.
321      * @constructor
322      * @param Object Will be used to fill attributes field
323      */
324     var JoinOrResumeButton = Y.Base.create("button", Wegas.Button, [], {
325         /** @lends Y.Wegas.Linkwidget# */
326         renderUI: function() {
327             JoinOrResumeButton.superclass.renderUI.apply(this);
328 
329             var entity = this.get("entity"),
330                     findInTeam = function(team) {
331                 return Y.Array.find(team.get("players"), function(p) {
332                     return p.get("userId") === Y.Wegas.Facade.User.get("currentUserId");
333                 });
334             }, findInGame = function(game) {
335                 return Y.Array.find(game.get("teams"), findInTeam);
336             };
337 
338             if (entity instanceof Y.Wegas.persistence.Team) {                   // 1st case: clicked on an team
339                 if (findInTeam(entity)) {
340                     this.set("label", "Resume");
341                     this.plug(Y.Plugin.OpenGameAction);
342                     return;
343                 } else if (findInGame(Y.Wegas.Facade.Game.cache.findById(entity.get("gameId")))) {
344                     this.set("disabled", true);
345                     return;
346                 }
347             } else if (findInGame(entity)) {
348                 this.set("label", "Resume");
349                 this.plug(Y.Plugin.OpenGameAction, {entity: this.get("entity")});
350                 return;
351             }
352             this.plug(Y.Plugin.OpenTabAction, {
353                 tabSelector: "#rightTabView",
354                 emptyTab: true,
355                 wchildren: [{
356                         type: "JoinTeam",
357                         entity: this.get("entity")
358                     }]
359             });
360         }
361     }, {
362         /** @lends Y.Wegas.Linkwidget */
363 
364         /**
365          * <p><strong>Attributes</strong></p>
366          * <ul>
367          *    <li>entity: the gamemodel or team entity, which the link will point to.</li>
368          * </ul>
369          *
370          * @field
371          * @static
372          */
373         ATTRS: {
374             entity: {}
375         }
376     });
377     Y.namespace("Wegas").JoinOrResumeButton = JoinOrResumeButton;
