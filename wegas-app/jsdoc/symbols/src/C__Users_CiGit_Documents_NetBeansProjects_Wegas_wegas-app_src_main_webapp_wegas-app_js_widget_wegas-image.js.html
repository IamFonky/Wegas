 13 YUI.add("wegas-image", function(Y) {
 14     "use strict";
 15     var CONTENTBOX = "contentBox", WImage;
 16     /**
 17      * @name Y.Wegas.WWImage
 18      * @extends Y.Widget
 19      * @borrows Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable
 20      * @class class to display simple image
 21      * @constructor
 22      * @description  Display a string (given as ATTRS) in content box
 23      */
 24     WImage = Y.Base.create("wegas-image", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 25         /** @lends Y.Wegas.WImage# */
 26         image: null,
 27         CONTENT_TEMPLATE: "<img style='width:inherit;height:inherit'></img>",
 28         /**
 29          * Lifecycle method
 30          * @function
 31          * @private
 32          */
 33         initializer: function() {
 34             this.publish("error", {fireOnce: true, async: true});
 35             this.publish("load", {fireOnce: true, async: true});
 36             this.image = this.get("contentBox");
 37         },
 38         getEditorLabel: function() {
 39             return this.get("url");
 40         },
 41         /**
 42          * Lifecycle method
 43          * @function
 44          * @private
 45          */
 46         renderUI: function() {
 47             this.image = this.get("boundingBox").one("img"); // !! IE 8 : this.image._node === this.get(CONTENTBOX)._node => false ...
 48             this.get("url");
 49         },
 50         /**
 51          * Lifecycle method
 52          * @function
 53          * @private
 54          */
 55         bindUI: function() {
 56             this.image.on("load", function(e) {
 57                 if (!this.CSSSize) { // adapt only without plugin
 58                     this.get("boundingBox").setStyles({width: this.image.width, height: this.image.height});
 59                 }
 60                 this.fire("load");
 61                 this.getEvent("load").fired = true;
 62             }, this);
 63             this.image.on("error", function(e) {
 64                 this.image.setAttribute("alt", "Image error");
 65                 this.image.setAttribute("src", "");
 66                 this.fire("error");
 67                 this.getEvent("error").fired = true;
 68             }, this);
 69         }
 70 
 71     }, {
 72         /** @lends Y.Wegas.WImage */
 73         EDITORNAME: "Image",
 74         FILEENTRY: Y.Wegas.Facade.File.get("source") + "read",
 75         ATTRS: {
 76             url: {
 77                 value: "",
 78                 type: "string",
 79                 setter: function(val) {
 80                     this.getEvent("load").fired = false;
 81                     this.getEvent("error").fired = false;
 82                     this.image.setAttribute("src", (val.indexOf("/") === 0) ?
 83                             this.constructor.FILEENTRY + val : //Wegas Filesystem
 84                             val);
 85                     return val;
 86                 },
 87                 _inputex: {
 88                     _type: "wegasimageurl"
 89                 }
 90             },
 91             complete: {
 92                 readOnly: true,
 93                 "transient": true,
 94                 getter: function() {
 95                     return this.image && this.image.getDOMNode() ? this.image.getDOMNode().complete : true;
 96                 }
 97             }
 98         }
 99     });
100     Y.namespace("Wegas").Image = WImage;
101 });
