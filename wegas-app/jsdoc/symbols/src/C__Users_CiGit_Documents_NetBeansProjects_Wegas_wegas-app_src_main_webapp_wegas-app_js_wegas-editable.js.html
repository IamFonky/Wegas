 15     "use strict";

 17     var Lang = Y.Lang;

 19     /**
 20      *  Add custom attributes to be used in ATTR param in static cfg.
 21      */
 22     Y.Base._ATTR_CFG.push("type", "properties", "_inputex", "optional", "format",
 23             "choices", "items", "enum", "default", "transient");
 24     Y.Base._ATTR_CFG_HASH = Y.Array.hash(Y.Base._ATTR_CFG);

 26     /**
 27      * @name Y.Wegas.Editable
 28      * @class Extension to be added to a Y.Widget, allowing edition on wegas entities.
 29      * @constructor
 30      */
 31     function Editable() {
 32     }
 33 
 34     Y.mix(Editable.prototype, {
 35         /** @lends Y.Wegas.Editable# */
 36 
 37         /**
 38          * Serialize to a json object. Method used <b>recursively</b> by JSON.stringify
 39          *
 40          * @function
 41          * @returns {Object}
 42          */
 43         toJSON: function() {
 44             var k, ret = this.getAttrs(),
 45                     attrCfgs = this.getAttrCfgs();
 46 
 47             for (k in ret) {
 48                 if (attrCfgs[k] && attrCfgs[k]["transient"]) {                  // Remove any transient attribute
 49                     delete ret[k];
 50                 }
 51                 //if(this.constructor.ATTRS[k] && this.constructor.ATTRS[k].value === ret[k]){ /* DEFAULT VALUE REMOVAL */
 52                 //    delete ret[k];
 53                 //}
 54             }
 55             return ret;                                                         // Return a copy of this's fields.
 56         },
 57         /**
 58          * Create a new JSON Object from this entity, filtered out by mask
 59          *
 60          * @fixme This method will parse the field and the clone it
 61          *
 62          * @function
 63          * @param {Array|String} mask or a list of string parameters
 64          * @return {Object} a filtered out clone
 65          */
 66         toObject: function(mask) {
 67             var masker;
 68             mask = Lang.isArray(mask) ? mask : Array.prototype.slice.call(arguments);
 69             masker = mask.length > 0 ? function(key, value) {
 70                 if (Y.Array.indexOf(mask, key) !== -1) {
 71                     return undefined;
 72                 } else {
 73                     return value;
 74                 }
 75             } : null;
 76             return Y.JSON.parse(Y.JSON.stringify(this), masker);
 77         },
 78         /**
 79          * Create a new Object from this entity
 80          * may be used by revive
 81          * @method clone
 82          * @return {Object} a clone
 83          */
 84         clone: function() {
 85             return this.toObject(["id", "variableInstances"]);
 86         },
 87         /**
 88          * Returns the form configuration associated to this object, to be used a an inputex object.
 89          * @param {Array} fieldsToIgnore (optional), don't create these inputs.
 90          */
 91         getFormCfg: function(fieldsToIgnore) {
 92             var i, form, schemaMap, attrCfgs, builder;
 93             fieldsToIgnore = (fieldsToIgnore || []);
 94             // form = forms[this.get('@class')] || forms[this.get("type")]
 95 
 96             form = form || this.constructor.EDITFORM;                           // And if no form is defined we check if there is a default one defined in the entity
 97 
 98             if (!form) {                          // If no edit form could be found, we generate one based on the ATTRS parameter.
 99                 attrCfgs = this.getAttrCfgs();
100 
101                 for (i in attrCfgs) {
102                     attrCfgs[i]["default"] = attrCfgs[i].value;// Use the value as default (useful form json object serialization)
103 
104                     if (attrCfgs[i]["transient"] || Y.Array.indexOf(fieldsToIgnore, i) > -1) {
105                         delete attrCfgs[i];
106                     }
107                 }
108 
109                 schemaMap = {
110                     Entity: {
111                         properties: attrCfgs
112                     }
113                 };
114 
115                 builder = new Y.inputEx.JsonSchema.Builder({
116                     schemaIdentifierMap: schemaMap,
117                     defaultOptions: {
118                         showMsg: true
119                     }
120                 });
121                 form = builder.schemaToInputEx(schemaMap.Entity);
122 
123                 form.fields.sort(function(a, b) {
124                     var a = a.index || 0;
125                     b = b.index || 0;
126                     return a - b;
127                 });
128             }
129             return form || [];
130         },
131         /**
132          * Returns the edition menu associated to this object, to be used a an inputex object.
133          *
134          * @function
135          * @param {type} data
136          * @returns {Array}
137          */
138         getMenuCfg: function(data) {
139             var menu = this.getStatic("EDITMENU", true)[0] || [];                // And if no form is defined we return the default one defined in the entity
140 
141             data = data || {};
142             data.entity = data.entity || this;
143             data.widget = data.widget || this;
144 
145             Editable.mixMenuCfg(menu, data);
146             return menu;
147         },
148         /**
149          * Returns the edition menu associated to this object, to be used a an wysiwyg editor.
150          * @function
151          * @param {type} data
152          * @returns {unresolved}
153          */
154         getMethodCfgs: function(data) {
155             var menu = this.getStatic("METHODS")[0] || {
156             };
157             return menu;
158         },
159         /**
160          *  Helper function that walks the class hierarchy and returns it's attributes
161          *  cfg (ATTRS), used in Y.Wegas.Entity.getFormCfg().
162          *  @function
163          *  @private
164          */
165         getAttrCfgs: function() {
166             return this._aggregateAttrs(this.getStatic("ATTRS"));
167         },
168         /**
169          *
170          * @function
171          * @private
172          * @param {type} key
173          * @param {type} withExtensions
174          * @returns {Array}
175          */
176         getStatic: function(key, withExtensions) {
177             var c = this.constructor, ret = [], i;
178 
179             while (c) {
180                 if (c[key]) {                                                  // Add to attributes
181                     ret[ret.length] = c[key];
182                 }
183                 if (withExtensions && c._yuibuild && c._yuibuild.exts) {
184                     for (i = 0; i < c._yuibuild.exts.length; i += 1) {
185                         if (c._yuibuild.exts[i][key]) {
186                             ret.push(c._yuibuild.exts[i][key]);
187                         }
188                     }
189                 }
190                 c = c.superclass ? c.superclass.constructor : null;
191             }
192             return ret;
193         },
194         /**
195          * Retrives editable's editorname or name
196          * @function
197          * @public
198          * @returns {String} static EDITORNAME or NAME
199          */
200         getType: function() {
201             return this.constructor.EDITORNAME || this.constructor.NAME;
202         },
203         getLabel: function() {
204             return null;
205         },
206         getEditorLabel: function() {
207             return this.getLabel();
208         },
209         /**
210          * Check if this widget is augmented (extended) by a specifique Widget
211          * @param {Class} extension to check for
212          * @returns {Boolean}
213          */
214         isAugmentedBy: function(extension) {
215             var self = this.constructor, i;
216             while (self._yuibuild) {
217                 for (i = 0; i < self._yuibuild.exts.length; i += 1) {
218                     if (self._yuibuild.exts[i] === extension) {
219                         return true;
220                     }
221                 }
222                 self = self.superclass.constructor;
223             }
224             return false;
225 
226         }
227     });
228     Y.mix(Editable, {
229         /** @lends Y.Wegas.Editable */
230         /**
231          *
232          * @param {type} elts
233          * @param {type} data
234          * @returns {undefined}
235          */
236         mixMenuCfg: function(elts, data) {
237             var i, j;
238             for (i = 0; i < elts.length; i += 1) {
239                 Y.mix(elts[i], data, true);                                     // Attach self and the provided datasource to the menu items, to allow them to know which entity to update
240 
241                 if (elts[i].children) {
242                     Editable.mixMenuCfg(elts[i].children, data);                // push data in children arg
243                 }
244                 if (elts[i].wchildren) {
245                     Editable.mixMenuCfg(elts[i].wchildren, data);               // push data in wchildren
246                 }
247                 if (elts[i].plugins) {
248                     for (j = 0; j < elts[i].plugins.length; j = j + 1) {
249                         elts[i].plugins[j].cfg = elts[i].plugins[j].cfg || {};
250                         Y.mix(elts[i].plugins[j].cfg, data, true);
251                         if (elts[i].plugins[j].cfg.children) {
252                             Editable.mixMenuCfg(elts[i].plugins[j].cfg.children, data); // push data in children arg
253                         }
254                         if (elts[i].plugins[j].cfg.wchildren) {
255                             Editable.mixMenuCfg(elts[i].plugins[j].cfg.wchildren, data);// push data in wchildren
256                         }
257                     }
258                 }
259             }
260         },
261         /**
262          * Load the modules from an Wegas widget definition
263          *
264          * @function
265          * @static
266          * @param {Object} cfg
267          * @param {Function} cb callback to be called when modules are loaded
268          */
269         use: function(cfg, cb) {
270             var modules = Y.Array.reject(Editable.getModulesFromDefinition(cfg), function(i) {
271                 return i === "";
272             });
273             if (modules.length > 0) {
274                 Y.log("Asynchronous loading for modules:" + modules.join(), "info", "Wegas.Editable");
275             }
276             modules.push(cb);
277             Y.use.apply(Y, modules);
278         },
279         /**
280          * Return recursively the inputex modules from their 'type' property using (modulesByType from loader.js)
281          * @function
282          * @static
283          * @private
284          * @param {type} cfg
285          * @returns {Array}
286          */
287         getRawModulesFromDefinition: function(cfg) {
288             var i, props, type = cfg.type || cfg["@class"],
289                     module = YUI_config.groups.wegas.modulesByType[type],
290                     modules = [],
291                     pushFn = function(field) {
292                         if (field) {
293                             modules = modules.concat(Editable.getModulesFromDefinition(field));
294                         }
295                     };
296             if (Y.Lang.isArray(cfg)) {
297                 return Y.Array.flatten(Y.Array.map(cfg, Editable.getModulesFromDefinition));
298             }
299 
300             if (module) {
301                 modules.push(module);
302             }
303 
304             props = ["children", "entities", "items"];                          // Revive array attributes
305             for (i = 0; i < props.length; i += 1) {
306                 if (cfg[props[i]]) {                                            // Get definitions from children (for Y.WidgetParent widgets)
307                     Y.Array.each(cfg[props[i]], pushFn);
308                 }
309             }
310             if (cfg.plugins) {                                                  // Plugins must be revived in the proper way
311                 Y.Array.each(cfg.plugins, function(field) {
312                     field.cfg = field.cfg || {
313                     };
314                     field.cfg.type = field.fn;
315                     modules = modules.concat(Editable.getModulesFromDefinition(field.cfg));
316                     delete field.cfg.type;
317                 });
318             }
319 
320             props = ["left", "right", "center", "top", "bottom"];               // Revive  objects attributes
321             for (i = 0; i < props.length; i = i + 1) {
322                 if (cfg[props[i]]) {
323                     modules = modules.concat(Editable.getModulesFromDefinition(cfg[props[i]]));
324                 }
325             }
326 
327             return modules;
328         },
329         /**
330          * Return unique modules definitions
331          * @function
332          * @static
333          * @private
334          * @param {Object} cfg
335          */
336         getModulesFromDefinition: function(cfg) {
337             var modules = Editable.getRawModulesFromDefinition(cfg);
338             return Y.Object.keys(Y.Array.hash(modules));
339         },
340         /**
341          *  This method takes a js object and recuresively instantiate it based on
342          *  on their @class attribute. Target class are found in namespace
343          *  Y.Wegas.persistence.
344          *
345          *  @static
346          *  @param {Object} data the object to revive
347          *  @return Y.Wegas.Widget the resulting entity
348          */
349         revive: function(data) {
350             var walk = function(value) {
351                 var k, v;
352                 if (Lang.isObject(value)) {
353                     for (k in value) {
354                         if (value.hasOwnProperty(k)) {
355                             value[k] = walk(value[k]);
356                         }
357                     }
358                     return Editable.reviver(value);
359                 } else {
360                     return value;                                               // Return raw original object
361                 }
362             };
363             return walk(data);
364         },
365         /**
366          *  Takes an js object and lookup the corresponding entity in the
367          *  Y.Wegas.persistence package, based on its @class or type attributes.
368          *
369          *  @function
370          *  @static
371          *  @param {Object} o the object to revive
372          *  @return {Y.Wegas.Widget} the resulting entity
373          */
374         reviver: function(o) {
375             if (o["@class"]) {
376                 return new (Y.Wegas.persistence[o["@class"]] || Y.Wegas.persistence.DefaultEntity)(o);
377             } else if (o.type) {
378                 return new (Y.Wegas.persistence[o.type] || Y.Wegas.persistence.DefaultEntity)(o);
379             }
380             return o;
381         },
382         /**
383          *
384          * Combine use and revive function to get a revived entity.
385          *
386          * @function
387          * @static
388          * @param {type} cfg
389          * @param {type} cb
390          */
391         useAndRevive: function(cfg, cb) {
392             Editable.use(cfg, Y.bind(function(cb) {                            // Load target class dependencies
393                 cb(Editable.revive(this));
394             }, cfg, cb));
395         },
396         /**
397          * Serialization purpose.
398          * Any null or empty String attribute won't serialize.
399          * Use this with getters.
400          * @static
401          * @public
402          * @function
403          * @param {Any} value
404          * @returns {Any|undefined}
405          */
406         removeNullValue: function(value) {
407             return (value === null || value === "") ? undefined : value;
408         }
409     });
410     Y.namespace("Wegas").Editable = Editable;
411 
412     Y.Wegas.use = Y.Wegas.Editable.use;                                         // Set up a shortcut
413
