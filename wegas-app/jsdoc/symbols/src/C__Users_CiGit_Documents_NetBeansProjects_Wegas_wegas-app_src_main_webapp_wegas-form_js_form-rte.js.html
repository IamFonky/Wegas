 14     "use strict";
 15 
 16     var CONTENTBOX = "contentBox", BOUNDINGBOX = "boundingBox",
 17             RichTextEditor;
 18 
 19     /**
 20      * 
 21      */
 22     RichTextEditor = Y.Base.create("form-rte", Y.Textarea, [], {
 23         renderUI: function() {
 24             RichTextEditor.superclass.renderUI.call(this);
 25             if (!RichTextEditor.init) {
 26                 RichTextEditor.init = true;
 27                 tinymce.init(this.get("tinymceCfg"));
 28             }
 29             Y.once("domready", function() {
 30                 //tinymce.createEditor(this.el.id, {});
 31                 tinymce.execCommand('mceAddEditor', false, this.get("inputNode").get("id"));
 32             }, this);
 33         },
 34         destructor: function() {
 35             tinymce.remove("#" + this.el.id);
 36         }
 37     }, {
 38         ATTRS: {
 39             inputNode: {
 40                 readonly: true,
 41                 getter: function() {
 42                     return this.get(CONTENTBOX).one("textarea");
 43                 }
 44             },
 45             value: {
 46                 setter: function(value, name, cfg) {
 47                     if (!cfg || !cfg.internal) {
 48                         this.get("inputNode").set("value", value);
 49 
 50                         //if (value) {

 52                         //            new RegExp("data-file=\"([^\"]*)\"", "gi"),

 55                         //}
 56 
 57                         var tmceI = tinyMCE.get(this.get("inputNode").get("id"));
 58                         if (tmceI) {
 59                             tmceI.setContent(value);
 60                         }
 61                     }
 62                     return value;
 63                 },
 64                 /**
 65                  * Get the html string
 66                  * @return {String} the html string
 67                  */
 68                 getter: function() {
 69                     //var path = Y.Plugin.CRDataSource.getFullpath("")
 70                     tinyMCE.triggerSave();
 71                     return this.get("inputNode").get("value");
 72 //                            .replace(new RegExp("((src|href)=\".*/rest/File/GameModelId/.*/read([^\"]*)\")", "gi"), "data-file=\"$3\"")// Replace absolute path with injector style path
 73 //                            .replace(new RegExp("((src|href)=\".*/rest/GameModel/.*/File/read([^\"]*)\")", "gi"), "data-file=\"$3\"");// Replace absolute path with injector style path
 74 
 75                 }
 76             },
 77             tinymceCfg: {

 79                     plugins: [
