230         addStateType: function(type) {
231             var region = this.get(CONTENT_BOX).one('.scrollable').get('region'),

233                     y = parseInt(region.height / 2 + this.scrollView.get('scrollY')),

241             this.save();

243         addState: function(x, y, id, entity) {
244             if (!this.jpLoaded) {
245                 return null;
246             }
247             var state, config = {
248                 sid: id,
249                 initial: (+id === +this.get(ENTITY).getInitialStateId()),
250                 x: x,
251                 y: y
252             };
253             if (entity) {
254                 config.entity = entity;
255             }
256             state = this.add(new Wegas.State(config)).item(0);
257             this.nodes[id.toString()] = state;
258             this.get(ENTITY).get(STATES)[id.toString()] = entity;
259             this.stateId = Math.max(this.stateId, parseInt(id) + 1);
260             return state;
261         },
262         save: function() {
263             this._saveTimer && this._saveTimer.cancel();
264             /**
265              * avoid multiple calls. Save last.
266              */
267             this._saveTimer = Y.later(100, this, function() {
268                 var entity = this.get(ENTITY),
269                         DEFAULTCB = {
270                             success: Y.bind(function(e) {
271                                 this._saveOngoing = false;
272                                 if (this._saveWaiting) {
273                                     this.save();
274                                 }
275 
276                                 this.highlightUnusedStates();
277                                 this.hideOverlay();
278 
279                             }, this),
280                             failure: Y.bind(function(e) {
281                                 this._saveOngoing = false;
282                                 if (this._saveWaiting) {
283                                     this.save();
284                                 }
285                                 this.showMessage("error", e.response.data.message);
286 
287                                 this.highlightUnusedStates();
288                                 this.hideOverlay();
289                             }, this)
290                         };
291                 if (entity) {
292                     //                    this.showOverlay();
293                     //                    if (this._saveOngoing) {
294                     //                        this._saveWaiting = true;
295                     //                        return;
296                     //                    }
297                     this._saveOngoing = true;
298                     this._saveWaiting = false;
299                     entity = Y.JSON.parse(Y.JSON.stringify(entity));
300                     if (entity.id) {
301                         Wegas.Facade.VariableDescriptor.cache.put(entity, {
302                             on: DEFAULTCB
303                         });
304                     } else {
305                         Wegas.Facade.VariableDescriptor.cache.post(entity, {
306                             on: DEFAULTCB
307                         });
308                     }
309                 }
310             });
311         },
312         zoom: function(event) {

320         setZoom: function(lvl, isFromSliderOrInit) {
321             this.currentZoom = lvl < StateMachineViewer.MIN_ZOOM ?
322                     StateMachineViewer.MIN_ZOOM :
323                     (lvl > StateMachineViewer.MAX_ZOOM ?
324                             StateMachineViewer.MAX_ZOOM :
325                             lvl);
326             this.get(CONTENT_BOX).one(".sm-zoom").setStyle('transform', 'scale(' + this.currentZoom + ')');
327             jp.setZoom(this.currentZoom);
328             this.btnZoomValue.set("label", parseInt(this.currentZoom * 100) + "%");
329             if (!isFromSliderOrInit) {
330                 this.sliderZoom.set("value", this.currentZoom * StateMachineViewer.FACTOR_ZOOM);
331             }

334         highlightCurrentState: function() {
335             var currentStateNode, sm = this.get(ENTITY);
336             if (!sm) {
337                 return;
338             }
339             this.get(BOUNDING_BOX).all(".currentState").removeClass("currentState");
340             currentStateNode = this.nodes[Wegas.Facade.VariableDescriptor.cache.findById(sm.get("id")).getInstance().get("currentStateId")];
341             if (currentStateNode) {
342                 currentStateNode.get(BOUNDING_BOX).addClass("currentState");
343             }
344         },
345         highlightUnusedStates: function() {
346             // Prepare vars
347             var currentState, i,
348                     initialNode = this.nodes[this.get(ENTITY).getInitialStateId()],
349                     listStates = Y.merge(this.nodes), // Prepare data
350                     listPath = initialNode ? [initialNode] : [];
351             this.get(BOUNDING_BOX).all(".unusedState").removeClass("unusedState");
352             // Follow the path
353             while (listPath.length > 0) {
354                 currentState = listPath.pop();
355                 // Test if state has been already visited
356                 if (listStates[currentState.get(SID)] !== undefined) {
357                     for (i in currentState.get(ENTITY).get("transitions")) {
358                         listPath.push(this.nodes[currentState.get(ENTITY).get("transitions")[i].get("nextStateId")]);
359                     }
360                 }
361                 delete listStates[currentState.get(SID)];
362             }
363             // Highlight
364             for (i in listStates) {
365                 listStates[i].get(BOUNDING_BOX).addClass("unusedState");
366             }
367         }

391             } else {
392                 return "";
393             }
