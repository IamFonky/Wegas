 13 YUI.add('datatable-csv', function(Y) {
 14     "use strict";
 15     var DatatableCSV, imgTest = new Image();
 16 
 17     /**
 18      * Plugin to export datatable as CSV.
 19      * @constructor
 20      * @extends Y.Plugin.Base
 21      * @name Y.Wegas.DatatableCSV
 22      */
 23     DatatableCSV = Y.Base.create("datatable-csv", Y.Plugin.Base, [], {
 24         /* @lends Y.Wegas.DatatableCSV# */
 25         /**
 26          * lifecycle method
 27          * @function
 28          * @private
 29          * @returns {undefined}
 30          */
 31         initializer: function() {
 32             this.bcsv = new Y.Button({
 33                 label: "CSV",
 34                 on: {
 35                     click: Y.bind(function() {
 36                         DatatableCSV.dataToWindow(this._toCSV());
 37                     }, this)
 38                 }
 39             }).render(this.get("host").get("boundingBox"));
 40             this.bdcsv = new Y.Button({
 41                 label: "Download CSV",
 42                 on: {
 43                     click: Y.bind(function() {
 44                         var csv = this._toCSV(),
 45                                 //node = this.get("host").get("boundingBox").one(".datatable-csv-file"),
 46                                 gm_name = Y.Wegas.Facade.GameModel.cache.getCurrentGameModel().get("name"),
 47                                 game_name = Y.Wegas.Facade.Game.cache.getCurrentGame().get("name"),
 48                                 name = gm_name + "-" + game_name + "-" + Y.Lang.now() + ".csv";
 49 
 50                         DatatableCSV.download("text/csv", name, csv);
 51                         //Disabled due to IE allowing limited data URI
 52                         //if (DatatableCSV.DATA_URI_SUPPORT) {
 53                         //    node.setAttribute("href", "data:text/csv;header=true," + encodeURIComponent(csv));
 54                         //    node.setAttribute("download", name);
 55                         //} else {
 56                         //    //FALLBACK : get a real browser
 57                         //    download("text/csv", name, csv);
 58                         //}
 59                     }, this)
 60                 }
 61             }).render(this.get("host").get("boundingBox"));
 62         },
 63         /**
 64          * Transformation method, datatable to CSV
 65          * @function
 66          * @private
 67          * @returns {undefined}
 68          */
 69         _toCSV: function() {
 70             var headers = Y.Array.map(this.get("host").get("columns"), function(i) {
 71                 return "\"" + (i.label || i.key) + "\"";
 72             }), keys = Y.Array.map(this.get("host").get("columns"), function(i) {
 73                 return i.key;
 74             }),
 75                     data = this.get("host").data.toArray(), records = [headers], i, j, fields;
 76             //Build lignes following key order.
 77             for (i = 0; i < data.length; i += 1) {
 78                 fields = [];
 79                 for (j = 0; j < keys.length; j += 1) {
 80                     fields.push("\"" + (
 81                             (data[i].get(keys[j]) === undefined || data[i].get(keys[j]) === null) ?
 82                             "" : data[i].get(keys[j])
 83                             ) + "\"");
 84                 }
 85                 records.push(fields);
 86 
 87             }
 88             return DatatableCSV.arraysToCSV(records);
 89 
 90         },
 91         /**
 92          * lifecycle method
 93          * @function
 94          * @private
 95          * @returns {undefined}
 96          */
 97         destructor: function() {
 98             if (this.bcsv) {
 99                 this.bcsv.destroy();
100             }
101             if (this.bdcsv) {
102                 this.bdcsv.destroy();
103             }
104         }
105     }, {
106         /* @lends Y.Wegas.DatatableCSV */
107         NS: "datatablecsv",
108         /**
109          * Field delimiter. As of RFC 4180 : "," by default
110          * @field
111          * @static
112          */
113         FIELD_DELIMITER: ",",
114         /**
115          * Record (lines) delimiter. As of RFC 4180 : "\r\n" (CRLF) by default
116          * @static
117          * @field
118          */
119         RECORD_DELIMITER: "\r\n",
120         /**
121          * Writes given data to window
122          * @static
123          * @function
124          * @param {contentType} data to write to window
125          * @param {contentType} winHandle (optional) window handle to write to. A new window is create if this parameter is missing.
126          * @returns {undefined}
127          */
128         dataToWindow: function(data, winHandle) {
129             var win = winHandle || window.open(), range, selection;
130             win.document.write("<pre>");
131             win.document.write("</pre>");
132             Y.one(win.document).one("pre").set("text", data);
133             //Select data
134             if (document.createRange && window.getSelection) {
135                 range = win.document.createRange();
136                 selection = win.getSelection();
137                 selection.removeAllRanges();
138                 range.selectNodeContents(win.document.body.firstChild);
139                 selection.addRange(range);
140             }
141         },
142         /**
143          * Convert array of arrays to CSV
144          * @private
145          * @param {Array} arrays array containing arrays [[], []]
146          * @returns {String} CSV string
147          */
148         arraysToCSV: function(arrays) {
149             return Y.Array.map(arrays, function(i) {
150                 return i.join(DatatableCSV.FIELD_DELIMITER);
151             }).join(DatatableCSV.RECORD_DELIMITER);
152         },
153         /**
154          * Send data to server to download it.
155          * @function
156          * @private
157          * @param {String} contentType file's content-type
158          * @param {String} name file's name
159          * @param {String} data file's content
160          * @returns {undefined}
161          */
162         download: function(contentType, name, data) {
163             var url = Y.Wegas.app.get("base") + "rest/Download/" + name,
164                     form = Y.Node.create('<form enctype="multipart/form-data" method="post" action="' + url + '" ><input type="hidden" name="data"><input type="hidden" name="ctype"></form>');
165             form.one("input[name=data]").getDOMNode().value = data;
166             form.one("input[name=ctype]").getDOMNode().value = contentType;
167             form.submit();
168             form.destroy();
169         },
170         DATA_URI_SUPPORT: undefined
171     });
172     Y.namespace("Wegas").DatatableCSV = DatatableCSV;
173 
174     imgTest.onload = imgTest.onerror = function() {
175         if (imgTest.width === 1 && imgTest.height === 1) {
176             DatatableCSV.DATA_URI_SUPPORT = true;
177         } else {
178             DatatableCSV.DATA_URI_SUPPORT = false;
179         }
180         imgTest = null;
181     };
182     imgTest.src = "data:image/gif;base64,R0lGODlhAQABAAD/ACwAAAAAAQABAAACADs=";
183 
184 });
