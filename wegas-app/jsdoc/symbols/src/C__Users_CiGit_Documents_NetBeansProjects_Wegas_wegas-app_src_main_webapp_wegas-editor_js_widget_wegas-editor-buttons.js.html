 17 YUI.add('wegas-editor-buttons', function(Y) {
 18     "use strict";
 19 
 20     var SelectGameButton, SelectPlayerButton;
 21     /**
 22      * A button that display all current game's player and that sets the
 23      * Wegas.Facade.Game.get('currentPlayerId') on click.
 24      *
 25      * @class Y.Wegas.SelectPlayerButton
 26      * @constructor
 27      * @extends Y.Wegas.Button
 28      * @param {Object} cfg The button config object
 29      */
 30     SelectPlayerButton = Y.Base.create("button", Y.Wegas.Button, [], {
 31         /** @lends Y.Wegas.SelectPlayerButton# */
 32 
 33         /**
 34          *  @function
 35          *  @private
 36          */
 37         bindUI: function() {
 38             SelectPlayerButton.superclass.bindUI.apply(this, arguments);
 39             this.plug(Y.Plugin.WidgetMenu);
 40 
 41             this.menu.on("button:click", function(e) {
 42                 Y.Wegas.Facade.Game.cache.set('currentPlayerId', e.target.get("data").get("id"));
 43             });
 44 
 45             Y.Wegas.Facade.Game.after("response", this.syncUI, this);
 46             Y.Wegas.Facade.Game.cache.after("currentPlayerIdChange", this.syncUI, this);
 47         },
 48         /**
 49          *  @function
 50          *  @private
 51          */
 52         syncUI: function() {
 53             SelectPlayerButton.superclass.bindUI.apply(this, arguments);
 54             var j, k, cTeam, menuItems = [],
 55                     cGame = Y.Wegas.Facade.Game.cache.getCurrentGame(),
 56                     cPlayer = Y.Wegas.Facade.Game.cache.getCurrentPlayer();
 57 
 58             this.set("label", "Current player: " + cPlayer.get("name"));      // Update the label
 59 
 60             for (j = 0; j < cGame.get("teams").length; j = j + 1) {
 61                 cTeam = cGame.get("teams")[j];
 62 
 63                 // if (cTeam.get("players").length == 0) {
 64                 //    continue;
 65                 // }
 66 
 67                 menuItems.push({
 68                     type: "Text",
 69                     content: "<b>" + cTeam.get("name") + "</b>"
 70                 });
 71 
 72                 for (k = 0; k < cTeam.get("players").length; k = k + 1) {
 73                     cPlayer = cTeam.get("players")[k];
 74                     menuItems.push({
 75                         type: "Button",
 76                         label: cPlayer.get("name"),
 77                         data: cPlayer
 78                     });
 79                 }
 80             }
 81 
 82             this.menu.set("children", menuItems);
 83         }
 84     }, {
 85         /** @lends Y.Wegas.SelectPlayerButton */
 86         ATTS: {
 87             entity: {}
 88         }
 89     });
 90     Y.namespace("Wegas").SelectPlayerButton = SelectPlayerButton;
 91 
 92     /**
 93      * @deprecated It should load game list on demand.
 94      *  
 95      * @class Y.Wegas.SelectGameButton
 96      * @constructor
 97      * @extends Y.Wegas.Button
 98      * @param {Object} cfg The button config object
 99      */
100     SelectGameButton = Y.Base.create("button", Y.Wegas.Button, [], {
101         /** @lends Y.Wegas.SelectGameButton# */
102         bindUI: function() {
103             SelectGameButton.superclass.bindUI.apply(this, arguments);
104             this.plug(Y.Plugin.WidgetMenu);
105 
106             Y.Wegas.Facade.GameIndex.after("response", this.syncUI, this);
107         },
108         syncUI: function() {
109             SelectGameButton.superclass.syncUI.apply(this, arguments);
110 
111             var j, menuItems = [],
112                     cGame = Y.Wegas.Facade.Game.cache.getCurrentGame(),
113                     games = Y.Wegas.Facade.GameIndex.cache.findAll();
114 
115             this.set("label", "Current game: " + cGame.get("name"));      // Update the label
116 
117             for (j = 0; j < games.length; j = j + 1) {
118                 menuItems.push({
119                     type: "Button",
120                     label: games[j].get("name"),
121                     plugins: [{
122                             fn: "OpenGameAction",
123                             cfg: {
124                                 target: "self",
125                                 entity: games[j]
126                             }
127                         }]
128                 });
129             }
130             this.menu.set("children", menuItems);
131         }
132     });
133     Y.namespace("Wegas").SelectGameButton = SelectGameButton;
134 
135 });
