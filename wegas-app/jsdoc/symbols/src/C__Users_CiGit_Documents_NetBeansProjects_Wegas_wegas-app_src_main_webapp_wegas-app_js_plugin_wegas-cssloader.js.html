 58             Y.io(url, {// Load the page css
 59                 context: this,
 60                 on: {
 61                     success: Y.bind(function(sheetId, tId, e) {
 62                         CSSLoader.updateStyleSheet(sheetId, e.responseText);
 63                     }, this, id),
 64                     failure: function(id, o) {

 74         updateStyleSheet: function(id, content) {
 75             if (CSSLoader.sheets[id]) {
 76                 CSSLoader.sheets[id].disable();
 77             }
 78             //CSSLoader.sheets[id] = new Y.StyleSheet(content);
 79             CSSLoader.sheets[id] = new Y.StyleSheet(content.replace(/\.\.\//g, ""));
 80         }
