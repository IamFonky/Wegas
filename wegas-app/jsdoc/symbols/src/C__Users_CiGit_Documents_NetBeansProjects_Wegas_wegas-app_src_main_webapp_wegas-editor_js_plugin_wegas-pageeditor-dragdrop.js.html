  9 /**
 10  * @fileOverview PageEditor drag and drop Extension
 11  * @author Cyril Junod <cyril.junod at gmail.com>
 12  */
 13 YUI.add("wegas-pageeditor-dragdrop", function(Y) {
 14     "use strict";
 15     var CONTENTBOX = "contentBox",
 16             BOUNDINGBOX = "boundingBox";
 17     /**
 18      * PageEditor Plugin Extension enabling position drag and drop.
 19      * @constructor
 20      */
 21     function PageEditorDD() {
 22         Y.onceAfter(this._initDD, this, "render");
 23         Y.after(this._alignDD, this, "showOverlay");
 24         Y.onceAfter(this._destruct, this, "destructor");
 25     }
 26     PageEditorDD.NAME = "wegas-pageeditor-dragdrop";
 27     PageEditorDD.prototype = {
 28         /* @lends Y.Wegas.PageEditorDD# */
 29         /**
 30          * @private
 31          * @returns {undefined}
 32          */
 33         _initDD: function() {
 34             if (!this.highlightOverlay) {
 35                 Y.error("PageEditorDD is an extension for PageEditor.");
 36             }
 37             this._ddNode = Y.Node.create("<div class='pageeditor-dragnode'></div>");
 38             this._ddNode.setStyles({
 39                 top: 0,
 40                 left: 0
 41             });
 42             this.highlightOverlay.get(CONTENTBOX).append(this._ddNode);
 43             this._ddNode.hide();
 44             this.highlightOverlay.after("visibleChange", function(e) {
 45                 if (e.newVal && this.overlayWidget && this.overlayWidget.CSSPosition) {
 46                     this._ddNode.show();
 47                 } else {
 48                     this._ddNode.hide();
 49                 }
 50             }, this);
 51             this._dd = new Y.DD.Drag({
 52                 node: this._ddNode
 53             });
 54             this._dd.plug(Y.Plugin.DDConstrained, {
 55                 constrain: this.get("host").get(CONTENTBOX),
 56                 cacheRegion: false                                              //scroll changes region
 57             }).plug(Y.Plugin.DDNodeScroll, {
 58                 node: this.get("host").get(CONTENTBOX)
 59             });
 60             this._ddNode.before("mousedown", function(e) {
 61                 this.detach();
 62                 this._dd.set("dragNode", this.overlayWidget.get(BOUNDINGBOX));
 63                 this._dd.con.set("constrain", this.get("host").get("widget").get(CONTENTBOX));
 64                 this._ddNode.show();
 65             }, this);
 66             this._ddNode.after("mouseup", function(e) {
 67                 this.bind();
 68             }, this);
 69             this._dd.before("drag:start", function(e) {
 70                 var node = this._dd.get("dragNode");
 71                 node.setXY(node.getXY()); //Init left, top in case they are missing
 72                 this._dd._initPos = {
 73                     left: parseInt(node.getComputedStyle("left"), 10),
 74                     right: parseInt(node.getComputedStyle("right"), 10),
 75                     top: parseInt(node.getComputedStyle("top"), 10),
 76                     bottom: parseInt(node.getComputedStyle("bottom"), 10)
 77                 };
 78                 this._dd.get("dragNode").setStyles({
 79                     bottom: null,
 80                     right: null,
 81                     width: node.getComputedStyle("width"),
 82                     height: node.getComputedStyle("height")
 83                 });
 84                 this.fixedOverlay(Y.Widget.getByNode(node));
 85             }, this);
 86             this._dd.after("drag:drag", function(){
 87                 this.fixedOverlay(this.shownOverlay._widget);
 88             }, this);
 89             this._dd.on("drag:end", function(e) {
 90                 var bb = this._dd.get("dragNode"),
 91                         widget = Y.Widget.getByNode(bb),
 92                         oldStyles = widget.CSSPosition.get("styles"),
 93                         newStyles = {},
 94                         style;
 95                 for (style in oldStyles) {
 96                     if (oldStyles.hasOwnProperty(style)) {
 97                         if (oldStyles[style] !== "") {
 98                             switch (style) {
 99                                 case "right":
100                                     newStyles[style] = this._dd._initPos[style] + this._dd._initPos.left - parseInt(bb.getStyle("left"), 10) + "px";
101                                     break;
102                                 case "bottom":
103                                     newStyles[style] = this._dd._initPos[style] + this._dd._initPos.top - parseInt(bb.getStyle("top"), 10) + "px";
104                                     break;
105                                 default:
106                                     newStyles[style] = bb.getComputedStyle(style);
107                             }
108                         } else {
109                             /* 
110                              * if no properties are defined horizontally or vertically 
111                              * specify respectively "left" or "top 
112                              */
113                             switch (style) {
114                                 case "left":
115                                     if (oldStyles.right === "") {
116                                         newStyles[style] = bb.getComputedStyle(style);
117                                     }
118                                     break;
119                                 case "top":
120                                     if (oldStyles.bottom === "") {
121                                         newStyles[style] = bb.getComputedStyle(style);
122                                     }
123                                     break;
124                                 default:
125                                     newStyles[style] = "";
126                             }
127 
128                         }
129                     }
130                 }
131                 widget.CSSPosition.set("styles", newStyles);
132                 /* 
133                  * Remove Size and set them through CSSSize if it exists
134                  */
135                 bb.setStyles({
136                     width: null,
137                     height: null
138                 });
139                 if (widget.CSSSize) {
140                     widget.CSSSize.set("styles", widget.CSSSize.get("styles"));
141                 }
142                 this.fixedOverlay(widget);
143                 this.saveCurrentPage();
144                 this.bind();
145                 this._syncWidgetEdition();
146             }, this);
147         },
148         /**
149          * Align drag node with PageEditor's overlay
150          * @private
151          * @returns {undefined}
152          */
153         _alignDD: function() {
154             var bb = this.overlayWidget.get(BOUNDINGBOX);
155             if (bb.getDOMNode()) {
156                 this._ddNode.setXY(bb.getXY());
157                 this._ddNode.setStyle("width", bb.getDOMNode().offsetWidth);
158                 this._ddNode.setStyle("height", bb.getDOMNode().offsetHeight);
159             }
160 
161         },
162         /**
163          * self destructor called after PageEditor's destructor
164          * @private
165          * @returns {undefined}
166          */
167         _destruct: function() {
168             Y.detach(this._alignDD, this, "showOverlay");
169             this._dd.detachAll();
170             this._ddNode.detachAll();
171             this._dd.destroy();
172             this._ddNode.destroy(true);
173         }
174 
175     };
176     Y.namespace("Wegas").PageEditorDD = PageEditorDD;
177 });
