 12 YUI.add("treeview-sortable", function(Y) {
 13     "use strict";
 14 
 15     var HOST = "host",
 16             NODE = 'node',
 17             CONTENTBOX = 'contentBox',
 18             TreeViewSortable;
 19 
 20     TreeViewSortable = Y.Base.create("treeview-sortable", Y.Plugin.Base, [], {
 21         initializer: function() {
 22             if (!(this.get(HOST) instanceof Y.TreeView)) {
 23                 Y.log("TreeView filter host must be a TreeView", "warn", "TreeViewFilter");
 24                 return;
 25             }
 26 
 27             this.afterHostEvent("render", function() {
 28                 var host = this.get(HOST),
 29                         cb = host.get(CONTENTBOX);
 30 
 31                 cb.setStyles({
 32                     overflowY: "auto",
 33                     overflowX: "hidden"
 34                 });
 35 
 36                 this.sortable = new NestedSortable({
 37                     container: cb,
 38                     //nodes: 'li',
 39                     nodes: 'li.treeview-draggable',
 40                     opacity: '.2',
 41                     invalid: ".wegas-editor-dummy",
 42                     moveType: "insert"
 43                             // handles: ['.yui3-treenode-content-icon', '.yui3-treeleaf-content-icon']
 44                             // opacityNode: "dragNode",
 45 
 46                 });
 47                 this.sortable.treeSortPlg = this;
 48                 //this.sortable.delegate.dd.plug(Y.Plugin.DDNodeScroll, {
 49                 //    node: cb,
 50                 //    horizontal: false
 51                 //});
 52                 var addedNode, newNode;
 53 
 54                 this.sortable.delegate.dd.on("drag:start", function(ev) {
 55                     newNode = Y.Node.create("<ul></ul>");
 56                     newNode.setAttribute("id", Y.guid());
 57                     this.sync();
 58                 }, this);
 59 
 60                 this.sortable.delegate.dd.on("drag:over", function(ev) {
 61                     if (addedNode !== undefined) {                              // remove it from where it was
 62                         addedNode.remove();
 63                     }
 64                     var dragNode = ev.drag.get("node"),
 65                             dropNode = ev.drop.get("node"),
 66                             found = this.testGroups(dragNode, dropNode),
 67                             tOl = dropNode.one("ul");                           // tOl is looking for a child ol below the li
 68 
 69                     //console.log("drag:over(", dropNode.get("nodeName").toLowerCase(), tOl ? "has tol" : "no tol",
 70                     //        found, dropNode._node.className,
 71                     //        dropNode.one(".yui3-treenode-content-label") ? dropNode.one(".yui3-treenode-content-label").getHTML() : "nolabel");
 72 
 73                     switch (dropNode.get("nodeName").toLowerCase()) {           // if we've over an li, add the new ol child block
 74                         case "li":
 75                             if (tOl) {                                          // try and append it to existing ol on the target
 76                                 try {
 77                                     if (found) {
 78                                         tOl.append(ev.drag.get("node"));
 79                                     }
 80                                 } catch (e) {
 81                                 }
 82                             } else {                                            // else add a new ol to the target
 83                                 try {
 84                                     return;
 85                                     if (found) {
 86                                         dropNode.append(newNode);               // try adding newNode
 87                                         newNode.append(ev.drag.get("node"));
 88                                         addedNode = newNode;
 89                                     }
 90                                 } catch (e) {
 91                                 }
 92                             }
 93                             break;
 94 
 95                         case "ul":                                              // if we're over an ol, just add this as a new li child
 96                             try {
 97                                 if (found) {
 98                                     dropNode.append(ev.drag.get("node"));
 99                                     return;
100 
101                                 }
102                             } catch (e) {
103                             }
104                             break;
105 
106                         default:
107                             break;
108                     }
109                 }, this);
110 
111                 this.sortable.delegate.dd.after('drag:end', function(ev) {
112                     var node = this.sortable.delegate.get('currentNode'),
113                             //  prev = node.previous(), next = node.next(),
114                             dragWidget = Y.Widget.getByNode(node),
115                             dropNode = node.get("parentNode"),
116                             dropWidget = Y.Widget.getByNode(dropNode),
117                             index = dropNode.get("children").indexOf(node),
118                             targetNode = ev.target.get("node");
119 
120                     //Y.log("onDragEnd()", "info", "Wegas.VariableTreeView");
121 
122                     addedNode = undefined;
123                     newNode = undefined;
124                     targetNode.removeAttribute("style");// DD somewhere sets some element styles, which mess up alignment somewhere in IE
125 
126                     dropWidget.add(dragWidget, index);                          // Update treeview
127                     this.fire("sort", {
128                         dragWidget: dragWidget,
129                         dropWidget: dropWidget,
130                         index: index
131                     });                                                         // Fire sorted event
132                     this.sync();                                                // Sync dummies
133                 }, this);
134                 //this.sortable.delegate.dd.after('drag:over', this.sync, this);
135             });
136             this.afterHostMethod("syncUI", this.sync);
137             this.afterHostEvent(["*:collapsedChange"], this.sync);
138         },
139         sync: function() {
140             var cb = this.get(HOST).get(CONTENTBOX), i,
141                     nodeGroups = this.get("nodeGroups");
142             // cb.all(".wegas-editor-dummy").remove(true);
143 
144             cb.all(".wegas-editor-dummy").each(function(n) {                    // Remove useless dummies
145                 if (n.ancestor("ul").get("children").size() > 1) {
146                     n.remove(true);
147                 }
148             });
149 
150             Y.Array.each(nodeGroups, function(item) {
151                 cb.all("." + item.nodeClass).addClass("treeview-draggable");          // Add class to all draggable nodes
152 
153                 cb.all(item.parentNode + " ul:empty")                                      // Add dummies to allow drag on empty nodes
154                         .append("<li class=\"yui3-widget yui3-treenode wegas-editor-dummy " + item.nodeClass + " \"><div class=\"content-header yui3-treenode-content-header\"><span class=\"yui3-treenode-content-label\" ><i>empty</i></span></div></li>");
155                 //.append('<li class="yui3-widget yui3-treenode wegas-editor-dummy wegas-editor-listitem yui3-dd-drop " tabindex="1"><div class="content-header yui3-treenode-content-header"><span class="yui3-treenode-content-label" ><i>empty</i></span></div></li>');
156 
157             });
158 
159 
160             for (i = 0; i < nodeGroups.length; i += 1) {
161             }
162             this.sortable.sync();
163         },
164         testGroups: function(dragNode, dropNode) {
165             var i, groups = this.get("nodeGroups");
166             if (groups) {
167                 return !!Y.Array.find(groups, function(item) {                // Added custom class mathing for node groups
168                     return dragNode.hasClass(item.nodeClass) && dropNode.hasClass(item.nodeClass);
169                 });
170             }
171             return false;
172         },
173         testDrag: function(dragNode) {
174             var groups = this.get("nodeGroups");
175             if (groups) {
176                 return !!Y.Array.find(groups, function(item) {
177                     return dragNode.hasClass(item.nodeClass);
178                 });
179             }
180             return false;
181         },
182         destructor: function() {
183             this.sortable.destroy();
184         }
185 
186     }, {
187         NAME: "TreeViewSortable",
188         NS: "sortable",
189         ATTRS: {
190             /*
191              * Defines groups of items to be moved and items on which they can be dropped, even when they are empty.
192              */
193             nodeGroups: {
194                 value: [{
195                         nodeClass: "yui3-widget",
196                         parentNode: ".yui3-widgget"
197                     }]
198             }
199         }
200     });
201     Y.namespace("Plugin").TreeViewSortable = TreeViewSortable;
202 
203     /**
204      * Extend so in works with nested lists
205      *
206      * @returns {undefined}
207      */
208     function NestedSortable() {
209         NestedSortable.superclass.constructor.apply(this, arguments);
210     }
211     Y.extend(NestedSortable, Y.Sortable, {
212         //_onDragStart: function(e) {
213         //    if (this.treeSortPlg.testDrag(e.drag.get(NODE))) {
214         //        NestedSortable.superclass._onDragStart.apply(this, e);
215         //    }
216         //},
217         _onDropEnter: function(e) {
218             // console.log("_onDropEnter(" + dropNode._node.className + ", " + dropNode.one(".yui3-treenode-content-label").getHTML() + ")");
219 
220             if (this.treeSortPlg.testGroups(e.drag.get(NODE), e.drop.get(NODE))) {
221                 NestedSortable.superclass._onDropEnter.apply(this, e);
222             }
223         },
224         _onDragOver: function(e) {
225             var dragNode = e.drag.get(NODE),
226                     dropNode = e.drop.get(NODE);
227 
228             //console.log("_onDragOver(", dropNode._node.className,
229             //        (dropNode.one(".yui3-treenode-content-label")) ?
230             //        dropNode.one(".yui3-treenode-content-label").getHTML() : "no label");
231 
232             if (!this.treeSortPlg.testGroups(dragNode, dropNode)) {
233                 return;
234             }
235 
236             // is drop a child of drag?  - this is the bit that's added:
237             if (e.drag.get(NODE).contains(e.drop.get(NODE))) {
238                 return;
239             }
240             NestedSortable.superclass._onDragOver.call(this, e);
241         }
242     });
243 });
