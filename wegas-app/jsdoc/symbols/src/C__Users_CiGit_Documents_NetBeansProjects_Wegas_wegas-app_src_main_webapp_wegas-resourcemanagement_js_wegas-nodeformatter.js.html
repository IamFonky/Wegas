 13 YUI.add('wegas-nodeformatter', function(Y) {
 14     "use strict";
 15 
 16     /**
 17      * @name Y.Wegas.NodeFormatter
 18      * @extends Y.Widget
 19      * @borrows Y.Wegas.Widget
 20      * @class return value in specifique node.
 21      * @constructor
 22      * @description returm value in specifique node. Available type:
 23      * Text node, image node, value box node, and position node.
 24      */
 25     var NodeFormatter = Y.Base.create("wegas-nodeformatter", Y.Base, [], {
 26         /** @lends Y.Wegas.NodeFormatter# */
 27 
 28         // ** Lifecycle Methods ** //
 29         /**
 30          * @function
 31          * @private
 32          * @param value
 33          * @param label
 34          * @param className
 35          * @return node
 36          * @description return a div node containing a node value and a node
 37          *  label corresponding with the given parameters.
 38          *  Returned node have the given class name.
 39          */
 40         makeNodeText: function(value, label, className) {
 41             var node = Y.Node.create('<div class="nodeformatter-properties"></div>');
 42             value = (value !== null) ? value : 'undefine';
 43             if (className) {
 44                 node.addClass(className);
 45             }
 46             if (label) {
 47                 node.append('<span class="label">' + label + '</span>');
 48             }
 49             node.append('<span class="value">' + value + '</span>');
 50             return node;
 51         },
 52         /**
 53          * @function
 54          * @private
 55          * @param attrs
 56          * @param className
 57          * @return node
 58          * @description return a div node containing an image node. This image
 59          * will have the given attrs parameters as attribute.
 60          * Returned node have the given class name.
 61          */
 62         makeNodeImage: function(attrs, className) {
 63             var k, node = new Y.Node.create('<div class="nodeformatter-img"></div>');
 64             node.append('<img></img>');
 65             if (className) {
 66                 node.one('img').addClass(className);
 67             }
 68             if (typeof attrs !== 'object') {
 69                 return node;
 70             }
 71             for (k in attrs) {
 72                 node.one('img').setAttribute(k, attrs[k]);
 73             }
 74             return node;
 75         },
 76         /**
 77          * @function
 78          * @private
 79          * @param value
 80          * @param maxVal
 81          * @param label
 82          * @param className
 83          * @return node
 84          * @description return a div node containing a node label, a node
 85          *  containing as many nodes as shown in value and a node shown the
 86          *  value and the max value.
 87          *  Returned node have the given class name.
 88          */
 89         makeNodeValueBox: function(value, maxVal, label, className) {
 90             var i, acc = [], node = new Y.Node.create('<div class="nodeformatter-valuebox"></div>');
 91             value = (typeof parseInt(value) === 'number') ? parseInt(value) : 0;
 92             maxVal = (maxVal || 'undefine');
 93             label = (label || 'undefine');
 94             if (className) {
 95                 node.addClass(className);
 96             }
 97             for (i = 0; i < value; i += 1) {
 98                 acc.push('<div class="box-unit"></div>');
 99             }
100             node.append('<div class="label">' + label + '</div>');
101             node.append('<span class="box-units">' + acc.join('') + '</span>');
102             node.append('<span class="box-value">(' + value + '<span class="box-valueMax">/' + maxVal + '</span>)</span>');
103             return node;
104         },
105         /**
106          * @function
107          * @private
108          * @param html
109          * @param selector
110          * @param value
111          * @param minVal
112          * @param invert
113          * @param className
114          * @return node
115          * @description return a div node containing the given html and
116          *  highlight a nodes according to the given values (a number).
117          *  To highlight a node this function count same type node as the
118          *   "selector" parameter (like 'li', 'div', 'p', 'img', etc...).
119          * The first value depending of the given 'minVal'
120          * The 'invert' parameter allow to count reversely.
121          * Returned node have the given class name.
122          */
123         makeNodePosition: function(html, selector, value, minVal, invert, className) {
124             var node = new Y.Node.create('<div class="nodeformatter-position"></div>');
125             minVal = (typeof parseInt(minVal) === 'number') ? parseInt(minVal) : 0;
126             value = (typeof parseInt(value) === 'number') ? parseInt(value) - minVal : -1 - minVal; // if value isn't a number, value egal minVal - 1 (and thus never selected)
127             invert = (invert === 'true' || invert === true) ? true : false;
128             if (className) {
129                 node.addClass(className);
130             }
131             node.append(html);
132             node.all(selector).each(function(n, i, q) { //n = node, i = iteration, q = number of iterations.
133                 i = (invert) ? (q.size() - 1 - i) : i;
134                 n.setAttribute('data-position', i + minVal); //keep a reference to real number position
135                 if (i < value) {
136                     n.addClass('previous');
137                 } else if (i === value) {
138                     n.addClass('current');
139                 } else {
140                     n.addClass('next');
141                 }
142             });
143             return node;
144         },
145         makeNodeLongText: function(cb, variable, value, className) {
146             Y.Wegas.Facade.VariableDescriptor.cache.getWithView(variable, "Extended", {// Retrieve the object from the server in Export view
147                 on: Y.Wegas.superbind({
148                     success: function(e) {
149                         cb.one("." + className).setContent(e.response.entity.get(value) || "No " + value);
150                     },
151                     failure: function(e) {
152                         cb.one("." + className).setContent("No " + value);
153                     }
154                 }, this)
155             });
156         }
157 
158     }, {
159         /*
160          * @lends Y.Wegas.NodeFormatter#
161          */
162         /**
163          * @field
164          * @static
165          * @description
166          * <p><strong>Method (none)</strong></p>
167          */
168         ATTRS: {}
169     });
170 
171     Y.namespace('Wegas').NodeFormatter = NodeFormatter;
172 });
