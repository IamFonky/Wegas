 31         /**
 32          * @function
 33          * @private
 34          * @description All button and fields are created.
 35          * For creating the field inputEx libary is used

 37         renderUI: function() {
 38             var cfg = this.get("dataTableCfg");
 39 
 40             cfg.columns = Y.Array.map(cfg.columns, function(c) {                // Add some default properties to columns
 41                 return Y.mix(c, {
 42                     allowHTML: true,
 43                     sortable: true
 44                 });
 45             });
 46             cfg = Y.mix(cfg, {//                                                // Add cfg default values
 47                 //data: data,
 48                 width: "100%"
 49             });
 50             this.dataTable = new Y.DataTable(cfg);                              // Render datatable
 51             this.dataTable.render(this.get(CONTENTBOX));
 52             this.dataTable.set('strings.emptyMessage', "<em><center><br /><br />" + this.get("emptyMessage") + "<br /><br /><br /></center></em>");
 53 
 54             this.get(CONTENTBOX).addClass("yui3-skin-wegas");
 55         },
 56         /**
 57          * @function
 58          * @private
 59          */
 60         bindUI: function() {
 61             var ds = this.get(DATASOURCE),
 62                     request = this.get("request");
 63             if (ds) {
 64                 this.updateHandler = ds.after("update", this.syncUI, this);                          // Listen updates on the target datasource
 65                 this.failureHandler = ds.after("failure", this.defaultFailureHandler, this);          // GLOBAL error message
 66 
 67                 if (request) {
 68                     ds.sendRequest(request);
 69                 }
 70             }
 71         },
 72         /**
 73          * @function
 74          * @private
 75          */
 76         syncUI: function() {
 77             Y.log("sync()", "info", "Wegas.LobbyDataTable");
 78 
 79             if (!this.get(DATASOURCE)) {
 80                 this.get(CONTENTBOX).append("Unable to find datasource");
 81                 return;
 82             }
 83 
 84             var ds = this.get(DATASOURCE),
 85                     data = this.genData(ds.cache.findAll());
 86 
 87             this.dataTable.set("data", data);
 88             this.hideOverlay();
 89         },
 90         destructor: function() {
 91             this.dataTable.destroy();
 92             this.updateHandler.detach();
 93             this.failureHandler.detach();
 94         },
 95         // *** Private Methods *** //
 96         /**
 97          * @function
 98          * @private
 99          */
100         genData: function(elements) {
101             var ret = [], i, entity, data;
102 
103             for (i in elements) {
104                 if (elements.hasOwnProperty(i)) {
105                     entity = elements[i];
106                     data = this.genEntityData(entity);
107                     if (data) {
108                         Y.mix(data, {//                                         // Add some default properties
109                             entity: entity,
110                             iconCSS: 'wegas-icon-' + entity.get("@class").toLowerCase()
111                         });
112                         ret.push(data);
113                     }
114                 }
115             }
116             return ret;
117         },
118         genEntityData: function(entity) {
119             switch (entity.get("@class")) {
120 
121                 case 'DebugGame':
122                     break;
123 
124                 case 'Game':
125                     return {
126                         name: entity.get(NAME),
127                         createdTime: entity.get("createdTime"),
128                         gameModelName: entity.get("gameModelName") || "",
129                         createdBy: entity.get("createdByName") || "undefined",
130                         playersCount: entity.get("playersCount")
131                                 //teamsCount: gameModel && gameModel.get("properties.freeForAll") ? -1 : entity.get("teams").length,
132                                 //token: entity.get("properties.freeForAll") ? entity.get("token") : "",
133                     };
134                     break;
135 
136                 case 'Team':
137                     return {
138                         name: entity.get(NAME),
139                         token: entity.get("token")
140                     };
141 
142                 case 'Player':
143                     return {
144                         name: entity.get(NAME)
145                     };
146 
147                 case 'GameModel':
148                     if (entity.get("canEdit")) {
149                         return {
150                             name: entity.get(NAME),
151                             createdBy: entity.get("createdByName"),
152                             createdTime: entity.get("createdTime")
153                         };
154                     }
155                     break;
156 
157                 case 'User':
158                     return {
159                         label: entity.get(NAME),
160                         entity: entity.getMainAccount()
161                     };
162 
163                 case 'Role':
164                     return {
165                         label: entity.get(NAME)
166                     };
167 
168                 default:
169                     Y.log("Unable to generate data for entity " + entity.get("@class") + '(' + entity.get(NAME) + ")");
170                     return null;
171             }
172         }
173     }, {

398     Plugin.EditorDTLink = Y.Base.create("admin-menu", Plugin.Base, [], {
399         initializer: function() {
400             this.afterHostEvent(RENDER, function() {
401                 var host = this.get(HOST);
402                 host.dataTable.delegate('click', function(e) {
403                     var rec = host.dataTable.getRecord(e.currentTarget),
404                             entity = rec.get("entity"),
405                             url = this.get("url");
406 
407                     if (entity instanceof Wegas.persistence.GameModel) {
408                         url += "gameModelId=" + entity.get("id");
409                     } else if (entity instanceof Wegas.persistence.Player) {
410                         url += "id=" + entity.get("id");
411                     } else if (entity instanceof Wegas.persistence.Team) {
412                         url += "teamId=" + entity.get("id");
413                     } else {
414                         url += "gameId=" + entity.get("id");
415                     }
416                     window.open(url);
417                     e.halt(true);
418                 }, '.yui3-datatable-data tr[data-yui3-record] a', this);
419             });
420         }
421     }, {
422         NS: "EditorDTLink",
423         NAME: "EditorDTLink",
424         ATTRS: {
425             url: {
426                 value: "play.html?"
427             }
428         }

431     /**
432      * @class Open a menu on right click, containing the admin edition field
433      * @constructor
434      */
435     Plugin.EditorDTContextMenu = Y.Base.create("admin-menu", Plugin.Base, [], {
436         initializer: function() {
437             this.onHostEvent("contextmenu", this.onTreeViewClick, this);
438 
439             this.menu = new Wegas.Menu();
440             this.menu.addTarget(this.get(HOST));
441             this.menu.render();
442         },
443         onTreeViewClick: function(e) {
444             var host = this.get(HOST),
445                     tr = e.domEvent.target.ancestor("tr"), // the Node for the TR clicked ...
446                     rec = host.dataTable.getRecord(tr), // the current Record for the clicked TR
447                     menuItems = this.get("children"),
448                     entity = rec.get("entity"),
449                     data = {
450                 entity: entity,
451                 dataSource: host.get(DATASOURCE)
452             },
453             menuItems = entity.getMenuCfg(data).slice(0);                       // Fetch menu items
454 
455             Y.Array.each(menuItems, function(i, itemIndex) {                    // @HACK Fix the submenu positioning
456                 Y.Array.each(i.plugins, function(p, index) {
457                     if (p.fn === "WidgetMenu") {
458                         menuItems[itemIndex] = Y.mix({}, menuItems[itemIndex]);
459                         menuItems[itemIndex].plugins = menuItems[itemIndex].plugins.slice(0);
460                         menuItems[itemIndex].plugins[index] = {
461                             fn: "WidgetMenu",
462                             cfg: Y.mix({
463                                 menuCfg: {
464                                     points: ["tl", "tr"]
465                                 },
466                                 event: "mouseenter"
467                             }, p.cfg)
468                         };
469                     }
470                 });
471             });
472 
473             if (menuItems) {
474                 e.domEvent.preventDefault();
475                 this.menu.destroyAll();
476                 this.menu.add(menuItems);                                       // Populate the menu with the elements associated to the
477                 this.menu.show();                                               // Move the right click position
478                 this.menu.set("xy", [e.domEvent.pageX, e.domEvent.pageY]);
479             } else {
480                 Y.log("Menu item has no target entity", "info", "Y.Plugin.EditorDTContextMenu");
481             }
482         }
483     }, {
484         NS: "contextmenu",
485         ATTRS: {
486             children: {}
487         }
488     });
