 13 YUI.add("wegas-panel-fileselect", function(Y) {
 14     "use strict";
 15     Y.namespace('Wegas').FileSelect = Y.Base.create("wegas-panel-fileselect", Y.Base, [], {
 16         initializer: function() {
 17             this.publish("fileSelected");
 18             this.publish("directorySelected");
 19             this.panel = new Y.Panel({
 20                 headerContent: this.get("title"),
 21                 bodyContent: '',
 22                 width: 600,
 23                 height: Y.DOM.winHeight() - 150,
 24                 zIndex: 99999,
 25                 modal: true,
 26                 render: true,
 27                 centered: true
 28             });
 29             this.explorer = new Y.Wegas.FileExplorer().render(this.panel.getStdModNode(Y.WidgetStdMod.BODY));
 30             this.explorer.treeView.filter.set("testFn", this.get("filter"));
 31             this.panel.get("boundingBox").addClass("wegas-panel-fileselect");
 32             this.bind();
 33         },
 34         bind: function() {
 35             this.explorer.on("*:fileSelected", function(e, path) {
 36                 e.halt(true);
 37                 this.fire("fileSelected", path);
 38             }, this);
 39             this.explorer.on("*:directorySelected", function(e, path) {
 40                 e.halt(true);
 41                 this.fire("directorySelected", path);
 42             }, this);
 43             this.panel.after("visibleChange", function() {
 44                 this.destroy();
 45             }, this);
 46         },
 47         destructor: function() {
 48             this.explorer.destroy();
 49             this.panel.destroy();
 50         }
 51     }, {
 52         ATTRS: {
 53             title: {
 54                 value: 'Choose a file from library'
 55             },
 56             filter: {
 57                 validator: Y.Lang.isFunction
 58             }
 59         }
 60     });
 61 });
