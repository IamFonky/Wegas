 12 YUI.add("wegas-inputex-rte", function(Y) {
 13     "use strict";
 14 
 15     var inputEx = Y.inputEx, RTEField;
 16 
 17     /**
 18      * @class Wrapper for the Rich Text Editor from YUI
 19      * @name Y.inputEx.Wegas.RTEField
 20      * @extends Y.inputEx.Textarea
 21      * @constructor
 22      * @param {Object} options
 23      */
 24     RTEField = function(options) {
 25         RTEField.superclass.constructor.call(this, options);
 26     };
 27 
 28     Y.extend(RTEField, inputEx.Textarea, {
 29         destroy: function() {
 30             if (this.editor) {
 31                 this.editor.destroy();
 32             } else {
 33                 Y.once("domready", function () {
 34                     this.editor.destroy();
 35                 }, this);
 36             }
 37             RTEField.superclass.destroy.call(this);
 38         },
 39         /**
 40          * Set the default values of the options
 41          * @param {Object} options Options object as passed to the constructor
 42          */
 43         setOptions: function(options) {
 44             RTEField.superclass.setOptions.call(this, options);
 45 
 46             this.options.opts = options.opts || {};
 47             this.options.typeInvite = null;
 48         },
 49         /**
 50          * Render the field using the YUI Editor widget
 51          */
 52         renderComponent: function() {
 53             RTEField.superclass.renderComponent.call(this);
 54             Y.once("domready", function() {
 55                 this.editor = new tinymce.Editor(this.el.id, {
 56                     plugins: [
 57                         "autolink autoresize link image lists code media table contextmenu paste"
 58                                 //textcolor wordcount autosave advlist charmap print preview hr anchor pagebreak spellchecker directionality
 59                     ],
 60                     toolbar1: "bold italic bullist | link image media | code",
 61                     // formatselect removeformat underline unlink forecolor backcolor anchor previewfontselect fontsizeselect styleselectspellchecker template
 62                     // contextmenu: "link image inserttable | cell row column deletetable | formatselect forecolor",
 63                     menubar: false,
 64                     statusbar: false,
 65                     relative_urls: false,
 66                     toolbar_items_size: 'small',
 67                     file_browser_callback: this.onFileBrowserClick,
 68                     setup: Y.bind(function(editor) {
 69                         editor.on('change', Y.bind(this.fireUpdatedEvt, this));
 70                         this.editor = editor;
 71                     }, this),
 72                     image_advtab: true,
 73                     autoresize_min_height: 35,
 74                     autoresize_max_height: 500,
 75                     content_css: [
 76                         Y.Wegas.app.get("base") + "wegas-editor/css/wegas-inputex-rte.css"
 77                     ],
 78                     style_formats: [{// Style formats
 79                             title: 'Title 1',
 80                             block: 'h1'
 81                         }, {
 82                             title: 'Title 2',
 83                             block: 'h2'
 84                                     // styles : {
 85                                     //    color : '#ff0000'
 86                                     // }
 87                         }, {
 88                             title: 'Title 3',
 89                             block: 'h3'
 90                         }, {
 91                             title: 'Normal',
 92                             inline: 'span'
 93                         }]}, tinymce.EditorManager);
 94 
 95                 //this.editor.on('change', Y.bind(this.sendUpdatedEvt, this));    // Update on editor update
 96                 this.editor.render();
 97 
 98                 //tinymce.createEditor(this.el.id, {});
 99                 //tinymce.execCommand('mceAddEditor', false, this.el.id);
100             }, this);
101         },
102         onFileBrowserClick: function(field_name, url, type, win) {
103             RTEField.filePanel = new Y.Wegas.FileSelect();
104 
105             RTEField.filePanel.after("*:fileSelected", function(e, path) {
106                 e.stopImmediatePropagation();
107                 e.preventDefault();
108 
109                 var win = RTEField.filePanel.win,
110                         field_name = RTEField.filePanel.field_name,
111                         targetInput = win.document.getElementById(field_name);
112                 targetInput.value = Y.Wegas.Facade.File.getPath() + path;        // update the input field
113 
114                 if (typeof (win.ImageDialog) !== "undefined") {                 // are we an image browser
115                     if (win.ImageDialog.getImageData) {                         // we are, so update image dimensions...
116                         win.ImageDialog.getImageData();
117                     }
118 
119                     if (win.ImageDialog.showPreviewImage) {                     // ... and preview if necessary
120                         win.ImageDialog.showPreviewImage(Y.Wegas.Facade.File.getPath() + path);
121                     }
122                 }
123                 if (win.Media) {                                                // If in an editor window
124                     win.Media.formToData("src");                                // update the data
125                 }
126                 RTEField.filePanel.destroy();
127             });
128             RTEField.filePanel.win = win;
129             RTEField.filePanel.field_name = field_name;
130             return false;
131         },
132         /**
133          * Set the html content
134          * @param {String} value The html string
135          * @param {boolean} sendUpdatedEvt (optional) Wether this setValue should fire the 'updated' event or not (default is true, pass false to NOT send the event)
136          */
137         setValue: function(value, sendUpdatedEvent) {
138             var tmceI = tinyMCE.get(this.el.id);
139 
140             if (value) {
141                 value = value.replace(
142                         new RegExp("data-file=\"([^\"]*)\"", "gi"),
143                         "src=\"" + Y.Wegas.Facade.File.getPath() + "$1\""
144                         + " href=\"" + Y.Wegas.Facade.File.getPath() + "$1\"");  // @hack Place both href and src so it will work for both <a> and <img> elements
145             }
146             RTEField.superclass.setValue.call(this, value, sendUpdatedEvent);
147 
148             if (tmceI) {
149                 tmceI.setContent(value);
150             }
151         },
152         /**
153          * Get the html string
154          * @return {String} the html string
155          */
156         getValue: function() {
157             tinyMCE.triggerSave();
158             //return RTEField.superclass.getValue.call(this).replace(reg, "data-file=\"$3\" $1");
159             return RTEField.superclass.getValue.call(this)
160                     .replace(new RegExp("((src|href)=\".*/rest/File/GameModelId/.*/read([^\"]*)\")", "gi"), "data-file=\"$3\"")// Replace absolute path with injector style path
161                     .replace(new RegExp("((src|href)=\".*/rest/GameModel/.*/File/read([^\"]*)\")", "gi"), "data-file=\"$3\"");// Replace absolute path with injector style path
162         }
163     });
164     inputEx.registerType("html", RTEField, []);                                 // Register this class as "html" type
165 });
