 12 YUI.add('wegas-widgettoolbar', function(Y) {
 13     "use strict";
 14 
 15     var BOUNDINGBOX = "boundingBox", HOST = "host",
 16             WidgetToolbar;
 17 
 18     /**
 19      *  @name Y.Wegas.WidgetToolbar
 20      *  @class Adds a toolbar to target Y.Widget
 21      *  @extends Y.Plugin.Base
 22      *  @constructor
 23      */
 24     WidgetToolbar = function() {
 25         WidgetToolbar.superclass.constructor.apply(this, arguments);
 26     };
 27 
 28     Y.extend(WidgetToolbar, Y.Plugin.Base, {
 29         /** @lends Y.Wegas.WidgetToolbar */
 30 
 31         // *** Lifecycle methods *** //
 32         /**
 33          * @function
 34          * @private
 35          */
 36         initializer: function() {
 37             this.children = [];
 38             this.render();
 39             //this.messageTimer = new Y.Wegas.Timer({
 40             //    duration: 2000
 41             //});
 42             //his.messageTimer.on("timeOut", this.setStatusMessage, this, "");
 43             //this.afterHostEvent("render", this.render, this);
 44 
 45             this.onHostEvent("*:message", function(e) {                         // Observe success messages,
 46                 if (e.level === "success" && !e.timeout) {
 47                     this.setStatusMessage(e.content);                           // to display in the toolbar
 48                     //this.messageTimer.reset();
 49                     e.halt(true);
 50                 }
 51             });
 52         },
 53         /**
 54          * @function
 55          * @private
 56          */
 57         destructor: function() {
 58             //this.messageTimer.destroy();
 59             this.menuBar.destroy();
 60         },
 61         // *** Private methods *** //
 62         /**
 63          * @function
 64          * @private
 65          */
 66         render: function() {
 67             var host = this.get(HOST),
 68                     bb = host.get(BOUNDINGBOX);
 69 
 70             bb.addClass("wegas-hastoolbar").prepend('<div class="wegas-toolbar"></div>');
 71             host.get('contentBox').addClass("wegas-toolbar-sibling");
 72 
 73             this.menuBar = new Y.Wegas.MenuBar({
 74                 children: this.get("children")
 75                         //srcNode: bb.one(".wegas-toolbar-header"),
 76                         //render: true
 77             });
 78             this.menuBar.addTarget(host);
 79             this.menuBar.render(bb.one(".wegas-toolbar"));
 80             this.menuBar.get(BOUNDINGBOX).addClass("wegas-toolbar-header");
 81 
 82             bb.one(".wegas-toolbar").append('<div class="wegas-toolbar-panel"></div>');
 83         },
 84         /**
 85          *
 86          * Adds a widget to the toolbar
 87          *
 88          * @function
 89          * @param {Y.Widget|Object} widget A widget instance or an widget cfg object to be instantiated
 90          * @return {Y.Widget} the newly created widget
 91          */
 92         add: function(widget) {
 93             return this.menuBar.add(widget);
 94         },
 95         /**
 96          * Returns a toolbar widget based on its index
 97          * @function
 98          * @param {Number} index
 99          * @returns {Y.Widget}
100          */
101         item: function(index) {
102             return this.menuBar.item(index);
103         },
104         size: function() {
105             return this.menuBar.size();
106         },
107         remove: function(index) {
108             return this.menuBar.remove(index);
109         },
110         removeAll: function() {
111             return this.menuBar.removeAll();
112         },
113         destroyAll: function() {
114             return this.menuBar.destroyAll();
115         },
116         /**
117          * @function
118          * @private
119          * @param txt
120          * @return boolean true is status is set.
121          * @description set content of the message.
122          */
123         setStatusMessage: function(txt) {
124             this.getStatusNode().setContent(txt);
125         },
126         /**
127          * @function
128          * @private
129          * @description clear message (see function 'showMessage')
130          */
131         emptyMessage: function() {
132             this.getStatusNode().empty();
133         },
134         /**
135          * @function
136          * @private
137          * @return Status node
138          * @description get the status node of the message.
139          * if 'wegas-status-message' doesn't exist, create and return it
140          */
141         getStatusNode: function() {
142             var statusNode = this.get("header").one(".wegas-status-message");
143             if (!statusNode) {
144                 statusNode = new Y.Node.create("<span class='wegas-status-message'></span>");
145                 this.get("header").append(statusNode);
146             }
147             return statusNode;
148         }
149     }, {
150         /** @lends Y.Plugin.WidgetToolbar */
151         NS: "toolbar",
152         NAME: "toolbar",
153         /**
154          * <p><strong>Config attributes</strong></p>
155          * <ul>
156          *    <li>children {Y.Widget[]} A list of all widgets contained in the toolbar</li>
157          *    <li>header {Y.Node} The main toolbar node</li>
158          *    <li>panel {Y.Node} The panel node, visible on mouse over</li>
159          * </ul>
160          *
161          * @field
162          * @static
163          */
164         ATTRS: {
165             children: {
166                 value: []
167             },
168             header: {
169                 lazyAdd: false,
170                 value: false,
171                 getter: function() {
172                     return this.get(HOST).get(BOUNDINGBOX).one(".wegas-toolbar-header");
173                 }
174             },
175             panel: {
176                 lazyAdd: false,
177                 value: false,
178                 getter: function() {
179                     return this.get(HOST).get(BOUNDINGBOX).one(".wegas-toolbar-panel");
180                 }
181             }
182         }
183     });
184     Y.namespace('Plugin').WidgetToolbar = WidgetToolbar;
185 
186 });
