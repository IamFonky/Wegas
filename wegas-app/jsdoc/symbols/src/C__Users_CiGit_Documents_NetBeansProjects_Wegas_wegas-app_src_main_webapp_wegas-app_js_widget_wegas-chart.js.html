 13 YUI.add('wegas-chart', function(Y) {
 14     var CONTENTBOX = 'contentBox',
 15             Chart = Y.Base.create("wegas-chart", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 16         bindUI: function() {
 17             this.dsUpdateHandler = Y.Wegas.Facade.VariableDescriptor.after("update", this.syncUI, this);
 18         },
 19         syncUI: function() {
 20             var vd, i, variables = this.get("variables");
 21             this.vdList = [];
 22             this.requestCounter = 0;
 23 
 24             for (i = 0; i < variables.length; i++) {
 25                 vd = Y.Wegas.Facade.VariableDescriptor.cache.find("name", variables[i].name);
 26                 if (!vd) {
 27                     this.showMessage("error", "Variable " + variables[i].name + " not found");
 28                     return;
 29                 }
 30                 vd.position = i;
 31                 vd.label = variables[i].label || variables[i].name;
 32 
 33                 this.vdList[i] = null;
 34                 this.historyRequest(vd);
 35             }
 36         },
 37         destructor: function() {
 38             if (this.chart) {
 39                 this.chart.destroy();
 40             }
 41             this.dsUpdateHandler.detach();
 42             Y.Wegas.DataSource.abort(this.historyRequestId);
 43         },
 44         historyRequest: function(vd) {
 45             this.historyRequestId = Y.Wegas.Facade.VariableDescriptor.cache.getWithView(vd.getInstance(), "Extended", {
 46                 on: {
 47                     success: Y.bind(function(e) {
 48                         var entity = e.response.entity;
 49                         entity.label = vd.label;
 50 
 51                         this.vdList[vd.position] = entity;
 52 
 53                         this.requestCounter += 1;
 54                         if (this.requestCounter === this.get("variables").length) {
 55                             this.createChart();
 56                         }
 57                     }, this),
 58                     failure: function(r) {
 59                         if (r.serverResponse.status === 0) {
 60                             Y.log("Abort history query", "info", "Y.Wegas.Chart");
 61                         } else {
 62                             Y.error("Error by loading history data");
 63                         }
 64                     }
 65                 }
 66             });
 67         },
 68         /**
 69          * Creat a YUI3 Charts combospline' with values of a resource's moral and confidence historic values.
 70          * If any resource is given, the chart will be not created.
 71          * @ Param NumberDescriptor numberDescriptor, the source of chart's values
 72          */
 73         createChart: function() {
 74             var i, cb = this.get(CONTENTBOX),
 75                     seriesCollection = [],
 76                     rawSeries = [];
 77 
 78             if (this.chart) {
 79                 this.chart.destroy();
 80             }
 81             if (this.vdList.length < 1)
 82                 return;
 83 
 84             for (i = 0; i < this.vdList.length; i++) {
 85                 seriesCollection.push({
 86                     yDisplayName: this.vdList[i].label
 87                 });
 88                 rawSeries.push(this.vdList[i].get("history"));
 89             }
 90 
 91             this.chart = new Y.Chart({
 92                 type: this.get("chartType"),
 93                 seriesCollection: seriesCollection,
 94                 // categoryType:"time",                                         // Start sur l'axe mais l'axe devient time
 95                 axes: {
 96                     values: {
 97                         minimum: this.get("minValue"),
 98                         maximum: this.get("maxValue")
 99                     }
100                 },
101                 legend: {
102                     styles: {
103                         gap: 0
104                     },
105                     position: this.get("legendPosition")
106                 },
107                 tooltip: {
108                     markerLabelFunction: function(categoryItem, valueItem, itemIndex, series, seriesIndex) {
109                         return new Y.Node.create('<div><div><p>' + valueItem.displayName + ': ' + valueItem.axis.get("labelFunction").apply(this, [valueItem.value]) + '</p></div></div>');
110                     }
111                 },
112                 dataProvider: this.getChartValues(this.findNumberOfValue(rawSeries), rawSeries),
113                 horizontalGridlines: this.get("horizontalGridlines"),
114                 verticalGridlines: this.get("verticalGridlines")
115             });
116             this.chart.render(cb);
117         },
118         findNumberOfValue: function(series) {
119             var i, number = 0;
120             if (!this.get("numberOfValue")) {
121                 for (i = 0; i < series.length; i++) {
122                     if (series[i].length > number) {
123                         number = series[i].length;
124                     }
125                 }
126                 return number;
127             } else {
128                 return this.get("numberOfValue");
129             }
130         },
131         /**
132          * Create series for the chart.
133          * i = numberOfValues
134          * For each series, If number of values is smaller than i, copy the last value to create a serie with i values.
135          * If number of values is greater than i, keep only the i last values.
136          * @param Integer numberOfValues, the number of value wanted in the series.
137          * @param Array rawSeries, an array of array of Integer.
138          */
139         getChartValues: function(numberOfValues, rawSeries) {
140             var i, j, fitSeries = [], serieRawData = [], serieFitData = [];
141 
142             for (i = 0; i < numberOfValues; i++) {
143                 serieFitData.push(i + 1);
144             }
145 
146             fitSeries.push(serieFitData.slice());
147             for (i = 0; i < rawSeries.length; i++) {
148                 serieRawData = rawSeries[i];
149                 serieFitData.length = 0;
150                 for (j = numberOfValues - 1; j >= 0; j--) {
151                     if (serieRawData.length - 1 >= j) {
152                         serieFitData.push(serieRawData[serieRawData.length - (j + 1)]);
153                     }
154                 }
155                 fitSeries.push(serieFitData.slice());
156             }
157 
158             if (fitSeries[0].length === 0) {
159                 for (i = 1; i < 10; i++) {
160                     fitSeries[0].push(i);
161                 }
162             }
163             return fitSeries;
164         }
165     }, {
166         EDITORNAME: "Chart",
167         ATTRS: {
168             /**
169              * The target variable, returned either based on the variableName attribute,
170              * and if absent by evaluating the expr attribute.
171              */
172             chartType: {
173                 type: "string",
174                 value: "combo",
175                 choices: ['combo', 'line'],
176                 _inputex: {
177                     label: "Chart type"
178                 }
179             },
180             variables: {
181                 _inputex: {
182                     _type: "list",
183                     useButtons: true,
184                     elementType: {
185                         type: "variableselect",
186                         label: "variable",
187                         classFilter: ["NumberDescriptor"]
188                     }
189                 }
190             },
191             minValue: {
192                 optional: true,
193                 _inputex: {
194                     _type: "integer",
195                     label: "Min. value",
196                     negative: true
197                 }
198             },
199             maxValue: {
200                 optional: true,
201                 _inputex: {
202                     _type: "integer",
203                     label: "Max. value",
204                     negative: true
205                 }
206             },
207             width: {
208                 type: "string",
209                 value: "250px"
210             },
211             height: {
212                 type: "string",
213                 value: "200px"
214             },
215             numberOfValue: {
216                 type: "Number",
217                 optional: "true",
218                 _inputex: {
219                     _type: "integer",
220                     label: "Number of value",
221                 }
222             },
223             legendPosition: {
224                 value: "bottom",
225                 type: "string",
226                 choices: ['bottom', 'left', 'right', 'top'],
227                 _inputex: {
228                     value: "bottom"
229                 }
230             },
231             horizontalGridlines: {
232                 value: true,
233                 type: "boolean",
234                 _inputex: {
235                     label: "Horizontal Gridlines"
236                 }
237             },
238             verticalGridlines: {
239                 value: true,
240                 type: "boolean",
241                 _inputex: {
242                     label: "Vertical Gridlines"
243                 }
244             }
245         }
246     });
247 
248     Y.namespace('Wegas').Chart = Chart;
249 });
