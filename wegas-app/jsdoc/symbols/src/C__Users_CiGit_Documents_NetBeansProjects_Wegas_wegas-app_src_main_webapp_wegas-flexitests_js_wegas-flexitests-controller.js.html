 13 YUI.add("wegas-flexitests-controller", function(Y) {
 14     "use strict";
 15     Y.Wegas.FlexitestsController = Y.Base.create("wegas-flexitests-controller", Y.Wegas.AbsoluteLayout, [Y.Wegas.Editable, Y.Wegas.Parent], {
 16         BOUNDING_TEMPLATE: "<div><div class='flexi-mask'></div></div>",
 17         /**
 18          * Lifecycle method
 19          * @function
 20          * @private
 21          * @returns {undefined}
 22          */
 23         initializer: function() {
 24             this.leftElement = null;
 25             this.rightElement = null;
 26             this.centerElement = null;
 27             this.mcq = null;
 28             this.maxSize = 0;
 29             this.ongoing = false;
 30             this.config = {};
 31             this.currentQuestionId = -1;
 32             this.questionToDo = [];
 33             this.startTime = null;
 34             this.events = [];
 35             this.publish("visibility-timer:restart", {
 36                 broadcast: 1
 37             });
 38             //this.get("contentBox").hide();
 39         },
 40         /**
 41          * Lifecycle method
 42          * @function
 43          * @private
 44          * @returns {undefined}
 45          */
 46         bindUI: function() {
 47             this.constructor.superclass.bindUI.apply(this);
 48             this.set("tabIndex", -1);
 49             this.after("*:currentLoadingChange", function(e) {
 50                 var i, noready = false;
 51                 for (i in e.newVal) {
 52                     if (e.newVal.hasOwnProperty(i)) {
 53                         noready = (noready || e.newVal[i]);
 54                         if (noready) {
 55                             break;
 56                         }
 57                     }
 58                 }
 59                 if (!noready) {
 60                     this.get("contentBox").show();
 61                     this.startStimuli();
 62                 }
 63             });
 64             this.on("*:clientResponse", function(e) {
 65                 if (this.ongoing) {
 66                     this.responseGiven(e.value);
 67                 }
 68             });
 69         },
 70         /**
 71          * Lifecycle method
 72          * @function
 73          * @private
 74          * @returns {undefined}
 75          */
 76         syncUI: function() {
 77             this.mcq = this.getChildById("flexi-mcq");
 78             var targetDescriptor = this.mcq.get("variable.evaluated"),
 79                     props, exist = false, i, j, done = 0;
 80             if (!targetDescriptor) {
 81                 this.showMessage("Unable to find target variable");
 82                 return;
 83             }
 84 
 85             this.mask();
 86             this.leftElement = this.getChildById("leftElement");
 87             this.rightElement = this.getChildById("rightElement");
 88             this.centerElement = this.getChildById("centerElement");
 89             this.fixPoint = this.get("contentBox").all(".fix-point");
 90             this.fixPoint.hide();
 91             props = targetDescriptor.getInstance().get("properties");
 92             this.maxSize = Math.max(this.leftElement.size(), this.rightElement.size(), this.centerElement.size());
 93             for (i = 0; i < this.maxSize; i += 1) {
 94                 exist = false;
 95                 for (j in props) {
 96                     if (props.hasOwnProperty(j)) {
 97                         if (+(Y.JSON.parse(props[j]).id) === +i) {
 98                             exist = true;
 99                             done += 1;
100                             break;
101                         }
102                     }
103                 }
104                 if (!exist) {
105                     this.questionToDo[i - done] = i;
106                 }
107             }
108             this.onceAfter("render", this.next, this);
109         },
110         responseGiven: function(response) {
111             var responseTime = Y.Lang.now() - this.startTime,
112                     reponseElement,
113                     elements = this.collectElements();
114             this.ongoing = false;
115             //this.mask();
116             elements.index = +this.maxSize - this.questionToDo.length;
117             elements.id = this.currentQuestionId;
118             elements.response = response;
119             elements.delay = responseTime;
120             reponseElement = this.centerElement.getActiveElement().flexiresponse;
121             if (reponseElement instanceof Y.Plugin.FlexiResponse &&
122                     reponseElement.get("value") === response) {
123                 elements.valid = true;
124                 this.mcq.success(responseTime);
125             } else {
126                 elements.valid = false;
127                 this.mcq.error(responseTime);
128             }
129             this.mcq.save(elements);
130             Y.later(+this.mcq.get("feedback"), this, this.next);
131         },
132         next: function() {
133             var onSuccess = Y.bind(function() {
134                 Y.later(+this.get("fixPoint"), this, this.createLoadingEvent);
135                 this.fixPoint.show();
136             }, this);
137             if (this.questionToDo.length === 0) {
138                 this.runEndScript();
139                 this.unmask();
140                 return;
141             }
142             this.mask();
143             this.set("currentLoading", {left: true, center: true, right: true});
144             this.currentQuestionId = this.generateNextId();
145             this.centerElement.set("element", +this.currentQuestionId);
146             this.leftElement.set("element", +this.currentQuestionId);
147             this.rightElement.set("element", +this.currentQuestionId);
148             if (this.get("popupAfter") > 0 &&
149                     (this.maxSize - this.questionToDo.length) !== 1 &&
150                     ((this.maxSize - this.questionToDo.length - 1) % this.get("popupAfter")) === 0) {
151                 this.get("boundingBox").emitDOMMessage("showPopup", {
152                     content: this.get("popupContent"),
153                     buttons: [{
154                             label: "Ok",
155                             action: function() {
156                                 this.exit();
157                                 onSuccess();
158                             }
159                         }]
160                 });
161             } else {
162                 onSuccess();
163             }
164 
165         },
166         collectElements: function() {
167             var elements = {},
168                     swaped = this.swapzone ? this.swapzone.swaped : false,
169                     left = this.leftElement.getActiveElement(),
170                     center = this.centerElement.getActiveElement(),
171                     right = this.rightElement.getActiveElement();
172             //if (swaped) {
173             //    elements.left = right.get("content") || right.get("url");
174             //    elements.right = left.get("content") || left.get("url");
175             //} else {
176             //    elements.left = left.get("content") || left.get("url");
177             //    elements.right = right.get("content") || right.get("url");
178             //}
179             //elements.center = center.get("content") || center.get("url");
180             /*Extract elements in results*/
181             if (swaped) {
182                 elements.left = "right";
183                 elements.right = "left";
184             } else {
185                 elements.left = "left";
186                 elements.right = "right";
187             }
188 
189             return elements;
190         },
191         /**
192          * Check elements to be loaded if they exist
193          * @returns {undefined}
194          */
195         createLoadingEvent: function() {
196             if (this.centerElement.getActiveElement() && this.centerElement.getActiveElement() instanceof Y.Wegas.Image) {
197                 this.centerElement.getActiveElement().onceAfter(["load", "error"], function(e) {
198                     this.set("currentLoading.center", false);
199                 }, this);
200             } else {
201                 this.set("currentLoading.center", false);
202             }
203             if (this.leftElement.getActiveElement() && this.leftElement.getActiveElement() instanceof Y.Wegas.Image) {
204                 this.leftElement.getActiveElement().onceAfter(["load", "error"], function(e) {
205                     this.set("currentLoading.left", false);
206                 }, this);
207             } else {
208                 this.set("currentLoading.left", false);
209             }
210             if (this.rightElement.getActiveElement() && this.rightElement.getActiveElement() instanceof Y.Wegas.Image) {
211                 this.rightElement.getActiveElement().onceAfter(["load", "error"], function(e) {
212                     this.set("currentLoading.right", false);
213                 }, this);
214             } else {
215                 this.set("currentLoading.right", false);
216             }
217         },
218         generateNextId: function() {
219             return this.get("random") ?
220                     this.questionToDo.splice(Math.round(Math.random() * (this.questionToDo.length - 1)), 1)[0] :
221                     this.questionToDo.shift();
222         },
223         startStimuli: function() {
224             this.get("boundingBox").focus();
225             this.fire("visibility-timer:restart");
226             Y.soon(Y.bind(this.unmask, this));
227             this.ongoing = true;
228         },
229         runEndScript: function() {
230             Y.Wegas.Facade.VariableDescriptor.sendRequest({
231                 request: "/Script/Run/" + Y.Wegas.Facade.Game.get('currentPlayerId'),
232                 cfg: {
233                     method: "POST",
234                     data: Y.JSON.stringify(this.get("endImpact"))
235                 },
236                 on: {
237                     failure: Y.bind(function(e) {
238                         Y.log("error", "Failed to store data", "Y.Wegas.FlexitestsController");
239                     }, this)
240                 }
241             });
242         },
243         mask: function() {
244             this.get("boundingBox").one(".flexi-mask").show();
245         },
246         unmask: function() {
247             this.fixPoint.hide();
248             this.get("boundingBox").one(".flexi-mask").hide();
249             this.startTime = Y.Lang.now();
250         },
251         getChildById: function(id) {
252             var returnItem = null;
253             this.some(function(item) {
254                 if (item.get("id") === id) {
255                     returnItem = item;
256                     return true;
257                 }
258             });
259             return returnItem;
260         },
261         toObject: Y.Wegas.AbsoluteLayout.prototype.toObject,
262         /**
263          * Lifecycle method
264          * @function
265          * @private
266          * @returns {undefined}
267          */
268         destructor: function() {
269             var i;
270             this.constructor.superclass.destructor.apply(this);
271             for (i = 0; i < this.events.length; i += 1) {
272                 this.events[i].detach();
273             }
274         }
275     }, {
276         EDITORNAME: "Flexitests Controller",
277         ATTRS: {
278             currentLoading: {
279                 value: {},
280                 "transient": true
281             },
282             fixPoint: {
283                 value: 2000,
284                 type: "number",
285                 _inputex: {
286                     label: "Show fix-point (ms)",
287                     description: "Time to wait before next question"
288                 }
289             },
290             random: {
291                 value: true,
292                 type: "boolean",
293                 _inputex: {
294                     label: "Play lists randomly"
295                 }
296             },
297             popupAfter: {
298                 value: 0,
299                 type: "number",
300                 _inputex: {
301                     label: "Show popup message",
302                     description: "After # question(s).0 or less to disable"
303                 }
304             },
305             popupContent: {
306                 value: "",
307                 type: "string",
308                 optional: true,
309                 format: "html",
310                 _inputex: {
311                     label: "Popup content"
312                 }
313             },
314             endImpact: {
315                 label: "End impact",
316                 _inputex: {
317                     _type: "script",
318                     description: "What should happen once there is no more questions",
319                     optional: true
320                 }
321             }
322         }
323     });
324     /*Inject specific plugins*/
325     Y.Wegas.FlexitestsController.ATTRS.plugins = Y.clone(Y.Wegas.FlexitestsController.ATTRS.plugins);
326     Y.Wegas.FlexitestsController.ATTRS.plugins._inputex.items.push({
327         type: "Button",
328         label: "Swap top zones",
329         data: "SwapZone"
330     });
331     Y.Plugin.FlexiResponse = Y.Base.create("wegas-flexi-response", Y.Plugin.Base, [Y.Wegas.Plugin, Y.Wegas.Editable], {
332     }, {
333         NS: "flexiresponse",
334         NAME: "flexiresponse",
335         EDITORNAME: "Flexitest Response",
336         ATTRS: {
337             value: {
338                 value: "",
339                 type: "string",
340                 _inputex: {
341                     label: "Response value",
342                     description: "Needed to match with user input"
343                 }
344             }
345         }
346     });
347     Y.Plugin.SwapZone = Y.Base.create("wegas-flexi-swapzone", Y.Plugin.Base, [Y.Wegas.Plugin, Y.Wegas.Editable], {
348         initializer: function() {
349             if (!(this.get("host") instanceof Y.Wegas.FlexitestsController)) {
350                 return;
351             }
352             this.swaped = false;
353             this.afterHostMethod("next", function() {
354                 var current = this.get("host").maxSize - this.get("host").questionToDo.length;
355                 if (current === 1 && Math.random() > 0.5) {
356                     this.swap();
357                 }
358                 if (current % this.get("after") === 0) {
359                     this.swap();
360                 }
361             });
362         },
363         swap: function() {
364             this.get("host").leftElement.get("contentBox").swap(this.get("host").rightElement.get("contentBox"));
365             this.swaped = !this.swaped;
366         },
367         destructor: function() {
368             if (this.swaped) {
369                 this.swap();
370             }
371         }
372     }, {
373         NS: "swapzone",
374         NAME: "swapzone",
375         EDITORNAME: "Flexitest swap zone",
376         ATTRS: {
377             after: {
378                 value: 1,
379                 type: "number",
380                 _inputex: {
381                     label: "Swap after",
382                     description: "Top zone swap after # question(s)"
383                 },
384                 setter: function(v) {
385                     return +v > 0 ? +v : 1;
386                 }
387             }
388         }
389     });
390 });
