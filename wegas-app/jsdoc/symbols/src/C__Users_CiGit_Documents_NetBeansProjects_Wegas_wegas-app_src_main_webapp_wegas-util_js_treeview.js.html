 27     RIGHTWIDGETSETTERFN = function(v) {
 28         var rightWidget = this.get("rightWidget"),
 29                 targetNode = this.get(BOUNDING_BOX).one(".yui3-tree-rightwidget");
 30         if (rightWidget !== v && rightWidget) {                             // Remove existing child
 31             if (rightWidget instanceof Y.Node) {                            // Case 1: Y.Node
 32                 rightWidget.remove();
 33             } else {
 34                 rightWidget.get(BOUNDING_BOX).remove(); // Case 2: Y.Widget
 35                 rightWidget.removeTarget(this);
 36                 this.set("parent", null);
 37             }
 38         }
 39         if (v) {                                                            // Append the new widget
 40             if (v instanceof Y.Node) {                                      // Case 1: Y.Node
 41                 v.appendTo(targetNode);
 42             } else {                                                        // Case 2: Y.Widget
 43                 if (v.get("rendered")) {
 44                     v.get(BOUNDING_BOX).appendTo(targetNode);
 45                 } else {
 46                     v.render(targetNode);
 47                 }
 48                 v.set("parent", this);
 49                 v.addTarget(this);
 50             }
 51         }
 52         return v;
 53     };
