 12 YUI.add('wegas-form', function(Y) {
 13     "use strict";
 14 
 15     /**
 16      * @name Y.Wegas.Form
 17      * @class  Class to submit a form, add a toolbar with buttons "submit" and
 18      * "cancel" to mangae forms.
 19      * @extends Y.Widget
 20      * @augments Y.WidgetChild
 21      * @augments Y.Wegas.Widget
 22      * @constructor
 23      */
 24     var inputEx = Y.inputEx,
 25             lang = Y.Lang,
 26             Form = Y.Base.create("wegas-form", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 27                 /**
 28                  * @lends Y.Wegas.Form#
 29                  */
 30 
 31                 // ** Lifecycle Methods ** //
 32                 /**
 33                  * @function
 34                  * @private
 35                  * @description plug a toolbar and publich "submit" event.
 36                  */
 37                 initializer: function() {
 38                     this.plug(Y.Plugin.WidgetToolbar);
 39                     this.publish("submit", {
 40                         emitFacade: true
 41                     });
 42                     this.publish("updated", {
 43                         emitFacade: false
 44                     });
 45                 },
 46                 /**
 47                  * @function
 48                  * @private
 49                  * @description
 50                  */
 51                 renderUI: function() {
 52                     Y.Array.each(this.get("buttons"), this.addButton, this);
 53                 },
 54                 /**
 55                  * @function
 56                  * @private
 57                  * @description call function "renderToolbar".
 58                  */
 59                 syncUI: function() {
 60                     this.set("cfg", this.get("cfg"));
 61                 },
 62                 /**
 63                  * @function
 64                  * @private
 65                  * @returns {undefined}
 66                  */
 67                 destructor: function() {
 68                     this.set("form", null);
 69                 },
 70                 addButton: function(b) {
 71                     switch (b.action) {
 72                         case "submit":
 73                             b.on = {
 74                                 click: Y.bind(function() {
 75                                     var form = this.get("form"),
 76                                             val = form.getValue();
 77 
 78                                     if (!form.validate()) {
 79                                         this.showMessageBis("error", "Some fields are not valid.");
 80                                         return;
 81                                     }
 82                                     if (val.valueselector) {
 83                                         val = val.valueselector;
 84                                     }
 85                                     this.fire("submit", {
 86                                         value: val
 87                                     });
 88                                 }, this)
 89                             };
 90                             break;
 91                         default:
 92                             b.on = {
 93                                 click: Y.bind(function(action) {
 94                                     this.fire(action);
 95                                 }, this, b.action)
 96                             };
 97                             break;
 98                     }
 99                     this.toolbar.add(new Y.Wegas.Button(b));
100                 },
101                 /**
102                  * @function
103                  * @private
104                  * @description set the given form to null
105                  */
106                 destroyForm: function() {
107                     this.get("form").destroy();
108                     this.set("form", null);
109                 },
110                 setCfg: function(val) {
111                     var cfg = Y.clone(val);                                             // Duplicate so val will be untouched while serializing
112                     Y.mix(cfg, {
113                         parentEl: this.get("contentBox"),
114                         type: "group"
115                     });                                                                  // Set up the form parentEl attribute, so it knows where to render
116 
117                     Y.inputEx.use(val, Y.bind(function(cfg) {                           // Load form dependencies
118                         var form = Y.inputEx(cfg);                                      // Initialize and render form
119                         form.setValue(this.get("values"), false);                       // Sync form with "values" ATTR
120                         this.set("form", form);
121                         form.on("updated", function(e) {
122                             this.fire("updated", e);
123                         }, this);
124                     }, this, cfg));
125                 }
126             }, {
127                 /** @lends Y.Wegas.Form */
128                 EDITORNAME: "Form",
129                 /**
130                  * <p><strong>Attributes</strong></p>
131                  * <ul>
132                  *    <li>values: values of fields of the form</li>
133                  *    <li>form: the form to manage (see YUI Form)</li>
134                  *    <li>cfg: configuation of the form (see YUI Form)</li>
135                  * </ul>
136                  *
137                  * @field
138                  * @static
139                  */
140                 ATTRS: {
141                     /**
142                      * Values of fields of the form
143                      */
144                     values: {
145                         "transient": true,
146                         value: {},
147                         setter: function(val) {
148                             if (this.get("form")) {
149                                 this.get("form").setValue(val, false);
150                             }
151                             return val;
152                         }
153                     },
154                     /**
155                      * The form to manage
156                      */
157                     form: {
158                         "transient": true,
159                         setter: function(val) {
160                             if (this.get("form")) {                                     // If there is alread a form instantiated, destroy it
161                                 this.get("form").destroy();
162                             }
163                             return val;
164                         }
165                     },
166                     /**
167                      * Configuation of the form
168                      */
169                     cfg: {
170                         validator: Y.Lang.isObject,
171                         setter: function(val) {
172                             this.setCfg(val);
173                             return val;
174                         },
175                         items: {
176                             test: {
177                                 type: "string"
178                             }
179                         },
180                         _inputex: {
181                             index: 8,
182                             _type: "group",
183                             legend: "Fields",
184                             fields: Y.inputEx.Group.groupOptions
185                         }
186                     },
187                     buttons: {
188                         value: [{
189                                 type: "Button",
190                                 action: "submit",
191                                 label: "<span class=\"wegas-icon wegas-icon-save\" ></span>Save"
192                             }
193                         ],
194                         _inputex: {
195                             _type: "hidden"
196                         }
197                     }
198                 }
199             });
200     Y.namespace("Wegas").Form = Form;
201 
202     /* Add relevant plugin*/
203     Y.Wegas.Form.ATTRS.plugins = Y.clone(Y.Wegas.Widget.ATTRS.plugins);
204     Y.Wegas.Form.ATTRS.plugins._inputex.items.push({
205         type: "Button",
206         label: "Save to",
207         data: "SaveObjectAction"
208     });
209 
210     /**
211      *  @hack So we can easily change classs on inputex fields
212      */
213     inputEx.Field.prototype.addClassName = function(className) {
214         Y.one(this.divEl).addClass(className);
215     };
216     inputEx.Field.prototype.removeClassName = function(className) {
217         Y.one(this.divEl).removeClass(className);
218     };
219 
220     Y.inputEx.Group.groupOptions.splice(0, 4);
221     Y.inputEx.Group.groupOptions[0].label = null;
222     Y.inputEx.Group.groupOptions[0].useButtons = true;
223 
224     /*
225      * @hack Automatically add the "optional" message when necessary
226      */
227     inputEx.StringField.prototype.setOptions = function(options) {
228         inputEx.StringField.superclass.setOptions.call(this, options);
229 
230         this.options.regexp = options.regexp;
231         this.options.size = options.size;
232         this.options.maxLength = options.maxLength;
233         this.options.minLength = options.minLength;
234         this.options.typeInvite = options.typeInvite;
235         if (!this.options.required && this.options.typeInvite === undefined) {  // @MODIFIED
236             this.options.typeInvite = "optional";
237         }
238         this.options.readonly = options.readonly;
239         this.options.autocomplete = lang.isUndefined(options.autocomplete) ?
240                 inputEx.browserAutocomplete :
241                 (options.autocomplete === false || options.autocomplete === "off") ? false : true;
242         this.options.trim = (options.trim === true) ? true : false;
243     };
244     /** 
245      * Modified to allow changes on the fly
246      */
247     inputEx.StringField.prototype.initEvents = function() {
248         //Y.on("change", this.onChange, this.el, this);
249         Y.on("valueChange", this.onChange, this.el, this);                      // @Modified
250 
251         if (Y.UA.ie > 0) { // refer to inputEx-95
252             var field = this.el;
253             Y.on("key", function(e) {
254                 field.blur();
255                 field.focus();
256             }, this.el, 'down:13', this);
257         }
258 
259         Y.on("focus", this.onFocus, this.el, this);
260         Y.on("blur", this.onBlur, this.el, this);
261         Y.on("keypress", this.onKeyPress, this.el, this);
262         Y.on("keyup", this.onKeyUp, this.el, this);
263     };
264     /**
265      * @hack Let inputex also get requirement from selectfields, lists
266      */
267     inputEx.getRawModulesFromDefinition = function(inputexDef) {
268         var type = inputexDef.type || 'string',
269                 module = YUI_config.groups.inputex.modulesByType[type],
270                 modules = [module || type],
271                 //set fields if they exist
272                 fields = inputexDef.fields
273                 //else see if we have elementType for lists - if neither then we end up with null
274                 || inputexDef.availableFields || [];
275 
276         if (inputexDef.elementType) {
277             fields.push(inputexDef.elementType);
278         }
279 
280         // recursive for group,forms,list,combine, etc...
281         Y.Array.each(fields, function(field) {
282             modules = modules.concat(this.getModulesFromDefinition(field));
283         }, this);
284 
285         // TODO: inplaceedit  editorField
286         return modules;
287     };
288 
289 });
