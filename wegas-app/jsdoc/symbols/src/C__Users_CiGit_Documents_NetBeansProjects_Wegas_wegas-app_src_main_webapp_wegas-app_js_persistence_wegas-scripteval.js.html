  8 YUI.add('wegas-scripteval', function(Y) {
  9     "use strict";
 10     var ScriptEval, Variable;
 11 
 12     function buildItems(entity, acc) {                                  // Recursively build items lists
 13         var j, items;
 14         if (entity instanceof Y.Wegas.persistence.ListDescriptor) {
 15             items = entity.get("items");
 16             acc.items = [];
 17             for (j in items) {
 18                 if (items.hasOwnProperty(j)) {
 19                     acc.items.push(Y.JSON.parse(Y.JSON.stringify(items[j].getInstance())));
 20                     buildItems(items[j], acc.items[acc.items.length - 1]);
 21                 }
 22             }
 23         }
 24     }
 25 
 26     ScriptEval = Y.Base.create("ScriptEval", Y.Plugin.Base, [], {
 27         context: null,
 28         upToDate: false,
 29         initializer: function() {
 30             this.context = {};
 31             this.upToDate = false;
 32             this.afterHostEvent("response", function(e) {
 33                 this.upToDate = false;
 34             }, this);
 35             this.publish("evaluated");
 36             this.publish("failure");
 37         },
 38         /**
 39          *  A localEval with server fallback.
 40          *
 41          *  @param script The script to evaluate
 42          *  @param cb A callback object, containing success, failure function or just a function as success callback. First parameter passed will be result
 43          */
 44         eval: function(script, cb) {
 45             var result, url;
 46             if (cb instanceof Function) {
 47                 cb = {
 48                     success: cb
 49                 };
 50             }
 51             try {
 52                 result = this.localEval(script);
 53                 this.fire("evaluated", result);
 54                 if (cb && cb.success instanceof Function) {
 55                     cb.success(result);
 56                 }
 57             } catch (error) {
 58                 this.run(null, {
 59                     cfg: {
 60                         method: "POST",
 61                         data: script,
 62                         headers: {
 63                             'Managed-Mode': 'false'
 64                         }
 65                     },
 66                     on: {
 67                         success: function(response) {
 68                             if (cb && cb.success instanceof Function) {
 69                                 cb.success(Y.JSON.parse(response.responseText));
 70                             }
 71                         },
 72                         failure: function(response) {
 73                             var result;
 74                             try {
 75                                 result = Y.JSON.parse(response.responseText);
 76                             } catch (e) {
 77                                 result = null;
 78                             }
 79                             if (cb && cb.failure instanceof Function) {
 80                                 cb.failure(result);
 81                             }
 82                         }
 83                     }
 84                 });
 85             }
 86         },
 87         run: function(script, cfg) {
 88             this.get("host").sendRequest(Y.mix(cfg, {
 89                 request: "/Script/Run/" + Y.Wegas.Facade.Game.get('currentPlayerId'),
 90                 cfg: {
 91                     method: "POST",
 92                     data: script
 93                 }
 94             }));
 95         },
 96         /**
 97          * Tries to evaluate the script locally, using variables cache
 98          * @param {String} script The script to eval localy
 99          * @return {Any} value locally evaluated
100          */
101         localEval: function(script) {
102             /*jslint evil: true */
103             if (!this.upToDate) {                                               //Only compute if new value
104                 this._buildContext();
105             }
106             if (script.indexOf("return ") === -1) {
107                 script = "return " + script;
108             }
109             return (new Function("with(this) { " + script + ";}")).call(this.context);
110         },
111         /**
112          * @function
113          * @private
114          * @returns {undefined}
115          */
116         _buildContext: function() {
117             var i, data = this.get("host").data;
118             this.upToDate = true;
119             this.context = {};
120 
121             for (i in data) {
122                 if (data.hasOwnProperty(i)) {
123                     this.context[data[i].get('name')] = Y.JSON.parse(Y.JSON.stringify(data[i].getInstance()));
124                     buildItems(data[i], this.context[data[i].get('name')]);
125 
126 //                if (data[i] instanceof Y.Wegas.persistence.ListDescriptor) {
127 //                    this.context[data[i].get('name')].items = [];
128 //                    for (j in data[i].get("items")) {
129 //                        this.context[data[i].get('name')].items.push(JSON.parse(JSON.stringify(data[i].get("items")[j].getInstance())));
130 //                    }
131 //                }
132                 }
133             }
134             /*SANDBOX*/
135             Y.mix(this.context, {
136                 window: undefined,
137                 Y: undefined,
138                 YUI: undefined
139             });
140             /*Extend functionalities (mirror server)*/
141             Y.mix(this.context, {
142                 VariableDescriptorFacade: Variable,
143                 Variable: Variable,
144                 self: Y.Wegas.Facade.Game.cache.getCurrentPlayer(),
145                 gameModel: Y.Wegas.Facade.GameModel.cache.getCurrentGameModel()
146             });
147         }
148 
149     }, {
150         NS: "script",
151         NAME: "scriptEval"
152     });
153     Variable = {
154         find: function(gameModel, name) {
155             return Y.Wegas.Facade.VariableDescriptor.cache.find("name", (Y.Lang.isString(gameModel)) ? gameModel : name);
156         }
157     };
158     Y.namespace('Plugin').ScriptEval = ScriptEval;
159 
160 });
