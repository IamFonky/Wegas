289         escape: function(t) {
290             return Y.JSON.stringify(this.sortObject(t), null, "\t");

292         sortObject: function(obj) {
293 
294             if (Y.Lang.isArray(obj)) {
295                 var sorted_obj = [];
296                 Y.Array.each(obj, function(i) {
297                     sorted_obj.push(this.sortObject(i));
298                 }, this);
299                 return sorted_obj;
300             } else if (Y.Lang.isObject(obj)) {
301 
302                 var keys = Y.Object.keys(obj),
303                         sorted_obj = {};
304 
305                 keys.sort();                                                        // sort keys
306 
307                 Y.Array.each(keys, function(item) {                               // create new array based on Sorted Keys
308                     sorted_obj[item] = this.sortObject(obj[item]);
309                 }, this);
310                 return sorted_obj;
311             } else {
312                 return obj;
313             }
