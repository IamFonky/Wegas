 14 YUI.add('wegas-pageloader', function(Y) {
 15     "use strict";
 16 
 17     var CONTENTBOX = 'contentBox', PageLoader;
 18     /**
 19      * @name Y.Wegas.PageLoader
 20      * @extends Y.Widget
 21      * @borrows Y.WidgetChild, Y.WidgetParent, Y.Wegas.Widget, Y.Wegas.Editable
 22      * @class  class loader of wegas's pages
 23      * @constructor
 24      * @description Load pages and request widget to render.
 25      */
 26     PageLoader = Y.Base.create("wegas-pageloader", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 27         /** @lends Y.Wegas.PageLoader# */
 28 
 29         // *** Private fields *** //
 30         /**
 31          * Current page id
 32          */
 33         currentPageId: null,
 34         // *** Lifecycle Methods ***/
 35         /**
 36          * @function
 37          * @private
 38          * @description Set variable with initials values.
 39          * Set page to default page
 40          * Keep a references of all loaded PageLoaders in PageLoader.pageLoaderInstances.
 41          */
 42         initializer: function() {
 43             this.handlers = [];
 44             PageLoader.pageLoaderInstances[this.get("pageLoaderId")] = this; // We keep a references of all loaded PageLoaders
 45             this.publish("contentUpdated", {emitFacade: false});
 46         },
 47         /**
 48          * @function
 49          * @private
 50          * @description bind function to events.
 51          * When page is updated, syncUI
 52          * When a response is fire by the VariableDescriptor, do sync
 53          * When the current player change, do sync
 54          * When an exception in fire, stop loading page, show error message.
 55          */
 56         bindUI: function() {
 57             this.handlers.push(Y.Wegas.Facade.VariableDescriptor.after("update", function() {// When the variable cache is update,
 58                 if (this.get("page") && "" + this.get("page.evaluated") !== "" + this.get('pageId')) {// and if the current page has change,
 59                     this.syncUI();                                              // sync the view
 60                 }
 61                 if (this.get("variable") && "" + this.get("variable.evaluated") !== "" + this.get('pageId')) {// @backwardcompatibilityand if the current page has change,
 62                     this.syncUI();                                              // sync the view
 63                 }
 64             }, this));
 65 
 66             //Y.Wegas.Facade.Page.after("response", this.syncUI, this);
 67             //this.handlers.push(Y.Wegas.Facade.Page.cache.after("pageUpdated", function(e) {
 68             //    if (e.page && ("" + e.page["@pageId"] === "" + this.get("pageId"))) {
 69             //        this.currentPageId = null; // @hack force update
 70             //        this.syncUI();
 71             //    }
 72             //}, this));
 73         },
 74         /**
 75          * @function
 76          * @private
 77          * @description Set pageId and displayed new page if the id is
 78          *  different that the current page id
 79          */
 80         syncUI: function() {
 81             var val = this.get("variable.evaluated"),
 82                     page = this.get("page.evaluated");
 83             if (page) {                                                         // If there is a page script
 84                 this.set("pageId", +page);                                      // display it
 85             } else if (val && val.getInstance().get('value')) {                 // @backwardcompatibility
 86                 this.set("pageId", val.getInstance().get('value'));
 87             } else if (this.get("defaultPageId") && !this.get("pageId")) {      //in case a defaultPageId is defined and no pageId is
 88                 this.set("pageId", this.get("defaultPageId"));
 89             } else {
 90                 this.set("pageId", this.get("pageId")); // Otherwise use pageId (in case the setter has not been called yet)
 91             }
 92 
 93         },
 94         /**
 95          * @function
 96          * @private
 97          * @description Destroy widget and detach all functions created by this widget
 98          * remove instance kept in PageLoader.pageLoaderInstances.
 99          */
100         destructor: function() {
101             if (this.get("widget")) {
102                 this.get("widget").destroy();
103             }
104             Y.Array.each(this.handlers, function(h) {
105                 h.detach();
106             });
107             delete PageLoader.pageLoaderInstances[this.get("pageLoaderId")];
108         },
109         /**
110          * reload current page from cache
111          * @function
112          * @public
113          */
114         reload: function() {
115             this.showOverlay();
116             this.currentPageId = null;
117             this.syncUI();
118         },
119         getEditorLabel: function() {
120             return this.get("pageLoaderId");
121         },
122         // *** Private Methods ***/
123         /**
124          * @function
125          * @private
126          * @param {String} pageId check for this page's ID.
127          * @return boolean
128          * @description Return true if an ancestor already loads pageId
129          */
130         ancestorWithPage: function(pageId) {                                     //Page loader mustn't load the page who contain itself.
131             var same = false;
132             this.get("boundingBox").ancestors("." + this.getClassName(), false).some(function(node) {
133                 var widget = Y.Widget.getByNode(node);
134                 if (+pageId === +widget.currentPageId || +pageId === +widget.get('variable.evaluated')) {
135                     same = true;
136                     this.showMessage("warn", "Pageloader [" + this.get("pageLoaderId") + "] tries to load page (" + pageId + ") already loaded by one of its ancestor[" + widget.get("pageLoaderId") + "].");
137                     return true;
138                 }
139             }, this);
140             return same;
141         }
142 
143     }, {
144         /** @lends Y.Wegas.PageLoader */
145         EDITORNAME: "Page display",
146         /**
147          * @field
148          * @static
149          * @description
150          * <p><strong>Attributes</strong></p>
151          * <ul>
152          *    <li>pageLoaderId: the id of this pageLoader</li>
153          *    <li>defaultPageId: the id of the default page to load</li>
154          *    <li>pageId: the page id to load</li>
155          *    <li>variable: a variable (or expression) which contain the id of the page to load
156          *    The target variable, returned either based on the name attribute,
157          *    and if absent by evaluating the expr attribute.
158          *    </li>
159          *    <li>widget: A widget to render in current page (transient)</li>
160          * </ul>
161          */
162         ATTRS: {
163             /**
164              * the id of this pageLoader
165              */
166             pageLoaderId: {
167                 type: "string",
168                 value: "maindisplayarea",
169                 //value: "PageLoader" + Y.Lang.now(), //generate a default pageLoaderId
170                 _inputex: {
171                     label: "Zone id"
172                 }
173             },
174             /**
175              * the id of the default page to load
176              */
177             defaultPageId: {
178                 type: "string",
179                 _inputex: {
180                     label: "Default page",
181                     _type: "pageselect",
182                     required: false
183                 }
184             },
185             /**
186              * the page id to load
187              */
188             pageId: {
189                 type: "string",
190                 "transient": true,
191                 setter: function(val) {
192                     if (!val || val === this.currentPageId || this.ancestorWithPage(val)) {// If the widget is currently being loaded,
193                         return val;                                             // do not continue
194                     }
195                     this.currentPageId = val;
196                     Y.log("Getting page", "log", "Wegas.PageLoader");
197                     Y.Wegas.Facade.Page.cache.getPage(val, Y.bind(function(widgetCfg) {// Retrieve page
198                         if (!widgetCfg) {
199                             return;
200                         }
201                         this.showOverlay();
202 
203                         if (this.get("widget")) {
204                             Y.log("Destroy previous widget", "log", "Wegas.PageLoader");
205                             this.get("widget").destroy();
206                             this.set("widget", null);
207                         }
208                         this.get(CONTENTBOX).empty();                           // Let the overlay appear during rendering
209 
210                         Y.Wegas.Widget.use(widgetCfg, Y.bind(function() {       // Load the subwidget dependencies
211                             try {
212                                 Y.log("Rendering new widget", "log", "Wegas.PageLoader");
213                                 var widget = Y.Wegas.Widget.create(widgetCfg);  // Render the subwidget
214                                 widget.render(this.get(CONTENTBOX));
215                                 widget['@pageId'] = widgetCfg['@pageId'];       // @HACK set up a reference to the page
216                                 this.set("widget", widget);
217                             } catch (e) {
218                                 this._state.add("widgetCfg", "value", widgetCfg);//@HACK : avoid some heavy computing, bypass setter!!! Warn, no events
219                                 this.get(CONTENTBOX).setContent("<center><i>Could not load sub page.</i></center>");
220                                 Y.log('renderUI(): Error rendering widget: ' + (e.stack || e), 'error', 'Wegas.PageLoader');
221                             } finally {
222                                 this.hideOverlay();
223                                 this.fire("contentUpdated");
224                             }
225                         }, this));
226                     }, this));
227                     return val;
228                 }
229             },
230             /**
231              * A variable (or expression) which contain the id of the page to load
232              * The target variable, returned either based on the name attribute,
233              * and if absent by evaluating the expr attribute.
234              * 
235              * @deprecated
236              */
237             variable: {
238                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER
239             },
240             /**
241              * 
242              */
243             page: {
244                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER
245             },
246             /**
247              * A widget to render in current page (transient)
248              */
249             widget: {
250                 "transient": true,
251                 setter: function(v) {
252                     if (this.get("widget")) {
253                         this.get("widget").removeTarget(this);
254                     }
255                     if (v) {
256                         v.addTarget(this);
257                     }                         // Event on the loaded widget will be forwarded
258                     return v;
259                 }
260             },
261             widgetCfg: {
262                 "transient": true,
263                 getter: function(val) {
264                     var p;
265                     if (this.get("widget")) {
266                         return Y.JSON.stringify(this.get("widget").toObject("@pageId"), null, "\t");
267                     } else if (val) {
268                         p = Y.clone(val);
269                         delete p['@pageId'];
270                         return Y.JSON.stringify(val, null, "\t");
271                     }
272                     return val;
273                 }
274             }
275         },
276         pageLoaderInstances: {},
277         find: function(id) {
278             return PageLoader.pageLoaderInstances[id];
279         }
280     });
281     Y.namespace('Wegas').PageLoader = PageLoader;
282 });
