 14 YUI.add('wegas-itemselector', function(Y) {
 15     "use strict";
 16 
 17     var CONTENTBOX = 'contentBox', ItemSelector;
 18     /**
 19      * @lends Y.Wegas.ItemSelector#
 20      */
 21     /**
 22      * @name Y.Wegas.ItemSelector
 23      * @extends Y.Widget
 24      * @borrows Y.Wegas.Widget, Y.Wegas.Editable, Y.Wegas.NodeFormatter
 25      * @class class to select single descriptor (from a list) to display desired variables from it.
 26      * @constructor
 27      * @description Display a given list of descriptor, each is selectable to display desired variables from it.
 28      */
 29     ItemSelector = Y.Base.create("wegas-itemselector", Y.Widget, [Y.Wegas.Widget, Y.Wegas.Editable, Y.Wegas.NodeFormatter], {
 30         /** @lends Y.Wegas.ItemSelector# */
 31 
 32         // *** Private fields *** //
 33         /**
 34          * Reference to each used functions
 35          */
 36         handlers: null,
 37         /**
 38          * The selected variable descriptor
 39          */
 40         currentItem: null,
 41         /**
 42          * The reference to the ScrollView widget
 43          */
 44         scrollView: null,
 45         // *** Lifecycle Methods *** //
 46         /**
 47          * @function
 48          * @private
 49          * @description Set variables with initials values.
 50          */
 51         initializer: function() {
 52             this.handlers = {};
 53         },
 54         /**
 55          * @function
 56          * @private
 57          * @description Select the first available variable instance and init
 58          * a ScrollView widget.
 59          */
 60         renderUI: function() {
 61             var i, variables, cb = this.get(CONTENTBOX);
 62             cb.append('<div class="selectors"></div>');
 63             cb.append('<div class="informations"></div>');
 64             if (!this.get('listVariables')) {
 65                 return;
 66             }
 67             variables = Y.Wegas.Facade.VariableDescriptor.cache.find("name", this.get('listVariables'));
 68             if (!variables || !variables.get('items')) {
 69                 return;
 70             }
 71             for (i = 0; i < variables.get('items').length; i++) {
 72                 if (variables.get('items')[i].getInstance().get('active') == null || variables.get('items')[i].getInstance().get('active') == true) {
 73                     this.currentItem = variables.get('items')[i];
 74                     break;
 75                 }
 76             }
 77             this.scrollView = new Y.ScrollView({
 78                 id: 'itemselector-scrollview',
 79                 srcNode: cb.one('.selectors'),
 80                 width: cb.one('.selectors').offsetwidth,
 81                 flick: {
 82                     minDistance: 20,
 83                     minVelocity: 0.6,
 84                     axis: 'x'
 85                 }
 86             });
 87         },
 88         /**
 89          * @function
 90          * @private
 91          * @description bind function to events.
 92          * When VariableDescriptorFacade is updated, do syncUI
 93          * When a "selector" div is clicked, set current item. PreventDevault to
 94          * prevent problem between click action and slide action (on SlidePanel)
 95          */
 96         bindUI: function() {
 97             var cb = this.get(CONTENTBOX);
 98             this.handlers.itemSelectorUpdate = Y.Wegas.Facade.VariableDescriptor.after("update", this.syncUI, this);
 99 
100             this.handlers.itemSelectorSelect = cb.one('.selectors').delegate('click', function(e) {
101                 var i, variables, name;
102                 e.preventDefault();
103                 if (e.target.ancestors('.selector').item(0)) {
104                     name = e.target.ancestors('.selector').item(0).getAttribute("data-name");
105                 } else {
106                     name = e.target.getAttribute("data-name");
107                 }
108                 variables = Y.Wegas.Facade.VariableDescriptor.cache.find("name", this.get('listVariables'));
109                 if (!variables || !variables.get('items'))
110                     return;
111 
112                 this.currentItem = Y.Array.find(variables.get('items'), function(item) {
113                     return item.get("name") === name;
114                 });
115                 this.syncUI();
116             }, '.selector', this);
117 
118         },
119         /**
120          * @function
121          * @private
122          * @description refresh displayed values.
123          */
124         syncUI: function() {
125             var cb = this.get(CONTENTBOX), variables;
126             if (!this.get('listVariables'))
127                 return;
128             variables = Y.Wegas.Facade.VariableDescriptor.cache.find("name", this.get('listVariables'));
129             if (!variables || !variables.get('items') || variables.get('items').length <= 0)
130                 return;
131             this.createSelector(cb, variables.get('items'));
132             this.createInformations();
133         },
134         /**
135          * @function
136          * @private
137          * @description Detach all functions created by this widget.
138          */
139         destructor: function() {
140             for (var k in this.handlers) {
141                 this.handlers[k].detach();
142             }
143         },
144         // *** Private Methods *** //
145         /**
146          * @function
147          * @private
148          * @param cb
149          * @param variables
150          * @description Delete selectors and, for all available instance,
151          * re-create a selector.
152          */
153         createSelector: function(cb, variables) {
154             var i, node = cb.one('.selectors'), selector;
155             node.empty();
156             for (i = 0; i < variables.length; i++) {
157                 if (variables[i].getInstance().get('active') == null || variables[i].getInstance().get('active') == true) {
158                     selector = Y.Node.create('<div class="selector" data-name="' + variables[i].get('name') + '"></div>');
159                     if (variables[i] === this.currentItem) {
160                         selector.addClass('current');
161                     }
162                     this.createDOMProperties(selector, variables[i], this.get('selectors'));
163                     node.append(selector);
164                 }
165             }
166             this.scrollView.render();
167         },
168         /**
169          * @function
170          * @private
171          * @param cb
172          * @description Delete all values relative to the selected
173          * instance and re-create them with the niew selected instance.
174          */
175         createInformations: function() {
176             var node = this.get(CONTENTBOX).one('.informations');
177             node.empty();
178             Y.Wegas.Facade.VariableDescriptor.cache.getWithView(this.currentItem, "Extended", {// just need to check if it causes bugs
179                 on: {
180                     success: Y.bind(function(e) {
181                         this.currentItem.set("description", e.response.entity.get("description"));// @hack how to mix two entities, one with the instance data and the other with descriptions, etc ?
182                         this.createDOMProperties(node, this.currentItem, this.get('informations'));
183                     }, this)
184                 }
185             });
186 
187         },
188         /**
189          * @function
190          * @private
191          * @param node
192          * @param variable
193          * @param attrs
194          * @description Create DOM nodes from selected instance and desired
195          * value to display from it. Some "mode" are available, display as :
196          * image: put the value as source (src) of the image.
197          * position: Use value to highlight a line in a given DOM list
198          * valuebox: Display equivalent number of div than the number indicates
199          *  by the value
200          * default : Display value as a simple text
201          * Label and classname can be added for each variable.
202          */
203         createDOMProperties: function(node, variable, attrs) {
204             var i, type, value, label, className, obj, child;
205             if (!node || !variable || !attrs)
206                 return;
207             for (i = 0; i < attrs.length; i++) {
208                 obj = attrs[i];
209                 if (typeof obj === 'object') {
210                     value = this.getVariableValue(variable, obj['name']);
211                     value = (value != null) ? value : obj['name'];
212                     label = (obj['label'] || null);
213                     type = (obj['type'] || 'undefine');
214                     className = (obj['className'] || null);
215                     switch (type) {
216                         case 'image' :
217                             child = this.makeNodeImage({
218                                 "data-file": value,
219                                 width: obj['height'],
220                                 height: obj['width']
221                             }, className);
222                             break;
223                         case 'position' :
224                             child = this.makeNodePosition(obj['html'], obj['selector'], value, obj['minVal'], obj['invert'], className);
225                             break;
226                         case 'valueBox' :
227                             child = this.makeNodeValueBox('' + value, obj['maxValue'], label, className);
228                             break;
229                         case 'longText':
230                             child = this.makeNodeText("<i>Loading</i>", label, className || value);
231                             this.makeNodeLongText(this.get(CONTENTBOX), variable, value, className || value);
232                             break;
233                         default :
234                             child = this.makeNodeText(value, label, className);
235                             break;
236                     }
237                     child.setAttribute('data-name', obj['name']);
238                     node.append(child);
239                 }
240             }
241         },
242         /**
243          * @function
244          * @private
245          * @param variable
246          * @param varName
247          * @return value
248          * @description return a value corresponding to its name
249          *  from a given variable
250          */
251         getVariableValue: function(variable, varName) {
252             var i, prop = this.get('searchInProperties'), value = null;
253             if (!variable || !varName)
254                 return value;
255             if (variable.get(varName) != null) {
256                 value = variable.get(varName);
257             } else if (variable.getInstance().get(varName) != null) {
258                 value = variable.getInstance().get(varName);
259             } else {
260                 for (i = 0; i < prop.length; i++) {
261                     if (variable.get(prop[i]) != null && variable.get(prop[i])[varName] != null) {
262                         value = variable.get(prop[i])[varName];
263                         break;
264                     }
265                     else if (variable.getInstance().get(prop[i]) != null && variable.getInstance().get(prop[i])[varName] != null) {
266                         value = variable.getInstance().get(prop[i])[varName];
267                         break;
268                     }
269                 }
270             }
271             if (parseFloat(value)) {
272                 value = parseFloat(value);
273             }
274             return value;
275         }
276 
277     }, {
278         /**
279          * @lends Y.Wegas.ItemSelector#
280          */
281         /**
282          * @field
283          * @static
284          * @description
285          * <p><strong>Attributes</strong></p>
286          * <ul>
287          *    <li>listVariables: A variable reference to a list of variable descriptor.</li>
288          *    <li>searchInProperties: Name of list (like properties) to seacrch in.</li>
289          *    <li>selectors: Array of values to display in selectors</li>
290          *    <li>informations: Array of values to display as information of a selected instance</li>
291          * </ul>
292          */
293         ATTRS: {
294             /**
295              * A variable reference to a list of variable descriptor
296              */
297             listVariables: {
298                 value: null,
299                 validator: function(s) {
300                     return s === null || Y.Lang.isString(s);
301                 }
302             },
303             /**
304              * Name of list (like properties) to seacrch in.
305              */
306             searchInProperties: {
307                 validator: Y.Lang.isArray,
308                 value: []
309             },
310             /**
311              * Array of values to display in selectors
312              */
313             selectors: {
314                 validator: Y.Lang.isArray,
315                 value: []
316             },
317             /**
318              * Array of values to display as information of a selected instance
319              */
320             informations: {
321                 validator: Y.Lang.isArray,
322                 value: []
323             }
324         }
325     });
326 
327     Y.namespace('Wegas').ItemSelector = ItemSelector;
328 });
