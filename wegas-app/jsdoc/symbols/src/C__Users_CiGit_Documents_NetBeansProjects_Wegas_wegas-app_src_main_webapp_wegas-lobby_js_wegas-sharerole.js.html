 12 YUI.add('wegas-sharerole', function(Y) {
 13     var CONTENTBOX = 'contentBox',
 14             ShareRole = Y.Base.create("wegas-sharerole", Y.Widget, [Y.WidgetChild, Y.Wegas.Editable, Y.Wegas.Widget], {
 15         /**
 16          *
 17          */
 18         CONTENT_TEMPLATE: "<div><div class=\"title\" >Option 2: Share link</div></div>",
 19         /**
 20          *
 21          */
 22         renderUI: function() {
 23             var cb = this.get(CONTENTBOX),
 24                     e = this.get("entity"),
 25                     //gameModel = (e instanceof Y.Wegas.persistence.Game) ? Y.Wegas.Facade.GameModel.cache.findById(e.get("gameModelId")) : e,
 26                     visibilityChoices = [
 27                 //{value: 'Private', label: 'Only people in the list can join'},
 28                 {value: 'Link', label: 'Anyone with the link can join'},
 29                 {value: 'Public', label: 'Everybody can join'}
 30             ];
 31 
 32             this.targetEntityId = (e instanceof Y.Wegas.persistence.GameModel) ? "gm" + e.get("id")
 33                     : "g" + e.get("id");
 34 
 35             this.visibility = new Y.inputEx.SelectField({
 36                 label: 'Accessibility',
 37                 choices: visibilityChoices,
 38                 parentEl: cb
 39             });
 40             (new Y.Node(this.visibility.divEl)).addClass("wegas-advanced-feature");// @fixme
 41 
 42             this.link = new Y.inputEx.StringField({
 43                 wrapperClassName: "inputEx-fieldWrapper wegas-link",
 44                 parentEl: cb,
 45                 description: 'Using this link, players need to log in or create an account, and will then be redirected to the game.',
 46                 value: Y.Wegas.app.get("base") + "game.html?token=" + this.get("entity").get("token")
 47             });
 48 
 49             this.syncLinkVisibility();
 50             this.requestPermissions();
 51 
 52 
 53             this.set("visible", false);                                         // @HACK Visibility depends on a node in the parent's tree 
 54             Y.on("available", function() {
 55                 var node = Y.one(".wegas-game-access select"),
 56                         updatVisibility = function() {
 57                     this.set("visible", node.get("value") === "ENROLMENTKEY");
 58                 };
 59                 updatVisibility.call(this);
 60                 node.on("valuechange", updatVisibility, this);
 61             }, ".wegas-game-access select", this);
 62         },
 63         bindUI: function() {
 64             this.visibility.on("updated", function(value) {
 65                 this.syncLinkVisibility(value);
 66                 Y.Wegas.Facade.User.cache.deleteAllRolePermissions(this.get('role'), this.targetEntityId);
 67 
 68                 var permission = Y.Array.find(this.get('permsList'), function(p) {
 69                     return p.name === value;
 70                 });
 71                 if (permission) {
 72                     Y.Wegas.Facade.User.sendRequest({// Send an add permission request
 73                         request: "/AddPermission/" + this.get('role') + "/" + permission.value + ":" + this.targetEntityId,
 74                         cfg: {
 75                             method: "POST"
 76                         }
 77                     });
 78                 }
 79             }, this);
 80 
 81             //this.link.disable();
 82             var inputNode = this.get("contentBox").one(".wegas-link input");
 83             inputNode.on("keypress", function(e) {
 84                 e.preventDefault();
 85             });                                                                 // Edition not allowed on the input node
 86             inputNode.on("click", inputNode.select, inputNode);                 // Select whole link on click
 87         },
 88         destructor: function() {
 89             this.link.destroy();
 90             this.visibility.destroy();
 91         },
 92         requestPermissions: function() {
 93             this.showOverlay();
 94             Y.Wegas.Facade.User.sendRequest({
 95                 request: "/FindPermissionByInstance/" + this.targetEntityId,
 96                 on: {
 97                     success: Y.bind(function(e) {
 98                         Y.Array.each(e.response.results.entities, function(role) {
 99                             if (role.get('val').name === this.get('role')) {
100                                 Y.Array.forEach(role.get('val').permissions, function(resultPerm) {
101                                     var splitedPerm = resultPerm.split(":");
102                                     Y.Array.forEach(this.get('permsList'), function(permFromList) {
103                                         if (splitedPerm[0] + ":" + splitedPerm[1] === permFromList.value) {
104                                             this.visibility.setValue(permFromList.name, false);
105                                             this.syncLinkVisibility();
106                                         }
107                                     }, this);
108                                 }, this);
109                             }
110                         }, this);
111                         this.hideOverlay();
112                     }, this),
113                     failure: Y.bind(this.defaultFailureHandler, this)
114                 }
115             });
116         },
117         syncLinkVisibility: function(selectValue) {
118             //if (this.visibility.getValue() === "Private") {
119             //    this.link.hide();
120             //} else {
121             //    this.link.show();
122             //}
123         }
124     }, {
125         ATTRS: {
126             role: {
127                 value: "Public"
128             },
129             entity: {},
130             permsList: {
131                 value: []
132             }
133         }
134     });
135     Y.namespace('Wegas').ShareRole = ShareRole;
136 
137 });
