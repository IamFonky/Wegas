1017     function loadModules(group) {
1018         var i, modules = group.modules,
1019                 module, type,
1020                 moduleName,
1021                 allModules = [];
1022         for (moduleName in modules) {                                           // Loop through all modules
1023             if (modules.hasOwnProperty(moduleName)) {
1024                 allModules.push(moduleName); // Build a list of all modules
1025 
1026                 module = modules[moduleName];
1027                 type = module.type || "js";
1028                 module.path = module.path || type + "/" + moduleName + "-min." + type;
1029 
1030                 if (module.ws_provides) {                                       // Build a reverse index on which module provides what type (Wegas)
1031                     if (Y.Lang.isArray(module.ws_provides)) {
1032                         for (i = 0; i < module.ws_provides.length; i = i + 1) {
1033                             YUI_config.groups.wegas.modulesByType[module.ws_provides[i]] = moduleName;
1034                         }
1035                     } else {
1036                         YUI_config.groups.wegas.modulesByType[module.ws_provides] = moduleName;
1037                     }
1038                 }
1039                 if (module.ix_provides) {                                       // Build a reverse index on which module provides what type (inputEx)
1040                     if (Y.Lang.isArray(module.ix_provides)) {
1041                         for (i = 0; i < module.ix_provides.length; i = i + 1) {
1042                             YUI_config.groups.inputex.modulesByType[module.ix_provides[i]] = moduleName;
1043                         }
1044                     } else {
1045                         YUI_config.groups.inputex.modulesByType[module.ix_provides] = moduleName;
1046                     }
1047                 }
1048 
1049             }
1050         }
1051         group.allModules = allModules;
1052     }
1053     YUI_config.groups.wegas.modulesByType = {};
1054 
1055     for (var i in YUI_config.groups) {
1056         if (i.indexOf("wegas") > -1) {
1057             loadModules(YUI_config.groups[i]);
1058         }
