 13 YUI.add("wegas-inputex-url", function(Y) {
 14     "use strict";
 15 
 16     var inputEx = Y.inputEx;
 17 
 18     /**
 19      * Adds an url regexp, and display the favicon at this url
 20      * @class inputEx.UrlField
 21      * @extends inputEx.StringField
 22      * @constructor
 23      * @param {Object} options inputEx.Field options object
 24      * <ul>
 25      *   <li>favicon: boolean whether the domain favicon.ico should be displayed or not (default is true, except for https)</li>
 26      * </ul>
 27      */
 28     Y.namespace("inputEx.Wegas").UrlField = function(options) {
 29         inputEx.Wegas.UrlField.superclass.constructor.call(this, options);
 30     };
 31 
 32     Y.extend(inputEx.Wegas.UrlField, inputEx.StringField, {
 33         filepanel: null,
 34         filter: null,
 35         /**
 36          * Adds the invalid Url message
 37          * @param {Object} options Options object as passed to the constructor
 38          */
 39         setOptions: function(options) {
 40             inputEx.Wegas.UrlField.superclass.setOptions.call(this, options);
 41             //this.options.favicon = lang.isUndefined(options.favicon) ? (("https:" == document.location.protocol) ? false : true) : options.favicon;
 42             // validate with url regexp
 43             // this.options.regexp = inputEx.regexps.url;
 44         },
 45         /**
 46          *
 47          */
 48         //getValue: function () {
 49         //    return {
 50         //        srcUrl: inputEx.Wegas.UrlField.superclass.getValue.call(this)
 51         //    }
 52         //},
 53         destructor: function() {
 54             if (this.filepanel) {
 55                 this.fileExplorer.destroy();
 56                 this.filepanel.destroy();
 57             }
 58         },
 59         /**
 60          * Adds a img tag before the field to display the favicon
 61          */
 62         render: function() {
 63             inputEx.Wegas.UrlField.superclass.render.call(this);
 64 
 65             this.fieldContainer.classList.add("inputEx-wegas-UrlField");
 66 
 67             this.imgButton = new Y.Button({
 68                 label: "<span class=\"wegas-icon wegas-icon-fileexplorer\"></span>",
 69                 on: {
 70                     click: Y.bind(this.showFileExplorer, this)
 71                 }
 72             }).render(this.fieldContainer);
 73         },
 74         showFileExplorer: function() {
 75             this.filepanel = new Y.Wegas.FileSelect({
 76                 filter: this.filter
 77             });
 78 
 79             this.filepanel.on("*:fileSelected", function(e, path) {
 80                 e.stopImmediatePropagation();
 81                 e.preventDefault();
 82                 this.filepanel.destroy();
 83                 this.setValue(path);
 84             }, this);
 85         }
 86     });
 87 
 88     inputEx.registerType("wegasurl", inputEx.Wegas.UrlField);                   // Register this class as "wegasurl" type
 89 
 90     Y.namespace("inputEx.Wegas").ImageUrlField = function(options) {
 91         inputEx.Wegas.ImageUrlField.superclass.constructor.call(this, options);
 92     };
 93 
 94     Y.extend(inputEx.Wegas.ImageUrlField, inputEx.Wegas.UrlField, {
 95         filter: function() {
 96             return (/image\/|application\/wfs-directory/).test(this.get("data.mimeType"));
 97         },
 98         /**
 99          * Adds a img tag before the field to display the favicon
100          */
101         render: function() {
102             inputEx.Wegas.ImageUrlField.superclass.render.call(this);
103             var cb = new Y.Node(this.divEl);
104             cb.append("<div class=\"preview\"></div>");
105         },
106         setValue: function(val) {
107             inputEx.Wegas.ImageUrlField.superclass.setValue.apply(this, arguments);
108             var cb = new Y.Node(this.divEl),
109                     previewNode = cb.one(".preview");
110             if (val && val.length > 0) {
111                 previewNode.setContent('<img data-file="' + val + '" style="max-width:100%;padding-top: 5px;" />');
112                 Y.Plugin.Injector.parser(previewNode.one("img"));               // Manually run parser, since it is not plugged on the editor
113             }
114         }
115     });
116 
117     inputEx.registerType("wegasimageurl", inputEx.Wegas.ImageUrlField);         // Register this class as "wegasurl" type
118 });
