 13 YUI.add('wegas-panel', function(Y) {
 14     "use strict";
 15     Y.namespace("Wegas").Panel = Y.Base.create("wegas-panel",
 16             Y.Widget,
 17             [Y.WidgetParent,
 18                 Y.WidgetPosition,
 19                 Y.WidgetStdMod,
 20                 Y.WidgetButtons,
 21                 Y.WidgetModality,
 22                 Y.WidgetPositionAlign,
 23                 Y.WidgetStack], {
 24         syncUI: function() {
 25             this.set("content", this.get("content"));
 26         },
 27         exit: function() {
 28             this.get("boundingBox").hide(true);
 29             Y.later(500, this, this.destroy);
 30         }
 31     }, {
 32         CSS_PREFIX: "wegas-panel",
 33         ATTRS: {
 34             align: {"transient": true},
 35             alignOn: {"transient": true},
 36             content: {
 37                 value: "",
 38                 type: "string",
 39                 format: "html",
 40                 setter: function(val) {
 41                     if (val) {
 42                         this.set("bodyContent", val);
 43                     }
 44                     return val;
 45                 }
 46             },
 47             bodyContent: {
 48                 "transient": true
 49             },
 50             buttons: {
 51                 "transient": true,
 52                 value: {
 53                     footer: [
 54                         {
 55                             name: 'proceed',
 56                             label: 'OK',
 57                             action: "exit"
 58                         }
 59                     ]
 60                 }
 61             },
 62             centered: {
 63                 value: true,
 64                 "transient": true
 65             },
 66             defaultButton: {"transient": true},
 67             fillHeight: {"transient": true},
 68             focusOn: {"transient": true},
 69             footerContent: {"transient": true},
 70             headerContent: {"transient": true},
 71             maskNode: {"transient": true},
 72             modal: {
 73                 value: false,
 74                 type: "boolean"
 75             },
 76             shim: {"transient": true},
 77             x: {"transient": true},
 78             xy: {"transient": true},
 79             y: {"transient": true},
 80             zIndex: {value: 100, "transient": true}
 81         }
 82 
 83     });
 84 });
