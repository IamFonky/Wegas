 28             var i, newObj = (o instanceof Array) ? [] : {};

 30                 if (i === 'clone')
 31                     continue;
 32                 if (o[i] && typeof o[i] === "object") {
 33                     newObj[i] = Wegas.Object.clone([i]);
 34                 } else
 35                     newObj[i] = o[i];
