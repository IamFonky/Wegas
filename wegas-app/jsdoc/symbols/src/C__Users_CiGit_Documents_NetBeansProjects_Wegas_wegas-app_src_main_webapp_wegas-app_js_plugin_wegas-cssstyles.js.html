 14 YUI.add('wegas-cssstyles', function(Y) {
 15     "use strict";
 16 
 17     var Wegas = Y.Wegas, CSSStyles;
 18     /**
 19      *  @class Add styles CSS styles
 20      *  @name Y.Plugin.CSSStyles
 21      *  @extends Y.Plugin.Base
 22      *  @constructor
 23      */
 24     CSSStyles = Y.Base.create("wegas-cssstyles", Y.Plugin.Base, [Wegas.Plugin, Wegas.Editable], {
 25         /** @lends Y.Plugin.CSSStyles */
 26 
 27         /**
 28          * Lifecycle methods
 29          * @function
 30          * @private
 31          */
 32         initializer: function() {
 33             this.styleList = [];
 34             if (this.get("host") instanceof Y.Widget) {
 35                 this.onceAfterHostEvent("render", function() {
 36                     this.setValue(this.get("styles"));
 37                 });
 38             } else if (this.get("host") instanceof Y.Node) {
 39                 this.setValue(this.get("styles"));
 40             } else {
 41                 Y.log("Host's type mistmach", "warn", "Y.Plugin.CSSStyles");
 42                 return;
 43             }
 44             this.after("stylesChange", function(e) {
 45                 this.removeStyle(e);
 46                 this.setValue(e.newVal);
 47             });
 48         },
 49         /**
 50          * @function
 51          * @private
 52          * @description remove a style
 53          */
 54         removeStyle: function(e) {
 55             var styleToRemove;
 56             for (styleToRemove in e.prevVal) {
 57                 if (e.prevVal.hasOwnProperty(styleToRemove) && !e.newVal.hasOwnProperty(styleToRemove)) {
 58                     this.nodeStyle(styleToRemove, "");
 59                     this.styleList.splice(Y.Array.indexOf(this.styleList, styleToRemove));
 60                 }
 61             }
 62         },
 63         /**
 64          * @function
 65          * @private
 66          */
 67         setStyle: function(newStylesList, style) {
 68             if (Y.Array.indexOf(this.styleList, style) === -1) {
 69                 this.styleList.push(style);
 70             }
 71             this.nodeStyle(style, newStylesList[style]);
 72         },
 73         /**
 74          * Destructor methods.
 75          * @function
 76          * @private
 77          */
 78         destructor: function() {
 79             var styleToRemove;
 80             for (styleToRemove in this.get("styles")) {
 81                 if (this.get("styles").hasOwnProperty(styleToRemove)) {
 82                     this.nodeStyle(styleToRemove, "");
 83                     this.styleList.splice(Y.Array.indexOf(this.styleList, styleToRemove));
 84                 }
 85             }
 86         },
 87         /**
 88          *
 89          * @private
 90          * @param {type} key Style to edit
 91          * @param {type} value Style's value
 92          * @returns {undefined}
 93          */
 94         nodeStyle: function(key, value) {
 95             if (this.get("host") instanceof Y.Widget) {
 96                 this.get("host").get(this.get("targetNode")).setStyle(key, value);
 97             } else if (this.get("host") instanceof Y.Node) {
 98                 return this.get("host").setStyle(key, value);
 99             }
100         },
101         /**
102          * @function
103          * @private
104          * @description setValue from style
105          */
106         setValue: function(styles) {
107             var style, value;
108             if (styles) {
109                 for (style in styles) {
110                     if (styles.hasOwnProperty(style)) {
111                         value = styles[style];
112                         if (value) {
113                             value = Y.Lang.trim(value);
114                             if (Y.Array.indexOf(CSSStyles.MEASURE_STYLE, style) > -1
115                                     && parseInt(value, 10).toString() === value) {
116                                 styles[style] = value + CSSStyles.MEASURE_SUFFIX;
117 
118                             }
119                         }
120                         this.setStyle(styles, style);
121                     }
122                 }
123             }
124         }
125     }, {
