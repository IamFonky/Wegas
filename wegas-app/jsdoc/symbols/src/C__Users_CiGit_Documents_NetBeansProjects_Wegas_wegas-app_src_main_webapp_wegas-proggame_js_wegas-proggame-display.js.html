 16     var ProgGameDisplay, GRIDSIZE = 32,
 17         speedToFrame = function(speed, x, y, toX, toY) {
 18             var dist;
 19             speed = (speed > 0) ? speed : 1;
 20             dist = Math.sqrt(Crafty.math.squaredDistance(x, y, toX, toY));
 21             return Math.round(((dist / GRIDSIZE) * (1000 / speed))) || 1;
 22         };

119             var object, entity, pos, i;

123                 case "resetLevel":

170                     //        return;
171                     //    }
172                     //    break;
173 
174                 case "Die":
175                 case "Trap":
176                     entity = this.getEntity(command.id);
177                     if (entity && entity["exec" + command.type]) {
178                         entity["exec" + command.type]();
179                         return;
180                     }
181                     break;
182 
183                 case "say":
184                     entity = this.getEntity(command.id);
185                     if (entity && typeof entity.say === 'function') {
186                         entity.say(command.text, command.duration);

188                     }

207                     entity = this.getEntity(command.id);

363                     this.h -= 1;
364                     this.y += 1;
365                 }).bind("TweenEnd", function() {
366                     var col = this.hit("Character");
367                     if (col) {
368                         if (this._currentReelId === "moveRight") {
369                             this.attr({
370                                 x: this._x - 6
371                             }).animate("gzRight");
372                             col[0].obj.attr({
373                                 x: col[0].obj._x + 6
374                             }).animate("gzLeft");
375                         } else { // All other moves. up/down/left
376                             this.attr({
377                                 x: this._x + 6
378                             }).animate("gzLeft");
379                             col[0].obj.attr({
380                                 x: col[0].obj._x - 6
381                             }).animate("gzRight");
382                         }
383                     }
384                 });
385         },

387             var POS = Y.clone(this.__coord);

650     Crafty.c("GridOffset", {
651         init: function() {
652             this.requires("2D");
653             var attr = this.attr;
654             this.attr = function(key, val) {
655                 var i;
656                 if (arguments.length === 2 && Y.Object.hasKey(this._offset, key)) {
657                     return attr.apply(this, [key, val + this._offset[key]]);
658                 }
659                 for (i in this._offset) {
660                     if (this._offset.hasOwnProperty(i)) {
661                         if (key[i]) {
662                             key[i] += this._offset[i];
663                         }
664                     }
665                 }
666                 return attr.apply(this, arguments);
667 
668             };
669             this._offset = {
670                 x: 0,
671                 y: 0
