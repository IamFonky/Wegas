  9 /**
 10  * @fileOverview PageEditor resize Extension
 11  * @author Cyril Junod <cyril.junod at gmail.com>

 14 YUI.add("wegas-pageeditor-resize", function(Y) {
 15     "use strict";
 16     var CONTENTBOX = "contentBox",
 17             BOUNDINGBOX = "boundingBox",
 18             WIDTH = "width",
 19             HEIGHT = "height";
 20     /**
 21      * PageEditor Plugin Extension enabling widget resize.
 22      * @constructor
 23      */
 24     function PageEditorResize() {
 25         Y.onceAfter(this._initResize, this, "render");
 26         Y.after(this._alignResize, this, "showOverlay");
 27         Y.once(this._resizeDestruct, this, "destructor");
 28         Y.after(this._setupIcon, this, "fixedOverlay");
 29     }
 30     PageEditorResize.NAME = "wegas-pageeditor-resize";
 31     PageEditorResize.prototype = {
 32         /* @lends Y.Wegas.PageEditorResize# */
 33         /**
 34          * @private
 35          * @returns {undefined}
 36          */
 37         _initResize: function() {
 38             if (!this.highlightOverlay) {
 39                 Y.error("PageEditorResize is an extension for PageEditor.");
 40             }
 41             this._resizeNode = Y.Node.create("<div class='pageeditor-resizenode'></div>");
 42             this._iconResizeNode = Y.Node.create("<div class='pageeditor-resizenode-icon'></div>");
 43             this._resizeNode.setStyles({
 44                 top: 0,
 45                 left: 0
 46             });
 47             this.highlightOverlay.get(CONTENTBOX).append(this._resizeNode);
 48             this._resizeNode.hide();
 49             this.shownOverlay.get(CONTENTBOX).append(this._iconResizeNode);
 50             this.highlightOverlay.after("visibleChange", function(e) {
 51                 if (e.newVal && this.overlayWidget && this.overlayWidget.CSSSize) {
 52                     this._resizeNode.show();
 53                 } else {
 54                     this._resizeNode.hide();
 55                 }
 56             }, this);
 57             this._resize = new Y.DD.Drag({
 58                 node: this._resizeNode
 59             });
 60             this._resize.plug(Y.Plugin.DDNodeScroll, {
 61                     node: this.get("host").get(CONTENTBOX)
 62                 });
 63                 //    .plug(Y.Plugin.DDConstrained, {
 64                 //    constrain: this.get("host").get(CONTENTBOX),
 65                 //    cacheRegion: false                                              //scroll changes region
 66                 //});
 67             this._resizeNode.before("mousedown", function(e) {
 68                 this.detach();
 69                 this._resizeNode.show();
 70                 this._dd.con.set("constrain", this.get("host").get("widget").get(CONTENTBOX));
 71                 this._resize._widget = this.shownOverlay._widget.get(BOUNDINGBOX);
 72             }, this);
 73             this._resizeNode.after("mouseup", function(e) {
 74                 this.bind();
 75             }, this);
 76             this._resize.on("drag:drag", function(e) {
 77                 var bb = this._resize._widget;
 78                 bb.setStyles({
 79                     width: parseInt(bb.getComputedStyle(WIDTH), 10) + e.info.delta[0],
 80                     height: parseInt(bb.getComputedStyle(HEIGHT), 10) + e.info.delta[1]
 81                 });
 82                 this.fixedOverlay(this.shownOverlay._widget);
 83             }, this);
 84             //var bindedFixedOverlay = Y.bind(this.fixedOverlay, this);
 85 //            this._resize.before("drag:start", function(e) {
 86 //                var bb = this._widget;
 87 //                bb.setStyles({
 88 //                    width: bb.getComputedStyle(WIDTH),
 89 //                    height: bb.getComputedStyle(HEIGHT)
 90 //                });
 91 //                bindedFixedOverlay(Y.Widget.getByNode(bb));
 92 //            });
 93             this._resize.on("drag:end", function(e) {
 94                 var widget = this.shownOverlay._widget, bb = widget.get(BOUNDINGBOX);
 95                 widget.CSSSize.setAttrs({
 96                     styles: {
 97                         width: bb.getComputedStyle(WIDTH),
 98                         height: bb.getComputedStyle(HEIGHT)
 99                     }
100                 });
101                 this._resize._widget = null;
102                 this.fixedOverlay(widget);
103                 this.saveCurrentPage();
104                 this.bind();
105                 this._syncWidgetEdition();
106             }, this);
107         },
108         /**
109          * Align drag node with PageEditor's overlay
110          * @private
111          * @returns {undefined}
112          */
113         _alignResize: function() {
114             var bb = this.shownOverlay.get(BOUNDINGBOX), pos = bb.getXY();
115             if (pos) {  //widget not destroyed
116                 pos[0] = pos[0] + bb.getDOMNode().offsetWidth - 12;
117                 pos[1] = pos[1] + bb.getDOMNode().offsetHeight - 12;
118                 this._resizeNode.setXY(pos);
119             }
120         },
121         _setupIcon: function() {
122             if (this.shownOverlay._widget.CSSSize) {
123                 this._iconResizeNode.show();
124             } else {
125                 this._iconResizeNode.hide();
126             }
127         },
128         /**
129          * self destructor called after PageEditor's destructor
130          * @private
131          * @returns {undefined}
132          */
133         _resizeDestruct: function() {
134             Y.detach(this._alignResize, this, "showOverlay");
135             Y.detach(this._setupIcon, this, "fixedOverlay");
136             this._resize.detachAll();
137             this._resizeNode.detachAll();
138             this._resize.destroy();
139             this._resizeNode.destroy(true);
140             this._iconResizeNode.destroy(true);
141         }
142 
143     };
144     Y.namespace("Wegas").PageEditorResize = PageEditorResize;
145 });
