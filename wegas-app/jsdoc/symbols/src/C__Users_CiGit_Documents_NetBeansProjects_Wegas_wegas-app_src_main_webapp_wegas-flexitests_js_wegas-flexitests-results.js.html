 13 YUI.add("wegas-flexitests-results", function(Y) {
 14     "use strict";
 15     var dateFormatter = function(o) {
 16         return (new Date(+o.value)).toLocaleString();
 17     }, UNWANTED_PROPS = function(item) {
 18         return item !== "date" && item !== "";
 19     }, getChildById = function(widget, id) {
 20         var returnItem = null;
 21         widget.some(function(item) {
 22             if (item.get("id") === id) {
 23                 returnItem = item;
 24                 return true;
 25             }
 26         });
 27         return returnItem;
 28     };
 29     Y.namespace("Wegas").FlexitestsResults = Y.Base.create("wegas-flexitests-results", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 30         CONTENT_TEMPLATE: "<div><div class='config'></div>"
 31                 + "<div class='results'>Collecting results</div>",
 32         syncUI: function() {
 33             var script = "getInstances([";
 34             try {
 35                 script += "'" + this.get("demographics.evaluated").get("name") + "',";
 36                 script += "'" + this.get("variable.evaluated").get("name") + "'";
 37             } catch (e) {
 38             } finally {
 39                 script += "]);";
 40             }
 41 
 42             if (!this.get("simpleMode")) {
 43                 this.get("contentBox").one(".config").setContent("Extracting test page " + this.get("testPage") + " configuration");
 44                 Y.Wegas.Facade.VariableDescriptor.sendRequest({
 45                     request: "/Script/Run/" + Y.Wegas.Facade.Game.get('currentPlayerId'),
 46                     cfg: {
 47                         method: "POST",
 48                         data: Y.JSON.stringify({
 49                             "@class": "Script",
 50                             language: "JavaScript",
 51                             content: script
 52                         })
 53                     },
 54                     on: {
 55                         success: Y.bind(function(e) {
 56                             Y.Wegas.Facade.Page.cache.getPage(this.get("testPage"), Y.bind(function(page) {
 57                                 Y.Wegas.Widget.use(page, Y.bind(function() {
 58                                     this.renderTable(Y.JSON.parse(e.data.response).entities, Y.Wegas.Widget.create(page));
 59                                     this._createConfig(Y.Wegas.Widget.create(page));
 60                                 }, this));
 61                             }, this));
 62 
 63                         }, this),
 64                         failure: Y.bind(function(e) {
 65                             Y.log("error", "Failed to retrieve data", "Y.Wegas.FlexitestsMCQ");
 66                         }, this)
 67                     }
 68                 });
 69 //                Y.Wegas.Facade.Page.cache.getPage(this.get("testPage"), Y.bind(function(page) {
 70 //                    Y.Wegas.Widget.use(page, Y.bind(function() {
 71 //                        this._createConfig(Y.Wegas.Widget.create(page));
 72 //                    }, this));
 73 //                }, this));
 74             } else {
 75                 this.renderResult(this.get("variable.evaluated"));
 76             }
 77         },
 78         renderResult: function(results) {
 79             if (!results) {
 80                 this.get("contentBox").one(".config").setHTML("Unable to find target variable.");
 81                 return;
 82             }
 83 
 84             var props = results.getInstance().get("properties"), i, sum = 0, correct = 0;
 85             for (i in props) {
 86                 if (props.hasOwnProperty(i)) {
 87                     sum += 1;
 88                     if (Y.JSON.parse(props[i]).valid) {
 89                         correct += 1;
 90                     }
 91                 }
 92             }
 93             this.get("contentBox").one(".config").empty();
 94             try {
 95                 this.get("contentBox").one(".results").setContent("Final result : " + correct + " on " + sum + " correct response" + (correct > 1 ? "s" : "") + " (" + (correct * 100 / (sum || 1)).toFixed(1) + "%)");
 96             } catch (e) {
 97             }
 98         },
 99         renderTable: function(results, page) {
100             var demographics = results[0],
101                     tests = results[1],
102                     table = this.get("contentBox").one(".results"),
103                     o, i, j, k, elements = {
104                 left: getChildById(page, "leftElement"),
105                 right: getChildById(page, "rightElement"),
106                 center: getChildById(page, "centerElement")
107             }, extractValue = function(element, qId) {
108                 var el = elements[element],
109                         selectElement = el.item(qId % el.size());
110                 if (selectElement) {
111                     return selectElement.get("content")
112                             || selectElement.get("url");
113                 } else {
114                     return "";
115                 }
116             };
117             delete demographics["@class"];
118             delete tests["@class"];
119             table.empty();
120             for (i in demographics) {
121                 if (demographics.hasOwnProperty(i)) {
122                     o = demographics[i].properties;
123                     j = Y.Array.filter(Y.Object.keys(o), UNWANTED_PROPS);
124                     for (k = 0; k < j.length; k++) {
125                         j[k] = {key: j[k], sortable: true};
126                     }
127                     if (this.resultTable) {
128                         this.resultTable.destroy();
129                     }
130                     this.resultTable = new Y.DataTable({columns: j.concat([
131                             {label: "order", key: "order"},
132                             {label: "Start time", key: "date", sortable: true,
133                                 formatter: dateFormatter
134                             },
135                             {label: "question id", key: "qid", sortable: true},
136                             "left",
137                             "center",
138                             "right",
139                             "response",
140                             {label: "Response time (ms)", key: "delay", sortable: true},
141                             "valid",
142                             {label: "total time (ms)", key: "totalTime"}
143                         ])});
144                     if (j.length !== 0) {
145                         break;
146                     }
147                 }
148             }
149             for (i in tests) {
150                 if (tests.hasOwnProperty(i)) {
151                     for (j in tests[i].properties) {
152                         if (tests[i].properties.hasOwnProperty(j)) {
153                             if (!demographics[i].properties.hasOwnProperty("totalTime")) {
154                                 demographics[i].properties.totalTime = 0;
155                             }
156                             demographics[i].properties.totalTime += Y.JSON.parse(tests[i].properties[j]).delay;
157                         }
158                     }
159                 }
160             }
161             for (i in tests) {
162                 if (tests.hasOwnProperty(i)) {
163                     for (j in tests[i].properties) {
164                         if (tests[i].properties.hasOwnProperty(j)) {
165                             o = Y.JSON.parse(tests[i].properties[j]);
166                             o.left = extractValue(o.left, o.id);
167                             o.right = extractValue(o.right, o.id);
168                             o.center = extractValue("center", o.id);
169                             o.order = j;
170                             /*Remove id as it should be unique*/
171                             o.qid = o.id;
172                             delete o.id;
173                             this.resultTable.addRow(Y.merge(demographics[i].properties, o));
174                         }
175                     }
176                 }
177             }
178 
179             this.resultTable.plug(Y.Wegas.DatatableCSV);
180             this.resultTable.render(table);
181         },
182         _extractConfig: function(widget) {
183             var cfg = {},
184                     getPos = function(element, config) {
185                 var style = element.CSSPosition ? element.CSSPosition.get("styles") : null;
186                 if (style === null) {
187                     return;
188                 } else {
189                     if (Y.Lang.isNumber(parseInt(style.top, 10))) {
190                         config.top = parseInt(style.top, 10);
191                     }
192                     if (Y.Lang.isNumber(parseInt(style.bottom, 10))) {
193                         config.bottom = parseInt(style.bottom, 10);
194                     }
195                     if (Y.Lang.isNumber(parseInt(style.left, 10))) {
196                         config.left = parseInt(style.left, 10);
197                     }
198                     if (Y.Lang.isNumber(parseInt(style.right, 10))) {
199                         config.right = parseInt(style.right, 10);
200                     }
201                 }
202             },
203                     getTimers = function(element, config) {
204                 if (element.hideafter) {
205                     config.hide = element.hideafter.get("time");
206                 }
207                 if (element.showafter) {
208                     config.show = element.showafter.get("time");
209                 }
210             },
211                     left = getChildById(widget, "leftElement"),
212                     center = getChildById(widget, "centerElement"),
213                     right = getChildById(widget, "rightElement");
214             cfg.right = {};
215             cfg.left = {};
216             cfg.center = {};
217             getPos(right, cfg.right);
218             getPos(left, cfg.left);
219             getPos(center, cfg.center);
220             getTimers(right, cfg.right);
221             getTimers(left, cfg.left);
222             getTimers(center, cfg.center);
223             return cfg;
224         },
225         _createConfig: function(cfg) {
226             var cfgTable = this.get("contentBox").one(".config"),
227                     i, cfgProps = [];
228             cfg = this._extractConfig(cfg);
229             cfgTable.empty();
230             for (i in cfg) {
231                 if (cfg.hasOwnProperty(i)) {
232                     cfgProps.push(Y.Object.keys(cfg[i]));
233                     cfg[i].key = i;
234                 }
235             }
236             if (this.configTable) {
237                 this.configTable.destroy();
238             }
239             this.configTable = new Y.DataTable({
240                 columns: [{label: "Element", key: "key"}].concat(Y.Array.dedupe(Y.Array.flatten(cfgProps))),
241                 data: Y.Object.values(cfg)
242             });
243             this.configTable.render(cfgTable);
244         },
245         destructor: function() {
246             if (this.resultTable) {
247                 this.resultTable.destroy();
248             }
249             if (this.configTable) {
250                 this.configTable.destroy();
251             }
252         }
253     }, {
254         EDITORNAME: "Flexitests results",
255         ATTRS: {
256             variable: {
257                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
258                 _inputex: {
259                     _type: "variableselect",
260                     legend: "Test results storage",
261                     description: "Test result storage",
262                     classFilter: ["ObjectDescriptor"]
263                 }
264             }, demographics: {
265                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
266                 _inputex: {
267                     _type: "variableselect",
268                     legend: "Demographic form storage",
269                     description: "Form variable storage",
270                     classFilter: ["ObjectDescriptor"]
271                 }
272             },
273             simpleMode: {
274                 value: false,
275                 type: "boolean",
276                 _inputex: {
277                     label: "Player mode"
278                 }
279             },
280             testPage: {
281                 type: "string",
282                 value: "3",
283                 _inputex: {
284                     label: "Test Page",
285                     _type: "pageselect",
286                     required: false,
287                     description: "Used to extract configuration"
288                 }
289             }
290 
291         }
292     });
293 });
