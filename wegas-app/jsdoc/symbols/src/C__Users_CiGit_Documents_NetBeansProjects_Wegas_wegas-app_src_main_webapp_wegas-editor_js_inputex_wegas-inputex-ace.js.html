 13 YUI.add('wegas-inputex-ace', function(Y) {
 14     "use strict";
 15 
 16     var inputEx = Y.inputEx;
 17 
 18     /**
 19      * Ace code editor field
 20      */
 21     inputEx.AceField = function(options) {
 22         inputEx.AceField.superclass.constructor.call(this, options);
 23     };
 24 
 25     Y.extend(inputEx.AceField, inputEx.Textarea, {
 26         /**
 27          * Set the default values of the options
 28          * @param {Object} options Options object as passed to the constructor
 29          */
 30         setOptions: function(options) {
 31             inputEx.AceField.superclass.setOptions.call(this, options);
 32 
 33             this.options.language = options.language || "javascript";
 34             this.options.theme = options.theme || "textmate";
 35             this.options.height = options.height || "75px";
 36         },
 37         disable: function() {
 38             this.editor.setReadOnly(true);
 39         },
 40         enable: function() {
 41             this.editor.setReadOnly(false);
 42         },
 43         /**
 44          * Render the field using the YUI Editor widget
 45          */
 46         renderComponent: function() {
 47             if (window.ace) {                                                   // Ace is present, launch it
 48                 this.el = Y.Node.create('<div>' + (this.options.value || "") + '</div>');//@fixme
 49                 this.fieldContainer.appendChild(this.el.getDOMNode());
 50 
 51                 this.el.setStyle("height", this.options.height);
 52                 this.editor = ace.edit(this.el.getDOMNode());
 53                 this.editor.setTheme("ace/theme/" + this.options.theme);
 54                 this.editor.setShowPrintMargin(false);
 55                 //this.editor.setOptions({
 56                 //    //enableBasicAutocompletion: true,
 57                 //    enableSnippets: true
 58                 //});
 59                 //this.editor.setHighlightActiveLine(false);
 60                 //this.editor.renderer.setHScrollBarAlwaysVisible(false);
 61 
 62                 this.session = this.editor.getSession();
 63                 this.session.setMode("ace/mode/" + this.options.language);
 64 
 65                 Y.Wegas.app.after("layout:resize", function() {                 // Every time the layout is resized (wegas editor only)
 66                     Y.once('domready', this.resize, this);                      // resize ace viewport
 67                 }, this.editor);
 68 
 69                 Y.after('windowresize', Y.bind(this.editor.resize, this.editor));// Every time window is resized, resize ace viewport
 70             } else {                                                            // Fallback with textarea
 71                 Y.log("Unable to find Ace libraries, falling back to text field", "error", "Wegas.Inputex.Ace");
 72                 inputEx.AceField.superclass.renderComponent.call(this);
 73             }
 74         },
 75         /**
 76          * Set the html content
 77          * @param {String} value The html string
 78          * @param {boolean} [sendUpdatedEvt] (optional) Wether this setValue should fire the 'updated' event or not (default is true, pass false to NOT send the event)
 79          */
 80         setValue: function(value, sendUpdatedEvt) {
 81             if (this.session) {
 82                 this.session.setValue(value);
 83                 if (sendUpdatedEvt !== false) {
 84                     // fire update event
 85                     this.fireUpdatedEvt();
 86                 }
 87             } else {    // fallback
 88                 return inputEx.AceField.superclass.setValue.apply(this, arguments);
 89             }
 90         },
 91         /**
 92          * Get the ace content
 93          * @return {String} the ace area content string
 94          */
 95         getValue: function() {
 96             if (this.session) {
 97                 return this.session.getValue();
 98             } else {
 99                 return inputEx.AceField.superclass.getValue.apply(this);
100             }
101         }
102     });
103 
104     inputEx.registerType("ace", inputEx.AceField, []);                          // Register this class as "html" type
105 });
