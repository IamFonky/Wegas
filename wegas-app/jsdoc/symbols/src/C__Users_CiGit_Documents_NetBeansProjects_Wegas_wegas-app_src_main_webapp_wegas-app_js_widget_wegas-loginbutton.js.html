 12 YUI.add("wegas-loginbutton", function(Y) {
 13     "use strict";
 14 
 15     /**
 16      * @name Y.Wegas.LoginButton
 17      * @extends Y.Wegas.Button
 18      * @class  Button with a defined behavior.
 19      * @constructor
 20      * @description Button with special label and menu with two
 21      * options : set user preferences or logout
 22      */
 23     var LoginButton = Y.Base.create("wegas-login", Y.Wegas.Button, [], {
 24         /** @lends Y.Wegas.LoginButton# */
 25 
 26         // *** Lifecycle Methods *** //
 27         /**
 28          * @function
 29          * @private
 30          * @description bind function to events.
 31          * Call widget parent to execute its proper bind function.
 32          * When UserFacade is updated, do syncUI
 33          * Add plugin menu with 2 options : open page "user preferences" and logout
 34          */
 35         bindUI: function() {
 36             Y.Wegas.LoginButton.superclass.bindUI.apply(this, arguments);
 37             
 38             this.handlers = {};
 39             this.handlers.userUpdate = Y.Wegas.Facade.User.after("update", this.syncUI, this);
 40             if (Y.Wegas.Facade.VariableDescriptor)
 41                 this.handlers.variableUpdate = Y.Wegas.Facade.VariableDescriptor.after("update", this.syncUI, this);
 42 
 43             if (this.menu) {                                                    // Don't add the plugin if it already exist.
 44                 return;
 45             }
 46 
 47             this.plug(Y.Plugin.WidgetMenu, {
 48                 children: [{
 49                         type: "Button",
 50                         label: "Preferences",
 51                         plugins: [{
 52                                 fn: "OpenPageAction",
 53                                 cfg: {
 54                                     subpageId: this.get("preferencePageId"), // @fixme
 55                                     targetPageLoaderId: this.get("targetPageLoader")
 56                                 }
 57                             }]
 58                     }, {
 59                         type: "Button",
 60                         label: "Logout",
 61                         plugins: [{
 62                                 fn: "OpenUrlAction",
 63                                 cfg: {
 64                                     url: "logout",
 65                                     target: "self"
 66                                 }
 67                             }]
 68                     }]
 69             });
 70         },
 71         /**
 72          * @function
 73          * @private
 74          * @description Call widget parent to execute its proper sync function.
 75          * Set label of this button with team and/or player name.
 76          */
 77         syncUI: function() {
 78             Y.Wegas.LoginButton.superclass.syncUI.apply(this, arguments);
 79 
 80             var cUser = Y.Wegas.Facade.User.get("currentUser"),
 81                     cPlayer = Y.Wegas.Facade.Game.cache.getCurrentPlayer(),
 82                     cTeam = Y.Wegas.Facade.Game.cache.getCurrentTeam(),
 83                     name = cUser.get("name") || "undefined",
 84                     mainAccount = cUser.getMainAccount(),
 85                     gameModel = Y.Wegas.Facade.GameModel.cache.getCurrentGameModel();
 86 
 87             if (mainAccount) {
 88                 name = "<img src=\"http://www.gravatar.com/avatar/" + mainAccount.get("hash") + "?s=28&d=mm\" />" + name;
 89             }
 90 
 91             if (mainAccount instanceof Y.Wegas.persistence.GuestJpaAccount) {   // If current account is a Guest,
 92                 this.menu.getMenu().item(0).hide();                             // hide the "Preference" button
 93             }
 94 
 95             if (!this.get('labelIsUser')) {
 96                 if (cPlayer) {
 97                     name = cPlayer.get("name");
 98                 }
 99                 if (cTeam) {
100                     name = cTeam.get("name") + " : " + name;
101                 }
102                 if (gameModel && gameModel.get("properties.freeForAll")) {
103                     name = cPlayer.get("name");
104                 }
105             }
106             if (!this.get("label")) {
107                 this.set("label", name);
108             }
109         },
110         destructor: function() {
111             for (var k in this.handlers) {
112                 this.handlers[k].detach();
113             }
114         }
115     }, {
116         /** @lends Y.Wegas.LoginButton */
117 
118         /**
119          * @field
120          * @static
121          * @description
122          * <p><strong>Attributes</strong></p>
123          * <ul>
124          *    <li>labelIsUser: Select what kind of label you want (user/team  or team/player)</li>
125          *    <li>preferencePageId: Id of the the page which contains widget userPreferences</li>
126          *    <li>targetPageLoader: Zone to display the page which contains widget userPreferences</li>
127          * </ul>
128          */
129         ATTRS: {
130             label: {
131                 "transient": true
132             },
133             /**
134              * Select what kind of label you want (user/team  or team/player)
135              */
136             labelIsUser: {
137                 value: false,
138                 validator: function(b) {
139                     return (b === 'true' || b === true);
140                 }
141             },
142             /**
143              * Id of the the page which contains widget userPreference
144              */
145             preferencePageId: {
146                 value: 1000                                                     //@fixme
147             },
148             /**
149              * targetPageLoader: Zone to display the page which contains widget userPreferences
150              */
151             targetPageLoader: {
152                 value: "maindisplayarea"
153             }
154         }
155     });
156     Y.namespace('Wegas').LoginButton = LoginButton;
157 
158 });
