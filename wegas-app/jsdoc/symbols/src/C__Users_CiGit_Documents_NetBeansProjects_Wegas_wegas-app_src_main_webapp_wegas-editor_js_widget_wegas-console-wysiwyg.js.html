 12 YUI.add('wegas-console-wysiwyg', function(Y) {
 13     'use strict';
 14 
 15     /**
 16      *  @class wysiwyg console for impacts 
 17      *  @name Y.Wegas.WysiwygConsole
 18      *  @extends Y.Wegas.Console
 19      *  @constructor
 20      */
 21     var CONTENTBOX = 'contentBox',
 22             WysiwygConsole, Plugin = Y.Plugin;
 23 
 24     WysiwygConsole = Y.Base.create("wegas-console-wysiwyg", Y.Wegas.Console, [Y.WidgetChild, Y.Wegas.Widget], {
 25         /**
 26          * @lends Y.Wegas.WysiwygConsole#
 27          */
 28         // ** Lifecycle Methods ** //
 29         /**
 30          * @function
 31          * @private
 32          * @description Set variables with initials values.
 33          */
 34         initializer: function() {
 35             this.handlers = [];
 36         },
 37         /**
 38          * @function
 39          * @private
 40          * @description create and render the Y.inputEx.WysiwygScript.
 41          */
 42         renderUI: function() {
 43             var cb = this.get(CONTENTBOX);
 44 
 45             this.get("boundingBox").addClass("wegas-form");
 46 
 47             this.plug(Plugin.WidgetToolbar);
 48 
 49             this.srcField = new Y.inputEx.WysiwygScript({
 50                 parentEl: cb
 51             });
 52             cb.append('<div class="results"></div>');
 53             this.srcField.el.rows = 8;
 54             this.srcField.el.cols = 100;
 55 
 56             this.renderRunButton();
 57 
 58             this.srcField.viewSrc.get("boundingBox").removeClass("inputEx-WysiwigScript-viewsrc");
 59 
 60             this.srcField.viewSrc.get("contentBox").append("Source");
 61             this.toolbar.add(this.srcField.viewSrc);
 62         },
 63         /**
 64          * @function
 65          * @private
 66          * @description bind function to events.
 67          * When parent tab change plug or unplug multiple selection plugin.
 68          */
 69         bindUI: function() {
 70             var treeView, editorTreeView = Y.Widget.getByNode("#leftTabView .wegas-editor-treeview-team"),
 71                     cGameModel = Y.Wegas.Facade.GameModel.cache.getCurrentGameModel(), i,
 72                     playerId, selected = 0;
 73 
 74             if (!editorTreeView) {
 75                 return;
 76             }
 77 
 78             treeView = editorTreeView.treeView;
 79 
 80             this.handlers.push(this.get("parent").on("selectedChange", function(e) {
 81                 selected = 0;
 82                 if (e.newVal !== 1) {
 83                     treeView.unplug(Plugin.CheckBoxTV);
 84                     this.removeCheckbox();
 85                     for (i = 0; i < treeView.size(); i += 1) {
 86                         if (treeView.item(i).get("selected")) {
 87                             selected = i;
 88                             break;
 89                         }
 90                     }
 91                     treeView.deselectAll();
 92                     // Check if a player or team is selected
 93                     if (!treeView.size()) {
 94                         return;
 95                     }
 96 
 97                     // Select only first team or player
 98                     if (cGameModel.get("properties.freeForAll")) {
 99                         playerId = treeView.item(selected).get("data").entity.get("id");
100                         treeView.item(selected).set("selected", 2);
101                     } else {
102                         while (!playerId) {
103                             if (treeView.item(selected).get("data").entity.get("players").length !== 0) {
104                                 playerId = treeView.item(selected).get("data").entity.get("players")[0].get("id");
105                             } else {
106                                 selected += 1;
107                             }
108                         }
109                         treeView.item(selected).selectAll();
110                     }
111                     Y.Wegas.Facade.Game.cache.set('currentPlayerId', playerId);
112                     playerId = null;
113                 } else {
114                     treeView.plug(Plugin.CheckBoxTV);
115                     this.addCheckbox();
116                 }
117             }, this));
118 
119         },
120         /**
121          * @function
122          * @private
123          * @description Create and render the button for run the script.
124          */
125         renderRunButton: function() {
126             var el = this.toolbar.get('header'), multiPlayerScript, playerList;
127 
128             this.runButton = new Y.Button({
129                 label: "<span class=\"wegas-icon wegas-icon-play\"></span>Run",
130                 on: {
131                     click: Y.bind(function() {
132                         playerList = this.playerList();
133                         if (playerList.length === 0) {
134                             return;
135                         }
136                         multiPlayerScript = {
137                             playerIdList: playerList,
138                             script: {
139                                 "@class": "Script",
140                                 language: "JavaScript",
141                                 content: this.srcField.getValue().content
142                             }
143                         };
144                         this.multiExecuteScript(multiPlayerScript);
145                     }, this)
146                 }
147             }).render(el);
148         },
149         /**
150          * @function
151          * @private
152          * @description Gives the list of teams or player selected in the
153          * treeview. If no treeview, only the current player is added in the list.
154          */
155         playerList: function() {
156             var treeview = Y.Widget.getByNode("#leftTabView .wegas-editor-treeview-team .yui3-treeview-content"),
157                     playerList = [], selection;
158 
159             if (!treeview) {
160                 playerList.push(Y.Wegas.Facade.Game.get('currentPlayerId'));
161                 return playerList;
162             }
163 
164             selection = treeview.get("selection") || new Y.ArrayList();
165 
166             if (Y.Wegas.Facade.GameModel.cache.getCurrentGameModel().get("properties.freeForAll")) {
167                 if (!selection.size()) {
168                     this.showMessageBis("info", "No player is selected.");
169                 }
170                 selection.each(function(item) {
171                     playerList.push(item.get("data.entity").get("id"));
172                 });
173             } else {
174                 if (!selection.size()) {
175                     this.showMessageBis("info", "No team is selected! This impact has not been applied");
176                 }
177                 selection.each(function(item) {
178                     var entity = item.get("data.entity");
179                     if (entity.get("players").length > 0) {
180                         playerList.push(entity.get("players")[0].get("id"));
181                     }
182                 });
183             }
184             return playerList;
185         },
186         /**
187          * @function
188          * @private
189          * @description adds the necessary elements for display the checkbox 
190          * and a button for select or deselct all teams/player with the corresponding events.
191          */
192         addCheckbox: function() {
193             var editorTreeview = Y.Widget.getByNode("#leftTabView .wegas-editor-treeview-team"), i;
194             this.emptyTeamList();
195             this.selectAll = new Y.Node.create("<span class='emptyCheckbox selectAll'>Select all</span>");
196             editorTreeview.toolbar.get("header").append(this.selectAll);
197             this.selectAll.on("click", function(e, editorTreeview) {
198                 if (this.selectAll.hasClass("yui3-treenode-selected")) {
199                     this.selectAll.removeClass("yui3-treenode-selected");
200                     editorTreeview.treeView.deselectAll();
201                 } else {
202                     this.selectAll.addClass("yui3-treenode-selected");
203                     editorTreeview.treeView.selectAll();
204                     for (i = 0; i < this.emptyTeam.length; i += 1) {
205                         editorTreeview.treeView.item(this.emptyTeam[i]).set("selected", 0);
206                     }
207                 }
208             }, this, editorTreeview);
209 
210             this.nodeClick = editorTreeview.treeView.on("nodeClick", function(e) {
211                 if (e.currentTarget.get("selection") && e.currentTarget.get("selection").size() >= (e.currentTarget.size() - this.emptyTeam.length)) {
212                     this.selectAll.addClass("yui3-treenode-selected");
213                 } else {
214                     this.selectAll.removeClass("yui3-treenode-selected");
215                 }
216                 if (e.node.get("data").entity.get("players") && !e.node.get("data").entity.get("players").length) {
217                     e.node.deselectAll();
218                 }
219             }, this);
220         },
221         /**
222          * @function
223          * @private
224          * @description removes all elements corresponding to checkboxes.
225          */
226         removeCheckbox: function() {
227             this.selectAll.remove();
228             this.nodeClick.detach();
229         },
230         /**
231          * @function
232          * @private
233          * @description checks if all teams has a player otherwise add a "noPlayer" class.
234          */
235         emptyTeamList: function() {
236             var editorTreeview = Y.Widget.getByNode("#leftTabView .wegas-editor-treeview-team"),
237                     i;
238             this.emptyTeam = [];
239             for (i = 0; i < editorTreeview.treeView.size(); i += 1) {
240                 if (editorTreeview.treeView.item(i).get("data").entity.get("players") && !editorTreeview.treeView.item(i).get("data").entity.get("players").length) {
241                     this.emptyTeam.push(i);
242                 }
243             }
244         },
245         /**
246          * @function
247          * @private
248          * @description Detach all functions created by this widget.
249          */
250         destructor: function() {
251             var i;
252             for (i = 0; i < this.handlers.length; i += 1) {
253                 this.handlers[i].detach();
254             }
255         }
256     });
257 
258     Y.namespace('Wegas').WysiwygConsole = WysiwygConsole;
259 });
