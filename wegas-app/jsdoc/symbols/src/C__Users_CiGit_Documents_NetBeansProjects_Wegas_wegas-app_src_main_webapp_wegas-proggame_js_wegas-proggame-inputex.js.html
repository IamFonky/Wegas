 11 YUI.add('wegas-proggame-inputex', function(Y) {
 12     "use strict";
 13 
 14     var inputEx = Y.inputEx,
 15             Alignable = Y.Base.create("proggame-alignable", Y.Widget, [Y.WidgetPosition, Y.WidgetPositionAlign, Y.WidgetStack, Y.WidgetPositionConstrain]),
 16             TILESIZE = 32;
 17 
 18     /**
 19      * ProgGameMap field
 20      */
 21     inputEx.ProgGameMap = function(options) {
 22         inputEx.ProgGameMap.superclass.constructor.call(this, options);
 23     };
 24 
 25     Y.extend(inputEx.ProgGameMap, inputEx.ListField, {
 26         setOptions: function(options) {
 27             options.elementType = {
 28                 type: "list",
 29                 useButtons: false,
 30                 elementType: {
 31                     type: "proggametile",
 32                     value: {x: 0, y: 0}
 33                 }
 34             };
 35             options.wrapperClassName = "inputEx-fieldWrapper wegas-inputex-proggamemap";
 36             inputEx.ProgGameMap.superclass.setOptions.call(this, options);
 37         },
 38         renderComponent: function() {
 39             inputEx.ProgGameMap.superclass.renderComponent.call(this);
 40 
 41             var node = new Y.Node(this.divEl),
 42                     addButtonNode = new Y.Node(this.addButton);                 // The add column link
 43 
 44             node.append("<div class=\"add-col\"><img />Add column</div>"),
 45                     node.one(".add-col").on("click", function(e) {
 46                 var i;
 47                 for (i = 0; i < this.subFields.length; i += 1) {
 48                     this.subFields[i].addElement();
 49                 }
 50             }, this);
 51             addButtonNode.wrap("<div class=\"add-row\"></div>");
 52             addButtonNode.get("parentNode")
 53                     .append("Add row")
 54                     .on("click", this.onAddButton, this);
 55             node.delegate("mousedown", function(e) {
 56                 e.halt(true);
 57                 this._paint = {
 58                     start: true,
 59                     x: Math.abs(parseInt(e.target.getStyle("backgroundPositionX"), 10) / TILESIZE),
 60                     y: Math.abs(parseInt(e.target.getStyle("backgroundPositionY"), 10) / TILESIZE)
 61                 };
 62             }, ".inputex-proggametile", this);
 63             node.delegate("mouseenter", function(e) {
 64                 e.halt(true);
 65                 if (this._paint && this._paint.start) {
 66                     e.target.fire("newValue", this._paint);
 67                 }
 68             }, ".inputex-proggametile", this);
 69             node.delegate("mouseup", function() {
 70                 this._paint = {start: false};
 71             }, ".inputex-proggametile", this);
 72         },
 73         onAddButton: function(e) {
 74             e.halt();
 75 
 76             // Prevent adding a new field if already at maxItems
 77             if (Y.Lang.isNumber(this.options.maxItems) && this.subFields.length >= this.options.maxItems) {
 78                 return;
 79             }
 80 
 81             var j, defaultValue = [],
 82                     mapWidth = (this.subFields.length > 0) ? this.subFields[0].getValue().length : 0;
 83 
 84             for (j = 0; j < mapWidth; j += 1) {
 85                 defaultValue.push({x: 0, y: 0});
 86             }
 87 
 88             // Add a field with no value:
 89             var subFieldEl = this.addElement(defaultValue);
 90 
 91             // Focus on this field
 92             subFieldEl.focus();
 93 
 94             // Fire updated !
 95             this.fireUpdatedEvt();
 96         }
 97     });
 98     inputEx.registerType("proggamemap", inputEx.ProgGameMap);                  // Register this class as "proggamemap" type
 99 
100     /**
101      * ProgGameTile field
102      */
103     inputEx.ProgGameTile = function(options) {
104         inputEx.ProgGameTile.superclass.constructor.call(this, options);
105     };
106 
107     Y.extend(inputEx.ProgGameTile, inputEx.Field, {
108         setValue: function(value, sendUpdatedEvent) {
109             inputEx.ProgGameTile.superclass.setValue.call(this, value, sendUpdatedEvent);
110             this.options.value = value || {x: 0, y: 0};
111             if (typeof value.x !== 'number')
112                 return;
113 
114             this.node.setStyles({
115                 backgroundPositionX: "-" + (this.options.value.x * TILESIZE) + "px",
116                 backgroundPositionY: "-" + (this.options.value.y * TILESIZE) + "px"
117             });
118         },
119         getValue: function() {
120             return this.options.value;
121         },
122         renderComponent: function() {
123             this.node = Y.Node.create("<div class=\"inputex-proggametile\"><div></div></div>");
124             //this.node.setStyles({
125             //      background-position:
126             //});
127             this.node.on("newValue", function(e) {
128                 this.setValue({x: e.x, y: e.y});
129             }, this);
130             this.node.after("click", function() {
131                 if (!this.alignable) {
132                     var i, j;
133 
134                     this.alignable = new Alignable({
135                         align: {
136                             node: this.node,
137                             points: ["tl", "bl"]
138                         },
139                         constrain: true,
140                         zIndex: 100
141                     });
142                     var cb = this.alignable.get("contentBox"),
143                             ct = ["<div class=\"table\">"],
144                             spriteSheet = Y.Wegas.ProgGameDisplay.SPRITESHEETS["TileSprite"];
145 
146                     for (i = 0; i < spriteSheet.height; i += 1) {
147                         ct.push("<div>");
148                         for (j = 0; j < spriteSheet.width; j += 1) {
149                             ct.push("<div data-position-x=\"" + j + "\" data-position-y=\"" + i + "\" ></div>");
150                         }
151                         ct.push("</div>");
152                     }
153                     ct.push("</div>");
154 
155 
156                     cb.set("innerHTML", ct.join(""));
157                     this.alignable.render();
158 
159                     cb.delegate("click", function(e) {
160                         this.setValue({
161                             x: +e.target.getAttribute("data-position-x"),
162                             y: +e.target.getAttribute("data-position-y")
163                         });
164                         this.alignable.hide();
165                         this.outHandler.detach();
166                     }, ".table div div", this);
167                 }
168 
169                 Y.on("domready", function() {
170                     this.outHandler = this.alignable.get("boundingBox").on("clickoutside", function() {
171                         if (this.alignable.get("visible")) {
172                             this.alignable.hide();
173                             this.outHandler.detach();
174                         }
175                     }, this);
176                 }, this);
177                 this.alignable.show();
178             }, this);
179             this.fieldContainer.appendChild(this.node.getDOMNode());
180         }
181     });
182 
183     inputEx.registerType("proggametile", inputEx.ProgGameTile, []);             // Register this class as "proggametile" type
184 });
