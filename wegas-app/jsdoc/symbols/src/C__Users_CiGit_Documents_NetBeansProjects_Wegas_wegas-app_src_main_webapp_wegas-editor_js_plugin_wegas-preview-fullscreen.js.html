 12 YUI.add("wegas-preview-fullscreen", function(Y) {
 13     "use strict";
 14 
 15     Y.namespace("Plugin").PreviewFullScreen = Y.Base.create("wegas-preview-fullscreen", Y.Plugin.Base, [], {
 16         initializer: function() {
 17             this.afterHostEvent("render", this.render);
 18         },
 19         render: function() {
 20             var host = this.get('host');
 21 
 22             if (host.toolbar) {
 23                 this.swapNode = Y.Node.create("<span></span>");
 24                 Y.one("body").append(this.swapNode);
 25                 //this.fullScreenButton = host.toolbar.add(new Y.ToggleButton({
 26                 //    label: "<span class='wegas-icon wegas-icon-fullscreen'></span>Fullscreen"
 27                 //})).item(0);
 28 
 29                 this.fullScreenButton = new Y.ToggleButton({
 30                     label: "<span class='wegas-icon wegas-icon-fullscreen'></span>Fullscreen"
 31                 });
 32                 this.fullScreenButton.render(host.toolbar.get("header"));
 33 
 34                 this.fullScreenButton.after("pressedChange", function() {
 35                     this.get("host").get("boundingBox").swap(this.swapNode);
 36                 }, this);
 37             }
 38         },
 39         destructor: function() {
 40             if (this.fullScreenButton) {
 41                 this.fullScreenButton.destroy();
 42             }
 43             if (this.swapNode) {
 44                 this.swapNode.destroy(true);
 45             }
 46         }
 47     }, {
 48         NS: "preview"
 49     });
 50 });
