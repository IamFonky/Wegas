 12 YUI.add('wegas-pusher-connector', function(Y) {
 13     "use strict";
 14     /**
 15      * PusherConnector singleton for each applicationKey
 16      * @name Y.Wegas.util.PusherConnector
 17      * @constructor
 18      * @param {Object} config, requires applicationKey
 19      * @returns {Instance}
 20      */
 21     var PusherDataSource = function() {
 22         PusherDataSource.superclass.constructor.apply(this, arguments);
 23     };
 24 
 25     Y.extend(PusherDataSource, Y.Wegas.DataSource, {
 26         /* @lends Y.Wegas.util.PusherConnector# */
 27         /*
 28          * life cycle method
 29          * @private
 30          * @function
 31          * @param {Object} cfg
 32          * @returns {undefined}
 33          */
 34         initializer: function(cfg) {
 35 
 36             this.pusherInit(cfg);
 37             //this.pusher = new Pusher('732a1df75d93d028e4f9');
 38 
 39         },
 40         pusherInit: function(cfg) {
 41             if (!window.Pusher) {
 42                 //Y.later(100, this.pusherInit, this, cfg);
 43                 return;
 44             }
 45             Pusher.log = Y.log;    // Enable pusher logging - don't include this in production
 46             document.WEB_SOCKET_DEBUG = true;// Flash fallback logging - don't include this in production
 47             this.pusher = new Pusher(cfg["applicationKey"]);
 48             this.pusher.connection.bind('error', function(err) {
 49                 if (err.data && err.data.code === 4004) {
 50                     Y.log("Pusher daily limit", "error", "Y.Wegas.util.PusherConnector");
 51                 }
 52             });
 53             this.gameChannel = this.pusher.subscribe('Game-' + Y.Wegas.Facade.Game.get("currentGameId"));
 54             this.teamChannel = this.pusher.subscribe('Team-' + Y.Wegas.Facade.Game.get("currentTeamId"));
 55             this.playerChannel = this.pusher.subscribe('Player-' + Y.Wegas.Facade.Game.get("currentPlayerId"));
 56             this.pusher.bind_all(Y.bind(this.eventReceived, this));
 57         },
 58         /**
 59          * @function
 60          * @private
 61          * @param {type} event
 62          * @param {type} data
 63          * @returns {undefined}
 64          */
 65         eventReceived: function(event, data) {
 66             if (event.indexOf("pusher:") !== 0) {                               //ignore pusher specific event
 67                 this.publish(event, {
 68                     emitFacade: false
 69                 });
 70                 this.fire(event, data);
 71             }
 72         },
 73         /**
 74          * @function
 75          * @public
 76          * @param {String} channel (Game | Team | Player)
 77          * @param {Object} data to send
 78          * @param {String} event to send
 79          * @returns {undefined}
 80          */
 81         triggerCustomEvent: function(channel, data, event) {
 82             var id;
 83             if (channel === "Game") {
 84                 id = Y.Wegas.Facade.Game.get("currentGameId");
 85             } else if (channel === "Team") {
 86                 id = Y.Wegas.Facade.Game.get("currentTeamId");
 87             } else {
 88                 id = Y.Wegas.Facade.Game.get("currentPlayerId");
 89             }
 90             this.sendRequest({
 91                 request: "Send/" + channel + "/" + id + "/" + event,
 92                 cfg: {
 93                     method: "POST",
 94                     headers: {
 95                         'Content-Type': 'application/json',
 96                         'Managed-Mode': 'false'
 97                     },
 98                     data: data
 99                 }
100             });
101         },
102         /*
103          * life cycle method
104          * @private
105          * @function
106          * @returns {undefined}
107          */
108         destructor: function() {
109             this.pusher.disconnect();
110 //            delete this.constructor.INSTANCES[this.get("applicationKey")];
111         }
112     }, {/* @lends Y.Wegas.util.PusherConnector */
113         /**
114          * Store running instances.
115          * @private
116          * @field
117          */
118         INSTANCES: {},
119         ATTRS: {
120             applicationKey: {
121                 initOnly: true,
122                 validator: Y.Lang.isString
123             }
124         }
125     });
126 
127     Y.namespace('Wegas').PusherDataSource = PusherDataSource;
128     //new PusherConnectorFactory({applicationKey: "732a1df75d93d028e4f9"});
129 });
