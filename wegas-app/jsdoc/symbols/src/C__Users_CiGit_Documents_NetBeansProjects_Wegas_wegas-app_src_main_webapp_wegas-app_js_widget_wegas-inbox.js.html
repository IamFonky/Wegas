 12 YUI.add('wegas-inbox', function(Y) {
 13     "use strict";
 14 
 15     var CONTENTBOX = 'contentBox',
 16             InboxDisplay,
 17             /**
 18              * define tab and content Templates, add new ones :
 19              * TEMPLATES:{TEMPLATE_NAME:{tab:'template', content:'template'}}
 20              *
 21              * specify the one to use in "template" ATTRS
 22              * !!COMPILED!! Template may be added after initialization in
 23              * Y.Wegas.InboxDisplay.prototype.TEMPLATES['name'] =
 24              * {tab: Y.Template.Micro.compile(tab_text_template),
 25              * content: Y.Template.Micro.compile(content_text_template)}
 26              */
 27             TEMPLATES = {
 28                 inbox: {
 29                     tab: "<div class=' <%=(this.get('unread') ? 'unread' : 'read')%>'><div class='left'><%= this.get('from') || '<i>No sender</i>' %> </div><div class='right'><%=this.get('subject')%></div></div>",
 30                     content: "<div class='msg-header'><div class='msg-subject'>Subject: <%=this.get('subject')%></div><div class='msg-from'>From: <%= this.get('from') || '<i>No sender</i>' %></div><div class='msg-attachement'><%= (this.get('attachements') && this.get('attachements').length) ? 'Attachements: ':'' %> <% Y.Array.each(this.get('attachements'), function(a){ %><a href='<%= a %>' data-file='<%= a %>'><%= a %></a>;<% }); %></div></div><div class='msg-body'><%== this.get('body') %></div>"
 31                 },
 32                 /**
 33                  * inbox without labels.
 34                  */
 35                 clean: {
 36                     tab: "<div class=' <%=(this.get('unread') ? 'unread' : 'read')%>'><div class='left'><%= this.get('from') || '<i>No sender</i>' %> </div><div class='right'><%=this.get('subject')%></div></div>",
 37                     content: "<div class='msg-header'><div class='msg-subject'><%=this.get('subject')%></div><div class='msg-from'><%= this.get('from') || '<i>No sender</i>' %></div><div class='msg-attachement'><% Y.Array.each(this.get('attachements'), function(a){ %><a href='<%= a %>' data-file='<%= a %>'><%= a %></a>;<% }); %></div></div><div class='msg-body'><%== this.get('body') %></div>"
 38                 }
 39             };
 40     /**
 41      * @name Y.Wegas.InboxDisplay
 42      * @extends Y.Widget
 43      * @borrows Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable
 44      * @class  class to manage e-mail
 45      * @constructor
 46      * @description Display and allow to manage e-mail sent to the current player
 47      */
 48     InboxDisplay = Y.Base.create("wegas-inbox", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 49         /**
 50          * @lends Y.Wegas.InboxDisplay#
 51          */
 52         // ** Private fields ** //
 53         /**
 54          *
 55          */
 56         readRequestTid: null,
 57         /**
 58          * Holds compiled templates. Add templates in scope variable TEMPLATES
 59          */
 60         TEMPLATES: (function() {
 61             var i, j, ret = {};
 62             for (i in TEMPLATES) {
 63                 if (TEMPLATES.hasOwnProperty(i)) {
 64                     ret[i] = {};
 65                     for (j in TEMPLATES[i]) {
 66                         if (TEMPLATES[i].hasOwnProperty(j)) {
 67                             ret[i][j] = Y.Template.Micro.compile(TEMPLATES[i][j]);
 68                         }
 69                     }
 70                 }
 71             }
 72             return ret;
 73         }()),
 74         // *** Lifecycle Methods *** //
 75         /**
 76          * @function
 77          * @private
 78          * @description Set variable with initials values.
 79          * Plug a toolbar widget (and add the delete button at this toolbar).
 80          */
 81         initializer: function() {
 82             this.dataSource = Y.Wegas.Facade.VariableDescriptor;
 83             this.handlers = {};
 84             this.jsTranslator = new Y.Wegas.JSTranslator();
 85         },
 86         /**
 87          * @function
 88          * @private
 89          * @description Render the TabView widget in the content box.
 90          */
 91         renderUI: function() {
 92             var cb = this.get(CONTENTBOX),
 93                     delBtn;
 94             this.tabView = new Y.TabView(); // TabView widget used to display messages
 95             this.tabView.render(cb); // and render it
 96             this.tabView.get("boundingBox").addClass("horizontal-tabview"); // Add class to have horizontal tabview
 97             cb.append("<div style='clear:both'></div>");
 98             if (this.toolbar) {
 99                 delBtn = new Y.Wegas.Button({
100                     label: "<span class='wegas-icon wegas-icon-cancel'></span>" + this.jsTranslator.getRB().Delete
101                 }); // Create delete mail button
102                 delBtn.on("click", function() { // On delete button click
103                     var selection = this.tabView.get("selection");
104                     if (selection && !selection.msg // If a valid mail tab is selected
105                             && confirm('The e-mail "' + selection.msg.get("subject") + '" will be deleted permanently. Continue?')) { // and user is sure
106                         this.deleteEmail(selection.msg); // destroy the message
107                     }
108                 }, this);
109                 this.toolbar.add(delBtn);
110             }
111         },
112         /**
113          * @function
114          * @private
115          * @description bind function to events.
116          * When selection change on tabView, call function "onTabSelected"
117          * When dataSource is updated, do syncUI
118          * When deleteButton is clicked, show an alert which allow to delete selected message
119          */
120         bindUI: function() {
121             this.tabView.after("selectionChange", this.onTabSelected, this);
122             this.handlers.dataUpdated = this.dataSource.after("update", function(e) {
123                 if (e.tId !== this.readRequestTid) {
124                     this.syncUI();
125                 }
126             }, this);
127         },
128         /**
129          * @function
130          * @private
131          * @description Clear and re-fill th TabView with current player message.
132          * display a message if there is no message.
133          * Re-select the current selected msg;
134          */
135         syncUI: function() {
136             var inboxDescriptor = this.get('variable.evaluated');
137             if (!inboxDescriptor) {
138                 this.tabView.add({
139                     label: '',
140                     content: '<center>Unable to find inbox variable</center>'
141                 });
142                 return;
143             }
144             Y.Wegas.Facade.VariableDescriptor.cache.getWithView(inboxDescriptor.getInstance(), "Extended", {
145                 on: {
146                     success: Y.bind(function(e) {
147                         this.updateTabView(e.response.entity.get("messages"));
148                     }, this)
149                 }
150             });
151         },
152         /**
153          * @function
154          * @private
155          * @description Destroy TabView and all eventHandlers created by this widget
156          */
157         destructor: function() {
158             this.tabView.destroy();
159             this.handlers.dataUpdated.detach();
160         },
161         // *** Private Methods *** //
162         /**
163          * Update tabView. Tab part.
164          * @private
165          * @param {Array of entity} entities
166          * @returns {undefined}
167          */
168         updateTabView: function(entities) {
169             var selection = this.tabView.get("selection"),
170                     oldMsg = selection && selection.msg;
171             this.tabView.destroyAll();
172             if (entities.length === 0) {
173                 this.tabView.add(new Y.Tab({
174                     label: '',
175                     content: '<center><i>You have no messages</i></center>'
176                 }));
177             }
178             Y.Array.each(entities, function(entity) {
179                 var tab = new Y.Tab({
180                     label: this.TEMPLATES[this.get("template")].tab(entity),
181                     content: "<div class=\"wegas-loading-div\"><div>"
182                 });
183                 this.tabView.add(tab);
184                 tab.msg = entity;
185                 if (oldMsg && oldMsg.get("id") === entity.get("id")) { // If the current tab was selected before sync,
186                     tab.set("selected", 2);
187                 }
188             }, this);
189             if (!this.tabView.get("selection")) { // Select the first tab by default
190                 this.tabView.selectChild(0);
191             }
192         },
193         /**
194          * Update specified tab's content. Forward provided entity to template.
195          * @private
196          * @param {Y.Tab} tab tab to update
197          * @param {Entity} entity entity to use for content
198          * @returns {undefined}
199          */
200         updateTabContent: function(tab, entity) {
201             tab.set("content", this.TEMPLATES[this.get("template")].content(entity));
202         },
203         /**
204          * @function
205          * @private
206          * @description Send a REST request to delete selected message
207          */
208         deleteEmail: function(msg) {
209             this.dataSource.sendRequest({
210                 request: "/Inbox/Message/" + msg.get("id"),
211                 cfg: {
212                     method: "DELETE"
213                 }
214             });
215         },
216         /**
217          * @function
218          * @private
219          * @description retrieve selected message's body and set this as "read" if
220          * tab has been selected longer than 2000 miliseconde.
221          */
222         onTabSelected: function(e) {
223             var tab = e.newVal;
224             if (this.timer) { // If there is an active unread message timer,
225                 this.timer.cancel(); // cancel it.
226             }
227             if (tab && tab.msg) {
228                 this.dataSource.sendRequest({//                                 // Retrieve the message body from the server
229                     request: "/Inbox/Message/" + tab.msg.get("id") + "?view=Extended",
230                     cfg: {
231                         updateCache: false
232                     },
233                     on: {
234                         success: Y.bind(function(e) {
235                             this.updateTabContent(tab, e.response.entity);
236                         }, this)
237                     }
238                 });
239                 if (tab.msg.get("unread")) { // If the message is currently unread,
240                     this.timer = Y.later(this.get("setToReadAfter") * 1000, this,
241                             function(tab) { // Send a request to mark it as read
242                                 Y.log("Sending message read update", "info", "InboxDisplay");
243                                 tab.get("contentBox").one(".unread").removeClass("unread").addClass("read"); // Immediately update view (before request)
244                                 tab.msg.set("unread", false); // Update the message (since there wont be no sync?)
245                                 this.readRequestTid = this.dataSource.sendRequest({// Send reqest to mark as read
246                                     request: "/Inbox/Message/Read/" + tab.msg.get("id"),
247                                     cfg: {
248                                         method: "PUT"
249                                     }
250                                 });
251                             }, tab);
252                 }
253             }
254         },
255         getEditorLabel: function() {
256             var variable = this.get("variable.evaluated");
257             if (variable) {
258                 return variable.getEditorLabel();
259             }
260             return null;
261         }
262     }, {
263         /**
264          * @lends Y.Wegas.InboxDisplay
265          */
266         EDITORNAME: "Inbox",
267         /**
268          * @field
269          * @static
270          * @description
271          * <p><strong>Attributes</strong></p>
272          * <ul>
273          *    <li>variable: The target variable, returned either based on the name
274          *     attribute, and if absent by evaluating the expr attribute.</li>
275          * </ul>
276          */
277         ATTRS: {
278             /**
279              * The target variable, returned either based on the name
280              * attribute, and if absent by evaluating the expr attribute.
281              */
282             variable: {
283                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
284                 _inputex: {
285                     _type: "variableselect",
286                     label: "variable",
287                     classFilter: ["InboxDescriptor"]
288                 }
289             },
290             /**
291              *
292              */
293             template: {
294                 value: "inbox",
295                 type: "string",
296                 choices: [{
297                         value: "inbox"
298                     }, {
299                         value: "clean",
300                         label: "No headers"
301                     }]
302             },
303             setToReadAfter: {
304                 value: 0.05,
305                 type: "number",
306                 _inputex: {
307                     label: "Mark as read after (s.)",
308                     wrapperClassName: "inputEx-fieldWrapper wegas-advanced-feature"
309                 }
310             }
311         }
312     });
313     Y.namespace('Wegas').InboxDisplay = InboxDisplay;
314 });
