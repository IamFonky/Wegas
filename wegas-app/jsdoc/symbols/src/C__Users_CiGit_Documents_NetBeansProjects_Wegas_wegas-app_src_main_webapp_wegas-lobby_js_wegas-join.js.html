 12 YUI.add('wegas-join', function(Y) {
 13     "use strict";
 14 
 15     /**
 16      * @name Y.Wegas.JoinTeam
 17      * @extends Y.Widget
 18      * @augments Y.WidgetChild
 19      * @augments Y.Wegas.Widget
 20      * @class class for join a team
 21      * @constructor
 22      * @description Allows just to join a team
 23      */
 24     var CONTENTBOX = "contentBox",
 25             JoinTeam = Y.Base.create("wegas-jointeam", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget], {
 26         /** @lends Y.Wegas.JoinTeam# */
 27         CONTENT_TEMPLATE: "<div><div class=\"title\"></div><div class=\"subtitle\"></div>"
 28                 + "<div class=\"description wegas-loading-div\"></div>"
 29                 + "<div class=\"teamselection\"></div>"
 30                 + "</div>",
 31         // *** Private fields *** //
 32 
 33         /**
 34          * @function
 35          * @private
 36          * @description All button and fields are created.
 37          * For creating the field inputEx libary is used
 38          */
 39         renderUI: function() {
 40             var cb = this.get(CONTENTBOX),
 41                     game = this.getTargetGame();
 42 
 43             this.joinButton = new Y.Button({//                                  // Render the button
 44                 label: "Join game",
 45                 render: cb,
 46                 visible: false
 47             });
 48 
 49             Y.Wegas.Facade.Game.cache.getWithView(game, "Extended", {//         // Get the game model full description
 50                 on: {
 51                     success: Y.bind(this.onGameRetrieved, this)
 52                 }
 53             });
 54         },
 55         bindUI: function() {
 56             this.joinButton.on("click", this.onJoinClick, this);                // On join button click
 57         },
 58         destructor: function() {
 59             if (this.teamField) {
 60                 this.teamField.destroy();
 61                 //this.playersField.destroy();
 62             }
 63             this.joinButton.destroy();
 64         },
 65         onGameRetrieved: function(e) {
 66             var emptyChoices, choices,
 67                     cb = this.get(CONTENTBOX),
 68                     teamSelectionNode = cb.one(".teamselection"),
 69                     entity = this.getTargetEntity(),
 70                     game = this.getTargetGame(),
 71                     teams = game.get("teams"),
 72                     gameModel = e.response.entity.get("gameModel"),
 73                     teamName = (entity instanceof Y.Wegas.persistence.Team) ? entity.get("name")
 74                     : game.get("name") + "-" + (game.get("teams").length);
 75 
 76             cb.one(".title").setHTML("" + gameModel.get("name") + " <br />" + game.get("name"));// Set game name
 77             cb.one(".subtitle").setHTML("Created by " + game.get("createdByName") + " " + Y.Wegas.Helper.smartDate(game.get("createdTime")));// Set game name
 78             cb.one(".description")
 79                     //.setHTML(e.response.entity.get("description") || "<em><center>No description available</em></center>")
 80                     .setHTML(e.response.entity.get("description"))
 81                     .removeClass("wegas-loading-div");
 82 
 83             var showTeamSelection = false,
 84                     showTeamCreation = false,
 85                     showTeamEdition = false;                                    // Default case, free for all games
 86 
 87             if (entity instanceof Y.Wegas.persistence.Team) {                   // If target entity is a team
 88                 if (gameModel.get("properties.freeForAll")) {                   // If game is free for all (no team)
 89                     this.sendJoinTeamRequest(entity.get("id"));                 // join it directly
 90                 } else if (entity.get("players").length === 0) {                // and this team is empty, (first connectin to this team)
 91                     showTeamEdition = true;                                     // display team composition edition
 92                 } else {
 93                     this.showMessageBis("error",
 94                             "This team has already been created. You can contact it's members so they can join you in.");
 95                 }
 96             } else if (entity instanceof Y.Wegas.persistence.Game && // If target entity is a game
 97                     !(gameModel && gameModel.get("properties.freeForAll"))) {   // and its game is not free for all (uses teams)
 98                 showTeamCreation = true;
 99             }
100             if (showTeamCreation || showTeamEdition) {
101                 teamSelectionNode.append("<br /><div class=\"title\">Team</div>");
102             }
103             this.showTeamEdition = showTeamEdition;
104             this.showTeamCreation = showTeamCreation;
105             this.showTeamSelection = showTeamSelection;
106 
107             if (showTeamCreation || showTeamEdition) {                          // Render Team creation option
108                 choices = Y.Array.map(teams, function(i) {                      // render team selection
109                     return {
110                         label: i.get("name"),
111                         value: i.get("id")
112                     };
113                 });
114 
115                 if (choices.length === 0) {
116                     choices.push({
117                         label: "No team created"
118                     });
119                     emptyChoices = true;
120                 }
121 
122                 //teamSelectionNode.append("<div style=\"margin:10px 0\"><span style=\"color: #505050;font-style: italic;padding-left: 20px;\">Team name:</span>   " + teamName + "</div>");
123 
124                 this.teamField = new Y.inputEx.MultipleOptions({//                Create team edition field
125                     parentEl: teamSelectionNode,
126                     fields: [{
127                             type: "hidden",
128                             label: "Team name",
129                             required: true,
130                             value: teamName
131                         }, {
132                             type: "select",
133                             choices: choices,
134                             label: "Select an existing team"
135                         }]
136                 });
137 
138                 if (emptyChoices) {                                             // Disable team selection if it's empty
139                     this.teamField.inputs[1].disable();
140                     //this.teamField.inputs[0].el.focus();
141                 }
142                 if (!showTeamSelection) {
143                     this.teamField.inputs[1].hide();                            // Disable team selection everywhere (temporary)
144                     //this.teamField.inputs[0].el.focus();
145                 }
146 
147             }
148             if (showTeamEdition || showTeamCreation) {
149                 this.teamEdition = new Y.Wegas.TeamEdition({
150                     render: teamSelectionNode,
151                     entity: entity
152                 });
153                 this.teamEdition.addExistingAccount(
154                         Y.Wegas.Facade.User.get("currentUser").getMainAccount());// Push  current user to the team's player list
155             }
156 
157             this.joinButton.set("visible", true);
158         },
159         onJoinClick: function() {
160             var entity = this.getTargetEntity(),
161                     selectedField = (this.teamField) ? this.teamField.getSelected() : null,
162                     name = (selectedField) ? selectedField.getValue() : null;
163 
164             if (this.showTeamEdition || this.showTeamCreation) {
165                 if (!this.teamEdition.playersField.validate()) {
166                     this.showMessage("error", "Invalid name, password or email");
167                     return;
168                 }
169                 if (name === "") {
170                     this.showMessage("error", "Enter a valid team name");
171                     return;
172                 }
173             }
174 
175             this.showOverlay();
176 
177             if (this.showTeamCreation) {                                        // If entity is a game token which allows team creation,
178                 if (this.teamId) {
179                     this.sendMultiJoinTeamRequest(this.teamId);
180                 } else {
181                     Y.Wegas.Facade.Game.sendRequest({//                             // create the team
182                         //request: "/" + entity.get("id") + "/CreateTeam/" + name,
183                         request: "/" + entity.get("id") + "/CreateTeam",
184                         cfg: {
185                             method: "POST",
186                             updateCache: false
187                         },
188                         on: {
189                             success: Y.bind(function(e) {
190                                 this.sendMultiJoinTeamRequest(e.response.entity.get("id"));// and then join it
191                             }, this),
192                             failure: Y.bind(function(e) {
193                                 this.hideOverlay();
194                                 this.showMessage("error", e.response.results.message || "Error creating team");
195                             }, this)
196                         }
197                     });
198                 }
199             } else if (this.showTeamEdition) {                                  // If joining
200                 if (name !== entity.get("name")) {                              // If team name was edited,
201                     entity.set("name", name);
202                     Y.Wegas.Facade.Game.sendRequest({//                         // update it
203                         request: "/Team/" + entity.get("id"),
204                         cfg: {
205                             method: "PUT",
206                             updateCache: false,
207                             data: entity.toObject()
208                         },
209                         on: {
210                             success: function() {
211                                 this.sendMultiJoinTeamRequest(entity.get("id"));// join the team
212                             },
213                             failure: Y.bind(this.defaultFailureHandler, this)
214                         }
215                     });
216                 } else {
217                     this.sendMultiJoinTeamRequest(entity.get("id"));            // join the team
218                 }
219             } else if (this.showTeamSelection) {
220                 // todo
221             } else {                                                            // If game is Free for all games or joining a team directly,
222                 this.sendTokenJoinGame(this.getTargetEntity().get("token"));    // use the token to join
223             }
224         },
225         getTargetEntity: function() {
226             var entity = this.get("entity");
227             if (Y.Lang.isArray(entity)) {
228                 return entity[0];
229             } else {
230                 return entity;
231             }
232         },
233         getTargetGame: function() {
234             var entity = this.get("entity");
235             if (Y.Lang.isArray(entity)) {
236                 if (entity[1]) {
237                     return entity[1];
238                 } else {
239                     return entity[0];
240                 }
241             } else if (entity instanceof Y.Wegas.persistence.Team) {
242                 return Y.Wegas.Facade.Game.cache.findById(entity.get("gameId"));
243             } else {
244                 return entity;
245             }
246         },
247         /**
248          * @function
249          * @private
250          * @description User rest request: rest/GameModel/1/Game/{gameModelID}/JoinTeam/{teamID}
251          */
252         sendJoinTeamRequest: function(teamId) {
253             this.showOverlay();
254             Y.Wegas.Facade.Game.sendRequest({
255                 request: "/JoinTeam/" + teamId,
256                 cfg: {
257                     updateCache: false
258                 },
259                 on: {
260                     success: Y.bind(function() {
261                         this.showMessage("success", "Game joined");
262 
263                         Y.fire("gameJoined", {
264                             gameId: this.getTargetGame().get("id"),
265                             game: this.getTargetGame()
266                         });
267                         this.hideOverlay();
268 
269                         this.get("contentBox").empty();
270 
271                         var rightTab = Y.Widget.getByNode("#rightTabView");     // Empty right tab on join
272                         rightTab && rightTab.destroyAll();
273                     }, this),
274                     failure: Y.bind(function(e) {
275                         this.hideOverlay();
276                         this.showMessage("error", "Error joining team");
277                     }, this)
278                 }
279             });
280         },
281         sendMultiJoinTeamRequest: function(teamId) {
282             Y.Wegas.Facade.Game.sendRequest({//                                 // Add all player to the list in the list to the target game
283                 request: "/JoinTeam/" + teamId,
284                 cfg: {
285                     method: "POST",
286                     data: this.teamEdition.getAccounts()
287                 },
288                 on: {
289                     success: Y.bind(this.onGameJoined, this),
290                     failure: Y.bind(function(e) {
291                         this.teamId = teamId;                                   // @hack
292                         this.defaultFailureHandler(e);
293                     }, this)
294                 }
295             });
296         },
297         /**
298          * @function
299          * @private
300          * @description All events are added to the buttons
301          * Create team button call rest url : rest/GameModel/{gameModelID}/Game/{gameID}/CreateTeam/{teamName}
302          */
303         onGameJoined: function(e) {
304             //this.showMessage("success", "Game joined");
305 
306             Y.fire("gameJoined", {
307                 gameId: e.response.entity.get("id"),
308                 game: e.response.entity
309                         //gameId: this.getTargetGame().get("id"),
310                         //game: this.getTargetGame()
311             });
312             this.hideOverlay();
313 
314             this.destructor();
315             this.get("contentBox").empty();
316 
317             var rightTab = Y.Widget.getByNode("#rightTabView");
318             rightTab && rightTab.destroyAll();
319         },
320         sendTokenJoinGame: function(token) {
321             this.showOverlay();
322             Y.Wegas.Facade.Game.sendRequest({
323                 request: "/JoinGame/" + token,
324                 cfg: {
325                     updateCache: !this.get("customEvent")
326                 },
327                 on: {
328                     success: Y.bind(function(e) {
329                         if (e.response.entity instanceof Y.Wegas.persistence.Team) { // If the returned value is a Team enity
330                             this.sendJoinTeamRequest(e.response.entity.get("id"));
331                         } else {
332                             // TODO
333                             this.hideOverlay();
334                         }
335                     }, this),
336                     failure: Y.bind(function(e) {
337                         this.showMessage("error", e.response.results.message || "Invalid token");
338                     }, this)
339                 }
340             });
341         }
342     }, {
343         ATTRS: {
344             entity: {},
345             customEvent: {
346                 value: false
347             }
348         }
349     });

390     var TeamEdition = Y.Base.create("wegas-editteam", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
391         CONTENT_TEMPLATE: "<div><div class=\"header yui3-g\">"
392                 + "<div class=\"yui3-u\">First name</div>"
393                 + "<div class=\"yui3-u\">Last name</div>"
394                 + "<div class=\"yui3-u\">Email</div>"
395                 + "<div class=\"yui3-u\">Password</div></div>"
396 
397                 + "<div class=\"uneditable-players\"></div></div>",
398         renderUI: function() {
399             var cb = this.get("contentBox"), gameId = Y.Widget.getByNode(this._parentNode).getTargetGame().get("id"),
400                     autoCompleteCfg = {
401                 type: "autocomplete",
402                 autoComp: {
403                     minQueryLength: 2,
404                     maxResults: 30,
405                     resultTextLocator: function(o) {
406                         return o.firstname + " " + o.lastname;
407                     },
408                     resultHighlighter: 'phraseMatch',
409                     source: Y.Wegas.app.get("base") + "rest/User/AutoCompleteFull/{query}/" + gameId,
410                     enableCache: true,
411                     resultListLocator: Y.bind(function(responses) {
412                         var i;
413                         Y.Array.each(this.otherAccounts, function(account) {
414                             for (i = 0; i < responses.length; i += 1) {
415                                 if (account.id === responses[i].id) {
416                                     responses.splice(i, 1);
417                                     break;
418                                 }
419                             }
420                         });
421                         Y.Array.each(this.playersField.subFields, function(user) {
422                             for (i = 0; i < responses.length; i += 1) {
423                                 if (user.getValue().email === responses[i].email) {
424                                     responses.splice(i, 1);
425                                     break;
426                                 }
427                             }
428                         });
429                         return responses;
430                     }, this)
431                 }
432             };
433 
434             this.otherAccounts = [];
435 
436             this.playersField = new Y.inputEx.ListField({//                     // Render team edition
437                 parentEl: cb,
438                 useButtons: true,
439                 elementType: {
440                     type: "group",
441                     fields: [{
442                             name: "@class",
443                             type: "hidden",
444                             value: "JpaAccount"
445                         },
446                         Y.mix({
447                             name: "firstname",
448                             required: true,
449                             typeInvite: "required",
450                             size: 13
451                         }, autoCompleteCfg),
452                         Y.mix({
453                             name: "lastname",
454                             typeInvite: "required",
455                             required: true,
456                             size: 13
457                         }, autoCompleteCfg),
458                         Y.mix({
459                             name: "email",
460                             //type: "email",
461                             typeInvite: "optional",
462                             //required: true,
463                             size: 13
464                         }, autoCompleteCfg), {
465                             name: "password",
466                             type: "password",
467                             //required: true,
468                             //typeInvite: "password",                           // typeInvite dont work on password in inputex
469                             size: 13
470                         }]
471                 }
472             });
473             cb.all("input[type=\"password\"]").setAttribute("placeholder", "required");// Put placeholder attribute on all password fields
474             cb.one(".inputEx-ListField").append(cb.one("img.inputEx-ListField-addButton"));// Move add button at the end of the list
475 
476             Y.on("domready", this.updateAutoCompletes, this);
477             cb.one("img.inputEx-ListField-addButton").on("click", function() {
478                 Y.later(10, this, this.updateAutoCompletes);
479             }, this);// Add proper callback on autocomplete
480         },
481         updateAutoCompletes: function() {
482             var i, j;
483             for (i = 0; i < this.playersField.subFields.length; i++) {
484                 for (j = 1; j < 4; j += 1) {
485                     var field = this.playersField.subFields[i].inputs[j];
486                     if (!field.wmodified) {
487                         field.yEl.ac.after("select", function(e) {
488                             this.setValue(e.result.raw);
489                         }, this.playersField.subFields[i]);
490                         field.wmodified = true;
491                     }
492                 }
493             }
494         },
495         addExistingAccount: function(account) {
496             var cb = this.get("contentBox"),
497                     firstname = (account instanceof Y.Wegas.persistence.GuestJpaAccount) ? account.getPublicName() :
498                     account.get("firstname");
499 
500             this.otherAccounts.push(account.toObject());
501 
502             cb.one(".uneditable-players").append("<div class=\"yui3-g\">"
503                     + "<div class=\"yui3-u\">" + (firstname || account.get("name")) + "</div>"
504                     + "<div class=\"yui3-u\">" + (account.get("lastname") || "") + "</div>"
505                     + "<div class=\"yui3-u\">*****</div>"
506                     + "<div class=\"yui3-u\">*****</div></div>");
507         },
508         getAccounts: function() {
509             return Y.Array.map(this.playersField.getValue().concat(this.otherAccounts), function(i) {
510                 delete i.name;
511                 return i;
512             });
513         }
514     }, {
515         ATTRS: {
516             entity: {}
517         }
518     });
519     Y.namespace('Wegas').TeamEdition = TeamEdition;
520     Y.inputEx.AutoComplete.prototype.buildAutocomplete = function() {
521         // Call this function only when this.el AND this.listEl are available
522         if (!this._nElementsReady) {
523             this._nElementsReady = 0;
524         }
525         this._nElementsReady++;
526         if (this._nElementsReady != 2)
527             return;
528 
529         this.yEl = Y.one(this.el)
530         this.yEl.plug(Y.Plugin.AutoComplete, this.options.autoComp);
531 
532         // Instantiate AutoComplete
533         this.yEl.ac.on("select", this.itemSelectHandler, this);
534         this.yEl.on("blur", this.onBlur, this);
535 
536         this.yEl.on("valueChange", function() {                                 // @MODIFIED
537             this.hiddenEl.value = this.el.value;
538             this.validate();
539         }, this);
540     };
