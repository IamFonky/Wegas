 11 YUI.add('wegas-websocketlistener', function(Y) {
 12     "use strict";
 13 
 14     var WebSocketListener = Y.Base.create("WebSocketListener", Y.Plugin.Base, [], {
 15         initializer: function() {
 16             var dataSource = Y.Wegas.Facade[this.get("dataSource")];
 17             if (dataSource) {
 18                 dataSource.on("EntityUpdatedEvent", this.onVariableInstanceUpdate, this);
 19             }
 20         },
 21         onVariableInstanceUpdate: function(data) {
 22             Y.log("Websocket event received.", "info", "Wegas.WebsocketListener");
 23             this.get("host").cache.onResponseRevived({
 24                 serverResponse: Y.Wegas.Editable.revive({
 25                     "@class": "ManagedModeResponseFilter$ServerResponse",
 26                     events: [data]
 27                 })
 28             });
 29         },
 30         destructor: function() {
 31         }
 32     }, {
 33         ATTRS: {
 34             dataSource: {
 35                 initOnly: true
 36             }
 37         },
 38         NS: "ws",
 39         NAME: "WebSocketListener"
 40     });
 41     Y.namespace('Plugin').WebSocketListener = WebSocketListener;
 42 
 43 });
