 12 YUI.add('wegas-pmg-bac', function(Y) {
 13     "use strict";
 14 
 15     /**
 16      *  @class add Bac
 17      *  @name Y.Plugin.Bac
 18      *  @extends Y.Plugin.Base
 19      *  @constructor
 20      */
 21     var Wegas = Y.Wegas,
 22             Bac = Y.Base.create("wegas-pmg-bac", Y.Plugin.Base, [Wegas.Plugin, Wegas.Editable], {
 23         /** @lends Y.Plugin.Bac */
 24 
 25         /**
 26          * Lifecycle methods
 27          * @function
 28          * @private
 29          */
 30         initializer: function() {
 31             this.onceAfterHostEvent("render", function() {
 32                 this.addBacColumn();
 33                 this.addInputField();
 34                 this.afterHostMethod("syncUI", this.addInputField);
 35                 this.onKeyUpEvent();
 36             });
 37             this.get("host").datatable.after("sort", this.addInputField, this);
 38         },
 39         addBacColumn: function() {
 40             this.get("host").datatable.addColumn({
 41                 key: 'bac',
 42                 label: "BAC"
 43             }, this.get("columnPosition"));
 44         },
 45         addInputField: function() {
 46             var i, dt = this.get("host").datatable, cell, field, taskDesc;
 47 
 48             for (i = 0; i < dt.data._items.length; i++) {
 49                 taskDesc = dt.data.item(i).get("descriptor");
 50                 cell = dt.getCell([i, this.get("columnPosition")]);
 51                 field = Y.Node.create("<input class='bacField'></input>");
 52                 field.set("value", taskDesc.getInstance().get("properties").bac);
 53                 cell.append(field);
 54             }
 55         },
 56         onKeyUpEvent: function() {
 57             var dt = this.get("host").datatable;
 58             this.changeHandle = this.get("host").datatable.delegate("change", function(e) {
 59                 var record = dt.getRecord(e.target), val = Y.Lang.trim(e.target.get("value"));
 60                 val = (val === "") ? "0" : val; //empty is 0
 61                 e.target.removeClass("invalid");
 62                 if (this.isValidField(val)) {
 63                     val = parseInt(val, 10) + ""; // Remove leading 0 : binary number
 64                     e.target.set("value", val);
 65                     if (record && record.get("instance").properties.bac + "" !== val) {
 66                         record.get("instance").properties.bac = val;
 67                         this.request(record);
 68                     }
 69                 } else {
 70                     e.target.addClass("invalid");
 71                 }
 72             }, ".bacField", this);
 73         },
 74         isValidField: function(value) {
 75             var regexp = /^[+]?\d+$/; //positive number
 76             if (!regexp.test(value)) {
 77                 this.get("host").showMessage("error", "\"" + value + "\" is not a positive integer");
 78                 return false;
 79             }
 80             return true;
 81         },
 82         request: function(taskDescriptor) {
 83             Y.Wegas.Facade.VariableDescriptor.sendRequest({
 84                 request: "/Script/Run/" + Y.Wegas.Facade.Game.get('currentPlayerId'),
 85                 cfg: {
 86                     method: "POST",
 87                     updateCache: false,
 88                     updateEvent: false,
 89                     data: Y.JSON.stringify({
 90                         "@class": "Script",
 91                         language: "JavaScript",
 92                         content: "importPackage(com.wegas.core.script);\nVariable.findByName(self.getGameModel(), '" + taskDescriptor.get("name") +"').getInstance(self).setProperty('bac', '" + taskDescriptor.get("instance").properties.bac + "');"
 93                     })
 94                 }
 95             });
 96         },
 97         destructor: function() {
 98             this.changeHandle.detach();
 99             this.get("host").datatable.removeColumn("bac");                     /* Time consuming */
100         }
101     }, {
102         ATTRS: {
103             columnPosition: {
104                 value: 2,
105                 _inputex: {
106                     _type: "integer",
107                     label: "Column position"
108                 }
109             }
110         },
111         NS: "bac",
112         NAME: "Bac"
113     });
114     Y.namespace("Plugin").Bac = Bac;
115 });
