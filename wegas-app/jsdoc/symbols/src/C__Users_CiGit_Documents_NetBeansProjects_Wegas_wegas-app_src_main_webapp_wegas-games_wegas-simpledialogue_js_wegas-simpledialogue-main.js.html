 12 YUI.add("wegas-simpledialogue", function(Y) {
 13     "use strict";
 14 
 15     var CONTENTBOX = "contentBox", SimpleDialogueMain;
 16 
 17     SimpleDialogueMain = Y.Base.create("wegas-teaching-main", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 18         CONTENT_TEMPLATE: '<div><div class="dialogue"><div class="talk"></div><div class="response"><ul class="responseElements"></ul></div></div></div>',
 19         initializer: function() {
 20             this.handlers = {};
 21         },
 22         bindUI: function() {
 23             this.handlers.update = Y.Wegas.Facade.VariableDescriptor.after("update", this.syncUI, this);
 24 
 25             this.get(CONTENTBOX).delegate('click', function(e) {
 26                 var no = parseInt(e.currentTarget.getAttribute("response_no"));
 27                 if (this.availableActions[no]) {
 28                     this.currentDialogue.doTransition(this.availableActions[no]);
 29                 }
 30             }, '.dialogue .response .responseElements li', this);
 31         },
 32         syncUI: function() {
 33             this.currentDialogue = this.get("dialogueVariable.evaluated");
 34 
 35             if (!this.currentDialogue) {
 36                 this.get(CONTENTBOX).one('.dialogue .talk').insert("Dialog variable could not be found");
 37                 return;
 38             }
 39             var state = this.currentDialogue.getCurrentState();
 40             this.displayText(state.get('text'));
 41             if (!state instanceof Y.Wegas.persistence.DialogueDescriptor) {
 42                 Y.log("State isn't a dialogue state.", 'error', 'SimpleDialogue');
 43                 return;
 44             }
 45             state.getAvailableActions(Y.bind(this.readStateContent, this));
 46         },
 47         destructor: function() {
 48             var i;
 49             for (i in this.handlers) {
 50                 this.handlers[i].detach();
 51             }
 52         },
 53         readStateContent: function(availableActions) {
 54             this.availableActions = availableActions;
 55             this.displayResponse(availableActions);
 56         },
 57         displayText: function(textParts) {
 58             this.get(CONTENTBOX).one('.dialogue .talk').setHTML("<p>" + textParts + "</p>");
 59         },
 60         displayResponse: function(availableActions) {
 61             var i, responseNode = this.get(CONTENTBOX).one('.dialogue .response .responseElements');
 62 
 63             if (!availableActions) {
 64                 return;
 65             }
 66             responseNode.empty(true);
 67             for (i = 0; i < availableActions.length; i++) {
 68                 responseNode.insert('<li response_no="' + i + '">' + availableActions[i].get('actionText') + '</li>');
 69             }
 70         }
 71     }, {
 72         ATTRS: {
 73             dialogueVariable: {
 74                 value: {
 75                     name: "simpleDialogue"
 76                 },
 77                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
 78                 _inputex: {
 79                     _type: "variableselect",
 80                     legend: "Dialogue",
 81                     classFilter: ["DialogueDescriptor"]
 82                 }
 83             }
 84         }
 85     });
 86     Y.namespace("Wegas").SimpleDialogueMain = SimpleDialogueMain;
 87 });
