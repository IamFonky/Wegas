 11 YUI.add('wegas-editor-form', function(Y) {
 12     "use strict";
 13 
 14     var CONTENTBOX = "contentBox", inputEx = Y.inputEx, Lang = Y.Lang,
 15             Wegas = Y.Wegas, Form, EditEntityForm;
 16 
 17     /**
 18      * @name Y.Wegas.Form
 19      * @class  Class to submit a form, add a toolbar with buttons "submit" and
 20      * "cancel" to manage forms.
 21      * @extends Y.Widget
 22      * @augments Y.WidgetChild
 23      * @augments Y.Wegas.Widget
 24      * @constructor
 25      */
 26     Form = Y.Base.create("wegas-form", Y.Widget, [Y.WidgetChild, Wegas.Widget, Wegas.Editable], {
 27         /**
 28          * @lends Y.Wegas.Form#
 29          */
 30         // ** Lifecycle Methods ** //
 31         /**
 32          * @function
 33          * @private
 34          * @description plug a toolbar and publich "submit" event.
 35          */
 36         initializer: function() {
 37             this.plug(Y.Plugin.WidgetToolbar);                                  // A toolbar where to place the buttons
 38             this.publish("submit", {
 39                 emitFacade: true
 40             });
 41             this.publish("updated", {
 42                 emitFacade: true
 43             });
 44             this.updateTimer = new Wegas.Timer({
 45                 duration: this.get("updateTimeout")
 46             });
 47         },
 48         /**
 49          * @function
 50          * @private
 51          * @description
 52          */
 53         bindUI: function() {
 54             this.on("update", this.updateTimer.reset, this.updateTimer);        // Whenever the form is updated, reset the update time
 55             this.updateTimer.on("timeOut", function() {
 56                 this.fire("updateTimeOut");
 57             }, this);
 58         },
 59         /**
 60          * @function
 61          * @private
 62          * @description call function "renderToolbar".
 63          */
 64         syncUI: function() {
 65             Y.Array.each(this.get("buttons"), this.addButton, this);            // Render the buttons
 66             this.renderForm(this.get("cfg"));                                   // Render the form
 67         },
 68         /**
 69          * @function
 70          * @private
 71          * @returns {undefined}
 72          */
 73         destructor: function() {
 74             if (this.form) {
 75                 this.form.destroy();
 76             }
 77             this.updateTimer.destroy();
 78         },
 79         renderForm: function(val) {
 80             inputEx.use(val, Y.bind(function(val) {                             // Load form dependencies
 81                 var cfg = Y.clone(val);                                         // Duplicate so val will be untouched upon serialization
 82                 Y.mix(cfg, {//                                                  // Add some default properties
 83                     parentEl: this.get(CONTENTBOX),
 84                     type: "group",
 85                     value: this.get("value", {internal: true})
 86                 });
 87                 if (this.form) {                                                // If there's already a form,
 88                     this.form.destroy();                                        // destroy it
 89                 }
 90                 this.form = inputEx(cfg);                                       // Initialize and render the new form
 91                 //this.form.setValue(this.get("value"), false);                 // Set the form value
 92                 this.form.on("updated", function() {//                          // Whenever a form is updated,
 93                     this.fire("update");                                        // throw an event
 94                 }, this);
 95 
 96                 this.form.parentWidget = this;                                  // @HACK some widgets need this reference
 97             }, this, val));
 98         },
 99         addButton: function(b) {
100             b.on = {
101                 click: Y.bind(function(action) {                                // Push click event to the button, depending on their action field
102                     if (!this.form.validate()) {
103                         this.showMessageBis("error", "Some fields are not valid");
104                         return;
105                     }
106                     this.fire(action, {
107                         value: this.get("value")
108                     });
109                 }
110                 , this, b.action)
111             };
112             this.toolbar.add(b);
113         }
114     }, {
115         /** @lends Y.Wegas.Form */
116         EDITORNAME: "Form",
117         /**
118          * <p><strong>Attributes</strong></p>
119          * <ul>
120          *    <li>value: value of fields of the form</li>
121          *    <li>form: the form to manage (see YUI Form)</li>
122          *    <li>cfg: configuation of the form (see YUI Form)</li>
123          * </ul>
124          *
125          * @field
126          * @static
127          */
128         ATTRS: {
129             /**
130              * Values of fields of the form
131              */
132             value: {
133                 "transient": true,
134                 setter: function(val) {
135                     if (this.form) {
136                         this.form.setValue(val, false);
137                     }
138                     return val;
139                 },
140                 getter: function(val, name, cfg) {
141                     if (this.form && (!cfg || !cfg.internal)) {
142                         return this.form.getValue();
143                     } else {
144                         return val;
145                     }
146                 }
147             },
148             /**
149              * Configuation of the form
150              */
151             cfg: {
152                 validator: Lang.isObject,
153                 setter: function(val) {
154                     this.renderForm(val);
155                     return val;
156                 },
157                 _inputex: {
158                     legend: "Fields",
159                     fields: inputEx.Group.groupOptions
160                 }
161             },
162             updateTimeout: {
163                 "transient": true,
164                 value: 1000
165             },
166             buttons: {
167                 value: [
168                     {
169                         type: "Button",
170                         action: "submit",
171                         label: "<span class=\"wegas-icon wegas-icon-save\" ></span>Save"
172                     }
173                 ]
174             }
175         }
176     });
177     //Y.namespace("Wegas").Form = Form;
178 
179     EditEntityForm = Y.Base.create("wegas-form", Form, [], {
180         bindUI: function() {
181             EditEntityForm.superclass.bindUI.call(this);
182 
183             this.on("update", function() {                                      // When the form is updated,
184                 if (this.form.validate()) {                                     // display appropriate message in the toolbar
185                     this.showMessageBis("success", "Saving...");
186                 } else {
187                     this.showMessageBis("success", "Unable to save");
188                 }
189             });
190             this.on("updateTimeOut", this.save);                                // 2 seconds after user has stopped updating, save changes
191             this.on("submit", this.save);                                       // Upon submit button click, save changes
192         },
193         /**
194          * @function
195          * @private
196          * @description call function "renderToolbar".
197          */
198         syncUI: function() {
199             var entity = this.get("entity");
200 
201             if ((Wegas.persistence.VariableDescriptor &&
202                     (entity instanceof Wegas.persistence.VariableDescriptor
203                             || entity instanceof Wegas.persistence.VariableInstance))// Those classes may not be loaded
204                     || entity instanceof Wegas.persistence.JpaAccount
205                     || entity instanceof Wegas.persistence.GameModel
206                     || entity instanceof Wegas.persistence.Game) {              // @fixme we may get extended mode for any entities, just need to check if it causes bugs
207                 this.showOverlay();
208                 this.get("dataSource").cache.getWithView(entity, "EditorExtended", {// Retrieve the entity from the source
209                     on: {
210                         success: Y.bind(function(e) {
211                             this.showUpdateForm(e.response.entity);             // and show the form
212                             this.hideOverlay();
213                         }, this)
214                     }
215                 });
216             } else {                                                            // Otherwise,
217                 this.showUpdateForm(entity);                                    // render the form directy
218             }
219         },
220         showUpdateForm: function(entity) {
221             this.set("value", entity.toObject());                               // Set the form value of the form,
222             this.set("cfg", this.get("cfg") || entity.getFormCfg());            // and then its fields
223 
224             var menuItems = entity.getMenuCfg({dataSource: this.get("dataSource")}).slice(1);
225             //var menuItems = Y.Array.filter(entity.getMenuCfg({dataSource: this.get("dataSource")}).slice(1), function(i) {
226             //    return (!i.label || (i.label.indexOf("New") < 0 && i.label.indexOf("Edit") < 0));
227             //});                                                               // Retrieve menu and remove the first item
228 
229             Y.Array.each(menuItems, function(i) {                               // @hack Add icons to some buttons
230                 switch (i.label) {
231                     case "Delete":
232                     case "New":
233                     case "New element":
234                     case "Copy":
235                     case "View":
236                     case "Open in editor":
237                     case "Open":
238                     case "Edit":
239                         i.label = '<span class="wegas-icon wegas-icon-' + i.label.replace(/ /g, "-").toLowerCase() + '"></span>' + i.label;
240                 }
241             });
242             this.toolbar.add(menuItems);                                        // Add menu items to the form
243         },
244         save: function() {
245             if (!this.form.validate()) {
246                 return;
247             }
248 
249             this.get("dataSource").cache.put(this.get("value"), {
250                 on: {
251                     success: Y.bind(function() {
252                         this.showMessageBis("success", "All changes saved");
253                     }, this),
254                     failure: Y.bind(this.defaultFailureHandler, this)
255                 }
256             });
257         }
258     }, {
259         ATTRS: {
260             entity: {},
261             dataSource: {
262                 getter: function(val) {
263                     if (Lang.isString(val)) {
264                         return Wegas.Facade[val];
265                     }
266                     return val;
267                 }
268             },
269             buttons: {
270                 value: []
271             }
272         }
273     });
274     Y.namespace('Wegas').EditEntityForm = EditEntityForm;
275 });
