 14 YUI.add('wegas-plugin', function(Y) {
 15     "use strict";
 16     var HOST = "host", Plugin = Y.Plugin, Wegas = Y.namespace("Wegas");
 17     /**
 18      *  @name Y.Wegas.Plugin
 19      *  @class Extension that adds editable capacities to plugins
 20      *  @extends Y.Plugin
 21      *  @constructor
 22      */
 23 
 24     Wegas.Plugin = function() {
 25     };
 26     Y.mix(Wegas.Plugin.prototype, {});
 27     Y.mix(Wegas.Plugin, {
 28         ATTRS: {
 29             host: {
 30                 "transient": true
 31             },
 32             initialized: {
 33                 "transient": true
 34             },
 35             destroyed: {
 36                 "transient": true
 37             }
 38         },
 39         /**
 40          * @function
 41          * @private
 42          * @static
 43          * @param {String} name
 44          * @return Status node
 45          * @description Get Class From plugin name. Hopefully a unique name ...
 46          */
 47         getPluginFromName: function(name) {
 48             var i;
 49             for (i in Y.Plugin) {
 50                 if (Y.Plugin[i].NAME === name) {
 51                     return "" + i;
 52                 }
 53             }
 54             return undefined;
 55         }
 56     });
 57     /**
 58      *  @name Y.Plugin.Action
 59      *  @extends Y.Plugin.Base
 60      *  @augments Y.Wegas.Plugin
 61      *  @augments Y.Wegas.Editable
 62      *  @class
 63      *  @constructor
 64      */
 65     var Action = Y.Base.create("wegas-actionplugin", Plugin.Base, [Wegas.Plugin, Wegas.Editable], {
 66         /** @lends Y.Plugin.Action */
 67 
 68         /**
 69          * @function
 70          * @private
 71          */
 72         initializer: function() {
 73             this.onHostEvent(this.get("targetEvent"), this.execute);
 74         },
 75         /**
 76          * @function
 77          * @protected
 78          */
 79         execute: function() {
 80             Y.error("Y.Plugin.Action.execute() is abstract, should be overriddent");
 81         }
 82     }, {
 83         NS: "wegas",
 84         NAME: "Action",
 85         ATTRS: {
 86             targetEvent: {
 87                 type: "string",
 88                 value: "click",
 89                 _inputex: {
 90                     _type: "hidden"
 91                 }
 92             }
 93         }
 94     });
 95     Plugin.Action = Action;
 96     /**
 97      *  @class
 98      *  @name Y.Plugin.OpenUrlAction
 99      *  @extends Y.Plugin.Action
100      *  @constructor
101      */
102     var OpenUrlAction = function() {
103         OpenUrlAction.superclass.constructor.apply(this, arguments);
104     };
105     Y.extend(OpenUrlAction, Action, {
106         execute: function() {
107             var url = this.get("url");
108             if (url.indexOf("http://") === -1) {
109                 url = Wegas.app.get("base") + url;
110             }
111             if (this.get("target") === "blank") {
112                 window.open(url);
113             } else {
114                 window.location.href = url;
115             }
116         }
117     }, {
118         NS: "openurlaction",
119         NAME: "OpenUrlAction",
120         ATTRS: {
121             url: {
122                 type: "string",
123                 _inputex: {
124                     //_type: "wegasurl"
125                     label: "Open url"
126                 }
127             },
128             /**
129              * Can be "self" or "blank"
130              */
131             target: {
132                 type: "string",
133                 value: "blank",
134                 choices: [{
135                         value: "blank",
136                         label: "In a new page"
137                     }, {
138                         value: "self",
139                         label: "In the same page"
140                     }],
141                 _inputex: {
142                     label: ""
143                 }
144             }
145         }
146     });
147     Plugin.OpenUrlAction = OpenUrlAction;
148     /**
149      *  @class
150      *  @name Y.Plugin.OpenPageAction
151      *  @extends Y.Plugin.Action
152      *  @module Wegas
153      *  @constructor
154      */
155     var OpenPageAction = function() {
156         OpenPageAction.superclass.constructor.apply(this, arguments);
157     };
158     Y.extend(OpenPageAction, Action, {
159         initializer: function() {
160             OpenPageAction.superclass.initializer.apply(this, arguments);
161             this.afterHostEvent("render", function() {
162                 var targetPageLoader = Wegas.PageLoader.find(this.get('targetPageLoaderId'));
163                 if (targetPageLoader) {
164                     targetPageLoader.onceAfter("render", function(e, targetPageLoader) {
165                         if ("" + targetPageLoader.get("pageId") === "" + this.subpage()) {
166                             this.get(HOST).set("selected", 1);
167                         }
168                     }, this, targetPageLoader);
169                 }
170             }, this);
171         },
172         execute: function() {
173             var targetPageLoader = Wegas.PageLoader.find(this.get('targetPageLoaderId'));
174             targetPageLoader.set("pageId", this.subpage());
175             this.get(HOST).set("selected", 1);
176         },
177         subpage: function() {
178             if (!this.get("subpageId")) {
179                 if (this.get("subpageVariable.evaluated")) {
180                     return this.get("subpageVariable.evaluated").getInstance().get("value");
181                 }
182             }
183             return this.get("subpageId");
184         }
185     }, {
186         NS: "OpenPageAction",
187         NAME: "OpenPageAction",
188         ATTRS: {
189             subpageId: {
190                 type: "string",
191                 _inputex: {
192                     label: "Open page",
193                     _type: "pageselect",
194                     required: false
195                 }
196             },
197             targetPageLoaderId: {
198                 type: "string",
199                 value: "maindisplayarea",
200                 _inputex: {
201                     label: "Target zone",
202                     //_type: "pageloaderselect",//@fixme There a bug with this widget when the target page is not loaded
203                     value: "maindisplayarea",
204                     wrapperClassName: 'inputEx-fieldWrapper wegas-advanced-feature'
205                 }
206             },
207             subpageVariable: {
208                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
209                 _inputex: {
210                     _type: "variableselect",
211                     wrapperClassName: "inputEx-fieldWrapper wegas-advanced-feature"
212                 }
213             }
214         }
215     });
216     Plugin.OpenPageAction = OpenPageAction;
217     /**
218      *  @class
219      *  @name Y.Plugin.ExecuteScriptAction
220      *  @extends Y.Plugin.Action
221      *  @constructor
222      */
223     var ExecuteScriptAction = function() {
224         ExecuteScriptAction.superclass.constructor.apply(this, arguments);
225     };
226     Y.extend(ExecuteScriptAction, Action, {
227         execute: function() {
228             var host = this.get(HOST);
229             if (!host.get("disabled")) {
230                 host.showOverlay();
231                 Wegas.Facade.VariableDescriptor.script.run(this.get("onClick"), {
232                     on: {
233                         success: Y.bind(host.hideOverlay, host),
234                         failure: Y.bind(host.defaultFailureHandler, host)
235                     }
236                 });
237             }
238         }
239     }, {
240         NS: "ExecuteScriptAction",
241         NAME: "ExecuteScriptAction",
242         ATTRS: {
243             onClick: {
244                 _inputex: {
245                     _type: "script",
246                     label: "On click"
247                 }
248             }
249         }
250     });
251     Plugin.ExecuteScriptAction = ExecuteScriptAction;
252     /**
253      *  @class
254      *  @name Y.Plugin.SaveObjectAction
255      *  @extends Y.Plugin.Action
256      *  @constructor
257      */
258     var SaveObjectAction = function() {
259         SaveObjectAction.superclass.constructor.apply(this, arguments);
260     };
261     Y.extend(SaveObjectAction, Action, {
262         execute: function(e) {
263             var overlayGuest,
264                     host = this.get(HOST),
265                     guest = host.get("root"),
266                     variable = this.get("variable.evaluated"),
267                     data = variable.get("name") + ".properties",
268                     script = this.get("clearStorage") ? data + ".clear();" : "", i;
269             if (guest.showOverlay && guest.hideOverlay) {
270                 overlayGuest = guest;
271                 overlayGuest.showOverlay();
272             }
273 
274             for (i in e.value) {
275                 script += data + ".put('" + (i + "").replace(/'/g, "\\'") + "','" + (e.value[i] + "").replace(/'/g, "\\'") + "');";
276             }
277 
278             Wegas.Facade.VariableDescriptor.sendRequest({
279                 request: "/Script/Run/" + Wegas.Facade.Game.get('currentPlayerId'),
280                 cfg: {
281                     method: "POST",
282                     data: Y.JSON.stringify({
283                         "@class": "Script",
284                         language: "JavaScript",
285                         content: script
286                     })
287                 },
288                 on: {
289                     success: function(r) {
290                         if (overlayGuest) {
291                             overlayGuest.hideOverlay();
292                         }
293                     },
294                     failure: function(r) {
295                         if (overlayGuest) {
296                             overlayGuest.hideOverlay();
297                         }
298                     }
299                 }
300             });
301         }
302     }, {
303         NS: "SaveObjectAction",
304         NAME: "SaveObjectAction",
305         ATTRS: {
306             variable: {
307                 /**
308                  * The target variable, returned either based on the name attribute,
309                  * and if absent by evaluating the expr attribute.
310                  */
311                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
312                 _inputex: {
313                     _type: "variableselect",
314                     legend: "Save to",
315                     classFilter: ["ObjectDescriptor"]
316                 }
317             },
318             targetEvent: {
319                 value: "submit"
320             },
321             clearStorage: {
322                 type: "boolean",
323                 value: true,
324                 _inputex: {
325                     label: "Replace Storage",
326                     description: "Will remove existing data. Else add them to the existing one."
327                 }
328             }
329         }
330     });
331     Plugin.SaveObjectAction = SaveObjectAction;
332 });
