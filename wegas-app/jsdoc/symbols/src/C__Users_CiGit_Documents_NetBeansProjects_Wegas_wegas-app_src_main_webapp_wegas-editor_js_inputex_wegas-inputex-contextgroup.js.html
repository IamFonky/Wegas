 13 YUI.add("wegas-inputex-contextgroup", function(Y) {
 14     "use strict";
 15 
 16     var contextGroup = function(options) {
 17         contextGroup.superclass.constructor.call(this, options);
 18     };
 19     Y.extend(contextGroup, Y.inputEx.Group, {
 20         setOptions: function(options) {
 21             var o = Y.clone(options, true), keys = [], i;
 22             this.contextKey = o.contextKey;
 23             this.fields = o.fields || [];
 24             for (i in o.fields) {
 25                 keys.push(o.fields[i].name);
 26             }
 27             Array.prototype.splice.apply(this.fields, [0, 0].concat([{type: "select", choices: keys}]));
 28             this.contextKeys = keys;
 29             contextGroup.superclass.setOptions.call(this, o);
 30         },
 31         renderFields: function() {
 32             contextGroup.superclass.renderFields.apply(this, arguments);
 33             this.inputs[0].on("updated", function(val) {
 34                 var value;
 35                 if (val === this.context || typeof this.context === "undefined") {
 36                     return;
 37                 }
 38                 value = this.getFieldByName(this.context).getValue();
 39                 value[this.contextKey] = val;
 40                 this.setValue(value);
 41             }, this);
 42             this.enableFields(this.contextKeys[0]);
 43         },
 44         setValue: function(val) {
 45             this.getFieldByName(val[this.contextKey]).setValue(val);
 46             this.enableFields(val[this.contextKey]);
 47 
 48         },
 49         enableFields: function(ctx) {
 50             var sel, i;
 51             this.context = ctx;
 52             for (i in this.fields) {
 53                 if (this.fields[i].name === this.context) {
 54                     sel = +i;
 55                 }
 56             }
 57 
 58             Y.Array.each(this.inputs, function(item, index) {
 59                 item.hide();
 60                 if (index === sel || index === 0) {
 61                     item.show();
 62                 }
 63             });
 64             this.inputs[0].setValue(this.context);
 65         },
 66         getValue: function() {
 67             return this.getFieldByName(this.context).getValue();
 68         }
 69     });
 70     Y.inputEx.registerType("contextgroup", contextGroup);
 71 });
