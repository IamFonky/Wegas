60      public void PlayerJoinTest() {
61          NumberDescriptor testNumber;
62          testNumber = new NumberDescriptor("number");
63          testNumber.setDefaultInstance(new NumberInstance(0));
64          testNumber.setScope(new TeamScope());
65  
66          NumberDescriptor testNumber2;
67          testNumber2 = new NumberDescriptor("number2");
68          testNumber2.setDefaultInstance(new NumberInstance(0));
69          testNumber2.setScope(new GameScope());
70  
71          TriggerDescriptor trigger = new TriggerDescriptor();
72          trigger.setScope(new TeamScope());
73          trigger.setDefaultInstance(new TriggerInstance());
74          trigger.setTriggerEvent(new Script("1===1"));
75          trigger.setPostTriggerEvent(new Script("number.value = " + FINAL_VALUE));
76          trigger.setOneShot(Boolean.TRUE);
77  
78          TriggerDescriptor trigger2 = new TriggerDescriptor();
79          trigger2.setScope(new TeamScope());
80          trigger2.setDefaultInstance(new TriggerInstance());
81          trigger2.setTriggerEvent(new Script("true"));
82          trigger2.setPostTriggerEvent(new Script("number2.value += 1 "));
83          trigger2.setOneShot(Boolean.FALSE);
84  
85          descriptorFacade.create(gameModel.getId(), testNumber);
86          descriptorFacade.create(gameModel.getId(), testNumber2);
87          descriptorFacade.create(gameModel.getId(), trigger);
88          descriptorFacade.create(gameModel.getId(), trigger2);
89  
90          Team team3 = new Team("test-team3");
91          Team team4 = new Team("test-team4");
92  
93          team3.setGame(game);
94  
95          team4.setGame(game);
96  
97          teamFacade.create(team3);
98  
99          teamFacade.create(team4);
100         Player testPlayer0 = new Player("TestPlayer0");
101         Player testPlayer1 = new Player("TestPlayer1");
102 
103         playerFacade.create(team.getId(), testPlayer0);
104         playerFacade.create(team4.getId(), testPlayer1);
105 
106         NumberDescriptor number = (NumberDescriptor) descriptorFacade.find(testNumber.getId());
107         /* CONTEXT? */
108         assert FINAL_VALUE == number.getValue(testPlayer0);
109         assert FINAL_VALUE == number.getValue(testPlayer1);
110 
111         /* REFRESH CONTEXT */
112         Assert.assertEquals(FINAL_VALUE, ((NumberInstance) instanceFacade.find(testNumber.getId(), testPlayer0)).getValue(), 0.0);
113         Assert.assertEquals(FINAL_VALUE, ((NumberInstance) instanceFacade.find(testNumber.getId(), testPlayer1)).getValue(), 0.0);
114 
115         /*
116          * Player created before variable -> state machines don't execute
117          */
118         Assert.assertEquals(0.0, ((NumberInstance) instanceFacade.find(testNumber.getId(), player2)).getValue(), 0.0);
119         /*
120          * Game Scope trigger increase for each player added after trigger creation
121          */
122         Assert.assertEquals(2, ((NumberInstance) instanceFacade.find(testNumber2.getId(), testPlayer0)).getValue(), 0.0);
123         /*
124          * add a player in not empty team then Reset, trigger will execute
125          */
126         playerFacade.create(team4.getId(), new Player("TestPlayer5"));
127         gameModelFacade.reset(gameModel.getId());
128         Assert.assertEquals(FINAL_VALUE, ((NumberInstance) instanceFacade.find(testNumber.getId(), testPlayer0)).getValue(), 0.0);
129         Assert.assertEquals(FINAL_VALUE, ((NumberInstance) instanceFacade.find(testNumber.getId(), testPlayer1)).getValue(), 0.0);
130         Assert.assertEquals(FINAL_VALUE, ((NumberInstance) instanceFacade.find(testNumber.getId(), player2)).getValue(), 0.0);
131 
132         /*
133          * trigger2 will execute numberOfPlayers times after a reset -> increment testNumber2 by the same amount.
134          * For each player
135          */
136         Assert.assertEquals(playerFacade.find(testPlayer0.getId()).getGame().getPlayers().size(), ((NumberInstance) instanceFacade.find(testNumber2.getId(), testPlayer0)).getValue(), 0.0);
137 
138         playerFacade.create(team4.getId(), new Player("TestPlayer6"));
139         Assert.assertEquals(playerFacade.find(testPlayer0.getId()).getGame().getPlayers().size(), ((NumberInstance) instanceFacade.find(testNumber2.getId(), testPlayer0)).getValue(), 0.0);
140         gameModelFacade.reset(gameModel.getId());
141         Assert.assertEquals(playerFacade.find(testPlayer0.getId()).getGame().getPlayers().size(), ((NumberInstance) instanceFacade.find(testNumber2.getId(), testPlayer0)).getValue(), 0.0);
142         /*
143          * Player added in empty team.
144          */
145         playerFacade.create(team3.getId(), new Player("TestPlayer7"));
146         Assert.assertEquals(playerFacade.find(testPlayer0.getId()).getGame().getPlayers().size(), ((NumberInstance) instanceFacade.find(testNumber2.getId(), testPlayer0)).getValue(), 0.0);
147         gameModelFacade.reset(gameModel.getId());
148         Assert.assertEquals(playerFacade.find(testPlayer0.getId()).getGame().getPlayers().size(), ((NumberInstance) instanceFacade.find(testNumber2.getId(), testPlayer0)).getValue(), 0.0);
149         playerFacade.create(team3.getId(), new Player("TestPlayer8"));
150         playerFacade.create(team3.getId(), new Player("TestPlayer9"));
151         playerFacade.create(team3.getId(), new Player("TestPlayer10"));
152         Assert.assertEquals(playerFacade.find(testPlayer0.getId()).getGame().getPlayers().size(), ((NumberInstance) instanceFacade.find(testNumber2.getId(), testPlayer0)).getValue(), 0.0);
