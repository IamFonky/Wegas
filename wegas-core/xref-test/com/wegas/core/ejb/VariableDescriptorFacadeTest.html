322         Assert.assertEquals(VARIABLENAME1, ((ListDescriptor) childrenDescriptors.get(0)).item(0).getName());// Check if item was successfully added
323 
324         vdf.move(vd1.getId(), list1.getId(), 1);                                // Move first item to second position
325         childrenDescriptors = vdf.findByGameModelId(gameModel.getId());         // Refresh
326         Assert.assertEquals(VARIABLENAME2, ((ListDescriptor) childrenDescriptors.get(0)).item(0).getName());
327 
328         // 2nd case: from root to descriptor
329         StringDescriptor vd3 = new StringDescriptor(VARIABLENAME3);

331         vdf.create(gameModel.getId(), vd3);
332 
333         vdf.move(vd3.getId(), list1.getId(), 0);                                // Move first item to index 0
334         childrenDescriptors = vdf.findByGameModelId(gameModel.getId());         // Refresh
335         Assert.assertEquals(VARIABLENAME3, ((ListDescriptor) childrenDescriptors.get(0)).item(0).getName());
336         Assert.assertEquals(1, childrenDescriptors.size());
337 
338         // 3rd case: from one descriptor to another
339         ListDescriptor list2 = new ListDescriptor(LISTNAME2);
340         list2.setDefaultInstance(new ListInstance());
341         vdf.create(gameModel.getId(), list2);
342 
343         vdf.move(vd3.getId(), list2.getId(), 0);                                // Move first item to index 0
344         childrenDescriptors = vdf.findByGameModelId(gameModel.getId());         // Refresh
345         Assert.assertEquals(VARIABLENAME3, ((ListDescriptor) childrenDescriptors.get(1)).item(0).getName());
346         Assert.assertEquals(2, ((ListDescriptor) childrenDescriptors.get(0)).size());
347 
348         vdf.remove(vd1.getId());                                                // Clean up
349         vdf.remove(vd2.getId());
350         vdf.remove(vd3.getId());
351         vdf.remove(list1.getId());
352         vdf.remove(list2.getId());

368         ListDescriptor list2 = new ListDescriptor(LISTNAME2, new ListInstance());

370 
371         ListDescriptor list3 = new ListDescriptor(LISTNAME3, new ListInstance());
372         vdf.createChild(list1.getId(), list3);
373 
374         ListDescriptor list4 = new ListDescriptor(LISTNAME4, new ListInstance());
375         vdf.createChild(list3.getId(), list4);
376

378         vdf.createChild(list4.getId(), nb);
379 
380         DescriptorListI duplicate =
381                 (DescriptorListI) vdf.duplicate(list1.getId());                 // Duplicate a root variable
382         Assert.assertEquals(10.0, ((NumberDescriptor) ((DescriptorListI) ((DescriptorListI) duplicate.item(1)).item(0)).item(0)).getInstance(player).getValue());
383 
384         duplicate = (DescriptorListI) vdf.duplicate(list3.getId());             // Duplicate a sub child variable
385         Assert.assertEquals(10.0, ((NumberDescriptor) ((DescriptorListI) duplicate.item(0)).item(0)).getInstance(player).getValue());
386 
387         GameModel duplicateGm = gameModelFacade.duplicateWithDebugGame(gameModel.getId());
388         DescriptorListI find = (DescriptorListI) vdf.find(duplicateGm, LISTNAME1);
389         Assert.assertEquals(10.0, ((NumberInstance) ((NumberDescriptor) ((DescriptorListI) ((DescriptorListI) find.item(1)).item(0)).item(0)).getScope().getVariableInstances().values().iterator().next()).getValue());
