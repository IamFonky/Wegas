141     /**
142      * Private method that initialise review instance when dispatching
143      *
144      * @param prd      the peer review descriptor
145      * @param author   peer review instance corresponding to the author
146      * @param reviewer peer review instance corresponding to the reviewer
147      * @return a brand new review, ready to be edited by the reviewer
148      */
149     private Review createReview(PeerReviewDescriptor prd, PeerReviewInstance author, PeerReviewInstance reviewer) {
150         Review r = new Review();
151         r.setReviewState(Review.ReviewState.DISPATCHED);
152         r.setAuthor(author);
153         r.setReviewer(reviewer);
154         for (EvaluationDescriptor ed : prd.getFeedback().getEvaluations()) {
155             EvaluationInstance ei = ed.createInstance();
156             ei.setFeedbackReview(r);
157             r.getFeedback().add(ei);
158         }
159         for (EvaluationDescriptor ed : prd.getFbComments().getEvaluations()) {
160             EvaluationInstance ei = ed.createInstance();
161             ei.setCommentsReview(r);
162             r.getComments().add(ei);
163         }
164         return r;
165     }

167     /**
168      * Called by the teacher, it will take each PeerReviewInstance matching the
169      * given peer review descriptor and dispatch them (who review who?)
170      *
171      * @param prd peer review descriptor to dispatch
172      */
173     public void dispatch(PeerReviewDescriptor prd) {
174         Collection<VariableInstance> values = prd.getScope().getVariableInstances().values();
175         List<PeerReviewInstance> pris = new ArrayList(values);
176         int numberOfReview;
177 
178         if (prd.getGameModel().getTemplate()) {
179             // Edit Scenario Case -> there is only one game (debug) and one player (TestPlayer)
180             // In this case, allow the player to review itself once
181             numberOfReview = 1;
182         } else {
183             /*
184              * Real Game: evict test instance(s)
185              *
186              * In case peerreview variable is game/gameModel scoped, there will be only one 
187              * instance for the whole game and such an instance will be evicted.... For the 
188              * time, it sounds like OK since such a review seems useless
189              */
190             for (Game game : prd.getGameModel().getGames()) {
191                 for (Team team : game.getTeams()) {
192                     if (team instanceof DebugTeam) {
193                         for (Player p : team.getPlayers()) {
194                             PeerReviewInstance instance = prd.getInstance(p);
195                             // Discared instance
196                             instance.setReviewState(PeerReviewDescriptor.ReviewingState.DISCARDED);
197                             pris.remove(instance);
198                             variableInstanceFacade.merge(instance);
199                         }
200                     }
201                 }
202             }
203             numberOfReview = Math.min(prd.getMaxNumberOfReview(), pris.size() - 1);
204             //throw WegasErrorMessage.error("Unable to dispatch reviews: there is not enough players");
205         }
206 
207         // Todo: evict dispatched instances
208         Collections.shuffle(pris);
209 
210         int i, j;
211         for (i = 0; i < pris.size(); i++) {
212             PeerReviewInstance author = pris.get(i);
213             if (author.getReviewState() == PeerReviewDescriptor.ReviewingState.SUBMITTED
214                     || author.getReviewState() == PeerReviewDescriptor.ReviewingState.NOT_STARTED) {
215                 logger.warn("Dispatch Author");
216                 List<Review> reviewed = author.getReviewed();
217                 for (j = 1; j <= numberOfReview; j++) {
218                     PeerReviewInstance reviewer = pris.get((i + j) % pris.size());
219                     Review r = createReview(prd, author, reviewer);
220                     reviewed.add(r);
221                     //reviewer.getToReview().add(r);
222                 }
223                 author.setReviewState(PeerReviewDescriptor.ReviewingState.DISPATCHED);
224             }
225         }
226         for (PeerReviewInstance pri : pris) {
227             variableInstanceFacade.merge(pri);
228             em.flush();
229         }
230     }

287     /**
288      * Save a review posted by a player
289      *
290      * @param pri   peer review instance containing the review
291      * @param other the review to merge within pri
292      * @return
293      */
294     public Review saveReview(PeerReviewInstance pri, Review other) {
295         Review review = this.findReview(other.getId());
296         PeerReviewInstance author = review.getAuthor();
297         PeerReviewInstance reviewer = review.getReviewer();
298 
299         /*
300          * if author is posting and only if review state is notified:
301          * update comments only
302          */
303         if (pri == author && review.getReviewState() == Review.ReviewState.NOTIFIED) {
304             mergeEvaluations(other.getComments());
305         }
306 
307         /*
308          * if reviewer is posting and only if review state is dispatched:
309          * update evaluation
310          */
311         if (pri == reviewer && review.getReviewState() == Review.ReviewState.DISPATCHED) {
312             mergeEvaluations(other.getFeedback());
313         }
314 
315         em.merge(review);
316         return review;
317     }

319     /**
320      * Submitting a review occurs twice in the whole process First time when the
321      * reviewer post his review. In this case, the review switch from DISPATCHED
322      * to REVIEWED. The second time is when the author post his comments, switch
323      * from NOTIFIED to COMPLETED
324      *
325      * @param review the review to submit
326      * @return review
327      */
328     public Review submitReview(Review review) {
329         Review r = this.findReview(review.getId());
330         PeerReviewInstance pri = this.getPeerReviewInstanceFromReview(r);
331         r = this.saveReview(pri, review);
332         if (r.getReviewState() == Review.ReviewState.DISPATCHED) {
333             r.setReviewState(Review.ReviewState.REVIEWED);
334             em.merge(r);
335         } else if (r.getReviewState() == Review.ReviewState.NOTIFIED) {
336             r.setReviewState(Review.ReviewState.COMPLETED);
337             em.merge(r);
338         }
339         return r;
340     }

353     /**
354      * Reviewing phase is over -> authors will be able to see feedbacks
355      *
356      * @param prd the PeerReviewDescriptor
357      */
358     public void notify(PeerReviewDescriptor prd) {
359         List<PeerReviewInstance> pris = new ArrayList(prd.getScope().getVariableInstances().values());
360         for (PeerReviewInstance pri : pris) {
361             for (Review review : pri.getReviewed()) {
362                 if (review.getReviewState() == Review.ReviewState.DISPATCHED || review.getReviewState() == Review.ReviewState.REVIEWED) {
363                     review.setReviewState(Review.ReviewState.NOTIFIED);
364                 }
365             }
366             variableInstanceFacade.merge(pri);
367             requestManager.addUpdatedInstance(pri);
368         }
369     }

385     /**

388      * @param prd

390     public void close(PeerReviewDescriptor prd) {
391         List<PeerReviewInstance> pris = new ArrayList(prd.getScope().getVariableInstances().values());
392         for (PeerReviewInstance pri : pris) {
393             for (Review review : pri.getReviewed()) {
394                 if (review.getReviewState() == Review.ReviewState.NOTIFIED
395                         || review.getReviewState() == Review.ReviewState.COMPLETED) {
396                     review.setReviewState(Review.ReviewState.CLOSED);
397                 }
398             }
399             variableInstanceFacade.merge(pri);
400             requestManager.addUpdatedInstance(pri);
401         }

418     /**
419      * Since PeerReviewDescriptor toReview variable is only referenced by its
420      * own private name on the JSON side, we have to resolve those name to
421      * effective VariableDescriptor
422      *
423      * Moreover, as the variable may not yet exists (especially when posting a
424      * whole GameModel) when the PeerReviewDescriptor is created, we'll have to
425      * wait to resolve such identifier.
426      *
427      * This is done by listening to DescriptorRevivedEvent
428      *
429      * @param event
430      */
431     public void descriptorRevivedEvent(@Observes DescriptorRevivedEvent event) {
432         if (event.getEntity() instanceof PeerReviewDescriptor) {
433             logger.debug("Received DescriptorRevivedEvent event");
434             PeerReviewDescriptor reviewD = (PeerReviewDescriptor) event.getEntity();
435             try {
436                 String toReviewName = reviewD.getImportedToReviewName();
437                 GameModel gameModel = reviewD.getGameModel();
438                 VariableDescriptor toReview = variableDescriptorFacade.find(gameModel, toReviewName);
439 
440                 reviewD.setToReview(toReview);
441             } catch (WegasNoResultException ex) {
442                 logger.error("Failed te revive ReviewDescriptor", ex);
443                 reviewD.setToReview(null);
444             }
445         }
