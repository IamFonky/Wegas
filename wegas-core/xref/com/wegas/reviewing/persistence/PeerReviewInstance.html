8   package com.wegas.reviewing.persistence;

10  import com.wegas.core.persistence.AbstractEntity;
11  import com.wegas.core.persistence.ListUtils;
12  import com.wegas.core.persistence.variable.VariableInstance;
13  import java.util.ArrayList;
14  import java.util.List;
15  import javax.persistence.CascadeType;
16  import javax.persistence.Entity;
17  import javax.persistence.OneToMany;
18  
19  /**
20   *
21   * Instance of the PeerReviewDescriptor variable
22   *
23   * Author:<br />
24   * - has to review several other authors: <code>toReview</code> Review
25   * list<br />
26   * - is reviewed by several other authors: <code>reviewed</code> Review list
27   *
28   * The review is in a specific state, see PeerReviewDescriptor
29   *
30   *
31   * @see PeerReviewDescriptor
32   *
33   * @author Maxence Laurent (maxence.laurent gmail.com)
34   */
35  @Entity
36  public class PeerReviewInstance extends VariableInstance {
37  
38      private static final long serialVersionUID = 1L;
39  
40      /**
41       * Current review state
42       */
43      private PeerReviewDescriptor.ReviewingState reviewState = PeerReviewDescriptor.ReviewingState.NOT_STARTED;
44  
45      /**
46       * List of review that contains feedback written by player owning this
47       */
48      @OneToMany(mappedBy = "reviewer", cascade = CascadeType.ALL, orphanRemoval = true)
49      private List<Review> toReview = new ArrayList<>();
50  
51      /**
52       * List of review that contains others feedback
53       */
54      @OneToMany(mappedBy = "author", cascade = CascadeType.ALL, orphanRemoval = true)
55      private List<Review> reviewed = new ArrayList<>();
56  
57      /**
58       * Get the current state of the review
59       *
60       * @return the current state
61       */
62      public PeerReviewDescriptor.ReviewingState getReviewState() {
63          return reviewState;
64      }
65  
66      /**
67       * Set the current state of the review
68       *
69       * @param state the new current state
70       */
71      public void setReviewState(PeerReviewDescriptor.ReviewingState state) {
72          this.reviewState = state;
73      }
74  
75      /**
76       * Get the list of feedback to write to review others authors
77       *
78       * @return the list of feedback
79       */
80      public List<Review> getToReview() {
81          return toReview;
82      }
83  
84      /**
85       * Set the list of feedback to write to review others authors
86       *
87       * @param toReview list of Review
88       */
89      public void setToReview(List<Review> toReview) {
90          this.toReview = toReview;
91      }
92  
93      /**
94       * Get the list of feedback written by others authors
95       *
96       * @return
97       */
98      public List<Review> getReviewed() {
99          return reviewed;
100     }
101 
102     /**
103      * Set the list of feedback written by others authors
104      *
105      * @param reviewed
106      */
107     public void setReviewed(List<Review> reviewed) {
108         this.reviewed = reviewed;
109     }
110 
111     @Override
112     public void merge(AbstractEntity a) {
113         if (a instanceof PeerReviewInstance) {
114             PeerReviewInstance o = (PeerReviewInstance) a;
115             this.setReviewState(o.getReviewState());
116             this.setReviewed(ListUtils.mergeLists(this.getReviewed(), o.getReviewed()));
117             this.setToReview(ListUtils.mergeLists(this.getToReview(), o.getToReview()));
118         }
119     }

121 }
