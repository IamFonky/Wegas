27  @Stateless
28  @LocalBean
29  public class RoleFacade extends BaseFacade<Role> {
30  
31      private static Logger logger = LoggerFactory.getLogger(RoleFacade.class);
32  
33      /**
34       *
35       */
36      public RoleFacade() {
37          super(Role.class);
38      }
39  
40      @Override
41      public void remove(Role role) {
42          // Strike out all members from the role to avoid pkey violation
43          Set<AbstractAccount> abstractAccounts = role.getAbstractAccounts();
44          logger.error("REMOVE ROLE " + role.getName() + " " + abstractAccounts.size() + " members");
45  
46          for (AbstractAccount aa : role.getAbstractAccounts()) {
47              logger.error("Account: " +  aa);
48              aa.removeRole(role);
49          }
50  
51          super.remove(role);
52      }
53  
54      /**
55       *
56       * @param name
57       * @return
58       * @throws WegasNoResultException role not found
59       */
60      public Role findByName(String name) throws WegasNoResultException {
61          try {
62              final TypedQuery<Role> query = getEntityManager().createNamedQuery("Role.findByName", Role.class);
63              query.setParameter("name", name);
64              return query.getSingleResult();
65          } catch (NoResultException ex) {
66              throw new WegasNoResultException(ex);
67          }
68      }
69  }
