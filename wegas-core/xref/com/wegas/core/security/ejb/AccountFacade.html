80      @Override
81      public AbstractAccount update(final Long entityId, final AbstractAccount account) {
82          if (!account.getUsername().equals("") && account.getUsername() != null) {// If the provided username is not null
83              try {
84                  AbstractAccount a = this.findByUsername(account.getUsername());
85                  if (!a.getId().equals(account.getId())) {                       // and we can find an account with the username which is not the one we are editing,
86                      throw WegasErrorMessage.error("This username is already in use");// throw an exception
87                  }
88              } catch (WegasNoResultException e) {
89                  // GOTCHA no username could be found, do not use
90              }
91          }
92  
93          AbstractAccount oAccount = super.update(entityId, account);
94  
95          if (SecurityUtils.getSubject().isPermitted("User:Edit:" + entityId)) {
96              Set<Role> revivedRoles = new HashSet<>();
97              for (Role r : account.getRoles()) {
98                  try {
99                      revivedRoles.add(roleFacade.find(r.getId()));
100                 } catch (EJBException e) {
101                     // not able to revive this role
102                 }
103             }
104             oAccount.setRoles(revivedRoles);
105         }
106 
107         return oAccount;
108     }

165     public List<JpaAccount> findByNameEmailOrUsername(String input) {
166         String[] tokens = input.split(" ");
167 
168         CriteriaBuilder cb = getEntityManager().getCriteriaBuilder();
169         CriteriaQuery cq = cb.createQuery();
170         Root<JpaAccount> jpaAccount = cq.from(JpaAccount.class);
171 
172         Predicate[] prs = {};
173 
174         List<Predicate> andPreds = new ArrayList<>();
175         for (String token : tokens) {
176             if (!token.isEmpty()) {
177                 token = "%" + token.toLowerCase() + "%";
178                 List<Predicate> orPreds = new ArrayList<>();
179                 orPreds.add(cb.like(cb.lower(jpaAccount.get("firstname")), token));
180                 orPreds.add(cb.like(cb.lower(jpaAccount.get("lastname")), token));
181                 orPreds.add(cb.like(cb.lower(jpaAccount.get("email")), token));
182                 orPreds.add(cb.like(cb.lower(jpaAccount.get("username")), token));
183 
184                 andPreds.add(cb.or(orPreds.toArray(prs)));
185             }
186         }
187 
188         cq.where(cb.and(andPreds.toArray(prs)));
189 
190         Query q = getEntityManager().createQuery(cq);
191         q.setMaxResults(MAXRESULT);
192         return (List<JpaAccount>) q.getResultList();
193     }

195     /**
196      *
197      * @param name
198      * @param withEmail
199      * @return
200      */
201     public List<JpaAccount> findByNameOrEmail(String name, boolean withEmail) {
202         ArrayList<JpaAccount> accounts = new ArrayList();
203         String splidedName[] = name.split(" ");
204         for (int i = 0; i < splidedName.length; i++) {
205             StringBuilder firstnameBuilder = new StringBuilder();
206             StringBuilder lastnameBuilder = new StringBuilder();
207             String firstname, lastname;
208             for (int ii = 0; ii <= i; ii++) {
209                 firstnameBuilder.append(splidedName[ii]).append(" ");
210             }
211             firstname = normalizeName(firstnameBuilder.toString());
212             if (i < splidedName.length - 1) {
213                 for (int ii = i + 1; ii < splidedName.length; ii++) {
214                     lastnameBuilder.append(splidedName[ii]).append(" ");
215                 }
216             }
217             lastname = normalizeName(lastnameBuilder.toString());
218             List<JpaAccount> tempAccount = this.findByNameOrEmailQuery("name", firstname, lastname);
219             accounts = this.compareExistingAccount(tempAccount, accounts);
220             tempAccount = this.findByNameOrEmailQuery("name", lastname, firstname);
221             accounts = this.compareExistingAccount(tempAccount, accounts);
222             if (withEmail && splidedName.length == 1) {
223                 tempAccount = this.findByNameOrEmailQuery("email", splidedName[i], null);
224                 accounts = this.compareExistingAccount(tempAccount, accounts);
225             }
226         }
227         return accounts;
228     }

230     private String normalizeName(String name) {
231         if (name.equals("")) {
232             return "%";
233         } else {
234             return name.substring(0, name.length() - 1);
235         }

267             if (!accounts.contains(tempAccount1)) {

293     public List<JpaAccount> getAutoCompleteFull(String value, Long gameId) {
294         List<JpaAccount> accounts = this.getAutoComplete(value);
295         for (int i = 0; i < accounts.size(); i++) {
296             JpaAccount ja = accounts.get(i);
297             getEntityManager().detach(ja);
298             ja.setEmail(ja.getEmail().replaceFirst("([^@]{1,4})[^@]*(@.*)", "$1****$2"));
299             try {
300                 Player p = playerFacade.findByGameIdAndUserId(gameId, ja.getUser().getId());
301                 if (ja.getUser() == p.getUser()) {
302                     accounts.remove(i);
303                 }
304             } catch (WegasNoResultException e) {
305                 //Gotcha
306             }
307         }
308         return accounts;

327     public List<Map> findAccountsByEmailValues(List<String> values) {
328         List<Map> returnValue = new ArrayList<>();
329         List<String> notValidValue = new ArrayList<>();
330         for (String value : values) {
331             try {
332                 Map account = new HashMap<>();
333                 JpaAccount a = findByEmail(value.trim());
334                 if (a.getFirstname() != null && a.getLastname() != null) {
335                     account.put("label", a.getFirstname() + " " + a.getLastname());
336                 } else {
337                     account.put("label", a.getEmail());
338                 }
339                 account.put("value", a.getId());
340                 returnValue.add(account);
341             } catch (WegasNoResultException e2) {
342                 notValidValue.add(value);
343             }
344         }
345         requestManager.addEvent(new WarningEvent("NotAddedAccount", notValidValue));
346         return returnValue;
