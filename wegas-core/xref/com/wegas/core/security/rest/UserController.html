170     @GET
171     @Path("FindAccountsByEmailValues")
172     public List<Map> findAccountsByEmailValues(@QueryParam("values") List<String> values) {
173         List<Map> returnValue = new ArrayList<>();
174         List<String> notValidValue = new ArrayList<>();
175         for (String value : values) {
176             try {
177                 Map account = new HashMap<>();
178                 JpaAccount a = accountFacade.findByEmail(value.trim());
179                 if (a.getFirstname() != null && a.getLastname() != null) {
180                     account.put("label", a.getFirstname() + " " + a.getLastname());
181                 } else {
182                     account.put("label", a.getEmail());
183                 }
184                 account.put("value", a.getId());
185                 returnValue.add(account);
186             } catch (PersistenceException e2) {
187                 notValidValue.add(value);
188             }
189         }
190         requestManager.addEvent(new WarningEvent("NotAddedAccount", notValidValue));
191         return returnValue;
192     }

284     public void login(@QueryParam("email") String email,
285             @QueryParam("password") String password,
286             @QueryParam("remember")
287             @DefaultValue("false") boolean remember,
288             @Context HttpServletRequest request, @Context HttpServletResponse response) throws ServletException, IOException {
289 
290         Subject subject = SecurityUtils.getSubject();
291 
292         //if (!currentUser.isAuthenticated()) {
293         UsernamePasswordToken token = new UsernamePasswordToken(email, password);
294         token.setRememberMe(remember);
295         try {
296             subject.login(token);                                               // try to log in.
297         } catch (AuthenticationException aex) {
298             if (checkGameAccountKeylogin(email, password)) {                    //Login failed, maybe create a GameAccount or fail
299                 GameAccountKey gameAccountKey = gameFacade.findGameAccountKey(email);
300                 subject.login(token);
301             } else {
302                 throw new WegasException("Email/password combination not found");
303             }
304         }
305     }

528     private Boolean checkGameAccountKeylogin(String key, String password) {
529         if (key.equals(password)) {                                             //Be kind, email password have to be the same...
530             /*
531              if there is a GameAccountKey, use it to create a new GameAccount 
532              */
533             try {
534                 GameAccountKey gameAccountKey = gameFacade.findGameAccountKey(key);
535                 if (gameAccountKey.getUsed()) {
536                     return false;
537                 }
538                 GameAccount gameAccount = new GameAccount();
539                 gameAccount.setEmail(key);
540                 gameAccount.setPassword(password);
541                 gameAccount.setGame(gameAccountKey.getGame());
542                 gameAccountKey.setUsed(Boolean.TRUE);
543                 this.signup(gameAccount);
544             } catch (SQLException | NoResultException ex) {
545                 return false;
546             }
547             return true;
548         }
549         return false;
550     }
