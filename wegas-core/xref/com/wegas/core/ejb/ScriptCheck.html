29  /**
30   * Script validation bean
31   *
32   * @author Cyril Junod <cyril.junod at gmail.com>
33   */
34  @Stateless
35  @LocalBean
36  @TransactionManagement(TransactionManagementType.BEAN)
37  public class ScriptCheck {
38  
39      private static final Logger logger = LoggerFactory.getLogger(ScriptCheck.class);
40      @Resource
41      private UserTransaction utx;
42      @EJB
43      private RequestFacade requestFacade;
44      @EJB
45      private ScriptFacade scriptFacade;
46  
47      /**
48       * Validate a given script, searching for errors in it.
49       *
50       * @param script the script to test
51       * @param player the player the script should run for
52       *
53       * @return Exception the exception found in script or null if none occured
54       */
55      public WegasScriptException validate(Script script, Player player) {
56          ScriptEngine engine = scriptFacade.instanciateEngine(player, script.getLanguage());
57  
58          return this.rollbackEval(engine, script, player.getId());
59  
60      }
61  
62      /**
63       * Execute a script before rolling it back.
64       *
65       * @param engine   the engine on which the script should execute
66       * @param script   the script
67       * @param playerId the player's id, needed to set up Java env.
68       *
69       * @return Exception the exception found in script or null if none occured
70       */
71      private WegasScriptException rollbackEval(ScriptEngine engine, Script script, Long playerId) {
72  
73          try {
74              utx.begin();
75              requestFacade.setPlayer(playerId);
76              try {
77                  engine.eval(script.getContent());
78              } catch (ScriptException ex) {
79                  logger.debug("Script Error: {} \n {}", script.getContent(), ex.getMessage(), ex.getStackTrace());
80                  return new WegasScriptException(script.getContent(), ex.getLineNumber(), ex.getMessage());
81              } catch (WegasRuntimeException ex) {                                // "Expected" Exception
82                  return null;
83              } catch (RuntimeException ex) {                                     // Java exception
84                  logger.debug("Script Error: {} \n {}", script.getContent(), ex.getMessage(), ex.getStackTrace());
85                  return new WegasScriptException(script.getContent(), ex.getMessage());
86              } finally {
87                  utx.rollback();
88              }
89          } catch (NotSupportedException | SystemException ex) {
90              logger.error("Transaction failed", ex);
91              throw WegasErrorMessage.error("I tried hardly, unfortunately without success");
92          }
93  
94          return null;
95      }
96  }
