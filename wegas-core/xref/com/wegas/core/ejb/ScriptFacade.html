76      /**
77       *
78       * Fires an engineInvocationEvent, which should be intercepted to customize
79       * engine scope.
80       *
81       * @param script
82       * @param arguments
83       * @return
84       * @throws ScriptException
85       * @throws WegasException
86       */
87      public Object eval(Script script, Map<String, AbstractEntity> arguments) throws ScriptException, WegasException {
88  
89          ScriptEngine engine = requestManager.getCurrentEngine();
90          if (engine == null) {
91              ScriptEngineManager mgr = new ScriptEngineManager();                // Instantiate the corresponding script engine
92  
93              try {
94                  engine = mgr.getEngineByName(script.getLanguage());
95                  // Invocable invocableEngine = (Invocable) engine;
96              } catch (NullPointerException ex) {
97                  logger.error("Could not find language", ex.getMessage(), ex.getStackTrace());
98                  throw new WegasException("Could not instantiate script engine for script" + script);
99              }
100             try {
101                 engineInvocationEvent.fire(
102                         new EngineInvocationEvent(requestManager.getPlayer(), engine));// Fires the engine invocation event, to allow extensions
103 
104             } catch (ObserverException ex) {
105                 throw (WegasException) ex.getCause();
106             }
107             requestManager.setCurrentEngine(engine);
108         }
109 
110         for (Entry<String, AbstractEntity> arg : arguments.entrySet()) {        // Inject the arguments
111             engine.put(arg.getKey(), arg.getValue());
112         }
113 
114         try {
115             return engine.eval(script.getContent());
116         } catch (ScriptException ex) {
117             logger.warn("{} in\n{}", ex.getMessage(), script.getContent());
118             requestManager.addException(
119                     new com.wegas.core.exception.ScriptException(script.getContent(), ex.getLineNumber(), ex.getMessage()));
120             throw new ScriptException(ex.getMessage(), script.getContent(), ex.getLineNumber());
121         }
122     }

124     /**
125      * Default customization of our engine: inject the script library, the root
126      * variable instances and some libraries.
127      *
128      * @param evt
129      * @throws ScriptException
130      * @throws WegasException
131      */
132     public void onEngineInstantiation(@Observes EngineInvocationEvent evt) throws ScriptException {
133         evt.getEngine().put("self", evt.getPlayer());                           // Inject current player
134         evt.getEngine().put("gameModel", evt.getPlayer().getGameModel());       // Inject current gameModel
135         evt.getEngine().put("Variable", variableDescriptorFacade);              // Inject the variabledescriptor facade
136         evt.getEngine().put("VariableDescriptorFacade", variableDescriptorFacade);// @backwardcompatibility
137         evt.getEngine().put("RequestManager", requestManager);                  // Inject the request manager
138         evt.getEngine().put("Event", event);                                    // Inject the Event manager
139         event.detachAll();
140 
141         for (Entry<String, GameModelContent> arg
142                 : evt.getPlayer().getGameModel().getScriptLibrary().entrySet()) { // Inject the script library
143             try {
144                 evt.getEngine().eval(arg.getValue().getContent());
145             } catch (ScriptException ex) {
146                 logger.warn("Error injecting script library: {} in\n{}", ex.getMessage(), arg.getValue());
147                 throw new ScriptException(ex.getMessage(), arg.getValue().getContent(), ex.getLineNumber());
148             }
149         }
150 
151         for (VariableDescriptor vd
152                 : evt.getPlayer().getGameModel().getChildVariableDescriptors()) { // Inject the variable instances in the script
153             VariableInstance vi = vd.getInstance(evt.getPlayer());
154             try {
155                 evt.getEngine().put(vd.getName(), vi);
156             } catch (IllegalArgumentException ex) {
157                 //logger.error("Missing name for Variable label [" + vd.getLabel() + "]");
158             }
159         }
160     }

163     /**
164      *
165      * @param scripts
166      * @param arguments
167      * @return
168      * @throws ScriptException
169      * @throws WegasException
170      */
171     public Object eval(List<Script> scripts, Map<String, AbstractEntity> arguments) throws ScriptException, WegasException {
172         if (scripts.isEmpty()) {
173             return null;
174         }
175         while (scripts.remove(null)) {
176         }                                                                        //remove null scripts
177         // @fixme test the most performant version
178         StringBuilder buf = new StringBuilder();
179         for (Script s : scripts) {                                              // Evaluate each script
180             try {
181                 buf.append(s.getContent());
182                 buf.append(";");
183             } catch (NullPointerException ex) {
184                 //script does not exist
185             }
186             //result = engine.eval(s.getContent());
187         }
188         return this.eval(new Script(buf.toString()));
189     }
