104 
105         ObjectMapper mapper = JacksonMapperProvider.getMapper();                // Retrieve a jackson mapper instance
106         GameModel gm = mapper.readValue(file, GameModel.class);                 // and deserialize file
107 
108         gm.setName(gameModelFacade.findUniqueName(gm.getName()));               // Find a unique name for this new game
109 
110         gameModelFacade.createWithDebugGame(gm);
111         return gm;

188     @PUT
189     @Path("{entityId: [1-9][0-9]*}/status/{status: [A-Z]*}")
190     public GameModel changeStatus(@PathParam("entityId") Long entityId, @PathParam("status") final GameModel.Status status) {
191         SecurityUtils.getSubject().checkPermission("GameModel:View:gm" + entityId);
192         GameModel gm = gameModelFacade.find(entityId);
193         Subject s = SecurityUtils.getSubject();
194         switch(status){
195             case LIVE:
196                 if (s.isPermitted("GameModel:View:gm" + gm.getId())
197                     || s.isPermitted("GameModel:Instantiate:gm" + gm.getId())
198                     || s.isPermitted("GameModel:Duplicate:gm" + gm.getId())) {
199                         gameModelFacade.live(gm);   
200                 }
201                 break;
202             case BIN:
203                 if (s.isPermitted("GameModel:Delete:gm" + entityId)){
204                     gameModelFacade.bin(gm);   
205                 }
206                 break;
207             case DELETE:
208                 if (s.isPermitted("GameModel:Delete:gm" + entityId)){
209                     gameModelFacade.delete(gm);   
210                 }
211                 break;
212 
213         }
214         return gm;
215     }
