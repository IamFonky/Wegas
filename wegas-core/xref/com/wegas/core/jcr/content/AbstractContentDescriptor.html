268      * @throws RepositoryException
269      */
270     public void sync() throws RepositoryException {
271         if (this.exist()) {                                                     //check existence then load it else create it
272             try {
273                 this.getContentFromRepository();
274             } catch (NullPointerException e) {
275                 if (!this.fileSystemAbsolutePath.equals("/")) {                 //Not a rootNode
276                     throw e;
277                 }
278             }
279             synched = true;
280         } else {
281             this.saveToRepository();
282             this.setContentToRepository();
283             synched = true;
284         }
285     }

384     /**
385      * Convert an absolute path (with or without WFS namespace) to path and name
386      * without namespace.
387      *
388      * @param absolutePath
389      */
390     //@XmlTransient
391     @JsonIgnore
392     private void parseAbsolutePath(String absolutePath) {
393         absolutePath = absolutePath.replaceAll(WFSConfig.WeGAS_FILE_SYSTEM_PREFIX, "");
394         if (!absolutePath.startsWith("/")) {
395             absolutePath = "/" + absolutePath;
396         }
397         if (absolutePath.equals("/")) {
398             this.name = "";
399             this.path = "/";
400         } else {
401             String[] dirs = absolutePath.split("/");
402             this.name = dirs[dirs.length - 1];
403             this.path = "/";
404 
405             for (int i = 1; i < dirs.length - 1; i++) {
406                 this.path += dirs[i];
407                 this.path += i < dirs.length - 2 ? "/" : "";
408             }
409         }
410     }

412     /**
413      * Convert name and path to a fileSystemAbsolutePath, including namespace ({@value WFSConfig#WeGAS_FILE_SYSTEM_PREFIX})
414      */
415     //@XmlTransient
416     @JsonIgnore
417     private void buildNamespaceAbsolutePath() {
418         if (this.path.equals("/")) {
419             this.fileSystemAbsolutePath = "/" + (this.name.equals("") ? "" : WFSConfig.WeGAS_FILE_SYSTEM_PREFIX + this.name);
420         } else {
421             this.fileSystemAbsolutePath = this.path.replaceAll("/(\\p{L})", "/" + WFSConfig.WeGAS_FILE_SYSTEM_PREFIX + "$1") + (this.name.equals("") ? "" : "/" + WFSConfig.WeGAS_FILE_SYSTEM_PREFIX + this.name);
422         }
423     }
