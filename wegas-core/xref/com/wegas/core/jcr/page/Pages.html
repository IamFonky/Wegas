47       * @throws RepositoryException
48       * @throws JSONException
49       */
50      public List<HashMap<String, String>> getIndex() throws RepositoryException {
51          final List<HashMap<String, String>> ret = new LinkedList<>();
52          if (!this.connector.exist(this.gameModelId)) {
53              return ret;
54          }
55          NodeIterator it = this.connector.listChildren(this.gameModelId);
56          Node n;
57          while (it.hasNext()) {
58              final HashMap<String, String> keyVal = new HashMap<>();
59              n = (Node) it.next();
60  
61              keyVal.put("id", n.getName());
62              if (n.hasProperty(Page.NAME_KEY)) {
63                  keyVal.put("name", n.getProperty(Page.NAME_KEY).getString());
64              }
65              if (n.hasProperty(Page.INDEX_KEY)) {
66                  keyVal.put("index", Long.toString(n.getProperty(Page.INDEX_KEY).getLong()));
67              }
68              ret.add(keyVal);
69          }
70          return ret;
71      }

211     private LinkedList<Page> getPages() throws RepositoryException {
212         final NodeIterator nodeIterator = this.connector.listChildren(this.gameModelId);
213         final LinkedList<Page> pages = new LinkedList<>();
214         while (nodeIterator.hasNext()) {
215             pages.add(new Page(nodeIterator.nextNode()));
216         }
217         return pages;
218     }
