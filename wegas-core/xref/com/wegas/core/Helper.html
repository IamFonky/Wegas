51      public static <T> T lookupBy(Context context, Class<T> type, Class service) throws NamingException {
52          try {
53              return (T) context.lookup("java:module/" + service.getSimpleName() + "!" + type.getName());
54          } catch (NamingException ex) {
55              try {
56                  return (T) context.lookup("java:global/classes/" + service.getSimpleName() + "!" + type.getName());
57              } catch (NamingException ex3) {
58                  try {
59                      return (T) context.lookup("java:global/embed-classes/" + service.getSimpleName() + "!" + type.getName());
60                  } catch (NamingException ex1) {
61                      try {
62                          return (T) context.lookup("java:global/cobertura/" + service.getSimpleName() + "!" + type.getName());
63                      } catch (NamingException ex2) {
64                          logger.error("Unable to retrieve to do jndi lookup on class: {}", type.getSimpleName());
65                          throw ex2;
66                      }
67                  }
68              }
69          }
70      }

108     private static String findUniqueName(final String name, List<String> usedNames, String pattern, String preSuff, String postSuff) {
109 
110         Pattern p = Pattern.compile(pattern);
111         Matcher matcher = p.matcher(name);
112 
113         int suff;
114         final String baseName;
115         if (matcher.matches()) {
116             baseName = matcher.group(1);
117             suff = Integer.parseInt(matcher.group(2)) + 1;
118         } else {
119             baseName = name;
120             suff = 2;
121         }
122 
123         String newName = name;
124         while (usedNames.contains(newName)) {
125             newName = baseName + preSuff + suff + postSuff;
126             suff++;
127         }
128 
129         return newName;
130     }

250     /**
251      * Encode a String to look like a JavaScript variable.
252      *
253      * @param name String to encode
254      * @return a String wich will be undestandable by JavaScript as a var
255      */
256     public static String camelCasify(String name) {
257         if (name.isEmpty()) {
258             throw new NullPointerException("Name is empty");
259         }
260         StringBuilder sb = new StringBuilder();
261         StringTokenizer st = new StringTokenizer(name);
262         String tmp;
263         Boolean first = true;
264         while (st.hasMoreTokens()) {                                            //CamelCase the name except first word (instance like)
265             tmp = st.nextToken();
266             if (first) {
267                 sb.append(tmp.substring(0, 1).toLowerCase());
268                 first = false;
269             } else {
270                 sb.append(tmp.substring(0, 1).toUpperCase());
271             }
272             sb.append(tmp.substring(1));
273             //sb.append(tmp.substring(1).toLowerCase());
274         }
275         return sb.toString();
276     }

321     /**
322      * Generate an alphanumeric token based on system time.
323      *
324      * @param maxLength Token maximum length. The shorter, the sooner it will
325      *                  collide.
326      * @return String token
327      */
328     public static String genToken(Integer maxLength) {
329         final String digits = "abcdefghijklmnopqrstuvwxyzABCDEFGHYJKLMNOPQRSTUVWXYZ1234567890";
330         final int digitSize = digits.length();
331         final StringBuilder sb = new StringBuilder();
332         int modulo;
333         long time = System.nanoTime();
334         while (time > 0) {
335             modulo = (int) (time % digitSize);
336             sb.append(digits.substring(modulo, modulo + 1));
337             time = time / digitSize;
338         }
339         return sb.toString().substring(0, Math.min(sb.length(), maxLength));
340     }

524     public static void recursiveDelete(File file) throws IOException {
525         if (file.isDirectory()) {
526             if (file.list().length == 0) {
527                 if (!file.delete()) {
528                     logger.warn("Failed to delete file {}", file.getName());
529                 }
530             } else {
531                 String files[] = file.list();
532                 for (String f : files) {
533                     File fileToDel = new File(file, f);
534                     recursiveDelete(fileToDel);
535                 }
536                 if (file.list().length == 0) {
537                     if (!file.delete()) {
538                         logger.warn("Failed to delete file {}", file.getName());
539                     }
540                 } else {
541                     throw new IOException("Could not empty " + file.getName());
542                 }
543             }
544 
545         } else {
546             if (!file.delete()) {
547                 logger.warn("Failed to delete file {}", file.getName());
548             }
549         }
550     }
