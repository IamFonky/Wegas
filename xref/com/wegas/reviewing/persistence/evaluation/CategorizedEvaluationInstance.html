8   package com.wegas.reviewing.persistence.evaluation;
9   
10  import com.wegas.core.persistence.AbstractEntity;
11  import javax.persistence.Column;
12  import javax.persistence.Entity;
13  
14  /**
15   * Evaluation instance corresponding to CategorizedEvaluationDescriptor
16   *
17   * @author Maxence Laurent (maxence.laurent at gmail.com)
18   */
19  @Entity
20  public class CategorizedEvaluationInstance extends EvaluationInstance {
21  
22      private static final long serialVersionUID = 1L;
23  
24      /**
25       * the chosen category (null means un-chosen)
26       */
27      @Column(name = "evaluationvalue")
28      private String value;
29  
30      /**
31       * get the chosen category
32       *
33       * @return the chosen category or null is not yet chosen
34       */
35      public String getValue() {
36          return value;
37      }
38  
39      /**
40       * Set the category
41       *
42       * @param category the category to set. If category does not match any
43       *                 category from the descriptor, category is set as NULL.
44       */
45      public void setValue(String category) {
46          if (this.getDescriptor() instanceof CategorizedEvaluationDescriptor) {
47              CategorizedEvaluationDescriptor descriptor = (CategorizedEvaluationDescriptor) this.getDescriptor();
48              if (descriptor.getCategories().contains(category)) {
49                  this.value = category;
50              } else {
51                  this.value = null;
52              }
53          } else {
54              this.value = null;
55          }
56      }
57  
58      @Override
59      public void merge(AbstractEntity a) {
60          if (a instanceof CategorizedEvaluationInstance) {
61              CategorizedEvaluationInstance o = (CategorizedEvaluationInstance) a;
62              super.merge(a);
63              this.setValue(o.getValue());
64          }
65      }
66  }
