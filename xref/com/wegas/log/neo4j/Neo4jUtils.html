55      /**
56       * Checks if the neo4j database is running. If not, it is not necessary to
57       * continue.
58       *
59       * @return true if the neo4j database is running, false otherwise
60       */
61      protected static boolean checkDataBaseIsRunning() {
62          if (NEO4J_SERVER_URL.isEmpty()) {
63              return false;
64          }
65          try {
66              Response response = getBuilder(NEO4J_SERVER_URL).get();
67              int status = response.getStatus();
68              response.close();
69              return status == 200;
70          } catch (ProcessingException ex) {
71              return false;
72          }
73      }

75      /**
76       * Sends a given query to the neo4j database and gets a response. The
77       * response is received as a string containing JSON format data.
78       *
79       * @param query the query to be submitted
80       * @return a string containing the query's answer
81       */
82      protected static String queryDBString(String query) {
83          final String qURL = NEO4J_SERVER_URL + "transaction/commit";
84          String entity = "{ \"statements\" : [ { \"statement\" : \"" + StringEscapeUtils.escapeJson(query) + "\" } ] }";
85          String result = null;
86          Response response = getBuilder(qURL).post(Entity.json(entity));
87          if (checkValidHttpResponse(response.getStatus())) {
88              result = response.readEntity(String.class);
89          }
90          response.close();
91          return result;
92      }

94      /**
95       * Extracts from the JSON result of a query the potential error message.
96       * If no message is found null is returned.
97       *
98       * @param result the result of the query
99       * @return the error message if an error occurred, null otherwise
100      */
101     protected static String extractErrorData(String result) {
102         ObjectMapper om = new ObjectMapper();
103         try {
104             JsonNode jn = om.readTree(result);
105             JsonNode jnErr = jn.path("errors");
106             Iterator<JsonNode> ite1 = jnErr.elements();
107             String err = "";
108             while (ite1.hasNext()) {
109                 JsonNode jn1 = ite1.next();
110                 err += ", " + jn1.path("message").asText();
111             }
112             if ("".equals(err)) return null;
113             return err.substring(2);
114         } catch (IOException ioe) {
115             logger.debug("Error in extractErrorData: " + ioe.getMessage());
116         }
117         return null;
118     }

120     /**
121      * Extracts the data part from the JSON result of a query. The data is
122      * returned as a list of string(s). If no data was found, an empty list is
123      * returned.
124      *
125      * @param result the result of a query
126      * @return the data list as a Json object
127      */
128     protected static ArrayNode extractListData(String result) {
129         ArrayNode on = objectMapper.createArrayNode();
130         ObjectMapper om = new ObjectMapper();
131         try {
132             JsonNode jn = om.readTree(result);
133             JsonNode jnRes = jn.path("results");
134             Iterator<JsonNode> ite1 = jnRes.elements();
135             while (ite1.hasNext()) {
136                 JsonNode jn1 = ite1.next();
137                 JsonNode jnDat = jn1.path("data");
138                 Iterator<JsonNode> ite2 = jnDat.elements();
139                 while (ite2.hasNext()) {
140                     JsonNode jn2 = ite2.next();
141                     JsonNode jnRow = jn2.path("row");
142                     Iterator<JsonNode> ite3 = jnRow.elements();
143                     while (ite3.hasNext()) {
144                         JsonNode jn3 = ite3.next();
145                         on.add(jn3);
146                     }
147                 }
148             }
149         } catch (IOException ioe) {
150             logger.debug("Error in extractListData: " + ioe.getMessage());
151         }
152         return on;
153     }

155     private static Invocation.Builder getBuilder(String URL) {
156         WebTarget target = client.target(URL);
157         if (Helper.isNullOrEmpty(NEO4J_BASIC_AUTH)) {
158             return target.request().accept(MediaType.APPLICATION_JSON);
159         } else {
160             return target.request().accept(MediaType.APPLICATION_JSON).header("Authorization", "Basic " + NEO4J_BASIC_AUTH);
161         }
162 
163     }

165     /**
166      * Checks the http status code. The successful operations return a status
167      * code between 200 and 299.
168      *
169      * @param status the status code to be tested
170      * @return true if the status code means success, false otherwise
171      */
172     private static boolean checkValidHttpResponse(int status) {
173         return ((status >= 200) && (status < 300));
