22  /**
23   *
24   * Faces component that print a StateMachine/Trigger State as xHTML. <br /> <br
25   * />
26   * <pre>
27   * <b>Usage:</b>
28   * &lt;<b>State</b> <b>value</b>="#{the State object}"
29   *        <b>stateID</b>="State number"
30   *        <b>player</b>="#{the player to print the state for (may be the test player)}"
31   *        <b>editorMode</b>="#{boolean : toggle editor or player export mode}" /%gt;
32   *
33   * editorMode: is used regardless currentUser permission (this is quite OK
34   *             for the time since this component is only included from a UIGameModel instance,
35   *             who has already checked such a permission...)
36   * </pre>
37   * 
38  * See WEB-INF/web.xml & WEB-INF/wegas-taglib.xml for tag and params definitions
39   *
40   * @author Maxence Laurent (maxence.laurent at gmail.com)
41   */
42  @FacesComponent("com.wegas.app.pdf.uicomponent.State")
43  public class UIState extends UIComponentBase {
44  
45      private Boolean editorMode;
46  
47      public UIState(){
48          super();
49      }
50  
51      public UIState(State state, Long id, Player player, Boolean editorMode, Boolean defaultValues) {
52          this();
53          getAttributes().put("value", state);
54          getAttributes().put("stateID", id);
55          getAttributes().put("player", player);
56          getAttributes().put("editorMode", editorMode);
57          getAttributes().put("defaultValues", defaultValues);
58      }
59  
60      @Override
61      public String getFamily() {
62          return "com.wegas.app.pdf.uicomponent.State";
63      }
64  
65      /**
66       * Print State Machine State Please use encodeAll();
67       *
68       * @param context
69       * @throws IOException
70       */
71      @Override
72      public void encodeBegin(FacesContext context) throws IOException {
73          super.encodeBegin(context);
74          ResponseWriter writer = context.getResponseWriter();
75  
76          State state = (State) getAttributes().get("value");
77          Long id = (Long) getAttributes().get("stateID");
78          editorMode = (Boolean) getAttributes().get("editorMode");
79          Boolean defaultValues = (Boolean) getAttributes().get("defaultValues");
80  
81          UIHelper.startDiv(writer, UIHelper.CSS_CLASS_VARIABLE_CONTAINER);
82          UIHelper.printText(context, writer, state.getClass().getSimpleName(), UIHelper.CSS_CLASS_VARIABLE_SUBTITLE);
83          UIHelper.printProperty(context, writer, UIHelper.TEXT_ID, id.toString());
84          UIHelper.printProperty(context, writer, UIHelper.TEXT_NAME, state.getLabel());
85  
86          if (state instanceof DialogueState) {
87              UIHelper.printPropertyTextArea(context, writer, UIHelper.TEXT_TEXT, ((DialogueState) state).getText(), false, editorMode);
88          }
89  
90          UIHelper.printPropertyImpactScript(context, writer, UIHelper.TEXT_ON_ENTER_IMPACT, state.getOnEnterEvent());
91  
92          UIHelper.startDiv(writer, UIHelper.CSS_CLASS_FOLDER);
93  
94          if (state.getTransitions().isEmpty()) {
95              UIHelper.printText(context, writer, "THERE IS NO TRANSITIONS", UIHelper.CSS_CLASS_ERROR);
96          } else {
97              for (Transition t : state.getTransitions()) {
98                  encode(context, writer, t);
99              }
100         }
101 
102         UIHelper.endDiv(writer);
103         UIHelper.endDiv(writer);
104     }
105 
106     public void encode(FacesContext context, ResponseWriter writer, Transition transition) throws IOException {
107         UIHelper.startDiv(writer, UIHelper.CSS_CLASS_VARIABLE_CONTAINER);
108         UIHelper.printText(context, writer, transition.getClass().getSimpleName(), UIHelper.CSS_CLASS_VARIABLE_SUBTITLE);
109 
110         UIHelper.printProperty(context, writer, UIHelper.TEXT_ID, transition.getId());
111         UIHelper.printProperty(context, writer, UIHelper.TEXT_INDEX, transition.getIndex());
112         UIHelper.printProperty(context, writer, UIHelper.TEXT_NEXT_STATE, transition.getNextStateId());
113 
114         if (transition instanceof DialogueTransition) {
115             UIHelper.printPropertyTextArea(context, writer, UIHelper.TEXT_TEXT, ((DialogueTransition) transition).getActionText(), true, editorMode);
116         }
117 
118         UIHelper.printPropertyScript(context, writer, UIHelper.TEXT_CONDITION, transition.getTriggerCondition());
119         UIHelper.printPropertyImpactScript(context, writer, UIHelper.TEXT_IMPACT_TEXT, transition.getPreStateImpact());
120 
121         UIHelper.endDiv(writer);
122     }
