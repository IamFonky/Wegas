25  public class ImpactPrinter {
26  
27      private final String script;
28  
29      /**
30       *
31       * @param script the script to print
32       */
33      public ImpactPrinter(String script) {
34          // Replace all new line by a semicolon  and ensure there is a ";" at the end
35          this.script = script.replaceAll("\r\n|\n|\r", ";") + ";";
36      }
37  
38      /**
39       * split the script into instructions
40       *
41       * ;, { and }
42       *
43       * @param script
44       * @return
45       */
46      private List<String> parse() {
47          List<String> instructions;
48          instructions = new ArrayList<>();
49  
50          // first char is never quoted nor escaped
51          boolean inQuotes = false;
52          boolean escaped = false;
53  
54          StringBuilder b = new StringBuilder();
55          for (char c : script.toCharArray()) {
56              switch (c) {
57                  case '{':
58                  case '}':
59                  case ';':
60                      b.append(c);
61                      if (!inQuotes) {
62                          instructions.add(b.toString());
63                          b = new StringBuilder();
64                      }
65                      break;
66                  case '\"': // detect unescaped quote
67                      if (!escaped) {
68                          inQuotes = !inQuotes;
69                      }
70                  default:
71                      b.append(c);
72                      break;
73              }
74              // if current char is '\' mark next char escaped (unless the '\' itself is escaped)
75              escaped = (!escaped && c == '\\');
76          }
77          instructions.add(b.toString());
78          return instructions;
79      }
80  
81      /**
82       * Impact Pretty Printer
83       *
84       * @param context
85       * @param writer
86       * @throws IOException
87       */
88      public void print(FacesContext context, ResponseWriter writer) throws IOException {
89  
90          List<String> instructions;
91          instructions = parse();
92          //instructions = script.split("(?s)[;\n](?=(?:(?:.*?(?<!\\\\)\"){2})*[^\"]*$)");
93          UIHelper.startTextArea(writer);
94          for (String instruction : instructions) {
95              // Avoid printg empty lines (i.e simple parser scrap...)
96              if (!instruction.matches("^\\s*[;]\\s*$")) {
97                  printInstruction(context, writer, instruction);
98              }
99          }
100         UIHelper.endTextArea(writer);
101     }
102 
103     /**
104      * Instruction pretty printer. Will try to detect common instruction pattern
105      * such Variable.find(x).add(5)
106      *
107      * @param context
108      * @param writer
109      * @param instruction
110      * @throws IOException
111      */
112     private void printInstruction(FacesContext context, ResponseWriter writer, String instruction) throws IOException {
113         Pattern pattern;
114         String str = instruction.trim();
115         //pattern = Pattern.compile("VariableDescriptorFacade.find\\(gameModel, \"*)\"\\)\\.(.*)\\(self, *(.*)\\)");
116 
117         if (!str.isEmpty()) {
118             // Unndent block (Parser create a new instruction after either a "{" or a "}", btw such a char always stands just before $)
119             if (str.matches(".*\\}$")) {
120                 UIHelper.endDiv(writer);
121             }
122 
123             pattern = Pattern.compile("(VariableDescriptorFacade|Variable)\\.find\\(gameModel, \"(.*)\"\\)\\.(.*)\\(self, *(.*)\\).*;$");
124 
125             Matcher matcher = pattern.matcher(str);
126 
127             if (matcher.matches()) {
128                 if (matcher.matches()) {
129                     //String group = matcher.group(0);
130                     String variableAlias = matcher.group(2);
131                     String operator = matcher.group(3);
132                     String value = matcher.group(4);
133 
134                     /*
135                      * Detect specific operation
136                      */
137                     if ("sendMessage".equals(operator)) { // e-mail like message
138                         String[] args;
139                         args = value.split("(?s)[,](?=(?:(?:.*?(?<!\\\\)\"){2})*[^\"]*$)"); // arg separation
140                         UIHelper.printMessage(context, writer, variableAlias, args[0], args[1], null, args[2], null);
141                                                               // TO           FROM     SUBJECT  DATE    BODY
142                     } else if ("sendDatedMessage".equals(operator)) { // dated e-mail like message
143                         String[] args;
144                         args = value.split("(?s)[,](?=(?:(?:.*?(?<!\\\\)\"){2})*[^\"]*$)"); // arg separation
145                         UIHelper.printMessage(context, writer, variableAlias, args[0], args[2], args[1], args[3], null);
146                                                               // TO           FROM     SUBJECT  DATE    BODY
147                     } else {
148                         String var = variableAlias;
149                         String op;
150                         switch (operator) {
151                             case "add":
152                                 op = "+";
153                                 var = variableAlias + " = " + variableAlias;
154                                 break;
155                             case "sub":
156                                 op = "-";
157                                 var = variableAlias + " = " + variableAlias;
158                                 break;
159                             case "set":
160                                 op = "=";
161                                 break;
162                             default:
163                                 op = operator;
164                                 break;
165                         }
166                         String print = var + " " + op + " " + value;
167                         UIHelper.printTextAreaText(context, writer, print, UIHelper.CSS_CLASS_PROPERTY_VALUE_TEXTAREA + " " + UIHelper.CSS_CLASS_SOURCE_CODE, true);
168                     }
169                 }
170             } else {
171                 UIHelper.printTextAreaText(context, writer, instruction, UIHelper.CSS_CLASS_PROPERTY_VALUE_TEXTAREA + " " + UIHelper.CSS_CLASS_SOURCE_CODE, true);
172             }
173 
174             // Indent block (Parser create a new instruction after a "{", implying such a char always stands just before $)
175             if (str.matches(".*\\{$")) {
176                 UIHelper.startDiv(writer, UIHelper.CSS_CLASS_INDENT_CODE);
177             }
178         }
179     }
