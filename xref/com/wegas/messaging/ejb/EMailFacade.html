32  public class EMailFacade {

36      /**
37       *
38       * @param to
39       * @param from
40       * @param subject
41       * @param body
42       * @param toType
43       * @param mimetype
44       * @throws javax.mail.MessagingException when something went wrong
45       */
46      public void send(String to, String from, String replyTo,
47              String subject, String body, RecipientType toType, String mimetype) throws MessagingException {
48  
49          Properties props = new Properties();
50          final String username = Helper.getWegasProperty("mail.smtp.username");
51          final String password = Helper.getWegasProperty("mail.smtp.password");
52  
53          props.put("mail.smtp.host", Helper.getWegasProperty("mail.smtp.host"));           // Attaching to default Session, or we could start a new one
54  
55          props.setProperty("mail.smtp.auth", Helper.getWegasProperty("mail.smtp.auth"));
56          props.put("mail.smtp.port", Helper.getWegasProperty("mail.smtp.port"));
57          if (Helper.getWegasProperty("mail.smtp.starttls.enable").equals("true")) {
58              props.put("mail.smtp.starttls.enable", "true");
59              props.put("mail.smtp.ssl.trust", Helper.getWegasProperty("mail.smtp.host"));
60          } else {
61              props.put("mail.smtp.socketFactory.class", "javax.net.ssl.SSLSocketFactory");
62              props.put("mail.smtp.socketFactory.port", Helper.getWegasProperty("mail.smtp.port"));
63              props.put("mail.smtp.ssl.trust", Helper.getWegasProperty("mail.smtp.host"));
64          }
65  
66          Session session = Session.getInstance(props, new Authenticator() {
67              @Override
68              protected PasswordAuthentication getPasswordAuthentication() {
69                  return new PasswordAuthentication(username, password);
70              }
71          });
72  
73          javax.mail.Message msg = new MimeMessage(session);
74  
75          if (replyTo != null){
76              msg.setHeader("Reply-To", replyTo);
77          }
78          msg.setFrom(new InternetAddress(from));
79          msg.setRecipients(toType, InternetAddress.parse(to, false));
80          msg.setSubject(subject);
81          msg.setContent(body, mimetype);
82          msg.setSentDate(new Date());
83          Transport.send(msg);
84      }
