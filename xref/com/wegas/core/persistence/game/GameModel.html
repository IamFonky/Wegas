549     @PostPersist
550     private void storePages() {
551         if (this.pages != null) {
552             try (final Pages pagesDAO = new Pages(this.id.toString())) {
553                 pagesDAO.delete();                                              // Remove existing pages
554                 for (Entry<String, JsonNode> p : this.pages.entrySet()) {       // Add all pages
555                     pagesDAO.store(new Page(p.getKey(), p.getValue()));
556                 }
557 
558             } catch (RepositoryException ex) {
559                 System.err.println("Failed to create repository for GameModel " + this.id);
560             }
561         }
562 
563     }
