10  import java.io.IOException;
11  import java.io.OutputStream;
12  import java.nio.charset.Charset;
13  import javax.ejb.Stateless;
14  import javax.ws.rs.Consumes;
15  import javax.ws.rs.POST;
16  import javax.ws.rs.Path;
17  import javax.ws.rs.PathParam;
18  import javax.ws.rs.WebApplicationException;
19  import javax.ws.rs.core.MediaType;
20  import javax.ws.rs.core.Response;
21  import javax.ws.rs.core.StreamingOutput;
22  import org.glassfish.jersey.media.multipart.FormDataParam;
23  
24  /**
25   *
26   * @author Cyril Junod <cyril.junod at gmail.com>
27   */
28  @Stateless
29  @Path("Download")
30  public class DownloadController {
31  
32      /**
33       *
34       * @param contentType
35       * @param filename
36       * @param data
37       * @return
38       */
39      @POST
40      @Consumes(MediaType.MULTIPART_FORM_DATA)
41      @Path("{filename}")
42      public static Response forward(@FormDataParam("ctype") String contentType, @PathParam("filename") String filename, @FormDataParam("data") final String data) {
43          return Response.ok(new StreamingOutput() {
44              @Override
45              public void write(OutputStream out) throws IOException, WebApplicationException {
46                  out.write(new byte[]{(byte) 0xEF, (byte) 0xBB, (byte) 0xBF}); //UTF8 BOM
47                  out.write(data.getBytes(Charset.forName("UTF-8")));
48              }
49          }, contentType).header("Content-Disposition", "attachment; filename=" + filename).build();
50      }
51  }
