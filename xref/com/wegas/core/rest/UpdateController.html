101     @GET
102     @Path("Encode/{gameModelId : ([1-9][0-9]*)}")
103     public String encode(@PathParam("gameModelId") Long gameModelId) {
104         List<VariableDescriptor> findAll = descriptorFacade.findByGameModelId(gameModelId);
105         for (VariableDescriptor vd : findAll) {
106             List<String> findDistinctNames = descriptorFacade.findDistinctNames(vd.getGameModel());
107             List<String> findDistinctLabels = descriptorFacade.findDistinctLabels(vd.getGameModel());
108             findDistinctNames.remove(vd.getName());
109             findDistinctLabels.remove(vd.getLabel());
110             Helper.setUniqueName(vd, findDistinctNames);
111             Helper.setUniqueLabel(vd, findDistinctLabels);
112             descriptorFacade.flush();
113         }
114         return "Finished";
115     }

122     @GET
123     @Path("UpdateScript/{gameModelId : ([1-9][0-9]*)}")
124     public String script(@PathParam("gameModelId") Long gameModelId) {
125         List<VariableDescriptor> findAll = descriptorFacade.findAll(gameModelId);
126         List<String> keys = new ArrayList<>();
127         List<String> values = new ArrayList<>();
128         for (VariableDescriptor vd : findAll) {
129             keys.add("VariableDescriptorFacade\\.find\\(" + vd.getId() + "\\)");
130             values.add("VariableDescriptorFacade.find(gameModel, \"" + vd.getName() + "\")");
131         }
132 
133         for (VariableDescriptor vd : findAll) {
134             if (vd instanceof ChoiceDescriptor) {
135                 ChoiceDescriptor choice = (ChoiceDescriptor) vd;
136                 for (Result r : choice.getResults()) {
137                     replaceAll(r.getImpact(), keys, values);
138                 }
139             } else if (vd instanceof StateMachineDescriptor) {
140                 StateMachineDescriptor fsm = (StateMachineDescriptor) vd;
141                 for (Map.Entry<Long, State> s : fsm.getStates().entrySet()) {
142                     replaceAll(s.getValue().getOnEnterEvent(), keys, values);
143                     for (Transition t : s.getValue().getTransitions()) {
144                         replaceAll(t.getPreStateImpact(), keys, values);
145                         replaceAll(t.getTriggerCondition(), keys, values);
146                     }
147                 }
148             }
149         }
150         return "Finished";
151     }

153     private static void replaceAll(Script script, List<String> a, List<String> b) {
154         if (script == null) {
155             return;
156         }
157         String s = script.getContent();
158         for (int i = 0; i < a.size(); i++) {
159             s = s.replaceAll(a.get(i), b.get(i));
160         }
161         script.setContent(s);
162     }

180     private String addVariable(GameModel gm, String json, String varName, String parentName) {
181         ObjectMapper mapper = JacksonMapperProvider.getMapper();
182         logger.error("Going to add " + parentName + "/" + varName + " variable");
183 
184         try {
185             // Does the variable already exists ? 
186             descriptorFacade.find(gm, varName);
187             logger.error("  -> variable " + varName + " exists : SKIP");
188             return "already exists";
189         } catch (WegasNoResultException ex) {
190             logger.error("  -> variable " + varName + " not found : PROCEED");
191         }
192 
193         try {
194             // assert the parent already exists ? 
195             descriptorFacade.find(gm, parentName);
196             logger.error("  -> variable " + parentName + " exists : PROCEED");
197         } catch (WegasNoResultException ex) {
198             logger.error("  -> variable " + parentName + " not found : FAILED");
199             return "parent not found";
200         }
201         try {
202             VariableDescriptor vd = mapper.readValue(json, VariableDescriptor.class);
203             descriptorController.createChild(gm.getId(), parentName, vd);
204             em.flush();
205             return "OK";
206         } catch (WegasNotFoundException ex) {
207             logger.error("Error white adding the variable : parent " + parentName + " not found", ex);
208             return "Parent (2) not found";
209         } catch (IOException ex) {
210             logger.error("Error While Reading JSON: " + json, ex);
211             return "JSON Error";
212         }
213     }

289     private List<Game> findNoDebugTeamGames() {
290         final CriteriaBuilder criteriaBuilder = em.getCriteriaBuilder();
291         final CriteriaQuery query = criteriaBuilder.createQuery();
292 
293         Root e = query.from(Game.class);
294         query.select(e);
295 
296         List<Game> games = em.createQuery(query).getResultList();
297         List<Game> noDebugTeamGames = new ArrayList<>();
298         for (Game g : games) {
299             if (!(g instanceof DebugGame)) {
300 
301                 List<Team> teams = g.getTeams();
302                 boolean hasDebugTeam = false;
303                 for (Team t : teams) {
304                     if (t instanceof DebugTeam) {
305                         hasDebugTeam = true;
306                         break;
307                     }
308                 }
309                 if (!hasDebugTeam) {
310                     noDebugTeamGames.add(g);
311                 }
312             }
313         }
314         return noDebugTeamGames;
315     }
