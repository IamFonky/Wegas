10  import java.io.IOException;
11  import java.lang.annotation.Annotation;
12  import java.lang.reflect.Method;
13  import java.util.ArrayList;
14  import java.util.Arrays;
15  import java.util.List;
16  import javax.ws.rs.container.ContainerRequestContext;
17  import javax.ws.rs.container.ContainerRequestFilter;
18  import javax.ws.rs.container.ResourceInfo;
19  import javax.ws.rs.core.Context;
20  import javax.ws.rs.core.Response;
21  import javax.ws.rs.ext.Provider;
22  import org.apache.shiro.SecurityUtils;
23  import org.apache.shiro.authz.Permission;
24  import org.apache.shiro.authz.annotation.RequiresAuthentication;
25  import org.apache.shiro.authz.annotation.RequiresPermissions;
26  import org.apache.shiro.authz.annotation.RequiresRoles;
27  import org.apache.shiro.authz.permission.WildcardPermission;
28  import org.apache.shiro.subject.Subject;
29  import org.slf4j.Logger;
30  import org.slf4j.LoggerFactory;

32  /**
33   *
34   * Assert current user has all sufficient right to use the targeted REST
35   * resource
36   *
37   * @author Maxence Laurent (maxence.laurent gmail.com)
38   */
39  @Provider
40  public class SecurityFilter implements ContainerRequestFilter {
41  
42      private static final Logger logger = LoggerFactory.getLogger(SecurityFilter.class);
43  
44      /*
45       * Post-matching filter knows the targeted resource by injecting 
46       * such a ResourceInfo
47       */
48      @Context
49      private ResourceInfo resourceInfo;
50  
51      private <T extends Annotation> T getAnnotation(Class<T> annotation, Class<? extends Object> klass, Method method) {
52          T a = method.getAnnotation(annotation);
53          if (a == null) {
54              a = klass.getAnnotation(annotation);
55          }
56          return a;
57      }
58  
59      @Override
60      public void filter(ContainerRequestContext crc) throws IOException {
61  
62          // Targeted Class & method
63          final Class<? extends Object> runtimeClass = resourceInfo.getResourceClass();
64          Method method = resourceInfo.getResourceMethod();
65  
66          Subject subject = SecurityUtils.getSubject();
67  
68          /*
69           * Is authentication required ?
70           */
71          RequiresAuthentication authRequired = this.getAnnotation(RequiresAuthentication.class, runtimeClass, method);
72  
73          if (authRequired != null) {
74              // Annotation found, assert subject is authenticated
75              if (!subject.isAuthenticated() && !subject.isRemembered()) {
76                  logger.error("Access denied for non-authenticted users");
77                  crc.abortWith(Response.status(Response.Status.UNAUTHORIZED).entity("Subject is not logged in").build());
78              }
79          }
80          /* 
81           * Check if specific roles are required 
82           */
83          RequiresRoles requiredRoles = this.getAnnotation(RequiresRoles.class, runtimeClass, method);
84  
85          if (requiredRoles != null) {
86              List<String> listOfRoles = Arrays.asList(requiredRoles.value());
87  
88              boolean[] boolRoles = subject.hasRoles(listOfRoles);
89              boolean roleVerified = false;
90  
91              for (boolean b : boolRoles) {
92                  if (b) {
93                      roleVerified = true;
94                      break;
95                  }
96              }
97              if (!roleVerified) {
98                  String msg = "Access denied. User doesn't have enough privilege Roles:"
99                          + listOfRoles + " to access this page.";
100                 logger.error("Access denied missing roles");
101                 crc.abortWith(Response.status(Response.Status.FORBIDDEN).entity(msg).build());
102             }
103         }
104         /*
105          * and lastly check for permissions
106          */
107         RequiresPermissions requiredPermissions = getAnnotation(RequiresPermissions.class, runtimeClass, method);
108 
109         if (requiredPermissions != null) {
110             List<String> listOfPermissionsString = Arrays.asList(requiredPermissions.value());
111 
112             List<Permission> listOfPermissions = new ArrayList<>();
113             for (String p : listOfPermissionsString) {
114                 listOfPermissions.add((Permission) new WildcardPermission(p));
115             }
116             boolean[] boolPermissions = subject.isPermitted(listOfPermissions);
117             boolean permitted = false;
118             for (boolean b : boolPermissions) {
119                 if (b) {
120                     permitted = true;
121                     break;
122                 }
123             }
124             if (!permitted) {
125                 String msg = "Access denied. User doesn't have enough privilege Permissions:"
126                         + listOfPermissions + " to access this page.";
127                 logger.error("Access denied: missing permissions");
128                 crc.abortWith(Response.status(Response.Status.FORBIDDEN).entity(msg).build());
129             }
130         }
131     }
