8   package com.wegas.core.rest.exception;
9   
10  import javax.ejb.EJBException;
11  import javax.enterprise.event.ObserverException;
12  import javax.persistence.PersistenceException;
13  import javax.transaction.RollbackException;
14  import javax.transaction.TransactionRolledbackException;
15  import javax.ws.rs.core.Response;
16  import org.eclipse.persistence.exceptions.DatabaseException;
17  import org.slf4j.Logger;
18  import org.slf4j.LoggerFactory;
19  
20  /**
21   *
22   * @author Francois-Xavier Aeberhard <fx@red-agent.com>
23   */
24  public abstract class AbstractExceptionMapper {
25  
26      final static private Logger logger = LoggerFactory.getLogger(AbstractExceptionMapper.class);
27  
28      /**
29       *
30       * @param exception
31       * @return
32       */
33      public static Response processException(Throwable exception) {
34          logger.warn("ProcessException: " + exception);
35  
36          if (exception instanceof RollbackException){
37             return Response.status(Response.Status.FORBIDDEN).entity(exception).build();
38         } else if (exception instanceof RollbackException
39                  || exception instanceof TransactionRolledbackException
40                  || exception instanceof ObserverException
41                  || exception instanceof PersistenceException
42                  //                || exception instanceof javax.persistence.PersistenceException
43                  || exception instanceof org.omg.CORBA.TRANSACTION_ROLLEDBACK) {
44              return processException(exception.getCause());
45  
46          } else if (exception instanceof EJBException) {
47              return processException(((EJBException) exception).getCausedByException());
48  
49          } else if (exception instanceof DatabaseException) {
50              DatabaseException dbe = (DatabaseException) exception;
51              return processException(dbe.getInternalException());
52          } else {
53              logger.error(exception.getLocalizedMessage());
54              return Response.status(Response.Status.BAD_REQUEST).entity(exception).build();
55          }
56      }
57  }
