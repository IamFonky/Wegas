145     /**

152     public Player findByGameId(Long gameId) throws WegasNoResultException {
153         Query getByGameId = getEntityManager().createQuery("SELECT player FROM Player player WHERE player.team.game.id = :gameId " +

220     /**
221      * Reset a player
222      *
223      * @param player the player to reset
224      */
225     public void reset(final Player player) {
226         // Need to flush so prepersit events will be thrown (for example Game will add default teams)
227         getEntityManager().flush();
228         player.getGameModel().propagateDefaultInstance(player);
229         getEntityManager().flush();
230         // Send an reset event (for the state machine and other)
231         resetEvent.fire(new ResetEvent(player));
232     }
